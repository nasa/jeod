93 /**
94  * Update the solid-body tidal delta-coefficients.
95  * \param[in] controls Gravity controls for planet
96  */
97 void
98 SphericalHarmonicsSolidBodyTides::update (
99  SphericalHarmonicsGravityControls & controls JEOD_UNUSED)
100 
101 {
102  double pfix_position[3];
103  double F = 0.0;
104  double deltaC20;
105 
106  for (unsigned int ii = 0; ii < num_tidal_bodies; ++ii) {
107  tidal_bodies_inertial[ii]->compute_position_from (*pfix,
108  pfix_position);
109 
110  double tidal_body_mu = tidal_bodies[ii]->grav_source->mu;
111 
112  double r = Vector3::vmag (pfix_position);
113  double phi = asin (pfix_position[2] / r);
114  double Rovr_3 = grav_source->radius / r;
115  Rovr_3 *= Rovr_3 * Rovr_3;
116 
117  F += tidal_body_mu / grav_source->mu* Rovr_3* sqrt (5.0) *
118  (1.5 * sin (phi) * sin (phi) - 0.5);
119  }
120 
121  deltaC20 = k2 / 5.0 * F;
122  dC20 = deltaC20;
123 
124  return;
