65 /**
66  * Update the solid-body tidal delta-coefficients.
67  * \param[in] controls Gravity controls for planet
68  */
69 void SphericalHarmonicsSolidBodyTides::update(SphericalHarmonicsGravityControls & controls JEOD_UNUSED)
70 {
71  double pfix_position[3];
72  double F = 0.0;
73  double deltaC20;
74 
75  for(unsigned int ii = 0; ii < num_tidal_bodies; ++ii)
76  {
77  tidal_bodies_inertial[ii]->compute_position_from(*pfix, pfix_position);
78 
79  double tidal_body_mu = tidal_bodies[ii]->grav_source->mu;
80 
81  double r = Vector3::vmag(pfix_position);
82  double phi = asin(pfix_position[2] / r);
83  double Rovr_3 = grav_source->radius / r;
84  Rovr_3 *= Rovr_3 * Rovr_3;
85 
86  F += tidal_body_mu / grav_source->mu * Rovr_3 * sqrt(5.0) * (1.5 * sin(phi) * sin(phi) - 0.5);
87  }
88 
89  deltaC20 = k2 / 5.0 * F;
90  dC20 = deltaC20;
91 }
