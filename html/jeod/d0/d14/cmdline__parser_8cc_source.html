191 /**
192  * Parse the command line.
193  * \return Parse status; zero=OK, non-zero=error.
194  * \param[in,out] argc The number of command line entries.
195  * \param[in,out] argv The command line entries.
196  */
197 int CmdlineParser::parse(int & argc, char ** argv)
198 {
199  IntList delete_list;
200  int status = 0;
201 
202  if(parsed)
203  {
204  MessageHandler::error(__FILE__,
205  __LINE__,
206  TestHarnessMessages::inconsistent_setup,
207  "CmdlineParser::parse() has already been called.");
208  return 1;
209  }
210 
211  unsigned int nopts = option_table.size();
212  unsigned int iopt = 0;
213  option * options = new option[nopts + 1];
214 
215  // Establish the options array based on the options added via
216  // the previous calls to CmdlineParser::add_XXX.
217  for(OptionTable::iterator iter = option_table.begin(); iter != option_table.end(); ++iter)
218  {
219  iter->second->set_default();
220 
221  BaseOptionValue * iter_opt = iter->second;
222  options[iopt].name = iter_opt->name.c_str();
223  if(iter->second->takes_argument())
224  {
225  options[iopt].has_arg = required_argument;
226  }
227  else
228  {
229  options[iopt].has_arg = no_argument;
230  }
231  options[iopt].flag = nullptr;
232  options[iopt].val = 1;
233  ++iopt;
234  }
235 
236  // The options array must be terminated with a null entry.
237  options[iopt].name = nullptr;
238  options[iopt].has_arg = 0;
239  options[iopt].flag = nullptr;
240  options[iopt].val = 0;
241 
242  // Parse the command line.
243  while(status == 0)
244  {
245  int cur_index = optind;
246  int option_index;
247  int optc = getopt_long_only(argc, argv, "", options, &option_index);
248 
249  // At the end: Exit the loop.
250  if(optc == -1)
251  {
252  break;
253  }
254 
255  // Unrecognized / ambiguous option: Report.
256  if(optc == '?')
257  {
258  MessageHandler::warn(__FILE__,
259  __LINE__,
260  TestHarnessMessages::invalid_option,
261  "Unrecognized command line option '%s'",
262  argv[cur_index]);
263  status = 2;
264  }
265 
266  // Missing argument: Report.
267  else if(optc == ':')
268  {
269  MessageHandler::warn(__FILE__,
270  __LINE__,
271  TestHarnessMessages::invalid_option,
272  "Missing argument for option '%s'",
273  argv[cur_index]);
274  status = 3;
275  }
276 
277  // getopt_long_only parse was OK: Process the option.
278  else
279  {
280  OptionTable::iterator iter(option_table.find(options[option_index].name));
281  assert(iter != option_table.end());
282  if(options[option_index].has_arg == no_argument)
283  {
284  iter->second->parse_value(nullptr);
285  delete_list.push_front(cur_index);
286  }
287  else
288  {
289  iter->second->parse_value(argv[cur_index + 1]);
290  delete_list.push_front(cur_index);
291  delete_list.push_front(cur_index + 1);
292  }
293  }
294  }
295 
296  // Delete the parsed options from argv, leaving only the unparsed items.
297  if(status == 0)
298  {
299  for(IntList::iterator iter = delete_list.begin(); iter != delete_list.end(); ++iter)
300  {
301  for(int darg = *iter; darg < argc; ++darg)
302  {
303  argv[darg] = argv[darg + 1];
304  }
305  --argc;
306  }
307  }
308 
309  // Cleanup: Mark the command line as parsed and delete temporaries.
310  parsed = true;
311 
312  delete[] options;
313  return status;
314 }

338  * Ensure that the option name has not yet been registered.
339  * \param[in] file __FILE_
340  * \param[in] line __LINE_
341  * \param[in] method_name The method name, used for error reporting.
342  * \param[in] option_name The option name, sans the leading hyphen.

344 bool CmdlineParser::check_registration(const char * file,
345  int line,
346  const char * method_name,
347  const std::string & option_name)
348 {
349  bool is_ok;
350 
351  // The option name must not be a duplicate entry.
352  // Failure is a non-fatal error; this lets the programmers see if they have
353  // goofed up elsewhere.
354  if(option_table.find(option_name) != option_table.end())
355  {
356  MessageHandler::error(file,
357  line,
358  TestHarnessMessages::inconsistent_setup,
359  "CmdlineParser::%s(\"%s\") error: Name is already registered.",
360  method_name,
361  option_name.c_str());
362 
363  is_ok = false;
364  }
365 
366  // Lookup is past end of table => option name has not been registered.
367  else
368  {
369  is_ok = true;
370  }
371 
372  return is_ok;
373 }
