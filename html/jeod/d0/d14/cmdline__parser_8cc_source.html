245 /**
246  * Parse the command line.
247  * \return Parse status; zero=OK, non-zero=error.
248  * \param[in,out] argc The number of command line entries.
249  * \param[in,out] argv The command line entries.
250  */
251 int
252 CmdlineParser::parse (
253  int & argc,
254  char ** argv)
255 {
256  IntList delete_list;
257  int status = 0;
258 
259  if (parsed) {
260  MessageHandler::error (
261  __FILE__, __LINE__, TestHarnessMessages::inconsistent_setup,
262  "CmdlineParser::parse() has already been called.");
263  return 1;
264  }
265 
266  unsigned int nopts = option_table.size();
267  unsigned int iopt = 0;
268  option * options = new option[nopts+1];
269 
270  // Establish the options array based on the options added via
271  // the previous calls to CmdlineParser::add_XXX.
272  for (OptionTable::iterator iter = option_table.begin();
273  iter != option_table.end();
274  ++iter) {
275  iter->second->set_default ();
276 
277  BaseOptionValue * iter_opt = iter->second;
278  options[iopt].name = iter_opt->name.c_str();
279  options[iopt].has_arg = iter->second->takes_argument() ?
280  required_argument : no_argument;
281  options[iopt].flag = NULL;
282  options[iopt].val = 1;
283  ++iopt;
284  }
285 
286  // The options array must be terminated with a null entry.
287  options[iopt].name = NULL;
288  options[iopt].has_arg = 0;
289  options[iopt].flag = NULL;
290  options[iopt].val = 0;
291 
292  // Parse the command line.
293  while (status == 0) {
294  int cur_index = optind;
295  int option_index;
296  int optc = getopt_long_only (argc, argv, "", options, &option_index);
297 
298  // At the end: Exit the loop.
299  if (optc == -1) {
300  break;
301  }
302 
303  // Unrecognized / ambiguous option: Report.
304  if (optc == '?') {
305  MessageHandler::warn (
306  __FILE__, __LINE__, TestHarnessMessages::invalid_option,
307  "Unrecognized command line option '%s'", argv[cur_index]);
308  status = 2;
309  }
310 
311  // Missing argument: Report.
312  else if (optc == ':') {
313  MessageHandler::warn (
314  __FILE__, __LINE__, TestHarnessMessages::invalid_option,
315  "Missing argument for option '%s'", argv[cur_index]);
316  status = 3;
317  }
318 
319  // getopt_long_only parse was OK: Process the option.
320  else {
321  OptionTable::iterator iter (
322  option_table.find(options[option_index].name));
323  assert (iter != option_table.end());
324  if (options[option_index].has_arg == no_argument) {
325  iter->second->parse_value (NULL);
326  delete_list.push_front (cur_index);
327  }
328  else {
329  iter->second->parse_value (argv[cur_index+1]);
330  delete_list.push_front (cur_index);
331  delete_list.push_front (cur_index+1);
332  }
333  }
334  }
335 
336  // Delete the parsed options from argv, leaving only the unparsed items.
337  if (status == 0) {
338  for (IntList::iterator iter = delete_list.begin();
339  iter != delete_list.end();
340  ++iter) {
341  for (int darg = *iter; darg < argc; ++darg) {
342  argv[darg] = argv[darg+1];
343  }
344  --argc;
345  }
346  }
347 
348  // Cleanup: Mark the command line as parsed and delete temporaries.
349  parsed = true;
350 
351  delete [] options;
352  return status;
353 }

381  * Ensure that the option name has not yet been registered.
382  * \param[in] file __FILE_
383  * \param[in] line __LINE_
384  * \param[in] method_name The method name, used for error reporting.

387 bool
388 CmdlineParser::check_registration (
389  const char * file,
390  int line,
391  const char * method_name,
392  const std::string & option_name)
393 {
394  bool is_ok;
395 
396  // The option name must not be a duplicate entry.
397  // Failure is a non-fatal error; this lets the programmers see if they have
398  // goofed up elsewhere.
399  if (option_table.find (option_name) != option_table.end()) {
400  MessageHandler::error (
401  file, line, TestHarnessMessages::inconsistent_setup,
402  "CmdlineParser::%s(\"%s\") error: Name is already registered.",
403  method_name, option_name.c_str());
404 
405  is_ok = false;
406  }
407 
408  // Lookup is past end of table => option name has not been registered.
409  else {
410  is_ok = true;
411  }
412 
413  return is_ok;
414 }
