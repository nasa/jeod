80 /**
81  * Initializes the radiation surface.
82  * \param[in] center_grav position of center of mass\n Units: M
83  */
84 void
85 RadiationSurface::initialize (
86  double center_grav[3])
87 {
88  if (num_facets == 0) {
89  MessageHandler::fail (
90  __FILE__, __LINE__, RadiationMessages::operational_setup_error, "\n"
91  "The initialization routine for Radiation Surface was called with no "
92  "facets. \n"
93  "This is likely an error, possibly caused by initializing the surface "
94  "before creating it. \n"
95  "If there are no facets in the radiation surface, eliminate the "
96  "radiation pressure model from the simulation. \n");
97  }
98 
99  for (ii_facet = 0; ii_facet < num_facets; ++ii_facet) {
100  facets[ii_facet]->initialize_geom (center_grav);
101  }
102 
103  for (ii_facet = 0; ii_facet < num_facets; ++ii_facet) {
104  if ((facets[ii_facet]->albedo > 1) || (facets[ii_facet]->albedo < 0)) {
105  MessageHandler::fail (
106  __FILE__, __LINE__, RadiationMessages::invalid_setup_error, "\n"
107  "The albedo from one of the facet %i is out of range. \n"
108  "Albedo represents a fraction of incident light, and must \n"
109  "be between 0 and 1.\n", ii_facet);
110  }
111  if ((facets[ii_facet]->diffuse > 1) || (facets[ii_facet]->diffuse < 0)) {
112  MessageHandler::fail (
113  __FILE__, __LINE__, RadiationMessages::invalid_setup_error, "\n"
114  "The diffuse parameter from facet %i is out of range. \n"
115  "Diffuse represents a fraction of reflected light, and must \n"
116  "be between 0 and 1.\n", ii_facet);
117  }
118  if ((facets[ii_facet]->thermal.emissivity > 1) ||
119  (facets[ii_facet]->thermal.emissivity < 0)) {
120  MessageHandler::fail (
121  __FILE__, __LINE__, RadiationMessages::invalid_setup_error, "\n"
122  "The emissivity from facet %i is out of range. \n"
123  "Emissivity represents the tendency towards a perfect black surface,\n"
124  "0 representing a perfectly insulating surface (no emission), \n"
125  "and 1 a perfect black body. \n"
126  "The value of thermal.emissivity must be between 0 and 1.\n",
127  ii_facet);
128  }
129  }
130 
131  return;
132 }

166 /**
167  * Turns facet memory into radiaiton facet memory.
168  * \param[in] facet pointer to the facet
169  * \param[in] factory pointer to the facet factory
170  * \param[in] params pointer to the generic facet parameter.
171  * \param[in] index index value in the facet list.
172  */
173 void
174 RadiationSurface::allocate_interaction_facet (
175  Facet* facet,
176  InteractionFacetFactory* factory,
177  FacetParams* params,
178  unsigned int index)
179 {
180  if (num_facets <= index) {
181  MessageHandler::fail (
182  __FILE__, __LINE__, RadiationMessages::invalid_setup_error, "\n"
183  "Attempting to allocate data for facet number %i, but the model has\n"
184  "allocated space for only %i facets (num_facets = %i).\n",
185  index, num_facets, num_facets);
186 
187  return;
188 
189  }
190 
191  /* need to temporarily save off the InteractionFacet returned before
192  dynamic casting it. If the dynamic cast fails, we want to destroy
193  the InteractionFacet so we don't get a memory leak */
194 
195  InteractionFacet* temp_facet = nullptr;
196 
197  temp_facet = factory->create_facet (facet, params);
198 
199  if (temp_facet == nullptr) {
200  MessageHandler::fail (
201  __FILE__, __LINE__, RadiationMessages::operational_setup_error, "\n"
202  "create_facet failed, returning a NULL pointer.\n");
203 
204  return;
205 
206  }
207 
208  RadiationFacet* temp_radiation_facet =
209  dynamic_cast<RadiationFacet*> (temp_facet);
210 
211 
212  if (temp_radiation_facet == nullptr) {
213 
214  // temp_facet can NOT be NULL, since it was already checked for above
215  JEOD_DELETE_OBJECT (temp_facet);
216 
217  MessageHandler::fail (
218  __FILE__, __LINE__, RadiationMessages::operational_setup_error, "\n"
219  "The InteractionFacet temp_facet exists, but it is not a\n"
220  "RadiationFacet; the cast to RadiationFacet failed.\n");
221  return;
222 
223  }
224 
225  facets[index] = temp_radiation_facet;
226 
227  return;
228
