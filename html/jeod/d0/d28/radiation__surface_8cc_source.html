66  * Initializes the radiation surface.
67  * \param[in] center_grav position of center of mass\n Units: M
68  */
69 void RadiationSurface::initialize(double center_grav[3])
70 {
71  if(num_facets == 0)
72  {
73  MessageHandler::fail(__FILE__,
74  __LINE__,
75  RadiationMessages::operational_setup_error,
76  "\n"
77  "The initialization routine for Radiation Surface was called with no "
78  "facets. \n"
79  "This is likely an error, possibly caused by initializing the surface "
80  "before creating it. \n"
81  "If there are no facets in the radiation surface, eliminate the "
82  "radiation pressure model from the simulation. \n");
83  }
84 
85  for(ii_facet = 0; ii_facet < num_facets; ++ii_facet)
86  {
87  facets[ii_facet]->initialize_geom(center_grav);
88  }
89 
90  for(ii_facet = 0; ii_facet < num_facets; ++ii_facet)
91  {
92  if((facets[ii_facet]->albedo > 1) || (facets[ii_facet]->albedo < 0))
93  {
94  MessageHandler::fail(__FILE__,
95  __LINE__,
96  RadiationMessages::invalid_setup_error,
97  "\n"
98  "The albedo from one of the facet %i is out of range. \n"
99  "Albedo represents a fraction of incident light, and must \n"
100  "be between 0 and 1.\n",
101  ii_facet);
102  }
103  if((facets[ii_facet]->diffuse > 1) || (facets[ii_facet]->diffuse < 0))
104  {
105  MessageHandler::fail(__FILE__,
106  __LINE__,
107  RadiationMessages::invalid_setup_error,
108  "\n"
109  "The diffuse parameter from facet %i is out of range. \n"
110  "Diffuse represents a fraction of reflected light, and must \n"
111  "be between 0 and 1.\n",
112  ii_facet);
113  }
114  if((facets[ii_facet]->thermal.emissivity > 1) || (facets[ii_facet]->thermal.emissivity < 0))
115  {
116  MessageHandler::fail(__FILE__,
117  __LINE__,
118  RadiationMessages::invalid_setup_error,
119  "\n"
120  "The emissivity from facet %i is out of range. \n"
121  "Emissivity represents the tendency towards a perfect black surface,\n"
122  "0 representing a perfectly insulating surface (no emission), \n"
123  "and 1 a perfect black body. \n"
124  "The value of thermal.emissivity must be between 0 and 1.\n",
125  ii_facet);
126  }
127  }

134 void RadiationSurface::allocate_array(unsigned int size)
135 {

149  facets = JEOD_ALLOC_CLASS_POINTER_ARRAY(size, RadiationFacet);

159 /**
160  * Turns facet memory into radiaiton facet memory.
161  * \param[in] facet pointer to the facet
162  * \param[in] factory pointer to the facet factory
163  * \param[in] params pointer to the generic facet parameter.
164  * \param[in] index index value in the facet list.
165  */
166 void RadiationSurface::allocate_interaction_facet(Facet * facet,
167  InteractionFacetFactory * factory,
168  FacetParams * params,
169  unsigned int index)
170 {
171  if(num_facets <= index)
172  {
173  MessageHandler::fail(__FILE__,
174  __LINE__,
175  RadiationMessages::invalid_setup_error,
176  "\n"
177  "Attempting to allocate data for facet number %i, but the model has\n"
178  "allocated space for only %i facets (num_facets = %i).\n",
179  index,
180  num_facets,
181  num_facets);
182 
183  return;
184  }
185 
186  /* need to temporarily save off the InteractionFacet returned before
187  dynamic casting it. If the dynamic cast fails, we want to destroy
188  the InteractionFacet so we don't get a memory leak */
189 
190  InteractionFacet * temp_facet = nullptr;
191 
192  temp_facet = factory->create_facet(facet, params);
193 
194  if(temp_facet == nullptr)
195  {
196  MessageHandler::fail(__FILE__,
197  __LINE__,
198  RadiationMessages::operational_setup_error,
199  "\n"
200  "create_facet failed, returning a NULL pointer.\n");
201 
202  return;
203  }
204 
205  auto * temp_radiation_facet = dynamic_cast<RadiationFacet *>(temp_facet);
206 
207  if(temp_radiation_facet == nullptr)
208  {
209  // temp_facet can NOT be NULL, since it was already checked for above
210  JEOD_DELETE_OBJECT(temp_facet);
211 
212  MessageHandler::fail(__FILE__,
213  __LINE__,
214  RadiationMessages::operational_setup_error,
215  "\n"
216  "The InteractionFacet temp_facet exists, but it is not a\n"
217  "RadiationFacet; the cast to RadiationFacet failed.\n");
218  return;
219  }
220 
221  facets[index] = temp_radiation_facet;
222 }
