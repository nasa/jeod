82 /**
83  * A RestartableStateIntegrator establishes the basic capabilities needed to
84  * make a state integrator a managed resource. This includes
85  * - Creating the integrator object,
86  * - Restoring the integrator's derivative function on restart, and
87  * - Destroying the integrator when the RestartableStateIntegrator instance
88  * goes out of scope.
89  *
90  * @tparam IntegratorType The type of integrator to be managed, either
91  * er7_utils::FirstOrderODEIntegrator or er7_utils::SecondOrderODEIntegrator.
92  *
93  * ### Usage:
94  *
95  * This base class template is not directly usable. One must instead use one of
96  * the three class templates that derive from this class template:
97  * - RestartableFirstOrderODEIntegrator to manage an
98  * er7_utils::FirstOrderODEIntegrator pointer;
99  * - RestartableGeneralizedDerivSecondOrderODEIntegrator to manage an
100  * er7_utils::SecondOrderODEIntegrator pointer for a generalized second
101  * order ODE, one in which the time derivative of generalized position
102  * is a function of generalized position and generalized velocity; and
103  * - RestartableGeneralizedStepSecondOrderODEIntegrator to manage an
104  * er7_utils::SecondOrderODEIntegrator pointer for a generalized second
105  * order ODE, one in which the time derivative of generalized position
106  * is a function of generalized position and generalized velocity; and
107  * - RestartableSimpleSecondOrderODEIntegrator to manage an
108  * er7_utils::SecondOrderODEIntegrator pointer for a simple second order
109  * ODE, one in which the time derivative of generalized position is
110  * the generalized velocity.
111  *
112  * Each state integrator to be used in some class needs a pair of data members
113  * declared in the definition of that class.
114  * The first of these pairs is a pointer to the appropriate state integrator
115  * type (either er7_utils::FirstOrderODEIntegrator or
116  * er7_utils::SecondOrderODEIntegrator). The second of the pairs
117  * of data members is an instance of the appropriate derived class of
118  * RestartableStateIntegrator that will manage the pointer. An example:
119  * @code
120  * class MyClass {
121  * ...
122  * er7_utils::SecondOrderODEIntegrator * integrator;
123  * RestartableSimpleSecondOrderODEIntegrator<3> integ_manager;
124  * ...
125  * };
126  * @endcode
127  * The pointer itself must be exposed to Trick for checkpoint and restart. The
128  * RestartableStateIntegrator-derived object should be hidden from Trick.
129  *
130  * The connection between the pointer and the manager for that pointer is made
131  * in the initializer lists of the constructors for the class. Use the
132  * RestartableStateIntegrator non-default constructor to tie the
133  * RestartableStateIntegrator object with the pointer it is to manage:
134  * @code
135  * MyClass::MyClass ()
136  * :
137  * ...
138  * integrator(nullptr),
139  * integ_manager(integrator),
140  * ...
141  * {
142  * // Body of MyClass constructor
143  * }
144  * @endcode
145  * Note that there the RestartableStateIntegrator default constructors exist
146  * only for the sake of the simulation engine. The default constructors do
147  * not create viable instances. One must use the non-default constructor to
148  * initialize RestartableStateIntegrator instances.
149  *
150  * The integrator itself is not created at construction time. Creating the
151  * integrator is the job of the create_integrator method. This method should
152  * be called at initialization time after having created the integration
153  * constructor and the integration controls objects.
154  *
155  * The counterpart to the create_integrator method is destroy_integrator.
156  * This method must be called prior to calling create_integrator if the
157  * integrator has already been created via a previous call to create_integrator.
158  * (For example, switching to a different integration technique).
159  * The destroy_integrator can be called at shutdown or destruction time,
160  * but this call is not essential. The RestartableStateIntegrator object
161  * will call this method internally when it goes out of scope.
162  *
163  * ### Class Design
164  *
165  * This class template uses the non-virtual interface (NVI) design pattern, aka
166  * the template method design pattern (no relation to C++ class templates).
167  * The public create_integrator and simple_restore member functions use the
168  * private virtual create_integrator_internal and simple_restore_internal
169  * functions to create and restore the integrators.
170  */
171 template<typename IntegratorType> class RestartableStateIntegrator : public SimpleCheckpointable
172 {
173 public:
174  // Note: The constructors are protected. This is not an instantiable class.
175 
176  /**
177  * Destructor.
178  */
179  ~RestartableStateIntegrator() override
180  {
181  destroy_integrator();
182  }
183 
184  /**
185  * Create the integrator to be managed.
186  * @param[in] generator Integrator constructor used to create the integrator.
187  * @param[in,out] controls Integration controls to be passed to the generator.
188  */
189  void create_integrator(const er7_utils::IntegratorConstructor & generator,
190  er7_utils::IntegrationControls & controls)
191  {
192  // Using an instance created with the default constructor is an error.
193  if(integrator_handle == nullptr)
194  {
195  MessageHandler::fail(__FILE__,
196  __LINE__,
197  IntegrationMessages::invalid_request,
198  "The pointer to the integrator object has not been set.");
199  return;
200  }
201 
202  // Destroy the integrator, if it exists.
203  destroy_integrator();
204 
205  // Create the integrator via the technique-specific creator.
206  *integrator_handle = create_integrator_internal(generator, controls);
207  }
208 
209  /**
210  * Destroy the integrator.
211  */
212  void destroy_integrator()
213  {
214  if(integrator_handle != nullptr)
215  {
216  er7_utils::Er7UtilsDeletable::delete_instance(*integrator_handle);
217  }
218  }
219 
220  /**
221  * Clear the pointer to the external integrator object.
222  * This currently (pre-Trick 13.0) needs to be called on restart because
223  * both pointers point to invalid objects.
224  */
225  void clear_integrator_reference()
226  {
227  integrator_handle = nullptr;
228  }
229 
230  /**
231  * Set the pointer to the external integrator object.
232  * This currently (pre-Trick 13.0) needs to be called on restart because
233  * the integrator_handle is not properly restored by checkpoint.
234  * @param[in,out] integ_ptr Reference to the external integrator object.
235  */
236  void set_integrator_reference(IntegratorType *& integ_ptr)
237  {
238  integrator_handle = &integ_ptr;
239  }
240 
241  /**
242  * Restore the integrator on restart.
243  * This currently (pre-Trick 13.0) needs to be called after calling
244  * set_integrator_reference.
245  */
246  void simple_restore() override
247  {
248  // Restore the integrator from the checkpoint-restored value in the
249  // external pointer and perform technique-specific actions on it.
250  if(integrator_handle != nullptr)
251  {
252  simple_restore_internal(*integrator_handle);
253  }
254  }

271  }

279  * @param[in,out] controls Integration controls to be passed to the generator.
280  * @return The constructed integrator.
281  */
282  virtual IntegratorType * create_integrator_internal(const er7_utils::IntegratorConstructor & generator,
283  er7_utils::IntegrationControls & controls) = 0;
284 
285  /**

288  * @param[in,out] integrator_ptr The integrator object to be restored
289  */
290  virtual void simple_restore_internal(IntegratorType * integrator_ptr JEOD_UNUSED) {}
291

294  /**
295  * Pointer to the containing object's integrator pointer.
296  */
