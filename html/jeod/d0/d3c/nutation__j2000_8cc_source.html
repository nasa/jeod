78  * Specific implementation of update_rotation, from the polymorphic
79  * pure virtual base class PlanetRotation
80  */
81 void NutationJ2000::update_rotation()
82 {
83  // This implements the Bond / Vallado implementation of J2000 nutation,
84  // as referenced in the documentation
85  // compute the fundamental arguments
86  // time2 is the square of the time, time3 is the cube
87  double time = current_time;
88  double time2 = 0.0;
89  double time3 = 0.0;
90 
91  time2 = time * time;
92  time3 = time2 * time;
93 
94  // the fundamental arguments are in degrees
95 
96  L = 134.9629813888889 + 477198.8673980555 * time + 0.008697222222222223 * time2 + 0.00001777777777777778 * time3;
97 
98  M = 357.5277233333333 + 35999.05034 * time - 0.00016027777777777778 * time2 - 0.000003333333333333333 * time3;
99 
100  F = 93.27191027777778 + 483202.0175380555 * time - 0.0036825 * time2 + 0.000003055555555555555 * time3;
101 
102  D = 297.8503630555556 + 445267.11148 * time - 0.001914166666666667 * time2 + 0.0000052777777777777778 * time3;
103 
104  omega = 125.0445222222222 - 1934.136260833333 * time + 0.00207083333333333 * time2 +
105  0.000002222222222222222 * time3;
106 
107  nutation_in_longitude = 0.0;
108  nutation_in_obliquity = 0.0;
109  for(unsigned int i = 0; i < num_coeffs; ++i)
110  {
111  double api = L_coeffs[i] * L + M_coeffs[i] * M + F_coeffs[i] * F + D_coeffs[i] * D + omega_coeffs[i] * omega;
112  api *= DEGTORAD;
113 
114  nutation_in_longitude += ((long_coeffs[i] + long_t_coeffs[i] * time)) * sin(api);
115 
116  nutation_in_obliquity += ((obliq_coeffs[i] + obliq_t_coeffs[i] * time)) * cos(api);
117 
118  } // for(unsigned int i = 0)
119 
120  // note that the numbers here have been converted from arcseconds to degrees
121 
122  epsilon_bar = 23.43929111111111 - 0.01300416666666667 * time - 0.00000016388888889 * time2 +
123  0.00000050361111111 * time3;
124  epsilon_bar *= DEGTORAD;
125 
126  // based on the JEOD 1.52 code, the nutations in obliquity and
127  // longitude are apparently in units of x 10^-4 arcseconds
128  // thus these conversions are necessary to get to radians
129  // also, to match numerical precesion of JEOD 1.52, some concessions
130  // (dividing by DEGTOSEC) are made to the old order of operations
131  nutation_in_obliquity = ((nutation_in_obliquity / 10000.0) / 3600.0) * DEGTORAD;
132 
133  double epsilon = 0.0;
134  epsilon = epsilon_bar + nutation_in_obliquity;
135 
136  nutation_in_longitude /= 10000.0;
137  double c_eps = cos(epsilon);
138  equa_of_equi = (nutation_in_longitude * c_eps) / 15.0;
139 
140  nutation_in_longitude = (nutation_in_longitude / 3600) * DEGTORAD;
141 
142  // populate the trig functions needed for the rotation matrix
143  double c_long = cos(nutation_in_longitude);
144  double s_long = sin(nutation_in_longitude);
145 
146  double s_eps = sin(epsilon);
147 
148  double c_eps_bar = cos(epsilon_bar);
149  double s_eps_bar = sin(epsilon_bar);
150 
151  Matrix3x3::identity(rotation);
152 
153  // populate the rotation matrix
154  rotation[0][0] = c_long;
155  rotation[0][1] = c_eps * s_long;
156  rotation[0][2] = s_eps * s_long;
157 
158  rotation[1][0] = -c_eps_bar * s_long;
159  rotation[1][1] = c_eps * c_long * c_eps_bar + s_eps * s_eps_bar;
160  rotation[1][2] = s_eps * c_long * c_eps_bar - c_eps * s_eps_bar;
161 
162  rotation[2][0] = -s_eps_bar * s_long;
163  rotation[2][1] = c_eps * c_long * s_eps_bar - s_eps * c_eps_bar;
164  rotation[2][2] = s_eps * s_eps_bar * c_long + c_eps * c_eps_bar;
165 }
