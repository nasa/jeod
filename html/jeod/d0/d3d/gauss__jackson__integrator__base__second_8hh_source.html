153  initialize_edit_integration_constants(double dt)

218 inline void GaussJacksonIntegratorBase<GaussJacksonTwoState, er7_utils::SecondOrderODEIntegrator>::mid_correct(
219  unsigned int coeff_idx, double dt, GaussJacksonTwoState & state)
220 {
221  double dtsq = dt * dt;

235 template<>
236 inline void GaussJacksonIntegratorBase<GaussJacksonTwoState, er7_utils::SecondOrderODEIntegrator>::predict(
237  double dt, const double * const * ER7_UTILS_RESTRICT ahist, GaussJacksonTwoState & state)
238 {
239  double dtsq = dt * dt;

254 /**
255  * Apply the corrector.
256  */
257 template<>
258 inline void GaussJacksonIntegratorBase<GaussJacksonTwoState, er7_utils::SecondOrderODEIntegrator>::correct(
259  double dt, const double * ER7_UTILS_RESTRICT acc, GaussJacksonTwoState & state)
260 {
261  double dtsq = dt * dt;
262  double * ER7_UTILS_RESTRICT first_csum = corrector_sum.first;
263  double * ER7_UTILS_RESTRICT second_csum = corrector_sum.second;
264  double * ER7_UTILS_RESTRICT first_dinv = delinv.first;
265  double * ER7_UTILS_RESTRICT second_dinv = delinv.second;
266  double * ER7_UTILS_RESTRICT velocity = state.first;
267  double * ER7_UTILS_RESTRICT position = state.second;
268  double vfact = velocity_corrector;
269  double pfact = position_corrector;
270 
271  for(unsigned int ii = 0; ii < size; ++ii)
272  {
273  double temp = first_csum[ii] + vfact * acc[ii];
274  velocity[ii] = dt * (first_dinv[ii] + temp);
275 
276  temp = second_csum[ii] + pfact * acc[ii];
277  position[ii] = dtsq * (second_dinv[ii] + temp);
278  }
279 }

285 inline bool GaussJacksonIntegratorBase<GaussJacksonTwoState, er7_utils::SecondOrderODEIntegrator>::test_for_convergence(
286  const GaussJacksonTwoState & state, double * ER7_UTILS_RESTRICT hist_data)
287 {
288  const double * ER7_UTILS_RESTRICT new_data = state.second;
289  bool passed = true;

293  if((error > absolute_tolerance) && (error > relative_tolerance * std::abs(new_data[ii])))

295  passed = false;

298  }
