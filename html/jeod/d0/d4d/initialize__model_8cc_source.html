86 /**
87  * Begin initialization of the JEOD manager model.
88  *
89  * \par Assumptions and Limitations
90  * - The user-input item selection table must have at most one selection
91  * rule for a given name. This limitation is an enforced constraint.
92  * \param[in,out] init Initialization data
93  * \param[in,out] time_mngr Time manager
94  */
95 void DynManager::initialize_model_internal(DynManagerInit & init, TimeManager & time_mngr)
96 {
97  // Set the mode.
98  mode = init.mode;
99 
100  if(mode != DynManagerInit::EphemerisMode_Ephemerides)
101  {
102  if(init.central_point_name.empty())
103  {
104  MessageHandler::fail(__FILE__,
105  __LINE__,
106  DynManagerMessages::invalid_name,
107  "Invalid value (empty string) for "
108  "DynManagerInit.central_point_name");
109  return;
110  }
111 
112  switch(mode)
113  {
114  // Empty space mode:
115  // Make an empty space ephemeris model to represent all of (empty) space.
116  case DynManagerInit::EphemerisMode_EmptySpace:
117  simple_ephemeris = JEOD_ALLOC_CLASS_OBJECT(EmptySpaceEphemeris, ());
118  break;
119 
120  // Single planet mode:
121  // Make an single-planet ephemeris model to represent space.
122  // Note: The simulation must define exactly one Planet object, give it the
123  // same name as the central point, and register it with the manager.
124  case DynManagerInit::EphemerisMode_SinglePlanet:
125  simple_ephemeris = JEOD_ALLOC_CLASS_OBJECT(SinglePlanetEphemeris, ());
126  break;
127  // Anything else is an error.
128  case DynManagerInit::EphemerisMode_Ephemerides:
129  default:
130  MessageHandler::fail(__FILE__, __LINE__, DynManagerMessages::inconsistent_setup, "Illegal mode value");
131  return;
132  }
133 
134  // There shouldn't be any ephemerides models registered with the
135  // ephemerides manager, but just in case, disable them and empty the list.
136  clear_added_ephemerides();
137 
138  // Initialize the simple ephemeris model.
139  simple_ephemeris->set_name(init.central_point_name);
140  simple_ephemeris->initialize_model(*this);
141 
142  // Tell the ephemeris manager to reject subsequent ephemerides.
143  disable_add_ephemeris();
144  }
145 
146  // Create the default integration group, if needed.
147  if(integ_groups.empty())
148  {
149  // Create the integrator constructor.
150  if(init.integ_constructor != nullptr)
151  {
152  integ_constructor = init.integ_constructor->create_copy();
153  }
154  else
155  {
156  if(init.jeod_integ_opt == er7_utils::Integration::Unspecified)
157  {
158  init.jeod_integ_opt = integ_interface->interpret_integration_type(init.sim_integ_opt);
159  if(init.jeod_integ_opt == er7_utils::Integration::Unspecified)
160  {
161  MessageHandler::fail(__FILE__,
162  __LINE__,
163  DynManagerMessages::inconsistent_setup,
164  "Integration type unspecified.");
165  return;
166  }
167  }
168  integ_constructor = er7_utils::IntegratorConstructorFactory::create(init.jeod_integ_opt);
169  }
170 
171  // Create the default integration group.
172  if(init.integ_group_constructor != nullptr)
173  {
174  default_integ_group = init.integ_group_constructor->create_group(*this,
175  *integ_constructor,
176  *integ_interface,
177  time_mngr.get_jeod_integration_time());
178  }
179  else
180  {
181  default_integ_group = JEOD_ALLOC_CLASS_OBJECT(DynamicsIntegrationGroup,
182  (*this,
183  *integ_constructor,
184  *integ_interface,
185  time_mngr.get_jeod_integration_time()));
186  }
187  }
188 }
