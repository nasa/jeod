96 /**
97  * Begin initialization of the JEOD manager model.
98  *
99  * \par Assumptions and Limitations
100  * - The user-input item selection table must have at most one selection
101  * rule for a given name. This limitation is an enforced constraint.
102  * \param[in,out] init Initialization data
103  * \param[in,out] time_mngr Time manager
104  */
105 void
106 DynManager::initialize_model_internal (
107  DynManagerInit & init,
108  TimeManager & time_mngr)
109 {
110 
111  // Set the mode.
112  mode = init.mode;
113 
114  if (mode != DynManagerInit::EphemerisMode_Ephemerides) {
115  if ((init.central_point_name == nullptr) ||
116  (init.central_point_name[0] == '\0')) {
117  MessageHandler::fail (
118  __FILE__, __LINE__, DynManagerMessages::invalid_name,
119  "Invalid value (NULL or empty string) for "
120  "DynManagerInit.central_point_name");
121  return;
122  }
123 
124 
125  switch (mode) {
126  // Empty space mode:
127  // Make an empty space ephemeris model to represent all of (empty) space.
128  case DynManagerInit::EphemerisMode_EmptySpace:
129  simple_ephemeris =
130  JEOD_ALLOC_CLASS_OBJECT (EmptySpaceEphemeris, ());
131  break;
132 
133  // Single planet mode:
134  // Make an single-planet ephemeris model to represent space.
135  // Note: The simulation must define exactly one Planet object, give it the
136  // same name as the central point, and register it with the manager.
137  case DynManagerInit::EphemerisMode_SinglePlanet:
138  simple_ephemeris =
139  JEOD_ALLOC_CLASS_OBJECT (SinglePlanetEphemeris, ());
140  break;
141  // Anything else is an error.
142  case DynManagerInit::EphemerisMode_Ephemerides:
143  default:
144  MessageHandler::fail (
145  __FILE__, __LINE__, DynManagerMessages::inconsistent_setup,
146  "Illegal mode value");
147  return;
148  }
149 
150  // There shouldn't be any ephemerides models registered with the
151  // ephemerides manager, but just in case, disable them and empty the list.
152  clear_added_ephemerides ();
153 
154  // Initialize the simple ephemeris model.
155  simple_ephemeris->set_name (init.central_point_name);
156  simple_ephemeris->initialize_model (*this);
157 
158  // Tell the ephemeris manager to reject subsequent ephemerides.
159  disable_add_ephemeris ();
160  }
161 
162 
163  // Create the default integration group, if needed.
164  if (integ_groups.empty()) {
165 
166  // Create the integrator constructor.
167  if (init.integ_constructor != nullptr) {
168  integ_constructor = init.integ_constructor->create_copy();
169  }
170  else {
171  if (init.jeod_integ_opt == er7_utils::Integration::Unspecified) {
172  init.jeod_integ_opt =
173  integ_interface->interpret_integration_type (init.sim_integ_opt);
174  if (init.jeod_integ_opt == er7_utils::Integration::Unspecified) {
175  // FIXME! This will die in the factory. Better to die here.
176  }
177  }
178  integ_constructor =
179  er7_utils::IntegratorConstructorFactory::create (
180  init.jeod_integ_opt);
181  }
182 
183 
184  // Create the default integration group.
185  if (init.integ_group_constructor != nullptr) {
186  default_integ_group =
187  init.integ_group_constructor->create_group (
188  *this,
189  *integ_constructor,
190  *integ_interface,
191  time_mngr.get_jeod_integration_time());
192  }
193  else {
194  default_integ_group =
195  JEOD_ALLOC_CLASS_OBJECT (
196  DynamicsIntegrationGroup,
197  (*this,
198  *integ_constructor,
199  *integ_interface,
200  time_mngr.get_jeod_integration_time()));
201  }
202  }
203 }
