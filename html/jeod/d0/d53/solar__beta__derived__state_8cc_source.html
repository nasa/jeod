84 /**
85  * Update the state.
86  */
87 void SolarBetaDerivedState::update()
88 {
89  // Return if the model is currently inactive or permanently disabled.
90  if((!active) || (sun == nullptr))
91  {
92  return;
93  }
94 
95  // Find the position of the sun with respect ot the planet.
96  sun->inertial.compute_position_from(planet->inertial, sun_wrt_planet);
97 
98  // Compute the relative state of the vehicle with respect to the planet.
99  subject->composite_body.compute_relative_state(planet->inertial, veh_wrt_planet);
100 
101  // Store the relative translational state.
102  RefFrameTrans * trans_state = &veh_wrt_planet.trans;
103  double ang_momentum[3];
104 
105  // Compute the angular momentum vector, which is a handy thing
106  // to have when calculating solar beta.
107  Vector3::cross(trans_state->position, trans_state->velocity, ang_momentum);
108 
109  // h = ang_momentum
110  // s = sun_wrt_planet
111  // |h| * |s| * sin(solar_beta) = h dot s
112  double dot_product = Vector3::dot(ang_momentum, sun_wrt_planet);
113  double h_mag_sq = Vector3::vmagsq(ang_momentum);
114  double s_mag_sq = Vector3::vmagsq(sun_wrt_planet);
115 
116  if(h_mag_sq < EPSILON)
117  {
118  MessageHandler::fail(__FILE__,
119  __LINE__,
120  DerivedStateMessages::divide_by_zero,
121  "The magnitude of the momentum of the vehicle tracked by solar "
122  "beta was too small, resulting in a divide by zero. The function "
123  "has failed.\n");
124  return;
125  }
126 
127  if(s_mag_sq < EPSILON)
128  {
129  MessageHandler::fail(__FILE__,
130  __LINE__,
131  DerivedStateMessages::divide_by_zero,
132  "The magnitude of the position of the sun with respect to the planet "
133  "referenced in solar beta was too small, resulting in a divide by "
134  "zero. The function has failed.\n");
135  return;
136  }
137 
138  solar_beta = dot_product / (sqrt(h_mag_sq * s_mag_sq));
139  solar_beta = asin(solar_beta);
140 }
