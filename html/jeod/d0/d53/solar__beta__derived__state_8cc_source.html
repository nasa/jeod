105 /**
106  * Update the state.
107  */
108 void
109 SolarBetaDerivedState::update (
110  void)
111 {
112 
113  // Return if the model is currently inactive or permanently disabled.
114  if ((! active) || (sun == nullptr)) {
115  return;
116  }
117 
118  // Find the position of the sun with respect ot the planet.
119  sun->inertial.compute_position_from(planet->inertial, sun_wrt_planet);
120 
121  // Compute the relative state of the vehicle with respect to the planet.
122  subject->composite_body.compute_relative_state(planet->inertial,
123  veh_wrt_planet);
124 
125  // Store the relative translational state.
126  RefFrameTrans* trans_state = &veh_wrt_planet.trans;
127  double ang_momentum[3];
128 
129  // Compute the angular momentum vector, which is a handy thing
130  // to have when calculating solar beta.
131  Vector3::cross(trans_state->position, trans_state->velocity, ang_momentum);
132 
133  // h = ang_momentum
134  // s = sun_wrt_planet
135  // |h| * |s| * sin(solar_beta) = h dot s
136  double dot_product = Vector3::dot(ang_momentum, sun_wrt_planet);
137  double h_mag_sq = Vector3::vmagsq(ang_momentum);
138  double s_mag_sq = Vector3::vmagsq(sun_wrt_planet);
139 
140  if(h_mag_sq < EPSILON) {
141 
142  MessageHandler::fail (
143  __FILE__, __LINE__, DerivedStateMessages::divide_by_zero,
144  "The magnitude of the momentum of the vehicle tracked by solar "
145  "beta was too small, resulting in a divide by zero. The function "
146  "has failed.\n");
147  return;
148 
149  }
150 
151  if(s_mag_sq < EPSILON) {
152 
153  MessageHandler::fail (
154  __FILE__, __LINE__, DerivedStateMessages::divide_by_zero,
155  "The magnitude of the position of the sun with respect to the planet "
156  "referenced in solar beta was too small, resulting in a divide by "
157  "zero. The function has failed.\n");
158  return;
159 
160  }
161 
162  solar_beta = dot_product / (sqrt(h_mag_sq * s_mag_sq));
163  solar_beta = asin(solar_beta);
164 }
