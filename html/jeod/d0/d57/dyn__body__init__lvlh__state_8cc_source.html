82 /**
83  * Apply the initializer: Construct the reference LVLH frame so the
84  * parent initializer can compute the vehicle's state relative to the
85  * vehicle's inertial frame.
86  * \param[in,out] dyn_manager Dynamics manager
87  */
88 void DynBodyInitLvlhState::apply(DynManager & dyn_manager)
89 {
90  LvlhFrame lvlh_frame; // -- Reference vehicle LVLH frame
91 
92  if(lvlh_object_ptr != nullptr)
93  {
94  lvlh_object_ptr->update();
95  reference_ref_frame = &lvlh_object_ptr->frame;
96  }
97  else
98  {
99  // Construct the LVLH frame.
100  lvlh_frame.set_subject_frame(ref_body->composite_body);
101  lvlh_frame.set_planet(*planet);
102  lvlh_frame.initialize(dyn_manager);
103  lvlh_frame.update();
104  // The position/velocity/attitude/rate data pertain to this newly
105  // constructed LVLH frame.
106  reference_ref_frame = &lvlh_frame.frame;
107  }
108 
109  if(lvlh_type == LvlhType::CircularCurvilinear)
110  {
111  // Curvilinear LVLH provided, must be converted to rectilinear
112  // Declare and initialize a temporary LVLH relative state.
113  LvlhRelativeDerivedState lvlh_state;
114 
115  // Set the direction sense to the only value that makes sense in this case
116  lvlh_state.direction_sense = RelativeDerivedState::ComputeSubjectStateinTarget;
117 
118  // Set the target frame for lvlh_state
119  lvlh_state.set_target_frame(*reference_ref_frame);
120 
121  // Store body action variables in a reference frame state
122  RefFrameState circ_rel_state;
123 
124  Vector3::copy(position, circ_rel_state.trans.position);
125  Vector3::copy(velocity, circ_rel_state.trans.velocity);
126 
127  orientation.compute_transform();
128  Matrix3x3::copy(orientation.trans, circ_rel_state.rot.T_parent_this);
129  circ_rel_state.rot.compute_quaternion();
130 
131  Vector3::copy(ang_velocity, circ_rel_state.rot.ang_vel_this);
132  circ_rel_state.rot.compute_ang_vel_products();
133 
134  // Convert provided curvilinear inputs into rectilinear. Note
135  // the result of the conversion is stored in lvlh_state.rel_state.
136  lvlh_state.convert_circ_to_rect(circ_rel_state);
137 
138  // Extract info into appropriate variables for application.
139  Vector3::copy(lvlh_state.rel_state.trans.position, position);
140  Vector3::copy(lvlh_state.rel_state.trans.velocity, velocity);
141  Vector3::copy(lvlh_state.rel_state.rot.ang_vel_this, ang_velocity);
142 
143  Matrix3x3::copy(lvlh_state.rel_state.rot.T_parent_this, orientation.trans);
144 
145  orientation.quat = lvlh_state.rel_state.rot.Q_parent_this;
146 
147  if(orientation.euler_sequence == Orientation::NoSequence)
148  {
149  orientation.euler_sequence = Orientation::Roll_Pitch_Yaw;
150  }
151 
152  Orientation::compute_euler_angles_from_matrix(orientation.trans,
153  orientation.euler_sequence,
154  orientation.euler_angles);
155  Orientation::compute_eigen_rotation_from_matrix(orientation.trans,
156  &(orientation.eigen_angle),
157  orientation.eigen_axis);
158  }
159  else if(lvlh_type == LvlhType::EllipticalCurvilinear)
160  {
161  // Not implemented yet as of March 2015; for now, fail.
162  MessageHandler::fail(__FILE__,
163  __LINE__,
164  BodyActionMessages::illegal_value,
165  "EllipticalCurvilinear option not yet implemented for DynBodyInitLvlhState");
166  // Not reached
167  return;
168  }
169  else if(lvlh_type != LvlhType::Rectilinear)
170  {
171  MessageHandler::fail(__FILE__,
172  __LINE__,
173  BodyActionMessages::illegal_value,
174  "No valid value of enum LvlhType indicated for DynBodyInitLvlhState");
175  // Not reached
176  return;
177  }
178 
179  // Compute the state relative to the parent given the user inputs
180  // relative to the reference reference frame.
181  apply_user_inputs();
182 
183  // Pass the message up the chain, with the reference frame for the
184  // state set to this newly-constructed LVLH frame.
185  DynBodyInitPlanetDerived::apply(dyn_manager);
186 
187  // The LVLH frame will go out of scope on returning from this function.
188  // The reference_ref_frame should no longer point to this frame.
189  reference_ref_frame = nullptr;
190 }
