55 /**
56  * initializes the default surface
57  */
58 void RadiationDefaultSurface::initialize()
59 {
60  double epsilon = 1.0E-12;
61 
62  RadiationBaseFacet::initialize();
63 
64  if((albedo < 0.0) && (diffuse < 0.0) && (rad_coeff < 0.0))
65  {
66  MessageHandler::fail(__FILE__,
67  __LINE__,
68  RadiationMessages::invalid_setup_error,
69  "\n"
70  "The surface reflectance has not been defined correctly.\n"
71  "Either the reflection (a.k.a. radiation) coefficient (Cr), or\n"
72  "both the albedo and diffuse parameters must be specified \n"
73  "and be within their appropriate ranges \n"
74  "(rad_coeff between 1.0 and 1.444,\n"
75  "albedo and diffuse both between 0.0 and 1.0).\n");
76  }
77 
78  if(albedo * diffuse < 0)
79  {
80  MessageHandler::fail(__FILE__,
81  __LINE__,
82  RadiationMessages::invalid_setup_error,
83  "\n"
84  "If specifying the surface reflectance by an albedo & diffuse \n"
85  "combination, both must be specified within the range 0 to 1.\n");
86  }
87 
88  if((rad_coeff > 0) && (albedo > 0))
89  { // hence diffuse must be > 0 also
90  // tried to define material properties and their combined value.
91  MessageHandler::fail(__FILE__,
92  __LINE__,
93  RadiationMessages::invalid_setup_error,
94  "\n"
95  "Both the radiation coefficient (rad_coeff) and either or both of\n"
96  "the albedo and diffuse parameters were defined.\n"
97  "Specify either radiation coefficient, or albedo and diffuse.\n");
98  }
99 
100  if(rad_coeff > 0)
101  { // defined by rad_coeff, albedo and diffuse not defined
102  if((rad_coeff > (13.0 / 9.0)) || (rad_coeff < 1.0))
103  {
104  MessageHandler::fail(__FILE__,
105  __LINE__,
106  RadiationMessages::invalid_setup_error,
107  "\n"
108  "The radiation coefficient (rad_coeff) was specified with a value\n"
109  "outside its legitimate range.\n"
110  "The radiation coefficient represents a multiplicative factor that\n"
111  "allows a reflective surface to be represented as a larger\n"
112  "non-reflective surface (and experience the same force).\n"
113  "For a spherical surface (such as this default surface), the \n"
114  "legitimate range of that value is from 1.0 to 1.44444 (i.e. 13/9).\n");
115  }
116  albedo = 9 * (rad_coeff - 1) / 4;
117  diffuse = 1.0;
118  if(albedo > 1.0)
119  { // This cannot happen, but just in case
120  MessageHandler::warn(__FILE__,
121  __LINE__,
122  RadiationMessages::operational_setup_error,
123  "\n"
124  "Radiation Coefficient was defined, and albedo was calculated \n"
125  "from it to have value %f, which is outside its range. \n"
126  "Adjusting value of albedo to 1.0.\n",
127  albedo);
128  albedo = 1.0;
129  }
130  if(albedo < 0.0)
131  { // This cannot happen, but just in case
132  MessageHandler::warn(__FILE__,
133  __LINE__,
134  RadiationMessages::operational_setup_error,
135  "\n"
136  "Radiation Coefficient was defined, and albedo was calculated \n"
137  "from it to have value %f, which is outside its range. \n"
138  "Adjusting value of albedo to 0.0.\n",
139  albedo);
140  albedo = 0.0;
141  }
142  }
143  else
144  { // defined by albedo and diffuse
145  if((albedo > 1.0) || (albedo < 0.0))
146  {
147  MessageHandler::fail(__FILE__,
148  __LINE__,
149  RadiationMessages::invalid_setup_error,
150  "\n"
151  "The albedo of the default surface is out of range. \n"
152  "Albedo represents a fraction of incident light, and must \n"
153  "be between 0 and 1.\n");
154  }
155  if((diffuse > 1.0) || (diffuse < 0.0))
156  {
157  MessageHandler::fail(__FILE__,
158  __LINE__,
159  RadiationMessages::invalid_setup_error,
160  "\n"
161  "The diffuse parameter for the default surface is out of range. \n"
162  "Diffuse represents a fraction of reflected light, and must \n"
163  "be between 0 and 1.\n");
164  }
165  }
166 
167  // Area definitions
168 
169  if((std::fpclassify(cx_area) != FP_ZERO) && (std::fpclassify(surface_area) != FP_ZERO))
170  {
171  MessageHandler::fail(__FILE__,
172  __LINE__,
173  RadiationMessages::invalid_setup_error,
174  "\n"
175  "Both the cross-sectional area (cx_area) and surface-area (surface_area)"
176  "\n were specified. Only one should be specified; its value is used to"
177  "\n calculate the other.\n");
178  }
179 
180  if(cx_area < epsilon)
181  {
182  cx_area = surface_area / 4.0;
183  }
184  if(surface_area < epsilon)
185  {
186  surface_area = cx_area * 4.0;
187  }
188  if(cx_area < epsilon)
189  {
190  MessageHandler::fail(__FILE__,
191  __LINE__,
192  RadiationMessages::invalid_setup_error,
193  "\n"
194  "The area of the default surface is 0.0. This is usually due to an \n"
195  "omission in defining the surface parameters.\n"
196  "One of the cross-sectional area (cx_area) or surface-area (surface_area)"
197  "\n should be specified to be non-zero.\n");
198  }
199 
200  thermal.initialize(temperature, surface_area);
201 }

203 /**
204  * Calculation of force and torque due to radiation pressure
205  *
206  * \par Assumptions and Limitations
207  * - Assumes that flux magnitude is positive.
208  * - Will only be called if flux magnitude is positive.
209  * - Assumes spherical surface.
210  * \param[in] flux_mag the magnitude of the incident flux
211  * \param[in] flux_struc_hat the unit vector in structural frame for the flux vector.
212  * \param[in] calculate_forces boolean indicating whether to calculate forces.
213  */
214 void RadiationDefaultSurface::incident_radiation(const double flux_mag,
215  const double flux_struc_hat[3],
216  const bool calculate_forces)
217 {
218  // energy flux
219  areaxflux_e = cx_area * flux_mag;
220  thermal.power_absorb += (1.0 - albedo) * areaxflux_e;
221 
222  if(!calculate_forces)
223  {
224  return;
225  }
226 
227  // momentum flux
228  areaxflux = areaxflux_e / speed_of_light;
229 
230  double tempvec1[3];
231  // absorption
232  Vector3::scale(flux_struc_hat, areaxflux * (1.0 - albedo), tempvec1);
233  Vector3::incr(tempvec1, F_absorption);
234 
235  // reflection
236  double ref_flux = areaxflux * albedo;
237 
238  // Diffuse reflection
239  Vector3::scale(flux_struc_hat, ref_flux * diffuse * 13.0 / 9.0, tempvec1);
240  Vector3::incr(tempvec1, F_diffuse);
241 
242  // Specular reflection
243  Vector3::scale(flux_struc_hat, ref_flux * (1.0 - diffuse), tempvec1);
244  Vector3::incr(tempvec1, F_specular);
245 }
