78 GravitySource * GravityManager::find_grav_source(const std::string & source_name) const
79 {
80  GravitySource * found_source = nullptr;

82 
83  if(source_name.empty())

109 void GravityManager::add_grav_source(GravitySource & source)

112  if(source.name.empty())
113  {
114  MessageHandler::fail(__FILE__,

118  return;

120 
121  // Body names must be unique. Check for conflicts.
122  if(find_grav_source(source.name) != nullptr)

126  GravityMessages::duplicate_entry,

129  return;
130  }
131 
132  // Save the body in the bodies container.
133  sources.push_back(&source);

174 // Compute gravitational attraction of gravitational bodies on the
175 // provided dynamic body.
176 // Note: This overload of GravityManager::gravitation is deprecated.
177 void GravityManager::gravitation(const double integ_pos[3], GravityInteraction & grav)
178 {
179  double total_grav_accel[3] = {0.0, 0.0, 0.0}; // M/s2 Total accel
180 
181  double total_grav_grad[3][3] = {
182  {0.0, 0.0, 0.0},
183  {0.0, 0.0, 0.0},
184  {0.0, 0.0, 0.0}
185  }; // 1/S2 Total gravity gradient
186 
187  double total_grav_pot[1] = {}; // -- Total gravitational potential
188 
189  unsigned int n_controls = grav.grav_controls.size(); // -- Number of controls for the dyn body
190 
191  unsigned int integ_idx = grav.integ_frame_index; // -- Vehicle integration frame index
192 
193  /* Compute the gravitational acceleration from each gravitational body
194  for which the vehicle has a control on the body and accumulate the
195  total gravitational acceleration. */
196  for(unsigned int ii = 0; ii < n_controls; ++ii)
197  {
198  GravityControls & control_ii = *(grav.grav_controls[ii]);
199  if(control_ii.active)
200  {
201  control_ii.gravitation(integ_pos,
202  integ_idx,
203  control_ii.grav_accel,
204  control_ii.grav_grad,
205  &control_ii.grav_pot);
206  Vector3::incr(control_ii.grav_accel, total_grav_accel);
207  Matrix3x3::incr(control_ii.grav_grad, total_grav_grad);
208  total_grav_pot[0] += control_ii.grav_pot;
209  }
210  }
211 
212  // Save the total acceleration, gradient in the gravity interaction instance.
213  Vector3::copy(total_grav_accel, grav.grav_accel);
214  Matrix3x3::copy(total_grav_grad, grav.grav_grad);
215  grav.grav_pot = total_grav_pot[0];
216 }
