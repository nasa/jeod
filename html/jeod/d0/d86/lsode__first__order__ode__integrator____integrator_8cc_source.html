343 void LsodeFirstOrderODEIntegrator::integrator_predict()
344 {
345  // 200
346  // Flag the Jacobian for needing an update if the following holds true:
347  // 1. The corrector method is NOT Functional Iteration AND EITHER
348  // 2a. The relative change since the last update exceeds the limit OR
349  // 2b. The number of steps taken since the update exceeds the limit.
350  // To avoid running through the second two tests every time for
351  // functional iteration, do that test first.
352 
353  update_jacobian = ((!control_data.is_corrector_method_functional_iteration()) &&

355  (num_steps_taken >= step_at_last_jacobian_update + control_data.max_num_steps_jacobian)));
356 
357  stage_target_time = stage_target_time + step_size;
