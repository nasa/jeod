340 LsodeFirstOrderODEIntegrator::integrator_predict()
341 {
342  // 200
343  // Flag the Jacobian for needing an update if the following holds true:
344  // 1. The corrector method is NOT Functional Iteration AND EITHER
345  // 2a. The relative change since the last update exceeds the limit OR
346  // 2b. The number of steps taken since the update exceeds the limit.
347  // To avoid running through the second two tests every time for
348  // functional iteration, do that test first.
349 
350  update_jacobian =

356
