38 /**
39  * Calculate the aggregate inertia matrix for a composite MassBody.
40  *
41  * \par Assumptions and Limitations
42  * - Rigid Bodies
43  * - Tree attachment structure
44  * - Must calculate from bottom to top of tree for meaningful results
45  */
46 void
47 MassBody::calc_composite_inertia (
48  void)
49 {
50  double offset_inertia[3][3]; // kg*M2 Inertia due to child body mass
51  double transform_inertia[3][3]; // kg*M2 Inertia due to child body inertia
52 
53 
54  // For composite bodies, the composite inertia tensor is the sum of all
55  // the inertia tensors for all the component masses shifted per the parallel
56  // axis theorem. The component masses comprise the core mass of this body
57  // alone plus the composite masses of each attached child body.
58 
59 
60  // The core and composite masses share a common body frame.
61  // Initialize the composite inertia as that of the core mass plus the
62  // inertia due to the core mass acting as a point mass at the core CoM.
63 
64  // Compute the inertia due to the core center of mass being offset from
65  // the composite center of mass.
66  compute_point_mass_inertia (core_properties.mass,
67  core_wrt_composite.position,
68  offset_inertia);
69 
70  // Compute the contribution to the composite inertia by the core mass.
71  Matrix3x3::add (offset_inertia, core_properties.inertia,
72  composite_properties.inertia);
73 
74  // Add each child body's contribution to the composite inertia.
75  for (auto* child_link : TreeLinksChildrenRange<MassBodyLinks>(links)) {
76  const MassBody* child = &(child_link->container());
77 
78  // Compute the inertia due to the core center of mass being offset from
79  // the composite center of mass.
80  compute_point_mass_inertia (child->composite_properties.mass,
81  child->composite_wrt_pbdy.position,
82  offset_inertia);
83 
84  // Transform the child composite inertia to this body's body frame.
85  Matrix3x3::transpose_transform_matrix (
86  child->composite_wrt_pbdy.T_parent_this,
87  child->composite_properties.inertia,
88  transform_inertia);
89 
90  // Add the child contributions to the composite inertia.
91  Matrix3x3::incr (offset_inertia, composite_properties.inertia);
92  Matrix3x3::incr (transform_inertia, composite_properties.inertia);
93  }
94 }
