37 /**
38  * Calculate the aggregate inertia matrix for a composite MassBody.
39  *
40  * \par Assumptions and Limitations
41  * - Rigid Bodies
42  * - Tree attachment structure
43  * - Must calculate from bottom to top of tree for meaningful results
44  */
45 void MassBody::calc_composite_inertia()
46 {
47  double offset_inertia[3][3]; // kg*M2 Inertia due to child body mass
48  double transform_inertia[3][3]; // kg*M2 Inertia due to child body inertia
49 
50  // For composite bodies, the composite inertia tensor is the sum of all
51  // the inertia tensors for all the component masses shifted per the parallel
52  // axis theorem. The component masses comprise the core mass of this body
53  // alone plus the composite masses of each attached child body.
54 
55  // The core and composite masses share a common body frame.
56  // Initialize the composite inertia as that of the core mass plus the
57  // inertia due to the core mass acting as a point mass at the core CoM.
58 
59  // Compute the inertia due to the core center of mass being offset from
60  // the composite center of mass.
61  compute_point_mass_inertia(core_properties.mass, core_wrt_composite.position, offset_inertia);
62 
63  // Compute the contribution to the composite inertia by the core mass.
64  Matrix3x3::add(offset_inertia, core_properties.inertia, composite_properties.inertia);
65 
66  // Add each child body's contribution to the composite inertia.
67  for(auto * child_link : TreeLinksChildrenRange<MassBodyLinks>(links))
68  {
69  const MassBody * child = &(child_link->container());
70 
71  // Compute the inertia due to the core center of mass being offset from
72  // the composite center of mass.
73  compute_point_mass_inertia(child->composite_properties.mass,
74  child->composite_wrt_pbdy.position,
75  offset_inertia);
76 
77  // Transform the child composite inertia to this body's body frame.
78  Matrix3x3::transpose_transform_matrix(child->composite_wrt_pbdy.T_parent_this,
79  child->composite_properties.inertia,
80  transform_inertia);
81 
82  // Add the child contributions to the composite inertia.
83  Matrix3x3::incr(offset_inertia, composite_properties.inertia);
84  Matrix3x3::incr(transform_inertia, composite_properties.inertia);
85  }
86 }
