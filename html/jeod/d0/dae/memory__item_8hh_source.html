79 /**
80  * A JeodMemoryItem contains metadata about some chunk of allocated memory.
81  * This is a simple datatype that contains POD elements only.
82  * All data members are private and are accessible only through getters;
83  * the members are essentially constant. The only way to change the values
84  * is via a wholesale copy.
85  */
86 class JeodMemoryItem
87 {
88  // Types
89 public:
90  /**
91  * Identifies by name the bit flags the comprise a JeodMemoryItem::flag.
92  */
93  enum Flags
94  {
95  PlacementNew = 1, /**< Was the item constructed with placement new?
96  There is no functional placement delete in C++. */
97  IsArray = 2, /**< Was the item an array constructed via new []?
98  This addresses the delete[] versus delete issue. */
99  IsGuarded = 4, /**< Is the allocated buffer surrounded by guard words?
100  This flag is always false in regular new mode. */
101  IsStructured = 8, /**< Is the item a class (versus a primitive type)?
102  Classes add several other twists. */
103  IsRegistered = 16, /**< Has the item been registered with the simulation
104  engine? */
105  CheckPointed = 32 /**< Reserved for future work,
106  as are flag bits 6 ando 7 (64 and 128). */
107  };
108 
109  // Static methods
110 private:
111  // Construct the flags for a new instance of a memory item
112  static uint8_t construct_flags(bool placement_new, bool is_array, bool is_guarded, bool is_structured);
113 
114  // Constructors, destructors, etc.
115 public:
116  // Default constructor.
117  JeodMemoryItem() = default;
118 
119  // Non-default constructor.
120  JeodMemoryItem(bool placement_new,
121  bool is_array,
122  bool is_guarded,
123  bool is_structured,
124  unsigned int nelems_in,
125  unsigned int type_idx,
126  unsigned int alloc_idx);
127 
128  // Destructor.
129  ~JeodMemoryItem() = default;
130 
131  // NOTE: The copy constructor and assignment operator are not masked out.
132  // The default implementations work just fine here; this is a POD structure.
133 
134  // Public accessor functions.
135  // Note: Only the unique_id and is_registered flag has a setter; these are
136  // not known at creation time.
137  // All others members do not have setters; the elements are immutable.
138  // Second note: The unique_id can only be set once.

153  uint32_t get_unique_id() const;

171  bool get_is_registered() const;
172 
173  // Access the checkpointed flag.
174  bool get_checkpointed() const;
175 
176  // Member data.
177  // Only 16 bytes!
178 private:
179  /**
180  * Number of elements in the allocated array.
181  */
182  uint32_t nelems{}; //!< trick_units(--)
183 
184  /**
185  * Allocation information index, max of 2^32-2 tracked locations.
186  * The allocation information is a string of the form "file.cc:line#" that
187  * indicates where in the code the data was allocated. The underlying string
188  * is maintained in the global memory manager's string table.
189  */
190  uint32_t alloc_info_index{}; //!< trick_units(--)
191 
192  /**
193  * Unique identifier, max of 2^32-2 allocations (zero is not used).
194  * The unique identifier forms the basis of the item name sent to the
195  * simulation engine for this memory item.
196  */
197  uint32_t unique_id{}; //!< trick_units(--)
198 
199  /**
200  * High order bits of the descriptor index.
201  * The descriptor_index specifies the type decriptor that describes the data.
202  * The underlying descriptor is maintained in the global memory manager's
203  * type descriptor table.
204  */
205  uint16_t descriptor_index_hi{}; //!< trick_units(--)
206 
207  /**
208  * Low order bits of the descriptor index.
209  */
210  uint8_t descriptor_index_lo{}; //!< trick_units(--)
211 
212  /**
213  * Flags indicating whether
214  * - The data was constructed with default new or placement new
215  * - The data was allocated as an array or as a single object
216  * - The allocated are guarded
217  * - The data is a structured or non-structured data type
218  * - The data has been checkpointed (future)
219  * - Plus three more future-use spares.
220  */
221  uint8_t flags{}; //!< trick_units(--)
222 };
