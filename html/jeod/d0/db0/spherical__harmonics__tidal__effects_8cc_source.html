104 
105  // Allocate arrays for this tidal effects object.
106  tidal_bodies = JEOD_ALLOC_CLASS_POINTER_ARRAY(num_tidal_bodies, Planet);
107  tidal_bodies_inertial = JEOD_ALLOC_CLASS_POINTER_ARRAY(num_tidal_bodies, RefFrame);
108 
109  for(unsigned int ii = 0; ii < num_tidal_bodies; ++ii)
110  {

112  if(var_init->tidal_body_names[ii].empty())
113  {
114  MessageHandler::fail(__FILE__,
115  __LINE__,

117  "tidal_body_name[%d] is an empty string.",
118  ii);

120  }
121 
122  // Find the Planet whose name matches the supplied name.
123  tidal_bodies[ii] = dyn_manager.find_planet(var_init->tidal_body_names[ii]);
124

126  if(tidal_bodies[ii] == nullptr)
127  {
128  MessageHandler::fail(__FILE__,
