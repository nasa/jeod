62 /**
63  * Write a character in the case of overflow of the write buffer.
64  * For now, the buffer always overflows.
65  * @return Status: EOF => failed
66  * \param[in] ch Character to be writter
67  */
68 std::streambuf::int_type SectionedOutputBuffer::overflow(std::streambuf::int_type ch)
69 {
70  std::streambuf::int_type result;
71 
72  // Protect against writes to an inoperable object.
73  if(!*this)
74  {
75  result = std::streambuf::traits_type::eof();
76  }
77 
78  // The response to EOF is to flush the (real) buffer.
79  else if(std::streambuf::traits_type::eq_int_type(ch, std::streambuf::traits_type::eof()))
80  {
81  try
82  {
83  result = file_buf->pubsync();
84  }
85  catch(...)
86  {
87  result = std::streambuf::traits_type::eof();
88  }
89  }
90 
91  // The response to non-EOF is write the character to the real buffer.
92  else
93  {
94  try
95  {
96  result = file_buf->sputc(std::streambuf::traits_type::to_char_type(ch));
97  }
98  catch(...)
99  {
100  result = std::streambuf::traits_type::eof();
101  }
102  }
103 
104  return result;
105 }

197 /**
198  * Activate the object.
199  * @note
200  * Using the object for writing prior to activation will write nothing.
201  * @return True if activated.
202  */
203 bool SectionedOutputStream::activate()
204 {
205  // Activating an invalid object is verboten.
206  if((manager == nullptr) || (stream == nullptr))
207  {
208  MessageHandler::error(__FILE__,
209  __LINE__,
210  SimInterfaceMessages::implementation_error,
211  "Illegal attempt to activate an invalid SectionedOutputStream.");
212  return false;
213  }
214 
215  // Activating an already-activated object is verboten.
216  if(is_active)
217  {
218  MessageHandler::error(__FILE__,
219  __LINE__,
220  SimInterfaceMessages::implementation_error,
221  "Illegal attempt to activate an already-active SectionedOutputStream");
222  return false;
223  }
224 
225  // There can only be one active object present at any one time.
226  if(!manager->register_writer(this))
227  {
228  MessageHandler::error(__FILE__,
229  __LINE__,
230  SimInterfaceMessages::implementation_error,
231  "Illegal attempt to activate a writer while one is active.");
232  return false;
233  }
234 
235  // All is cool. Write the section header to the file and activate.
236  // Note: The destructor eventually writes the corresponding section trailer.
237 
238  // Preface the header with blank lines unless we're at start of file.
239  // Note: The static cast is needed to support old versions of GNU c++.
240  if(static_cast<std::size_t>(stream->tellp()) != 0)
241  {
242  *stream << "\n\n";
243  }
244 
245  // Write the section header proper to the file.
246  *stream << *section_start << tag << "\n" << std::endl;
247 
248  // Activate the stream buffer and the stream.
249  sectbuf.activate(*stream);
250  is_active = true;
251 
252  return true;
253 }

255 /**
256  * Deactivate the object.
257  * @note
258  * Deactivation is undoable.
259  */
260 void SectionedOutputStream::deactivate()
261 {
262  // Write the section trailer to the file if OK.
263  if(!!*this)
264  {
265  *stream << "\n" << *section_end << tag << std::endl;
266  }
267 
268  // Deregister this object as the active writer.
269  if(is_active)
270  {
271  manager->deregister_writer(this);
272  }
273  // Deactivate the sectioned output stream buffer.
274  // Note: This will flush buffered output if this is a well-behaved stream.
275  sectbuf.deactivate();
276 
277  // Mark the object as (permanently) inactive.
278  manager = nullptr;
279  stream = nullptr;
280  is_active = false;
281 }
