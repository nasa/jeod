79 /**
80  * Compute the minimum interpolation quaternion
81  * between a start quarternion and end quaternion
82  * \param[in] q1 Starting quaternion
83  * \param[in] q2 Ending quaternion
84  * \param[in] T Interpolation coefficient between
85  * 0.0 and 1.0 representing a rotational scale factor
86  * between the intial and final quaternion. When the compute_slerp
87  * method is used in a loop to rotate an object from a start and end
88  * orientation, a smaller step or change in T results in a smoother
89  * object rotation
90  */
91 Quaternion Quaternion::compute_slerp(Quaternion & q1, Quaternion & q2, const double T)
92 {
93  // SLERP is only valid for unit quaternions
94  q1.normalize();
95  q2.normalize();
96 
97  // The SLERP Algorithm is as follows
98  // qm = ( q1*sin((1-T)*theta) + q2*sin(T*theta) ) / sin(theta)
99 
100  Quaternion qm;
101 
102  double cosTheta = (q1.scalar * q2.scalar) + Vector3::dot(q1.vector, q2.vector);
103  // Ensure shortest angle is used
104  if(cosTheta < 0)
105  {
106  // q2 = -q2;
107  q2.scalar = -q2.scalar;
108  q2.vector[0] = -q2.vector[0];
109  q2.vector[1] = -q2.vector[1];
110  cosTheta = -cosTheta;
111  }
112 
113  // If q1 = q2 or q1 = -q2 then theta = 0
114  // Therefore, qm = q1
115  if(std::abs(cosTheta) > 1.0 || Numerical::compare_exact(std::abs(cosTheta), 1.0))
116  {
117  return q1;
118  }
119 
120  // Intermediate calcs. Angles are in radians
121  double theta = std::acos(cosTheta);
122  double sinTheta = std::sin(theta);
123  // If theta = 180 deg, result is undefinted
124  if(std::abs(sinTheta) < 0.0000000000001)
125  {
126  MessageHandler::fail(__FILE__,
127  __LINE__,
128  QuatMessages::undefined,
129  "The quaternions provided result in an undefined quaternion interpolation.");
130  }
131  double ratio1 = std::sin((1 - T) * theta) / sinTheta;
132  double ratio2 = std::sin(T * theta) / sinTheta;
133 
134  qm.scalar = ((q1.scalar * ratio1) + (q2.scalar * ratio2));
135  for(int i = 0; i < 3; i++)
136  {
137  qm.vector[i] = (q1.vector[i] * ratio1 + q2.vector[i] * ratio2);
138  }
139 
140  // Normalize the resulting interpolation quaternion
141  qm.normalize();
142 
143  return qm;
144  // all done!
145 }
