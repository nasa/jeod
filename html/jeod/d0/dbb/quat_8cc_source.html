87 /**
88  * Compute the minimum interpolation quaternion
89  * between a start quarternion and end quaternion
90  * \param[in] q1 Starting quaternion
91  * \param[in] q2 Ending quaternion
92  * \param[in] T Interpolation coefficient between
93  * 0.0 and 1.0 representing a rotational scale factor
94  * between the intial and final quaternion. When the compute_slerp
95  * method is used in a loop to rotate an object from a start and end
96  * orientation, a smaller step or change in T results in a smoother
97  * object rotation
98  */
99  Quaternion Quaternion::compute_slerp (
100  Quaternion & q1, Quaternion & q2, const double T)
101 {
102  // SLERP is only valid for unit quaternions
103  q1.normalize();
104  q2.normalize();
105 
106  // The SLERP Algorithm is as follows
107  // qm = ( q1*sin((1-T)*theta) + q2*sin(T*theta) ) / sin(theta)
108 
109  Quaternion qm;
110 
111  double cosTheta = (q1.scalar * q2.scalar) +
112  Vector3::dot(q1.vector, q2.vector);
113  // Ensure shortest angle is used
114  if (cosTheta < 0)
115  {
116  //q2 = -q2;
117  q2.scalar = -q2.scalar;
118  q2.vector[0] = -q2.vector[0];
119  q2.vector[1] = -q2.vector[1];
120  cosTheta = -cosTheta;
121  }
122 
123  // If q1 = q2 or q1 = -q2 then theta = 0
124  // Therefore, qm = q1
125  if (std::abs(cosTheta) > 1.0 || Numerical::compare_exact(std::abs(cosTheta),1.0))
126  {
127  return q1;
128  }
129 
130  // Intermediate calcs. Angles are in radians
131  double theta = std::acos(cosTheta);
132  double sinTheta = std::sin(theta);
133  // If theta = 180 deg, result is undefinted
134  if (std::abs(sinTheta) < 0.0000000000001 )
135  {
136  MessageHandler::fail (
137  __FILE__, __LINE__, QuatMessages::undefined,
138  "The quaternions provided result in an undefined quaternion interpolation.");
139  }
140  double ratio1 = std::sin((1 - T) * theta) / sinTheta;
141  double ratio2 = std::sin(T * theta) / sinTheta;
142 
143  qm.scalar = ((q1.scalar * ratio1) + (q2.scalar * ratio2));
144  for (int i=0; i<3; i++)
145  {
146  qm.vector[i] = (q1.vector[i] * ratio1 + q2.vector[i] * ratio2);
147  }
148 
149  // Normalize the resulting interpolation quaternion
150  qm.normalize();
151 
152  return qm;
153  // all done!
154 
155 }
