81  * Build NED frame state based on current reference point information.
82  */
83 void NorthEastDown::build_ned_orientation()
84 {
85  // Compute the sine and cosine of the latitude and longitude used to specify
86  // the NED frame orientation
87  double sinlat = 0.0;
88  double coslat = 0.0;
89  double sinlon = 0.0;
90  double coslon = 0.0;
91 
92  if(this->altlatlong_type == spherical)
93  {
94  sinlat = std::sin(sphere_coords.latitude);
95  coslat = std::cos(sphere_coords.latitude);
96  sinlon = std::sin(sphere_coords.longitude);
97  coslon = std::cos(sphere_coords.longitude);
98  }
99  else if(this->altlatlong_type == elliptical)
100  {
101  sinlat = std::sin(ellip_coords.latitude);
102  coslat = std::cos(ellip_coords.latitude);
103  sinlon = std::sin(ellip_coords.longitude);
104  coslon = std::cos(ellip_coords.longitude);
105  }
106  // Any other value is undefined/invalid
107  else
108  {
109  MessageHandler::fail(__FILE__,
110  __LINE__,
111  PlanetFixedMessages::invalid_request,
112  "AltLatLong set used to define North-East-Down frame not specified.");
113  return;
114  }
115 
116  // Construct the NED frame's orientation
117  ned_frame.state.rot.T_parent_this[0][0] = -sinlat * coslon;
118  ned_frame.state.rot.T_parent_this[0][1] = -sinlat * sinlon;
119  ned_frame.state.rot.T_parent_this[0][2] = coslat;
120  ned_frame.state.rot.T_parent_this[1][0] = -sinlon;
121  ned_frame.state.rot.T_parent_this[1][1] = coslon;
122  ned_frame.state.rot.T_parent_this[1][2] = 0;
123  ned_frame.state.rot.T_parent_this[2][0] = -coslat * coslon;
124  ned_frame.state.rot.T_parent_this[2][1] = -coslat * sinlon;
125  ned_frame.state.rot.T_parent_this[2][2] = -sinlat;
126  ned_frame.state.rot.compute_quaternion();
127  Vector3::initialize(ned_frame.state.rot.ang_vel_this);
128  ned_frame.state.rot.ang_vel_mag = 0.0;
129 }
