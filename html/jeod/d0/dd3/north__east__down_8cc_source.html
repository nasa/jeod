112 /**
113  * Build NED frame state based on current reference point information.
114  */
115 void NorthEastDown::build_ned_orientation(
116  void)
117 {
118  // Compute the sine and cosine of the latitude and longitude used to specify
119  // the NED frame orientation
120  double sinlat = 0.0;
121  double coslat = 0.0;
122  double sinlon = 0.0;
123  double coslon = 0.0;
124 
125  if (this->altlatlong_type == spherical) {
126  sinlat = std::sin (sphere_coords.latitude);
127  coslat = std::cos (sphere_coords.latitude);
128  sinlon = std::sin (sphere_coords.longitude);
129  coslon = std::cos (sphere_coords.longitude);
130  }
131  else if (this->altlatlong_type == elliptical) {
132  sinlat = std::sin (ellip_coords.latitude);
133  coslat = std::cos (ellip_coords.latitude);
134  sinlon = std::sin (ellip_coords.longitude);
135  coslon = std::cos (ellip_coords.longitude);
136  }
137  // Any other value is undefined/invalid
138  else {
139  MessageHandler::fail (
140  __FILE__, __LINE__, PlanetFixedMessages::invalid_request,
141  "AltLatLong set used to define North-East-Down frame not specified.");
142  return;
143  }
144 
145  // Construct the NED frame's orientation
146  ned_frame.state.rot.T_parent_this[0][0] = -sinlat * coslon;
147  ned_frame.state.rot.T_parent_this[0][1] = -sinlat * sinlon;
148  ned_frame.state.rot.T_parent_this[0][2] = coslat;
149  ned_frame.state.rot.T_parent_this[1][0] = -sinlon;
150  ned_frame.state.rot.T_parent_this[1][1] = coslon;
151  ned_frame.state.rot.T_parent_this[1][2] = 0;
152  ned_frame.state.rot.T_parent_this[2][0] = -coslat * coslon;
153  ned_frame.state.rot.T_parent_this[2][1] = -coslat * sinlon;
154  ned_frame.state.rot.T_parent_this[2][2] = -sinlat;
155  ned_frame.state.rot.compute_quaternion ();
156  Vector3::initialize (ned_frame.state.rot.ang_vel_this);
157  ned_frame.state.rot.ang_vel_mag = 0.0;
158 
159  return;
160 }
