174 void
175 EmptySpaceEphemeris::set_name (

177 {
178  std::string sname(new_name);
179  static const char dotInrtlName[] = ".inertial";
180  static const char inrtlName[] = "inertial";
181 
182  // Set the name of the central point and central frame.
183  // The central point and frame must be named "xxx.inertial".
184  if ((sname.length() >= 10) &&
185  (sname.compare(sname.length() - 9, sizeof(dotInrtlName), dotInrtlName) == 0)) {
186  central_point.set_name (new_name);
187  central_frame.set_name (new_name);
188  }
189  else {
190  central_point.set_name (new_name, inrtlName);
191  central_frame.set_name (new_name, inrtlName);
192  }
193 
194  // Pass the message to the parent.
195  SinglePointEphemeris::set_name (new_name);
196 }

354 void
355 SinglePlanetEphemeris::ephem_initialize (
356  EphemeridesManager & ephem_manager)
357 {
358 
359  // Nothing to do if the model is inactive.
360  if (! active) {
361  return;
362  }
363 
364  unsigned int nplanets = ephem_manager.get_num_planets();
365  if (nplanets != 1) {
366  MessageHandler::fail (
367  __FILE__, __LINE__, EphemeridesMessages::inconsistent_setup,
368  "%s planets registered with the EphemeridesManager; "
369  "exactly one is needed in single planet mode",
370  nplanets == 0 ? "No" : "More than one");
371 
372  central_point.disable ();
373  deactivate ();
374 
375  return;
376  }
377 
378  // At this point, the planet with same name as the model must exist and
379  // the central point must have a target frame.
380  if ((ephem_manager.find_base_planet (identifier) == nullptr) ||
381  (central_point.get_target_frame() == nullptr)) {
382  MessageHandler::error (
383  __FILE__, __LINE__, EphemeridesMessages::inconsistent_setup,
384  "Could not find planet named \"%s\".\n"
385  "Shutting down the SinglePlanetEphemeris.",
386  identifier);
387  central_point.disable ();
388  deactivate ();
389 
390  return;
391  }
