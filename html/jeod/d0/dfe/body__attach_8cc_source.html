95 /**
96  * A derived class presumably has performed the attachment, which may
97  * not have worked, and forwarded the apply call to this method.
98  * This method acts on the status from that child class attachment.
99  * \param[in,out] dyn_manager Jeod manager
100  */
101 void BodyAttach::apply(DynManager & dyn_manager)
102 {
103  std::string subject_name("Unknown");
104  std::string parent_name("Unknown");
105  if(dyn_parent != nullptr)
106  {
107  parent_name = dyn_parent->name.get_name();
108  }
109  else
110  {
111  if(mass_parent != nullptr)
112  {
113  parent_name = mass_parent->name.get_name();
114  }
115  else if(ref_parent != nullptr)
116  {
117  parent_name = ref_parent->get_name();
118  }
119  }
120 
121  if(dyn_subject != nullptr)
122  {
123  subject_name = dyn_subject->name.get_name();
124  }
125  else
126  {
127  if(mass_subject != nullptr)
128  {
129  subject_name = mass_subject->name.get_name();
130  }
131  }
132 
133  // Attachment succeeded: Debug.
134  if(succeeded)
135  {
136  MessageHandler::debug(__FILE__,
137  __LINE__,
138  BodyActionMessages::trace,
139  "%s: %s attached to %s.",
140  action_identifier.c_str(),
141  subject_name.c_str(),
142  parent_name.c_str());
143  }
144 
145  // Attachment failed: Terminate the sim if terminate_on_error is set.
146  else if(terminate_on_error)
147  {
148  MessageHandler::fail(__FILE__,
149  __LINE__,
150  BodyActionMessages::fatal_error,
151  "'%s' failed to attach '%s' to '%s'. "
152  "The terminate_on_failure flag set and an attachment error occurred.\n"
153  "The attachment error described above is fatal per this setting.",
154  action_identifier.c_str(),
155  subject_name.c_str(),
156  parent_name.c_str());
157  }
158 
159  // Attachment failed, terminate_ not set: Tell the user about the problem.
160  else
161  {
162  MessageHandler::error(__FILE__,
163  __LINE__,
164  BodyActionMessages::not_performed,
165  "%s failed to attach %s to %s.",
166  action_identifier.c_str(),
167  subject_name.c_str(),
168  parent_name.c_str());
169  }
170 
171  // Forward the action up the chain.
172  BodyAction::apply(dyn_manager);
173 }
