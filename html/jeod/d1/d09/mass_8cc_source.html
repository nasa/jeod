77  : links(*this)
78 {
79  JEOD_REGISTER_CLASS(MassBody);
80  JEOD_REGISTER_CLASS(MassPoint);

82  // Initialize the connectivities of the contained MassPoint objects. The
83  // three MassProperties objects form a tree with the structure_point as
84  // the root of this tree and the core_ and composite_properties as children
85  // of the structure_point.
86  core_properties.attach(structure_point);
87  composite_properties.attach(structure_point);
88 
89  // The MassBody object contains additional representations of the core and
90  // composite frames as mass points to aid in the computation of composite
91  // properties. These are the core properties with respect to the composite
92  // properties and the composite properties with respect to the parent mass
93  // body's structural and composite body frames.
94  // These latter two start unattached. The former needs to be attached.
95  core_wrt_composite.attach(composite_wrt_pbdy);
96 
97  // The core and composite body frames share the same axes.
98  // Note: This statement could be omitted as the MassPointState constructor
99  // initializes the transformation matrix to identity.
100  // The initialization is intentionally replicated here intentionally.
101  Matrix3x3::identity(core_wrt_composite.T_parent_this);
102 }

137 /**
138  * Destroy a MassBody object.
139  */
140 MassBody::~MassBody()
141 {
142  // Denote that the parent's (if any) composite properties need an update.
143  set_update_flag();
144 
145  // If frame has children, all those children and their progeny contain
146  // pointers to this frame. Sever those links by making each child a root.
147  while(links.has_children())
148  {
149  links.child_tail()->container().detach();
150  }
151 
152  // Sever the links from the parent and sibling nodes as well.
153  if(!links.is_root())
154  {
155  bool success = detach();
156  // Assume failure because this or attachment belongs to DynBody
157  if(!success)
158  {
159  /*
160  * FATAL!
161  * Can't find a DynManager, can't attempt DynBody disconnect.
162  * Try to print a message.
163  * NOTE:
164  * During shutdown destruction, the MessageHandler may not be
165  * available!
166  */
167  MessageHandler::fail(__FILE__,
168  __LINE__,
169  MassBodyMessages::invalid_detach,
170  "MassBody '%s' is attached to something "
171  "at shutdown, but failed to detach!",
172  name.c_str());
173  }
174  }
175 
176  // Detach and free the allocated MassPoints.
177  while(!mass_points.empty())
178  {
179  MassPoint * point = mass_points.back();
180  point->detach();
181  JEOD_DELETE_OBJECT(point);
182  mass_points.pop_back();
183  }
184 }

191 void MassBody::initialize_mass(const MassPropertiesInit & properties, const std::vector<MassPointInit *> & points)
192 {
193  // Denote that the composite properties need an update.
194  set_update_flag();
195 
196  // Initialize the core mass properties.
197  properties.initialize_mass_properties(core_properties);
198 
199  mass_properties_initialized = true;
200 
201  // The core and composite frames have the same alignment wrt structure.
202  composite_properties.Q_parent_this = core_properties.Q_parent_this;
203  Matrix3x3::copy(core_properties.T_parent_this, composite_properties.T_parent_this);
204 
205  // Add the mass points.
206  for(size_t ii = 0; ii < points.size(); ii++)
207  {
208  MassPointInit * point = points.at(ii);
209  if(dyn_owner == nullptr)
210  {
211  add_mass_point(*point);
212  }
213  else
214  {
215  dyn_owner->add_mass_point(*point);
216  }
217  }
218 }

303  * Find the mass point with the given name.
304  * @return Mass point
305  * \param[in] pt_name mass point name
306  */
307 const MassPoint * MassBody::find_mass_point(const std::string & pt_name) const
308 {
309  std::size_t search_offset = name.size() + 1;
310  const MassPoint * found_point = nullptr;
311  std::string pt_suffix;
312 
313  // Only search if the name is valid. (No else; found point is already NULL)
314  if(!pt_name.empty())
315  {
316  pt_suffix = name.suffix(pt_name);
317 
318  // Search for the point.
319  auto name_compare = [&search_offset, &pt_suffix](const MassPoint * point)
320  {
321  return point->name.ends_with(search_offset, pt_suffix);
322  };
323 
324  const auto iter = std::find_if(mass_points.begin(), mass_points.end(), name_compare);
325 
326  if(iter != mass_points.end())
327  {
328  found_point = *iter;
329  }
330  }
331 
332  return found_point;
333 }

335 /**
336  * Add a mass point to the list of such.
337  * \param[in] mass_point_init Mass point spec
338  */
339 void MassBody::add_mass_point(const MassPointInit & mass_point_init)
340 {
341  const std::string & pt_suffix = name.suffix(mass_point_init.name);
342  MassPoint * mass_point;
343 
344  // Sanity checks:
345  // 1. The MassPoint must have a name.
346  if(pt_suffix.empty())
347  {
348  MessageHandler::fail(__FILE__,
349  __LINE__,
350  MassBodyMessages::invalid_name,
351  "Attempt to add a MassPoint with an empty name "
352  "to the MassPoint list for MassBody '%s'.",
353  name.c_str());
354 
355  // Not reached
356  return;
357  }
358 
359  // 2. The MassPoint's name must be unique.
360  else if(find_mass_point(mass_point_init.name) != nullptr)
361  {
362  MessageHandler::fail(__FILE__,
363  __LINE__,
364  MassBodyMessages::invalid_name,
365  "Attempt to add a MassPoint with duplicate name '%s' "
366  "to the MassPoint list for MassBody '%s'.",
367  mass_point_init.name.c_str(),
368  name.c_str());
369 
370  // Not reached
371  return;
372  }
373  // No else: All checks passed.
374 
375  // Allocate the point.
376  mass_point = JEOD_ALLOC_CLASS_OBJECT(MassPoint, ());
377 
378  // Construct the mass point's name as body_name.suffix.
379  mass_point->name.set_name(name.get_name(), pt_suffix);
380  mass_point->name.freeze_name();
381 
382  // Link the mass point to the structural frame.
383  mass_point->attach(structure_point);
384 
385  // Complete initialization of the mass point.
386  mass_point_init.initialize_mass_point(*mass_point);
387 
388  // Add the mass point to the list of such.
389  mass_points.push_back(mass_point);
390 }
