158 bool BodyAction::validate_body_inputs(DynBody *& dyn_body_in, MassBody *& mass_body_in,
159  const std::string & body_base_name, bool allow_failure)
160 {
161  bool is_good_body = false;
162  // Sanity check: Subject must not be null. Check for bad scoobies
163  if( mass_body_in==nullptr && dyn_body_in==nullptr )
164  {
165  if(!allow_failure) {
166  MessageHandler::fail (
167  __FILE__, __LINE__, BodyActionMessages::null_pointer,
168  "%s failed:\n"
169  "The %s body was not assigned",
170  action_identifier.c_str(), body_base_name.c_str());
171  }
172  }
173  else if( mass_body_in!=nullptr && dyn_body_in!=nullptr )
174  {
175  if(mass_body_in != &dyn_body_in->mass)
176  {
177  if(!allow_failure) {
178  MessageHandler::fail (
179  __FILE__, __LINE__, BodyActionMessages::fatal_error,
180  "%s failed:\n"
181  "Both mass_%s and dyn_%s are defined,\n"
182  "only one may be set. This should not be able to happen.",
183  action_identifier.c_str(), body_base_name.c_str(), body_base_name.c_str());
184  }
185  }
186  }
187 
188  if (dyn_body_in != nullptr)
189  {
190  mass_body_in = &dyn_body_in->mass;
191  is_good_body = true;
192  }
193  else if (mass_body_in != nullptr)
194  {
195  dyn_body_in = mass_body_in->dyn_owner;
196  is_good_body = true;
197  }
198  return is_good_body;
199 }
