118 bool BodyAction::validate_body_inputs(DynBody *& dyn_body_in,
119  MassBody *& mass_body_in,
120  const std::string & body_base_name,
121  bool allow_failure)
122 {
123  bool is_good_body = false;
124 
125  if(dyn_body_in == nullptr)
126  {
127  if(mass_body_in == nullptr)
128  {
129  // Sanity check: Subject must not be null. Check for bad scoobies
130  if(!allow_failure)
131  {
132  MessageHandler::fail(__FILE__,
133  __LINE__,
134  BodyActionMessages::null_pointer,
135  "%s failed:\n"
136  "The %s body was not assigned",
137  action_identifier.c_str(),
138  body_base_name.c_str());
139  }
140  }
141  else
142  {
143  dyn_body_in = mass_body_in->dyn_owner;
144  is_good_body = true;
145  }
146  }
147  else
148  {
149  if(mass_body_in == nullptr)
150  {
151  mass_body_in = &dyn_body_in->mass;
152  is_good_body = true;
153  }
154  else
155  {
156  if(mass_body_in != &dyn_body_in->mass)
157  {
158  if(!allow_failure)
159  {
160  MessageHandler::fail(__FILE__,
161  __LINE__,
162  BodyActionMessages::fatal_error,
163  "%s failed:\n"
164  "Both mass_%s and dyn_%s are defined,\n"
165  "only one may be set. This should not be able to happen.",
166  action_identifier.c_str(),
167  body_base_name.c_str(),
168  body_base_name.c_str());
169  }
170  }
171  else
172  {
173  is_good_body = true;
174  }
175  }
176  }
177 
178  return is_good_body;
179 }
