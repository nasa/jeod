71 void GaussJacksonStateMachine::reset()
72 {
73  // "Reset" is the starting point of the state machine.
74  // Any history collected is now trash, so history_length is set to zero.
75  // Setting history_size to one makes it so that the next call to perform_step
76  // will result in a state transition out of Reset (to Priming).

84  current_order = initial_order;

88  cycle_start_time = 0.0;
89

91  at_reinitialize = false;
92  at_order_change = false;
93  at_end_of_tour = false;

103 void GaussJacksonStateMachine::perform_step()
104 {
105  at_downsample = false;
106  at_reinitialize = false;
107  at_order_change = false;
108  at_end_of_tour = false;
109 
110  ++history_length;
111 
112  if(history_length == history_size)
113  {
114  transition_state();
115  }
116 
117  if(fsm_state == BootstrapEdit)
118  {
119  steps_since_reset = history_length * step_increment;
120  cycle_start_time = double(steps_since_reset) / double(tour_count);
121  }
122  else
123  {
124  cycle_start_time = double(steps_since_reset) / double(tour_count);
125  steps_since_reset += step_increment;
126  assert((steps_since_reset == history_length * step_increment));
127  at_end_of_tour = (steps_since_reset % tour_count) == 0;
128  }
129 }

131 void GaussJacksonStateMachine::transition_state()
132 {
133  switch(fsm_state)
134  {
135  // Reset:
136  // Transition to Priming, with a history length of initial_order+1 needed to
137  // make the next transition (which will be to BootstrapEdit).
138  case Reset:
139  fsm_state = Priming;
140  current_order = 0;
141  history_size = initial_order + 1;
142  steps_since_reset = 0;
143  break;
144 
145  // Priming:
146  // Transition to BootstrapEdit.
147  case Priming:
148  exit_priming();
149  break;
150 
151  // BootstrapEdit:
152  // Transition out of edit if the edit succeeded, redo otherwise.
153  case BootstrapEdit:
154  exit_bootstrap_edit();
155  break;
156 
157  // BootstrapStep:
158  // Transition to BootstrapEdit, but with added twists of
159  // downsampling/step doubling and/or increasing the order.
160  case BootstrapStep:
161  exit_bootstrap_step();
162  break;
163 
164  // Operational: You can checkout but you can never leave.
165  case Operational:
166  break;
167 
168  default:
169  assert("State machine called incorrectly" == nullptr);
170  break;
171  }
172 }

197 void GaussJacksonStateMachine::exit_bootstrap_edit()
198 {
199  assert(history_length == history_size);
200  assert(history_size == current_order + 1);
201  assert(steps_since_reset == step_increment * (history_length - 1));
202 
203  at_reinitialize = true;
204 
205  // Edit failed:
206  // Redo the edit, but now using the newly collected derivatives.
207  if(bootstrap_edit_redo_needed)
208  {
209  bootstrap_edit_redo_needed = false;
210  history_length = 1;
211  ++correction_iterations;
212  fsm_state = BootstrapEdit;
213  }
214 
215  // Edit passed, and at final order, final step size:
216  // Transition to Operational.
217  else if((scale_factor == 1) && (current_order == final_order))
218  {
219  fsm_state = Operational;
220  }
221 
222  // Edit passed, but not at both final order, final step size:
223  // Transition to BootstrapStep.
224  else
225  {
226  fsm_state = BootstrapStep;
227 
228  // Set the history size to that needed for the action at completion
229  // of BootstrapStep, which will be to bump order by two,
230  // downsample and double the step size, or both.
231  if(scale_factor == 1)
232  {
233  // Change order only:
234  // Need history_size-1 = current_order + 2
235  history_size = (current_order + 2) + 1;
236  }
237  else if(current_order == final_order)
238  {
239  // Downsample only:
240  // Need (history_size-1)/2 = current_order
241  history_size = 2 * current_order + 1;
242  }
243  else
244  {
245  // Downsample and change order:
246  // Need (history_size-1)/2 = current_order+2
247  history_size = 2 * (current_order + 2) + 1;
248  }
249  }
250 }

252 void GaussJacksonStateMachine::exit_bootstrap_step()
253 {
254  at_reinitialize = true;
255 
256  // Downsample if not at the final step size.
257  if(scale_factor != 1)
258  {
259  at_downsample = true;
260  history_size = (history_size + 1) / 2;
261  scale_factor /= 2;
262  step_increment *= 2;
263  cycle_scale *= 2.0;
264  }
265 
266  // Increase the order if not at the final order.
267  if(current_order != final_order)
268  {
269  at_order_change = true;
270  current_order += 2;
271  }
272 
273  // At least one of the above should be the case.
274  assert(at_downsample || at_order_change);
275 
276  // Corrections enabled: Transition to BootstrapEdit.
277  if(max_correction_iterations > 0)
278  {
279  bootstrap_edit_redo_needed = false;
280  history_length = 1;
281  correction_iterations = 1;
282  fsm_state = BootstrapEdit;
283  }
284 
285  // Corrections disabled: Act as if edit has succeeded.
286  else
287  {
288  bootstrap_edit_redo_needed = false;
289  exit_bootstrap_edit();
290  }
291 }
