26 /*------------------------------------------------------------------------------
27  * Overview
28  * The goal is to find a unitary quaternion 'q' such that 'q' yields the
29  * same
30  * vector transformations as does the transformation matrix 'a'. The
31  * quaternion
32  * is unique save for an ambiguity in the sign of all the elements. The sign
33  * will be chosen such that the scalar part of the quaternion is
34  * non-negative.
35  *
36  * Nomenclature
37  * a Transformation matrix
38  * q Left quaternion based on a
39  * qs The scalar part of the quaternion
40  * qv The vector part of the quaternion
41  * qv_i Element i of qv
42  * a_ij Element i,j of a
43  * tr Trace of the transformation matrix: a_00 + a_11 + a_22
44  * t_i a_ii - (a_jj + a_kk), k!=j!=i
45  * d_i a_kj - a_jk, eps_ijk = 1
46  * s_ij a_ij + a_ji
47  * t Rotation angle
48  * cost Cosine of t
49  * sint Sine of t
50  * u Rotation axis unit vector
51  * eps_ijk Permutation tensor
52  *
53  * Given a single axis rotation t from reference frame 1 to reference frame
54  * 2,
55  * the transformation matrix from frame 1 to frame 2 is
56  * a_ii = 1 - (1-cost)(1-u_i^2)
57  * = cost + (1-cost) u_i^2
58  * a_ij = (1-cost) u_i u_j + sint eps_ijk u_k i!=j
59  * The left transformation quaternion from frame 1 to frame 2 is
60  * qs = cos(t/2)
61  * qv_i = -sin(t/2) u_i
62  * The left quaternion can be expressed in terms of the transformation
63  * matrix
64  * by using the half-angle formulae
65  * cost = 2 cos^2(t/2) - 1
66  * sint = 2 cos(t/2) sin(t/2)
67  * 1-cost = 2 sin^2(t/2)
68  * With a little manipulation,
69  * qs = sqrt(1+tr)/2
70  * = d_i/sqrt(1+t_i)/2
71  * qv_i = sqrt(1+t_i)/2
72  * = s_ij/sqrt(1+t_j)/2
73  * = s_ik/sqrt(1+t_k)/2
74  * The above equations are subject to precision loss / divide-by-zero
75  * problems.
76  * To avoid precision loss, the forms best suited to the given matrix will
77  * be selected. Denote
78  * Method -1:
79  * qix2 = sqrt(1+tr)
80  * fact = 0.5 / qix2
81  * qs = 0.5 * qsx2
82  * qv_i = d_i * fact
83  * Methods 0-2:
84  * qix2 = sqrt(1+t_i)
85  * fact = 0.5 / qix2
86  * s = sign(d_i)
87  * qs = s d_i fact
88  * qv_i = s qix2 / 2
89  * Method -1 is selected if the trace dominates the diagonal elements of
90  * the matrix. Method 0 is selected if the a_00 dominates the trace and the
91  * other two elements. Methods 1 and 2 are selected when a_11 or a_22
92  * dominates.
93  *----------------------------------------------------------------------------*/

107 /**
108  * Compute the parent-to-child left quaternion from the input
109  * transformation matrix.
110  *
111  * \par Assumptions and Limitations
112  * - Matrix is orthonormal.
113  * \param[in] T Transformation matrix
114  */
115 void Quaternion::left_quat_from_transformation(const double T[3][3])
116 {
117  double tr; /* Trace of input transformation matrix */
118  double tmax; /* Max of tr, diagonal elements */
119  double qix2; /* sqrt(1+max(tr,t_i)) */
120  double fact; /* 0.5/qix2 */
121  int meth; /* Index of tmax in t (-1 if trace dominates) */
122  int ii;
123 
124  /* Compute the trace of the matrix. */
125  tr = T[0][0] + T[1][1] + T[2][2];
126 
127  /* Find the largest of the trace (meth = -1) and the three diagonal
128  * elements of 'a' (meth = 0, 1, or 2). */
129  meth = -1;
130  tmax = tr;
131  for(ii = 0; ii < 3; ii++)
132  {
133  if(T[ii][ii] > tmax)
134  {
135  meth = ii;
136  tmax = T[ii][ii];
137  }
138  }
139 
140  /* Use method -1 when no diagonal element dominates the trace. */
141  if(meth == -1)
142  {
143  qix2 = std::sqrt(1.0 + tr);
144  fact = 0.5 / qix2;
145  scalar = 0.5 * qix2;
146  vector[0] = fact * (T[2][1] - T[1][2]);
147  vector[1] = fact * (T[0][2] - T[2][0]);
148  vector[2] = fact * (T[1][0] - T[0][1]);
149 
150  /* Use method 0,1, or 2 based on the dominant diagonal element. */
151  }
152  else
153  {
154  ii = meth;
155  int jj = (ii + 1) % 3;
156  int kk = (jj + 1) % 3;
157 
158  double di = T[kk][jj] - T[jj][kk]; /* a_kj - a_jk */
159  qix2 = sqrt(1.0 + T[ii][ii] - (T[jj][jj] + T[kk][kk]));
160  if(di < 0.0)
161  {
162  qix2 = -qix2;
163  }
164  fact = 0.5 / qix2;
165  vector[ii] = 0.5 * qix2;
166  vector[jj] = fact * (T[ii][jj] + T[jj][ii]);
167  vector[kk] = fact * (T[ii][kk] + T[kk][ii]);
168  scalar = fact * di;
169  }
170 }
