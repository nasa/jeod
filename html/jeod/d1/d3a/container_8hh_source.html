300  /**
301  * Perform a checkpoint-restart action that will, in part, restore the object
302  * to its state at the time of the checkpoint.
303  *
304  * In the case of a JeodContainer, the actions are "clear", "insert", and
305  * "cleanup". The checkpoint writer automatically creates an initial "clear"
306  * entry as the first entry in the checkpoint file for a JeodCheckpointable
307  * object and a "cleanup" entry as the final entry. An "insert" entry is
308  * created for each element in the container's contents.
309  */
310  int perform_restore_action(const std::string & action_name, const std::string & action_value) override
311  {
312  int return_code;
313 
314  // "clear" action response is to clear the contents of the object.
315  if(action_name.compare("clear") == 0)
316  {
317  this->clear();
318  return_code = 0;
319  }
320 
321  // "insert" action response is to insert the value in the contents.
322  else if(action_name.compare("insert") == 0)
323  {
324  perform_insert_action(action_value);
325  return_code = 0;
326  }
327 
328  // "cleanup" action response is to remove detritus created during restart.
329  else if(action_name.compare("cleanup") == 0)
330  {
331  perform_cleanup_action(action_value);
332  return_code = 0;
333  }
334 
335  // All other actions are illegal and should never happen.
336  else
337  {
338  return_code = -1;
339  }
340 
341  return return_code;
342  }
