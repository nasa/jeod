141 /**
142  * Open the checkpoint output file.
143  */
144 void BasicJeodTrickSimInterface::open_checkpoint_file()
145 {
146  std::string output_file_name;
147 
148  if(!trick_memory_interface.is_checkpoint_restart_supported())
149  {
150  return;
151  }
152 
153  if(checkpoint_file_name.empty())
154  {
155  output_file_name = trick_memory_interface.get_trick_checkpoint_file(true);
156 
157  if(output_file_name.empty())
158  {
159  output_file_name = "jeod_checkpoint";
160  }
161  else
162  {
163  std::size_t last_slash = output_file_name.find_last_of('/');
164  std::size_t insert_pos = 0;
165  if(last_slash != output_file_name.npos)
166  {
167  insert_pos = last_slash + 1;
168  }
169  output_file_name.insert(insert_pos, "jeod_");
170  }
171  }
172 
173  else
174  {
175  output_file_name = checkpoint_file_name;
176  }
177 
178  checkpoint_writer = new CheckPointOutputManager(output_file_name, section_start, section_end);
179 }
