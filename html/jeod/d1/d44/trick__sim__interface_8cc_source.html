166 /**
167  * Open the checkpoint output file.
168  */
169 void
170 BasicJeodTrickSimInterface::open_checkpoint_file (
171  void)
172 {
173  std::string output_file_name;
174 
175  if (! trick_memory_interface.is_checkpoint_restart_supported()) {
176  return;
177  }
178 
179  if (checkpoint_file_name.empty()) {
180  output_file_name =
181  trick_memory_interface.get_trick_checkpoint_file (true);
182 
183  if (output_file_name.empty()) {
184  output_file_name = "jeod_checkpoint";
185  }
186  else {
187  std::size_t last_slash =
188  output_file_name.find_last_of ('/');
189  std::size_t insert_pos =
190  (last_slash == output_file_name.npos) ? 0 : last_slash + 1;
191  output_file_name.insert (insert_pos, "jeod_");
192  }
193  }
194 
195  else {
196  output_file_name = checkpoint_file_name;
197  }
198 
199  checkpoint_writer = new CheckPointOutputManager (
200  output_file_name, section_start, section_end);
201 }
