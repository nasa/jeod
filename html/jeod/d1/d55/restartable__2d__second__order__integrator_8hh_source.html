131  * @param[in] dyn_dt Dynamic time step, in dynamic time seconds.

135  * @param[in,out] velocity Generalized velocity vector.

137  *

140  er7_utils::IntegratorResult integrate(double dyn_dt,
141  unsigned int target_stage,

147  }

149  /**

153  * The integrator needs to restart from scratch when such events occur.

155  void reset_integrator()

157  integrator->reset_integrator();
