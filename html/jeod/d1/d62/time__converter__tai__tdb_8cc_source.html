20 ASSUMPTIONS AND LIMITATIONS:
21  ((The equation for calculating TDB, per page 42 in the citation above,
22  requires Julian Date to 2 decimal places. Since the various timescales
23  differ only in the 3rd decimal place, this equation is valid whether
24  TAI, UTC, UT1, or TT is used.
25  In this representation, the TAI JD is used.
26  This ambiguity produces a potential difference of 0.01 degrees in the
27  position of Earth on its orbit, which produces a potential difference
28  on the order of 0.1 microseconds in the calculation of TDB.
29  The method used is therefore accurate only to 0.1 microseconds.))

110 /******************************************************************************
111  Function: set_a_to_b_offset
112  Purpose: Calculate and set the a_to_b_offset
113  ******************************************************************************/
114 void TimeConverter_TAI_TDB::set_a_to_b_offset()
115 {
116  double g = (M_PI / 180) * (357.53 + 0.9856003 * (tai_ptr->trunc_julian_time - tai_ptr->tjt_at_epoch));
117  /* -- g parameter used in calculating TDB */
118  // NB - Astronomical Almanac provides conversion from TT to TDB;
119  // it uses (JD - 2451545.0), or (TT_JD - TT_JD_at_J2000).
120  // TT and TAI tick at the same rate, so the same number is obtained by
121  // taking (TAI_tjt - TAI_tjt_at_J2000).
122  a_to_b_offset = 0.001658 * sin(g) + 0.000014 * sin(2.0 * g);
123 }

135 /******************************************************************************
136  Function: convert_b_to_a
137  Purpose: Convert from TimeTDB to TimeTAI
138  ******************************************************************************/
139 void TimeConverter_TAI_TDB::convert_b_to_a()
140 {
141  // Initial guess - iterative solution
142  tai_ptr->set_time_by_seconds(prev_tai_seconds + (tdb_ptr->seconds - prev_tdb_seconds));
143 
144  while(true)
145  {
146  nSteps++;
147  nIter++;
148  set_a_to_b_offset();
149  double dtai = (tdb_ptr->seconds - tai_ptr->seconds) - (a_to_b_offset - a_to_b_offset_epoch);
150  tai_ptr->set_time_by_seconds(tai_ptr->seconds + dtai);
151  if(nSteps > 5 || std::abs(dtai / tai_ptr->seconds) < 1.0e-15)
152  {
153  prev_tdb_seconds = tdb_ptr->seconds;
154  prev_tai_seconds = tai_ptr->seconds;
155  break;
156  }
157  }
158 }
