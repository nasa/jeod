43 /**
44  * Normalize the quaternion,
45  * making the scalar part of the quaternion non-negative.
46  */
47 void Quaternion::normalize()
48 {
49  double qmagsq = norm_sq();
50  double diff1 = 1.0 - qmagsq;
51  double fact;
52 
53  // Compute the normalization fact, nominally 1/sqrt(qmagsq).
54  // Computational shortcut: Approximate as 2/(1+qmagsq)
55  // To second order, the error in the approximation is diff1^2/8.
56  // The approximation is valid if this error is smaller than numerical
57  // precision. A double IEEE floating point number has a 53 bit mantissa plus
58  // an implied 1 to the left of the binary point. The validity limit is thus
59  // sqrt(8*2^(-54)) = 2.107342e-08, to the accuracy of the appoximation.
60  if((diff1 > -2.107342e-08) && (diff1 < 2.107342e-08))
61  {
62  fact = 2.0 / (1.0 + qmagsq);
63  }
64  else
65  {
66  fact = 1.0 / std::sqrt(qmagsq);
67  }
68 
69  // Negate the scale factor if needed to make the scalar part non-negative.
70  if(scalar < 0.0)
71  {
72  fact = -fact;
73  }
74 
75  // Scale the quaternion by the above normalization factor.
76  scale(fact);
77 }
