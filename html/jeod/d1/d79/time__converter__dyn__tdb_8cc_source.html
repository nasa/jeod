70 /**
71  * Initialize the converter.
72  * \param[in] parent_ptr Time used to initialize the converter
73  * \param[in] child_ptr Other Time used to initialize the converter
74  * \param[in] int_dir Conversion direction: +1 a=parent; -1 b=parent; 0 error
75  */
76 void
77 TimeConverter_Dyn_TDB::initialize (
78  JeodBaseTime * parent_ptr,
79  JeodBaseTime * child_ptr,
80  const int int_dir)
81 {
82  // NOT const Time *; will be casting to different type.
83 
84  // Direction = 1 => Translate from parent (TimeDyn) to child (TimeTDB) ONLY
85  // Note: This converter will only support conversions from Dyn to TDB, but
86  // must be initialized from TDB. If TDB has not been initialized at this
87  // point, the verify_setup routines will fail the simulation.
88 
89  if (int_dir == 1) {
90  verify_setup (child_ptr, parent_ptr, int_dir);
91 
92  // Convert the parent to a TimeDyn, ensuring that this conversion works.
93  dyn_ptr = dynamic_cast<TimeDyn *> (parent_ptr);
94  if (dyn_ptr == nullptr) {
95  MessageHandler::fail (
96  __FILE__, __LINE__, TimeMessages::invalid_setup_error,
97  "\n Object '%s' is not an object of type %s, as was expected when"
98  "initializing the converter with int_dir = 1. \n"
99  "TimeDyn", parent_ptr);
100  }
101 
102  // Convert the child to a TimeTDB, ensuring that this conversion works.
103  tdb_ptr = dynamic_cast<TimeTDB *> (child_ptr);
104  if (tdb_ptr == nullptr) {
105  MessageHandler::fail (
106  __FILE__, __LINE__, TimeMessages::invalid_setup_error,
107  "\n Object '%s' is not an object of type %s, as was expected when"
108  "initializing the converter with int_dir = 1.\n"
109  "TimeTDB", child_ptr);
110  }
111  }
112 
113  // All other int_dir values are *ALWAYS* invalid.
114  else {
115  MessageHandler::fail (
116  __FILE__, __LINE__, TimeMessages::invalid_setup_error,
117  "\nIllegal value of int_dir in Dyn->TDB initializer.\n");
118  }
119 
120  // Compute the initial offset.
121  if (std::fpclassify(dyn_ptr->seconds) != FP_ZERO) {
122  MessageHandler::fail (
123  __FILE__, __LINE__, TimeMessages::initialization_error, "\n"
124  "DynTime must be 0 for the Dyn-TDB converter to be "
125  "initialized.\n");
126  }
127  a_to_b_offset = tdb_ptr->seconds;
128 
129  initialized = true;
130 
131  return;
132 }
