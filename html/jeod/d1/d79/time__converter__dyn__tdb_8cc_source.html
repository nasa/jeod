71 void TimeConverter_Dyn_TDB::initialize(JeodBaseTime * parent_ptr, JeodBaseTime * child_ptr, const int int_dir)
72 {
73  // NOT const Time *; will be casting to different type.
74 
75  // Direction = 1 => Translate from parent (TimeDyn) to child (TimeTDB) ONLY
76  // Note: This converter will only support conversions from Dyn to TDB, but
77  // must be initialized from TDB. If TDB has not been initialized at this
78  // point, the verify_setup routines will fail the simulation.
79 
80  if(int_dir == 1)
81  {
82  verify_setup(child_ptr, parent_ptr, int_dir);
83 
84  // Convert the parent to a TimeDyn, ensuring that this conversion works.
85  dyn_ptr = dynamic_cast<TimeDyn *>(parent_ptr);
86  if(dyn_ptr == nullptr)
87  {
88  MessageHandler::fail(__FILE__,
89  __LINE__,
90  TimeMessages::invalid_setup_error,
91  "\n Object '%s' is not an object of type %s, as was expected when"
92  "initializing the converter with int_dir = 1. \n"
93  "TimeDyn",
94  parent_ptr);
95  return;
96  }
97 
98  // Convert the child to a TimeTDB, ensuring that this conversion works.
99  tdb_ptr = dynamic_cast<TimeTDB *>(child_ptr);
100  if(tdb_ptr == nullptr)
101  {
102  MessageHandler::fail(__FILE__,
103  __LINE__,
104  TimeMessages::invalid_setup_error,
105  "\n Object '%s' is not an object of type %s, as was expected when"
106  "initializing the converter with int_dir = 1.\n"
107  "TimeTDB",
108  child_ptr);
109  return;
110  }
111  }
112 
113  // All other int_dir values are *ALWAYS* invalid.
114  else
115  {
116  MessageHandler::fail(__FILE__,
117  __LINE__,
118  TimeMessages::invalid_setup_error,
119  "\nIllegal value of int_dir in Dyn->TDB initializer.\n");
120  return;
121  }
122 
123  // Compute the initial offset.
124  if(std::fpclassify(dyn_ptr->seconds) != FP_ZERO)
125  {
126  MessageHandler::fail(__FILE__,
127  __LINE__,
128  TimeMessages::initialization_error,
129  "\n"
130  "DynTime must be 0 for the Dyn-TDB converter to be "
131  "initialized.\n");
132  return;
133  }
134  a_to_b_offset = tdb_ptr->seconds;
135 
136  initialized = true;
137 }
