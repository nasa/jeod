58 // Set the integration frame for this body and all its child bodies.
59 void
60 DynBody::set_integ_frame ( // Return: -- Void
61  EphemerisRefFrame & new_integ_frame) // In: -- New integration frame
62 {
63 
64  // Body does not yet have an integration frame:
65  // Link each of the three primary frames (core, composite, structure) and
66  // each of the vehicle point frames to the provided integration frame.
67  if (integ_frame == nullptr) {
68  new_integ_frame.add_child (core_body);
69  new_integ_frame.add_child (composite_body);
70  new_integ_frame.add_child (structure);
71  for (std::list<BodyRefFrame *>::iterator it = vehicle_points.begin();
72  it != vehicle_points.end();
73  ++it) {
74  BodyRefFrame * point = *it;
75  new_integ_frame.add_child (*point);
76  }
77  }
78 
79  // Body has an existing integration frame:
80  // Free resources (unsubscribe), re-link the primary frames and the
81  // registered vehicle points to the new integration frame, and
82  // reset the integrators.
83  // NOTE WELL: This uses the low-level reset_parent(). It does not
84  // update state.
85  else {
86  dyn_manager->unsubscribe_to_frame (*integ_frame);
87 
88  core_body.reset_parent (new_integ_frame);
89  composite_body.reset_parent (new_integ_frame);
90  structure.reset_parent (new_integ_frame);
91  for (std::list<BodyRefFrame *>::iterator it = vehicle_points.begin();
92  it != vehicle_points.end();
93  ++it) {
94  BodyRefFrame * point = *it;
95  point->reset_parent (new_integ_frame);
96  }
97 
98  get_dynamics_integration_group()->reset_integrators();
99  }
100 
101 
102  // Propagate the change to all child bodies.
103  for (std::list<DynBody *>::iterator it = dyn_children.begin();
104  it != dyn_children.end();
105  ++it) {
106  DynBody * child = *it;
107  child->set_integ_frame (new_integ_frame);
108  }
109 
110 
111  // Add a subscription to the integration frame. The subscription will stay
112  // in effect until the body switches integration frames.
113  dyn_manager->subscribe_to_frame (new_integ_frame);
114 
115  // Keep the gravity interaction model in sync with this body.
116  grav_interaction.set_integ_frame (new_integ_frame, *dyn_manager);
117 
118  // Set the integration frame for this body.
119  integ_frame = &new_integ_frame;
120 
121  return;
122 }

146 // Switch to the specified integration frame.
147 void
148 DynBody::switch_integration_frames ( // Return: -- Void
149  EphemerisRefFrame & new_integ_frame) // In: -- New integration frame
150 {
151  RefFrameState new_state;
152 
153  // Pass the buck to the root body.
154  // FIXME: Should calling this for a non-root body be an error?
155  if (dyn_parent != nullptr) {
156  dyn_parent->switch_integration_frames (new_integ_frame);
157  return;
158  }
159 
160  // Validate the provided frame.
161  if (! dyn_manager->is_integ_frame (new_integ_frame)) {
162  MessageHandler::fail (
163  __FILE__, __LINE__, DynBodyMessages::invalid_frame,
164  "Reference frame '%s' sent to DynBody '%s'\n"
165  "is not a valid integration frame.",
166  new_integ_frame.get_name(), name.c_str());
167 
168  // Not reached
169  return;
170  }
171 
172 
173  // Update the integrated state from the sources.
174  update_integrated_state ();
175 
176  // Compute the integrated frame's state wrt the new integration frame.
177  integrated_frame->compute_relative_state (new_integ_frame, new_state);
178 
179  // Set the integration frame.
180  set_integ_frame (new_integ_frame);
181 
182  // Update the integrated state and propagate downward.
183  integrated_frame->state = new_state;
184  propagate_state();
185 }

307 /**
308  * Integrate the translational and rotational state and propagate
309  * the integrated state to derived states.
310  * @param[in] dyn_dt Dynamic time step, in dynamic time seconds.
311  * @param[in] target_stage The stage of the integration process
312  * that the integrator should try to attain.
313  * @return The status (time advance, pass/fail status) of the integration.
314  */
315 er7_utils::IntegratorResult
316 DynBody::integrate (
317  double dyn_dt,
318  unsigned int target_stage)
319 {
320  er7_utils::IntegratorResult status (false);
321 
322  if(!frame_attach.isAttached()) {
323 
324  // Integrate the translational state if enabled to do so.
325  if (translational_dynamics) {
326  integ_results_merger.merge_integrator_result (
327  trans_integ (dyn_dt, target_stage),
328  status);
329  }
330 
331  // Integrate the rotational state if enabled to do so.
332  if (rotational_dynamics) {
333  integ_results_merger.merge_integrator_result (
334  rot_integ (dyn_dt, target_stage),
335  status);
336  }
337  }
338  else
339  {
340  // Get the integ_frame to parent_frame relative state
341  RefFrameState X_iframe_to_pframe;
342  frame_attach.get_parent_frame()->compute_relative_state(*integ_frame, X_iframe_to_pframe);
343  // Add the parent_frame to struct offset from the attachment
344  RefFrameState X_iframe_to_struct(X_iframe_to_pframe);
345  X_iframe_to_pframe.incr_right(frame_attach.get_attach_offset());
346  // Reset the structure state to the new attached state relative to integ_frame
347  set_state(RefFrameItems::Pos_Vel_Att_Rate, X_iframe_to_pframe, structure);
348  }
349 
350  // Mark the state as updated.
351  initialized_states.set (RefFrameItems::Pos_Vel_Att_Rate);
352 
353  // Propagate the integrated state to other state descriptions.
354  propagate_state ();
355 
356  return status;
357 }

380 /**
381  * Integrate the rotational state of a DynBody.
382  * @param[in] dyn_dt Dynamic time step, in dynamic time seconds.
383  * @param[in] target_stage The stage of the integration process
384  * that the integrator should try to attain.
385  * @return The status (time advance, pass/fail status) of the integration.
386  */
387 er7_utils::IntegratorResult
388 DynBody::rot_integ (
389  double dyn_dt,
390  unsigned int target_stage)
391 {
392  RefFrameRot & rot_state = composite_body.state.rot;
393 
394  // Integrate the rotational state.
395  er7_utils::IntegratorResult status =
396  rot_integrator.integrate (
397  dyn_dt, target_stage,
398  derivs.rot_accel,
399  rot_state.ang_vel_this, rot_state.Q_parent_this);
400 
401  // Normalize the integrated quaternion.
402  rot_state.Q_parent_this.normalize_integ ();
403 
404  // Compute the corresponding transformation matrix.
405  rot_state.Q_parent_this.left_quat_to_transformation (
406  rot_state.T_parent_this);
407 
408  // Compute the quaternion derivative.
409  rot_state.Q_parent_this.compute_left_quat_deriv (
410  rot_state.ang_vel_this, derivs.Qdot_parent_this);
411 
412  // Compute the angular velocity magnitude and unit vector.
413  rot_state.ang_vel_mag = Vector3::vmag (rot_state.ang_vel_this);
414  if (std::fpclassify(rot_state.ang_vel_mag) != FP_ZERO) {
415  Vector3::scale (rot_state.ang_vel_this,
416  1.0 / rot_state.ang_vel_mag,
417  rot_state.ang_vel_unit);
418  }
419  else {
420  Vector3::initialize (rot_state.ang_vel_unit);
421  }
422 
423  return status;
424 }
