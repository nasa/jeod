63 // Set the integration frame for this body and all its child bodies.
64 void DynBody::set_integ_frame( // Return: -- Void
65  EphemerisRefFrame & new_integ_frame) // In: -- New integration frame
66 {
67  // Body does not yet have an integration frame:
68  // Link each of the three primary frames (core, composite, structure) and
69  // each of the vehicle point frames to the provided integration frame.
70  if(integ_frame == nullptr)
71  {
72  new_integ_frame.add_child(core_body);
73  new_integ_frame.add_child(composite_body);
74  new_integ_frame.add_child(structure);
75  for(auto point : vehicle_points)
76  {
77  new_integ_frame.add_child(*point);
78  }
79  }
80 
81  // Body has an existing integration frame:
82  // Free resources (unsubscribe), re-link the primary frames and the
83  // registered vehicle points to the new integration frame, and
84  // reset the integrators.
85  // NOTE WELL: This uses the low-level reset_parent(). It does not
86  // update state.
87  else
88  {
89  dyn_manager->unsubscribe_to_frame(*integ_frame);
90 
91  core_body.reset_parent(new_integ_frame);
92  composite_body.reset_parent(new_integ_frame);
93  structure.reset_parent(new_integ_frame);
94  for(auto point : vehicle_points)
95  {
96  point->reset_parent(new_integ_frame);
97  }
98 
99  get_dynamics_integration_group()->reset_integrators();
100  }
101 
102  // Propagate the change to all child bodies.
103  for(auto child : dyn_children)
104  {
105  child->set_integ_frame(new_integ_frame);
106  }
107 
108  // Add a subscription to the integration frame. The subscription will stay
109  // in effect until the body switches integration frames.
110  dyn_manager->subscribe_to_frame(new_integ_frame);
111 
112  // Keep the gravity interaction model in sync with this body.
113  grav_interaction.set_integ_frame(new_integ_frame, *dyn_manager);
114 
115  // Set the integration frame for this body.
116  integ_frame = &new_integ_frame;
117 }

141 // Switch to the specified integration frame.
142 void DynBody::switch_integration_frames( // Return: -- Void
143  EphemerisRefFrame & new_integ_frame) // In: -- New integration frame
144 {
145  RefFrameState new_state;
146 
147  // Pass the buck to the root body.
148  if(dyn_parent != nullptr)
149  {
150  dyn_parent->switch_integration_frames(new_integ_frame);
151  return;
152  }
153 
154  // Validate the provided frame.
155  if(!dyn_manager->is_integ_frame(new_integ_frame))
156  {
157  MessageHandler::fail(__FILE__,
158  __LINE__,
159  DynBodyMessages::invalid_frame,
160  "Reference frame '%s' sent to DynBody '%s'\n"
161  "is not a valid integration frame.",
162  new_integ_frame.get_name().c_str(),
163  name.c_str());
164 
165  // Not reached
166  return;
167  }
168 
169  // Update the integrated state from the sources.
170  update_integrated_state();
171 
172  // Compute the integrated frame's state wrt the new integration frame.
173  integrated_frame->compute_relative_state(new_integ_frame, new_state);
174 
175  // Set the integration frame.
176  set_integ_frame(new_integ_frame);
177 
178  // Update the integrated state and propagate downward.
179  integrated_frame->state = new_state;
180  propagate_state();
181 }

205 /**
206  * Create the translational and rotational integrators for a
207  * DynBody. A DynBody integrates forces and torques
208  * in the body frame and forces induced by changes in mass properties.
209  * @param[in] generator Integrator constructor to be used to create
210  * state integrators.
211  * @param[in] controls The integration ontrols created the integrator
212  * constructor's create_integration_controls method.
213  * @param[in] time_mngr The JEOD time manager object.
214  */
215 void DynBody::create_body_integrators(const er7_utils::IntegratorConstructor & generator,
216  er7_utils::IntegrationControls & controls,
217  const JeodIntegrationTime & time_mngr)
218 {
219  // Create the translational state integrator.
220  trans_integrator.create_integrator(generator, controls);
221 
222  // Create the rotational state integrator, if enabled.
223  if(!three_dof)
224  {
225  // Validate the user-specified technique against the
226  // user-specified integration method.
227  rotation_integration = GeneralizedSecondOrderODETechnique::validate_technique(generator,
228  rotation_integration,
229  __FILE__,
230  __LINE__,
231  "DynBody",
232  name.c_str());
233 
234  // Create the rotational state integrator.
235  rot_integrator.create_integrator(rotation_integration, generator, controls);
236  }
237 
238  // Configure the object that merges results to use the integration technique
239  // corresponding to the generator.
240  integ_results_merger.configure(generator);
241 
242  // Store the time manager used for time stamping the integrated frame
243  time_manager = &time_mngr;
244 }

297 /**
298  * Integrate the translational and rotational state and propagate
299  * the integrated state to derived states.
300  * @param[in] dyn_dt Dynamic time step, in dynamic time seconds.
301  * @param[in] target_stage The stage of the integration process
302  * that the integrator should try to attain.
303  * @return The status (time advance, pass/fail status) of the integration.
304  */
305 er7_utils::IntegratorResult DynBody::integrate(double dyn_dt, unsigned int target_stage)
306 {
307  er7_utils::IntegratorResult status(false);
308 
309  if(!frame_attach.isAttached())
310  {
311  // Integrate the translational state if enabled to do so.
312  if(translational_dynamics)
313  {
314  integ_results_merger.merge_integrator_result(trans_integ(dyn_dt, target_stage), status);
315  }
316 
317  // Integrate the rotational state if enabled to do so.
318  if(rotational_dynamics)
319  {
320  integ_results_merger.merge_integrator_result(rot_integ(dyn_dt, target_stage), status);
321  }
322  }
323  else
324  {
325  // Get the integ_frame to parent_frame relative state
326  RefFrameState X_iframe_to_pframe;
327  frame_attach.get_parent_frame()->compute_relative_state(*integ_frame, X_iframe_to_pframe);
328  // Add the parent_frame to struct offset from the attachment
329  RefFrameState X_iframe_to_struct(X_iframe_to_pframe);
330  X_iframe_to_pframe.incr_right(frame_attach.get_attach_offset());
331  // Reset the structure state to the new attached state relative to integ_frame
332  set_state(RefFrameItems::Pos_Vel_Att_Rate, X_iframe_to_pframe, structure);
333  }
334 
335  // Mark the state as updated.
336  initialized_states.set(RefFrameItems::Pos_Vel_Att_Rate);
337 
338  // Propagate the integrated state to other state descriptions.
339  propagate_state();
340 
341  return status;
342 }

361 /**
362  * Integrate the rotational state of a DynBody.
363  * @param[in] dyn_dt Dynamic time step, in dynamic time seconds.
364  * @param[in] target_stage The stage of the integration process
365  * that the integrator should try to attain.
366  * @return The status (time advance, pass/fail status) of the integration.
367  */
368 er7_utils::IntegratorResult DynBody::rot_integ(double dyn_dt, unsigned int target_stage)
369 {
370  RefFrameRot & rot_state = composite_body.state.rot;
371 
372  // Integrate the rotational state.
373  er7_utils::IntegratorResult status = rot_integrator.integrate(dyn_dt,
374  target_stage,
375  derivs.rot_accel,
376  rot_state.ang_vel_this,
377  rot_state.Q_parent_this);
378 
379  // Normalize the integrated quaternion.
380  rot_state.Q_parent_this.normalize_integ();
381 
382  // Compute the corresponding transformation matrix.
383  rot_state.Q_parent_this.left_quat_to_transformation(rot_state.T_parent_this);
384 
385  // Compute the quaternion derivative.
386  rot_state.Q_parent_this.compute_left_quat_deriv(rot_state.ang_vel_this, derivs.Qdot_parent_this);
387 
388  // Compute the angular velocity magnitude and unit vector.
389  rot_state.ang_vel_mag = Vector3::vmag(rot_state.ang_vel_this);
390  if(std::fpclassify(rot_state.ang_vel_mag) != FP_ZERO)
391  {
392  Vector3::scale(rot_state.ang_vel_this, 1.0 / rot_state.ang_vel_mag, rot_state.ang_vel_unit);
393  }
394  else
395  {
396  Vector3::initialize(rot_state.ang_vel_unit);
397  }
398 
399  return status;
400 }
