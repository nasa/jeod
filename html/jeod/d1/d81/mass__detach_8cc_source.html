49 /**
50  * Detach the two bodies, 'this' and the argument body, from each other such
51  * such that the detachment occurs between the superior body and the immediate
52  * child along the path from the inferior body to the superior body.
53  * Returns true to indicate success, false to indicate failure.
54  *
55  * \par Assumptions and Limitations
56  * - Instances of more derived classes, with presumably more involved dynamics,
57  * are situated higher in the mass tree than are more basic instances.
58  * For example, a simple MassBody can be a child of a DynBody, but not the
59  * other way around.
60  * - The attachment in the mass tree between the immediate child and the superior
61  * body is assumed to reflect a real physical attachment.
62  * @return Success flag
63  * \param[in,out] mass_body The other body
64  */
65 bool
66 MassBody::detach (
67  MassBody & mass_body)
68 {
69  MassBody * detaching_body = nullptr;
70  MassBody * parent_body = nullptr;
71 
72 
73  // Find the body that is to be detached from its immediate parent.
74  // First search from 'this' up for the input mass_body, and if that fails
75  // search from the input mass_body up for 'this'. Both searches failing is
76  // an error.
77 
78 
79  for (auto* link : TreeLinksAscendRange<MassBodyLinks>(links)) {
80  MassBody* iter_body = &(link->container());
81  if (iter_body == &mass_body) {
82  parent_body = &mass_body;
83  break;
84  }
85  detaching_body = iter_body;
86  }
87 
88  // parent_body is null => First search failed. Do the second search.
89  if (parent_body == nullptr) {
90  for (auto* link : TreeLinksAscendRange<MassBodyLinks>(mass_body.links)) {
91  MassBody* iter_body = &(link->container());
92  if (iter_body == this) {
93  parent_body = this;
94  break;
95  }
96  detaching_body = iter_body;
97  }
98  }
99 
100  // parent_body is still null => both searches failed.
101  // There is no way to perform the requested detachment.
102  // This is a non-fatal error.
103  if (parent_body == nullptr) {
104  MessageHandler::inform (
105  __FILE__, __LINE__, MassBodyMessages::invalid_detach,
106  "\nWarning: Unable to find detach point between\n"
107  "bodies '%s' and '%s'. No detachment was performed.",
108  name.c_str(), mass_body.name.c_str());
109 
110  return false;
111  }
112 
113 
114  // Detach the detaching body from its immediate parent, returning the
115  // status from that method as the status from this method.
116  return (detaching_body->detach ());
117 }

121  * Detach a mass body from its immediate parent.
122  * Returns true to indicate success, false to indicate failure.
123  *
124  * \par Assumptions and Limitations
125  * - Instances of more derived classes, with presumably more involved dynamics,
126  * are situated higher in the mass tree than are more basic instances.
127  * For example, a simple MassBody can be a child of a DynBody, but not the
128  * other way around.
129  * - The attachment in the mass tree between the immediate child and the superior
130  * body is assumed to reflect a real physical attachment.
131  * @return Success flag
132  */
133 bool
134 MassBody::detach (
135  void)
136 {
137  MassBody * parent = links.parent();
138  bool success;
139 
140  // If parent is a DynBody, call appropriate detach call
141  DynBody * parent_dyn_body = parent->dyn_owner;
142  if(parent_dyn_body != nullptr)
143  {
144  success = parent_dyn_body->remove_mass_body(*this);
145  } else {
146 
147  // Perform the detachment, but only if it is valid to do so.
148  if (detach_validate (parent, true)) {
149 
150  // Tell the child (this) to break the logical links between the two.
151  detach_sever_links (*parent);
152 
153  // Tell the parent to update properties/state/...
154  parent->detach_update_properties (*this);
155 
156  // Note that the detachment was performed.
157  success = true;
158  }
159 
160  // Validation failed. Note that the detachment was not performed.
161  else {
162  success = false;
163  }
164  }
165 
166  return success;
167 }

316 /**
317  * Update parent and child properties to reflect that they are detached.
318  *
319  * Extensibility comments --
320  * - This method is sent to the parent body of the detachment after
321  * the child body has severed the logical connectivity between
322  * the parent body and child body.
323  * - The generic purpose of this method is to update any physical
324  * properties that change as a result of the detachment.
325  * - Any class that overrides this method must either invoke this method
326  * or perform the actions performed herein.
327  *
328  * \par Assumptions and Limitations
329  * - The detachment is valid and logical connectivity has been severed.
330  * Neither assumption is checked.
331  * \param[in,out] child The child body; the body newly detached from this body.
332  */
333 void
334 MassBody::detach_update_properties (
335  MassBody & child)
336 {
337 
338  // Update the mass properties from the root down to this body.
339  set_update_flag ();
340  get_root_body_internal()->update_mass_properties ();
341 
342  // Re-initialize the child's auxiliarly attachment info.
343  child.structure_point.initialize_mass_point ();
344  child.composite_wrt_pstr.initialize_mass_point ();
345  child.composite_wrt_pbdy.initialize_mass_point ();
346 
347  // Root bodies need a valid inverse inertia tensor.
348  // Child bodies don't, so calculate the child's inverse inertia tensor.
349  if (child.compute_inverse_inertia &&
350  (child.composite_properties.mass > 0.0)) {
351  Matrix3x3::invert_symmetric(child.composite_properties.inertia,
352  child.composite_properties.inverse_inertia);
353  } else {
354  Matrix3x3::initialize (child.composite_properties.inverse_inertia);
355  }
356 
357  return;
358 }
