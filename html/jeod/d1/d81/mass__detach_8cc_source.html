47 /**
48  * Detach the two bodies, 'this' and the argument body, from each other such
49  * such that the detachment occurs between the superior body and the immediate
50  * child along the path from the inferior body to the superior body.
51  * Returns true to indicate success, false to indicate failure.
52  *
53  * \par Assumptions and Limitations
54  * - Instances of more derived classes, with presumably more involved dynamics,
55  * are situated higher in the mass tree than are more basic instances.
56  * For example, a simple MassBody can be a child of a DynBody, but not the
57  * other way around.
58  * - The attachment in the mass tree between the immediate child and the superior
59  * body is assumed to reflect a real physical attachment.
60  * @return Success flag
61  * \param[in,out] mass_body The other body
62  */
63 bool MassBody::detach(MassBody & mass_body)
64 {
65  MassBody * detaching_body = nullptr;
66  MassBody * parent_body = nullptr;
67 
68  // Find the body that is to be detached from its immediate parent.
69  // First search from 'this' up for the input mass_body, and if that fails
70  // search from the input mass_body up for 'this'. Both searches failing is
71  // an error.
72 
73  for(auto * link : TreeLinksAscendRange<MassBodyLinks>(links))
74  {
75  MassBody * iter_body = &(link->container());
76  if(iter_body == &mass_body)
77  {
78  parent_body = &mass_body;
79  break;
80  }
81  detaching_body = iter_body;
82  }
83 
84  // parent_body is null => First search failed. Do the second search.
85  if(parent_body == nullptr)
86  {
87  for(auto * link : TreeLinksAscendRange<MassBodyLinks>(mass_body.links))
88  {
89  MassBody * iter_body = &(link->container());
90  if(iter_body == this)
91  {
92  parent_body = this;
93  break;
94  }
95  detaching_body = iter_body;
96  }
97  }
98 
99  // parent_body is still null => both searches failed.
100  // There is no way to perform the requested detachment.
101  // This is a non-fatal error.
102  if(parent_body == nullptr || detaching_body == nullptr)
103  {
104  MessageHandler::inform(__FILE__,
105  __LINE__,
106  MassBodyMessages::invalid_detach,
107  "\nWarning: Unable to find detach point between\n"
108  "bodies '%s' and '%s'. No detachment was performed.",
109  name.c_str(),
110  mass_body.name.c_str());
111 
112  return false;
113  }
114 
115  // Detach the detaching body from its immediate parent, returning the
116  // status from that method as the status from this method.
117  return (detaching_body->detach());
118 }

120 /**
121  * Detach a mass body from its immediate parent.
122  * Returns true to indicate success, false to indicate failure.
123  *
124  * \par Assumptions and Limitations
125  * - Instances of more derived classes, with presumably more involved dynamics,
126  * are situated higher in the mass tree than are more basic instances.
127  * For example, a simple MassBody can be a child of a DynBody, but not the
128  * other way around.
129  * - The attachment in the mass tree between the immediate child and the superior
130  * body is assumed to reflect a real physical attachment.
131  * @return Success flag
132  */
133 bool MassBody::detach()
134 {
135  MassBody * parent = links.parent();
136  bool success;
137 
138  // If parent is a DynBody, call appropriate detach call
139  DynBody * parent_dyn_body = parent->dyn_owner;
140  if(parent_dyn_body != nullptr)
141  {
142  success = parent_dyn_body->remove_mass_body(*this);
143  }
144  else
145  {
146  // Perform the detachment, but only if it is valid to do so.
147  if(detach_validate(parent, true))
148  {
149  // Tell the child (this) to break the logical links between the two.
150  detach_sever_links(*parent);
151 
152  // Tell the parent to update properties/state/...
153  parent->detach_update_properties(*this);
154 
155  // Note that the detachment was performed.
156  success = true;
157  }
158 
159  // Validation failed. Note that the detachment was not performed.
160  else
161  {
162  success = false;
163  }
164  }
165 
166  return success;
167 }

169 /**
170  * Validate whether the pending detachment is legal.
171  *
172  * Extensibility comments --
173  * - Overriding this method doesn't make a whole lot of sense.
174  * - Any class that overrides this method must either invoke this
175  * method
176  * or perform the actions performed herein.
177  *
178  * \par Assumptions and Limitations
179  * - The subject body, this, must be a root body. This is not checked.
180  * @return Validity indicator
181  * \param[in] parent The parent body; the body from which this body is to be detached.
182  * \param[in] generate_message Generate message if invalid?
183  */
184 bool MassBody::detach_validate(const MassBody * parent, bool generate_message) const
185 {
186  bool is_valid;
187 
188  // Validate that the child can detach from the parent in terms of
189  // - Connectivity (detach_validate_parent, sent to child)
190  // - Whatever (detach_validate_child, sent to parent).
191  is_valid = (detach_validate_parent(parent, generate_message) &&
192  parent->detach_validate_child(*this, generate_message));
193 
194  return is_valid;
195 }

210 bool MassBody::detach_validate_parent(const MassBody * parent, bool generate_message) const
211 {
212  bool is_valid = true;
213 
214  // From a MassBody perspective, the only problem with detachment is
215  // when a root body attempts to detach.
216  is_valid = (parent != nullptr);
217  if(parent == nullptr)
218  {
219  is_valid = false;
220  if(generate_message)
221  {
222  MessageHandler::inform(__FILE__,
223  __LINE__,
224  MassBodyMessages::invalid_detach,
225  "\nWarning: Body '%s' is a root body and cannot be detached.",
226  name.c_str());
227  }
228  }
229  else if(parent->dyn_manager != nullptr && parent->dyn_manager->find_dyn_body(parent->name.get_name()) != nullptr)
230  {
231  is_valid = false;
232  if(generate_message)
233  {
234  MessageHandler::warn(__FILE__,
235  __LINE__,
236  MassBodyMessages::invalid_detach,
237  "\nWarning: Body '%s' refers to a DynBody and cannot be detached "
238  "through MassBody detach methods.",
239  parent->name.c_str());
240  }
241  }
242 
243  return is_valid;
244 }

298 /**
299  * Update parent and child properties to reflect that they are detached.
300  *
301  * Extensibility comments --
302  * - This method is sent to the parent body of the detachment after
303  * the child body has severed the logical connectivity between
304  * the parent body and child body.
305  * - The generic purpose of this method is to update any physical
306  * properties that change as a result of the detachment.
307  * - Any class that overrides this method must either invoke this method
308  * or perform the actions performed herein.
309  *
310  * \par Assumptions and Limitations
311  * - The detachment is valid and logical connectivity has been severed.
312  * Neither assumption is checked.
313  * \param[in,out] child The child body; the body newly detached from this body.
314  */
315 void MassBody::detach_update_properties(MassBody & child)
316 {
317  // Update the mass properties from the root down to this body.
318  set_update_flag();
319  get_root_body_internal()->update_mass_properties();
320 
321  // Re-initialize the child's auxiliarly attachment info.
322  child.structure_point.initialize_mass_point();
323  child.composite_wrt_pstr.initialize_mass_point();
324  child.composite_wrt_pbdy.initialize_mass_point();
325 
326  // Root bodies need a valid inverse inertia tensor.
327  // Child bodies don't, so calculate the child's inverse inertia tensor.
328  if(child.compute_inverse_inertia && (child.composite_properties.mass > 0.0))
329  {
330  Matrix3x3::invert_symmetric(child.composite_properties.inertia, child.composite_properties.inverse_inertia);
331  }
332  else
333  {
334  Matrix3x3::initialize(child.composite_properties.inverse_inertia);
335  }
336 }
