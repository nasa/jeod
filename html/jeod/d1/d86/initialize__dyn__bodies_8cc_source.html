56 {
57  // Initialize mass body mass properties and
58  // perform initialization-time attachments.
59  perform_mass_body_initializations(nullptr);
60  perform_mass_attach_initializations();
61 
62  // Update the body mass trees.
63  for(std::vector<DynBody *>::const_iterator it = dyn_bodies.begin(); it != dyn_bodies.end(); ++it)
64  {
65  DynBody * body = *it;
66  if(body->is_root_body())
67  {
68  body->mass.compute_inverse_inertia = true;
69  body->mass.update_mass_properties();
70  }
71  }
72

75  perform_dyn_body_initializations(nullptr);

111 /**
112  * Initialize all queued body actions that derive from MassBodyInit
113  * and apply those that are immediately ready to be applied.
114  * \param[in,out] body Body to be initialized
115  */
116 void DynManager::perform_mass_body_initializations(MassBody * body)
117 {
118  // Initialize and then apply body actions that derive from MassBodyInit.
119  // Note that the increment is in (and must be in) the loop body as
120  // actions are to be deleted from the queue once applied.
121  for(auto it = body_actions.begin(); it != body_actions.end();
122  /* No increment -- increment is in loop body */)
123  {
124  BodyAction * action = *it;
125 
126  // Only process instances of MassBodyInit and its derived classes.
127  // If MassBody is NULL (default), then process all ready actions.
128  // If MassBody is specified, process only the ready associated actions.
129  if((dynamic_cast<MassBodyInit *>(action) != nullptr) &&
130  ((body == nullptr) || (action->is_same_subject_body(*body))))
131  {
132  // Initialize the action.
133  action->initialize(*this);
134 
135  // Perform the action if it is ready.
136  if(action->is_ready())
137  {
138  // Apply the action ...
139  action->apply(*this);
140 
141  // ... and delete it from the queue.
142  body_actions.erase(it++);
143  }
144 
145  // Attachment is not ready: Skip over it.
146  // A MassBodyInit action might not be ready because
147  // - The user has set the active flag to false.
148  // For example, the subject MassBody object might not be used
149  // in this particular simulation run.
150  // - The action is a user-defined extension to MassBodyInit.
151  // That the action is not ready is not our problem.
152  // Whatever the reason, the response here is to pass over the action.
153  else
154  {
155  ++it;
156  }
157  }
158 
159  // Simply skip over actions that do not derive from MassBodyInit.
160  else
161  {
162  ++it;
163  }
164  }
165 }

216 /**
217  * Initialize dynamic bodies.
218  * \param[in,out] body Body to be initialized
219  */
220 void DynManager::perform_dyn_body_initializations(DynBody * body)
221 {
222  // Initialize the queued actions that derive from DynBodyInit.
223 
224  for(auto action : body_actions)
225  {
226  // Initialize the action if it derives from DynBodyInit.
227  if(dynamic_cast<DynBodyInit *>(action) != nullptr)
228  {
229  // If body is NULL (default), then initialize any action.
230  // If body is specified, initialize only the associated action
231  if((body == nullptr) || (action->is_same_subject_body(body->mass)))
232  {
233  // Initialize the action.
234  action->initialize(*this);
235  }
236  }
237  }
238 
239  // If body is NULL (default):
240  // Perform remaining DynBodyInit actions.
241  // Keep trying to perform pending actions of type DynBodyInit until
242  // the queue is drained of all such initializations (nominal completion) or
243  // the unprocessed actions of this type can never be performed (abort).
244  bool more_inits_to_do = true; // Set true to enter loop
245  bool any_inits_performed = true; // Set true to enter loop
246  while(more_inits_to_do && any_inits_performed)
247  {
248  more_inits_to_do = any_inits_performed = false;
249 
250  for(auto it = body_actions.begin(); it != body_actions.end();
251  /* No increment -- increment is in loop body */)
252  {
253  BodyAction * action = *it;
254 
255  // Only process instances of DynBodyInit and its derived classes.
256  if((dynamic_cast<DynBodyInit *>(action) != nullptr) &&
257  ((body == nullptr) || (action->is_same_subject_body(body->mass))))
258  {
259  // Perform the action if the initializer is ready.
260  // Note: Body initializations that are not ready will presumably
261  // be performed on some subsequent pass through this double loop.
262  if(action->is_ready())
263  {
264  // Note that an initialization was performed, which will force
265  // another pass through the outer loop if some initializations
266  // remain unprocessed after completing the inner loop.
267  any_inits_performed = true;
268 
269  // Apply the action ...
270  action->apply(*this);
271 
272  // ... and delete it from the queue.
273  // Note that a side effect (via it++) is to advance the iterator.
274  body_actions.erase(it++);
275 
276  // Update the ephemerides if the above did something weird.
277  if((body == nullptr) && regenerate_ref_frame_tree)
278  {
279  update_ephemerides();
280  }
281  }
282 
283  // Initializer is *not* ready:
284  // Skip over this initializer for this pass.
285  else
286  {
287  // Note that some initializations remain unprocessed.
288  // This initializer will be applied on a subsequent pass if other
289  // initializers still in the queue set the requisite knowledge
290  // needed by this initializer.
291  more_inits_to_do = true;
292 
293  // Advance to the next action.
294  ++it;
295  }
296  }
297 
298  // Action is not a DynBodyInit. Skip over it.
299  else
300  {
301  ++it;
302  }
303  }
304  }
305 
306  // Active initializations that could not be performed are a non-fatal error.
307  if(more_inits_to_do)
308  {
309  more_inits_to_do = false;
310 
311  // Report the initializers that didn't run.
312  for(auto action : body_actions)
313  {
314  auto * body_init = dynamic_cast<DynBodyInit *>(action);
315 
316  // Action is a DynBodyInit: Report failure to initialize.
317  if((body_init != nullptr) && body_init->active)
318  {
319  body_init->report_failure();
320  more_inits_to_do = true;
321  }
322  }
323  }
324 
325  if(more_inits_to_do)
326  {
327  MessageHandler::error(__FILE__,
328  __LINE__,
329  DynManagerMessages::inconsistent_setup,
330  "Unable to perform all active initializations.");
331  }
332 }

334 /**
335  * Ensure that all of the required states have been set.
336  */
337 void DynManager::check_for_uninitialized_states()
338 {
339  bool uninitialized_bodies;
340 
341  // Check that all bodies have been initialized.
342  uninitialized_bodies = false;
343  for(std::vector<DynBody *>::const_iterator it = dyn_bodies.begin(); it != dyn_bodies.end(); ++it)
344  {
345  DynBody * body = *it;
346  RefFrameItems missing_states;
347 
348  // First denote all of the required states for this body as missing.
349  // Which states are required depends on which aspects of the
350  // state are to be propagated.
351  if(body->translational_dynamics && body->rotational_dynamics)
352  {
353  missing_states.set(RefFrameItems::Pos_Vel_Att_Rate);
354  }
355  else if(body->translational_dynamics)
356  {
357  missing_states.set(RefFrameItems::Pos_Vel);
358  }
359  else if(body->rotational_dynamics)
360  {
361  missing_states.set(RefFrameItems::Att_Rate);
362  }
363 
364  // Now mark the items that have been set as present.
365  missing_states.remove(body->get_initialized_states().get());
366 
367  // Any missing items are problematic.
368  if(!missing_states.is_empty())
369  {
370  uninitialized_bodies = true;
371  MessageHandler::error(__FILE__,
372  __LINE__,
373  DynManagerMessages::inconsistent_setup,
374  "Body '%s' %s not initialized",
375  body->name.c_str(),
376  missing_states.to_string().c_str());
377  }
378  }
379 
380  // That required states were not initialized is a fatal error.
381  if(uninitialized_bodies)
382  {
383  MessageHandler::fail(__FILE__, __LINE__, DynManagerMessages::inconsistent_setup, "Uninitialized states");
384  }
385 }
