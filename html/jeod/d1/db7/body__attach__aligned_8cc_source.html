83 /**
84  * Initialize the core mass properties of the subject MassBody.
85  * \param[in,out] dyn_manager Jeod manager
86  */
87 void BodyAttachAligned::apply(DynManager & dyn_manager)
88 {
89  if(dyn_parent != nullptr)
90  {
91  if(dyn_subject != nullptr) // DynBody to DynBody
92  {
93  succeeded = dyn_parent->attach_child(parent_point_name, subject_point_name, *dyn_subject);
94  }
95  else // MassBody subbody/subassembly to DynBody
96  {
97  succeeded = dyn_parent->add_mass_body(parent_point_name, subject_point_name, *mass_subject);
98  }
99  }
100  else if(mass_parent != nullptr)
101  {
102  if(dyn_subject != nullptr) // DynBody to MassBody -- ILLEGAL
103  {
104  succeeded = false;
105  }
106  else // MassBody/MassBody assembly to MassBody
107  {
108  succeeded = mass_subject->attach_to(subject_point_name, parent_point_name, *mass_parent);
109  }
110  }
111  else if(ref_parent != nullptr)
112  {
113  if(dyn_subject != nullptr) // DynBody to RefFrame
114  {
115  // Construct the affine transformation from the attach point on the child
116  // body to the parent ref frame.
117  // Per JEOD convention, this is always defined as a zero offset with a
118  // corresponding 180 degree yaw orientation.
119  // Thus, the point is initialized with zero offset and identity transform.
120  double vec[3], mat[3][3];
121  Vector3::initialize(vec);
122  Matrix3x3::identity(mat);
123  mat[0][0] = -1.0;
124  mat[1][1] = -1.0;
125  succeeded = dyn_subject->attach_to_frame(subject_point_name, parent_point_name, vec, mat);
126  }
127  else // MassBody to RefFrame -- ILLEGAL
128  {
129  succeeded = false;
130  }
131  }
132  else // Should be unreachable
133  {
134  succeeded = false;
135  }
136 
137  // Forward the action up the chain. The parent class deals with status.
138  BodyAttach::apply(dyn_manager);
139 }
