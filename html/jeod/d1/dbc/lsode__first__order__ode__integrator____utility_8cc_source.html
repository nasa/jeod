128 /**
129  * returns the RMS value of {V dot W}, where V and W are N-vectors.
130  *
131  * Modified version of DVNORM
132  *
133  * The only places DVNORM is used, it is multiplying some array by the
134  * error_weight array (arrays.error_weight) across control_data.num_odes terms.
135  * These values are fixed for our application, and do not need to be passed in.
136  *
137  * We provide two implementations, one for a one-dimensional array, and one for
138  * a two-dimensional array in which the first index is the variable.
139  * \param v array
140  */
141 double LsodeFirstOrderODEIntegrator::magnitude_of_weighted_array(const double * v)
142 {
143  double sum = 0.0;
144  for(unsigned int ii = 0; ii < control_data.num_odes; ii++)
145  {
146  double mult = v[ii] * arrays.error_weight[ii];
147  sum += (mult * mult);
148  }
149  return sqrt(sum / control_data.num_odes);
150 }

170 /**
171  * Factors a double array (arrays.lin_alg) by Gaussian elimination.
172  *
173  * Modified version of DGEFA.
174  */
175 int LsodeFirstOrderODEIntegrator::gauss_elim_factor()
176 {
177  // DGEFA was only called once. While it had two indices passed in, they were
178  // both the same (N = control_data.num_odes). The array passed in was
179  // WM(3) = arrays.lin_alg.
180  // Also, IWM(21)=pivots was passed in, and IER passed out.
181  // num_odes, arrays.lin_alg, and pivots are all class variables so do not
182  // need to be passed. The return value (IER) is not reproduced, but is
183  // passed back by the return statement instead.
184 
185  int info = 0;
186  if(control_data.num_odes >= 2) // inverse go to 70;
187  {
188  for(unsigned int k = 0; k < control_data.num_odes - 1; k++) // do 60
189  {
190  // find l = pivot index;
191  // Original had IDAMAX + K -1.
192  // If nth tested element is max, IDAMAX returns n, while
193  // index_of_max_magnitude returns index = (n-1)+k
194  unsigned int l = index_of_max_magnitude(control_data.num_odes - k, arrays.lin_alg, k);
195  arrays.pivots[k] = l;
196  // zero pivot implies this column already triangularized
197  if(std::fpclassify(arrays.lin_alg[l][k]) != FP_ZERO) // inverse go to 40;
198  {
199  // interchange if necessary
200  if(l != k) // inverse go to 10;
201  {
202  double temp_val = arrays.lin_alg[l][k];
203  arrays.lin_alg[l][k] = arrays.lin_alg[k][k];
204  arrays.lin_alg[k][k] = temp_val;
205  // 10
206  }
207  // compute multipliers
208  double t = -1.0 / arrays.lin_alg[k][k];
209  for(unsigned int ii = 0; ii < control_data.num_odes - k - 1; ii++)
210  {
211  arrays.lin_alg[k + 1 + ii][k] *= t;
212  }
213  // row elimination with column indexing
214  for(unsigned int j = k + 1; j < control_data.num_odes; j++)
215  {
216  // do 30
217  t = arrays.lin_alg[l][j];
218  if(l != k) // inverse go to 20;
219  {
220  arrays.lin_alg[l][j] = arrays.lin_alg[k][j];
221  arrays.lin_alg[k][j] = t;
222  // 20
223  }
224  for(unsigned int ii = 0; ii < control_data.num_odes - k - 1; ii++)
225  {
226  arrays.lin_alg[k + 1 + ii][j] += (t * arrays.lin_alg[k + 1 + ii][k]);
227  }
228  // 30
229  }
230  // go to 50 using if-else structure.
231  // 40
232  }
233  else
234  {
235  info = k;
236  // 50
237  }
238  // 60
239  }
240  // 70
241  }
242 
243  arrays.pivots[control_data.num_odes - 1] = control_data.num_odes - 1;
244  if(std::fpclassify(arrays.lin_alg[control_data.num_odes - 1][control_data.num_odes - 1]) == FP_ZERO)
245  {
246  info = control_data.num_odes - 1;
247  }
248  return info;
249 }

252  * Solves the equation Y' = A Y, with A = arrays.lin_alg.
253  */
254 void LsodeFirstOrderODEIntegrator::linear_solver()
255 {
256  double t;
257  // job = 0 , solve a * x = b;
258  // first solve l*y = b;
259  if(control_data.num_odes > 1) // inverse go to 30;
260  {
261  for(unsigned int k = 0; k < control_data.num_odes - 1; k++) // do 20
262  {
263  unsigned int l = arrays.pivots[k];
264  t = y[l];
265  if(l != k) // inverse go to 10;
266  {
267  y[l] = y[k];
268  y[k] = t;
269  // 10
270  }
271  for(unsigned int ii = 0; ii < control_data.num_odes - k - 1; ii++)
272  {
273  y[k + 1 + ii] += (t * arrays.lin_alg[k + 1 + ii][k]);
274  }
275  // 20
276  }
277  // 30
278  }
279  // now solve u*x = y;
280  for(int k = control_data.num_odes - 1; k >= 0; k--) // do 40
281  {
282  y[k] /= arrays.lin_alg[k][k];
283  t = -y[k];
284  for(int ii = 0; ii < k; ii++)
285  {
286  y[ii] += (t * arrays.lin_alg[ii][k]);
287  }
288  // 40
289  }
290  // go to 100 using if-else structure
291  // 50
292 }

338 unsigned int LsodeFirstOrderODEIntegrator::index_of_max_magnitude(unsigned int num_points,
339  double ** array,
340  int start_ix)
341 {
342  unsigned int idamax = start_ix;
343  double max_value = std::abs(array[start_ix][start_ix]);
344 
345  for(unsigned int ii = 1; ii < num_points; ii++)
346  {
347  double test_value = std::abs(array[start_ix + ii][start_ix]);
348  if(test_value > max_value)
349  {
350  max_value = test_value;
351  idamax = ii;
352  }
353  }
354  return idamax;
355 }
