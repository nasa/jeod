73 /**
74  * Specifies the orientiation of one reference frame with respect to another.
75  * There are many competing charts on the rotation group.
76  * This class provides means for representing rotations as Euler rotations,
77  * transformation matrices, left transformation quaternions, and eigen
78  * rotations. The class also provides mechanisms for converting these
79  * representations into the alternative representations.
80  */
81 class Orientation

106  // Sequences named by axes
