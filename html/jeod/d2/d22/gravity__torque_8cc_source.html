90 GravityTorque::update (
91  void)

93 
94  if (active == false) {
95  torque[0] = 0.0;
96  torque[1] = 0.0;
97  torque[2] = 0.0;
98  return;
99  }
100 
101  if (subject_body == nullptr) {
102  MessageHandler::fail (
103  __FILE__, __LINE__, GravityTorqueMessages::initialization_error,
104  "GravityTorque object was not properly initialized.");
105  return;
106  }
107 
108  double g[3][3];
109  double inertia[3][3];
110 
111  // inertia tensor (in body frame)
112  Matrix3x3::copy (subject_body->mass.composite_properties.inertia, inertia);
113 
114  // g = transform gradient matrix from inertial to body
115  Matrix3x3::transform_matrix (
116  subject_body->composite_body.state.rot.T_parent_this,
117  subject_body->grav_interaction.grav_grad, g);
118 
119  // torque in body frame
120  torque[0] = g[1][2] * (inertia[2][2] - inertia[1][1]) -
121  g[0][2] * inertia[0][1] +
122  g[0][1] * inertia[0][2] -
123  inertia[1][2] * (g[2][2] - g[1][1]);
124 
125  torque[1] = g[0][2] * (inertia[0][0] - inertia[2][2]) +
126  g[1][2] * inertia[0][1] -
127  g[0][1] * inertia[1][2] -
128  inertia[0][2] * (g[0][0] - g[2][2]);
129 
130  torque[2] = g[0][1] * (inertia[1][1] - inertia[0][0]) -
131  g[1][2] * inertia[0][2] +
132  g[0][2] * inertia[1][2] -
133  inertia[0][1] * (g[1][1] - g[0][0]);
134 
135  // Transform to structure.
136  Vector3::transform_transpose (
137  subject_body->mass.composite_properties.T_parent_this, torque);
138 
139  return;
