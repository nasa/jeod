45 /**
46  * Contains data needed to construct a transformation matrix given a sequence
47  * of Euler angles and to extract a sequence of Euler angles from a matrix.
48  * See Orientation::compute_euler_angles_from_matrix for details.
49  */
50 struct EulerInfo
51 {
52  /**
53  * The axes about which the rotations are performed in the order in which
54  * the rotations are performed, with X=0, Y=1, Z=2. For example, an XYZ or
55  * roll pitch yaw sequence is {0,1,2} while a ZXZ sequence is {2,0,2}.
56  */
57  unsigned int indices[3]; //!< trick_units(--)
58 
59  /**
60  * The initial element of the sequence for aerodynamics sequences,
61  * but the index of the omitted axis for astronomical sequences.
62  * For example, the omitted axis in a ZXZ sequence is Y=1.
63  */
64  unsigned int alternate_x; //!< trick_units(--)
65 
66  /**
67  * The final element of the sequence for aerodynamics sequences,
68  * but the index of the omitted axis for astronomical sequences.
69  */
70  unsigned int alternate_z; //!< trick_units(--)
71 
72  /**
73  * Indicates whether the 3-axis rotation sequence generated by replacing the
74  * final element of the sequence with the one axis not specified by the
75  * first two elements of the sequence is an even (true) permutation or odd
76  * permutation (false) of XYZ.
77  * The alternative 3-axis sequence is identical to the original sequence in
78  * the case of aerodynamics sequences. The astronomical ZXZ sequence becomes
79  * ZXY via this replacement rule. Since ZXY is an even permutation of XYZ,
80  * the is_even_permutation member for a ZXZ sequence is true.
81  */
82  bool is_even_permutation; //!< trick_units(--)
83 
84  /**
85  * True if the sequence is an aerodynamics sequence such as XYZ;
86  * false for an astronomical sequence such as ZXZ.
87  */
88  bool is_aerodynamics_sequence; //!< trick_units(--)
89 };

112 /**
113  * Compute the left transformation quaternion from the Euler sequence.
114  * The quaternion is formed by generating a sequence of three simple
115  * quaternions corresponding to the three rotations. The composite
116  * quaternion is the reverse-order product of these three simple quaternions.
117  * \param[in] euler_sequence Euler sequence
118  * \param[in] euler_angles Euler angles\n Units: r
119  * \param[out] quat Resultant quaternion
120  */
121 void Orientation::compute_quaternion_from_euler_angles(EulerSequence euler_sequence,
122  const double euler_angles[3],
123  Quaternion & quat)
124 {
125  // Validate the value of the euler_sequence member.
126  // Note that an invalid value means the object is left unchanged.
127  if((euler_sequence < EulerXYZ) || (euler_sequence > EulerZYZ))
128  {
129  MessageHandler::error(__FILE__,
130  __LINE__,
131  OrientationMessages::invalid_enum,
132  "The euler_sequence data member has not been set or is invalid; "
133  "value=%d",
134  static_cast<int>(euler_sequence));
135  return;
136  }
137 
138  const unsigned int * axes = Euler_info[euler_sequence].indices;
139  Quaternion q[3], q21;
140 
141  for(int ii = 0; ii < 3; ii++)
142  {
143  double htheta = 0.5 * euler_angles[ii];
144  double cosht = std::cos(htheta);
145  double sinht = std::sin(htheta);
146  q[ii].scalar = cosht;
147  q[ii].vector[axes[ii]] = -sinht;
148  }
149 
150  q[2].multiply(q[1], q21);
151  q21.multiply(q[0], quat);
152  quat.normalize();
153 }

155 /**
156  * Compute the transformation matrix from the Euler sequence.
157  * The matrix is formed by generating a sequence of three simple transformation
158  * matrices corresponding to the three rotations. The composite transformation
159  * matrix is the reverse-order product of these three simple matrices.
160  * \param[in] euler_sequence Euler sequence
161  * \param[in] euler_angles Euler angles\n Units: r
162  * \param[out] trans Resultant transformation matrix
163  */
164 void Orientation::compute_matrix_from_euler_angles(EulerSequence euler_sequence,
165  const double euler_angles[3],
166  double trans[3][3])
167 {
168  // Validate the value of the euler_sequence member.
169  // Note that an invalid value means the object is left unchanged.
170  if((euler_sequence < EulerXYZ) || (euler_sequence > EulerZYZ))
171  {
172  MessageHandler::error(__FILE__,
173  __LINE__,
174  OrientationMessages::invalid_enum,
175  "The euler_sequence data member has not been set or is invalid; "
176  "value=%d",
177  static_cast<int>(euler_sequence));
178  return;
179  }
180  const unsigned int * axes = Euler_info[euler_sequence].indices;
181  double m[3][3][3], m21[3][3];
182 
183  for(int ii = 0; ii < 3; ii++)
184  {
185  Matrix3x3::initialize(m[ii]);
186  double sin_theta = std::sin(euler_angles[ii]);
187  double cos_theta = std::cos(euler_angles[ii]);
188  switch(axes[ii])
189  {
190  case 0:
191  m[ii][0][0] = 1.0;
192  m[ii][1][1] = cos_theta;
193  m[ii][1][2] = sin_theta;
194  m[ii][2][1] = -sin_theta;
195  m[ii][2][2] = cos_theta;
196  break;
197  case 1:
198  m[ii][1][1] = 1.0;
199  m[ii][0][0] = cos_theta;
200  m[ii][0][2] = -sin_theta;
201  m[ii][2][0] = sin_theta;
202  m[ii][2][2] = cos_theta;
203  break;
204  case 2:
205  m[ii][2][2] = 1.0;
206  m[ii][0][0] = cos_theta;
207  m[ii][0][1] = sin_theta;
208  m[ii][1][0] = -sin_theta;
209  m[ii][1][1] = cos_theta;
210  break;
211  default:
212  break;
213  }
214  }
215 
216  Matrix3x3::product(m[2], m[1], m21);
217  Matrix3x3::product(m21, m[0], trans);
218 }

220 /**
221  * Extract an Euler sequence from the transformation matrix.
222  * A transformation matrix constructed from an XYZ Euler sequence
223  * is of the form @f[
224  * \left[\array{ccc}
225  * \cos\psi\cos\theta & \cdots & \cdots \\
226  * -\sin\psi\cos\theta & \cdots & \cdots \\
227  * \sin\theta & -\cos\theta\sin\phi & \cos\theta\cos\phi
228  * \endarray\right]
229  * @f]
230  * Note that the [2][0] element of the matrix depends on theta only.
231  * The other two elements of the leftmost column are simple terms that depend on
232  * theta and psi only, and the other two elements of the bottommost row are
233  * simple terms that depend on theta and phi only.
234  * Those five elements are the key to extracting an XYZ Euler sequence from a
235  * transformation matrix.
236  * The same principle applies to all twelve of the Euler sequences:
237  * Five key elements contain all of the information needed to extract the
238  * desired sequence. The location and form of those key elements of course
239  * depends on the sequence.
240  *
241  * A problem arises in the above when cos(theta) is zero, or nearly so. This
242  * siutation is called 'gimbal lock'. Those four elements used to determine phi
243  * and psi are zero or nearly so. Fortunately That ugly stuff isn't so ugly in
244  * the case of gimbal lock. Once again looking at the matrix generated from an
245  * XYZ Euler sequence, when theta=pi/2 the matrix becomes @f[
246  * \left[\array{ccc}
247  * 0 & \sin(\phi+\psi) & -cos(\phi+\psi) \\
248  * 0 & \cos(\phi+\psi) & \sin(\phi+\psi) \\
249  * 1 & 0 & 0
250  * \endarray\right]
251  * @f]
252  * In this case there no way to determine both phi and psi; all that can be
253  * determined is their sum. One way to overcome this problem is to arbitrarily
254  * set one of those angles to an arbitrary value such as zero. That is the
255  * approach used in this method. This arbitrary setting enables an XYZ Euler
256  * sequence to be extracted from the matrix even in the case of gimbal lock.
257  * The same principle once again applies to all twelve sequences.
258  *
259  * In summary, for a transformation matrix corresponding to an XYZ sequence,
260  * - The [2][0] element of the matrix specifies theta.
261  * - The [1][0] and [0][0] elements of the matrix specify psi.
262  * - The [2][1] and [2][2] elements of the matrix specify phi.
263  * These psi and phi values are valid only when gimbal lock is not present.
264  * - The [1][2] and [1][1] elements of the matrix specify phi in the
265  * case of gimbal lock.
266  *
267  * Extending this analysis to the remaining eleven sequences provides the
268  * essential information needed to extract the desired Euler angles from a
269  * transformation matrix. This information is captured in the EulerInfo
270  * array Euler_info defined at the head of this file. With a
271  * reference <tt>info</tt> to the appropriate element of this array,
272  * - The [info.indices[2]][info.indices[0]] element of the matrix
273  * specifies the angle theta.
274  * - The [info.indices[1]][info.indices[0]] and
275  * [info.alternate_x][info.indices[0]] elements of the matrix
276  * specify the angle psi when gimbal lock is not present.
277  * - The [info.indices[2]][info.indices[1]] and
278  * [info.indices[2]][info.alternate_z] elements of the matrix
279  * specify the angle phi when gimbal lock is not present.
280  * - The [info.indices[1]][info.alternate_z] and
281  * [info.indices[1]][info.indices[1]] elements of the matrix
282  * specify angle phi when gimbal lock is present.
283  *
284  * \par Assumptions and Limitations
285  * - To within numerical accuracy, the transformation matrix in the
286  * Orientation object @e is a proper transformation matrix:
287  * - The magnitude of each row and column vector is nearly one.
288  * - The inner product of any two different rows / two different columns of
289  * the matrix nearly zero.
290  * - The determinant of the matrix is nearly one.
291  * - An element whose value is outside the range [-1,1] is only slightly
292  * outside that range and the deviation is numerical.
293  * \param[in] trans Transformation matrix
294  * \param[in] euler_sequence Euler sequence
295  * \param[out] euler_angles Resultant Euler angles\n Units: r
296  */
297 void Orientation::compute_euler_angles_from_matrix(const double trans[3][3],
298  EulerSequence euler_sequence,
299  double euler_angles[3])
300 {
301  // Validate the value of the euler_sequence member.
302  // Note that an invalid value means the object is left unchanged.
303  if((euler_sequence < EulerXYZ) || (euler_sequence > EulerZYZ))
304  {
305  MessageHandler::error(__FILE__,
306  __LINE__,
307  OrientationMessages::invalid_enum,
308  "The euler_sequence data member has not been set or is invalid; "
309  "value=%d",
310  static_cast<int>(euler_sequence));
311  return;
312  }
313  const EulerInfo & info = Euler_info[euler_sequence]; // See above
314 
315  double phi; // First Euler angle
316  double theta; // Second Euler angle
317  double psi; // Third Euler angle
318 
319  double cos_phi; // cos(phi) * extra stuff * sign
320  double sin_phi; // sin(phi) * extra stuff * sign
321  double cos_psi; // cos(psi) * extra stuff * sign
322  double sin_psi; // sin(psi) * extra stuff * sign
323  double theta_val; // sin(theta), -sin(theta) or cos(theta)
324  double alt_theta_val1; // Conceptually, 1-theta_val^2
325  double alt_theta_val2; // Conceptually, 1-theta_val^2
326  double alt_theta_val; // Conceptually, 1-theta_val^2
327 
328  // Extract the key elements from the matrix assuming that this is not
329  // a gimbal lock situation.
330  // The trans[info.indices[2]][info.indices[0]] element is
331  // * sin(theta) for even permutation aerodynamics sequences,
332  // * -sin(theta) for odd permutation aerodynamics sequences, or
333  // * cos(theta) for all astronomical sequences.
334  theta_val = trans[info.indices[2]][info.indices[0]];
335 
336  // Get terms containing the sines and cosines of the first and third Euler
337  // angle times sin(theta) or cos(theta) (and sometimes negated).
338  sin_phi = trans[info.indices[2]][info.indices[1]];
339  cos_phi = trans[info.indices[2]][info.alternate_z];
340  sin_psi = trans[info.indices[1]][info.indices[0]];
341  cos_psi = trans[info.alternate_x][info.indices[0]];
342 
343  // Compute alternative theta values based on the above four terms.
344  alt_theta_val1 = std::sqrt(sin_phi * sin_phi + cos_phi * cos_phi);
345  alt_theta_val2 = std::sqrt(sin_psi * sin_psi + cos_psi * cos_psi);
346  alt_theta_val = 0.5 * (alt_theta_val1 + alt_theta_val2);
347 
348  // theta_val is -sin(theta) for odd permutation aerodynamics sequences.
349  // Negate to get rid of the minus sign.
350  if(info.is_aerodynamics_sequence && (!info.is_even_permutation))
351  {
352  theta_val = -theta_val;
353  }
354 
355  // Compute theta.
356  if(alt_theta_val < std::fabs(theta_val))
357  {
358  double alt_theta = std::asin(alt_theta_val);
359 
360  if(info.is_aerodynamics_sequence)
361  {
362  if(theta_val < 0.0)
363  {
364  theta = -0.5 * M_PI + alt_theta;
365  }
366  else
367  {
368  theta = 0.5 * M_PI - alt_theta;
369  }
370  }
371  else
372  {
373  if(theta_val < 0.0)
374  {
375  theta = M_PI - alt_theta;
376  }
377  else
378  {
379  theta = alt_theta;
380  }
381  }
382  }
383  else
384  {
385  if(info.is_aerodynamics_sequence)
386  {
387  theta = std::asin(theta_val);
388  }
389  else
390  {
391  theta = std::acos(theta_val);
392  }
393  }
394 
395  // Not in a gimbal lock situation:
396  // Compute sin_phi, cos_phi, sin_psi, and cos_psi.
397  // - These are not the sine and cosine of the Euler angles phi and psi.
398  // Rather, they are sin(phi) etc. scaled by a common positive number.
399  // - Key elements of the matrix are of the form
400  // sign*cos(theta)*sin(phi) etc.
401  // - The trick then is to find these key elements and then ensure that
402  // sign*cos(theta) is positive for each of the four values.
403  // - The specifics of the sign correction are depend on whether the
404  // sequence is an aerodynamical or astronomical sequence.
405  if(alt_theta_val > gimbal_lock_threshold)
406  {
407  // Correct signs for aerodynamic sequences.
408  if(info.is_aerodynamics_sequence)
409  {
410  // The sine values have the wrong sign for right-handed aero sequences.
411  if(info.is_even_permutation)
412  {
413  sin_phi = -sin_phi;
414  sin_psi = -sin_psi;
415  }
416  }
417 
418  // Correct signs for astronomical sequences.
419  else
420  {
421  // A cosine term has the wrong sign in the case of astro sequences.
422  // The term with the wrong sign is cos_phi for even permutations but
423  // cos_psi for odd permutations.
424  if(info.is_even_permutation)
425  {
426  cos_phi = -cos_phi;
427  }
428  else
429  {
430  cos_psi = -cos_psi;
431  }
432  }
433 
434  // Compute phi and psi.
435  phi = std::atan2(sin_phi, cos_phi);
436  psi = std::atan2(sin_psi, cos_psi);
437  }
438 
439  // In a gimbal lock situation:
440  // All that can be determine is the difference between / sum of phi and psi.
441  // Arbitrarily setting psi to zero resolves this issue.
442  else
443  {
444  // Compute sin_phi and cos_phi with the same constraint on the
445  // common scale factor as outlined above.
446  sin_phi = trans[info.indices[1]][info.alternate_z];
447  cos_phi = trans[info.indices[1]][info.indices[1]];
448 
449  // The sine value has the wrong sign for odd sequences.
450  if(!info.is_even_permutation)
451  {
452  sin_phi = -sin_phi;
453  }
454 
455  // Compute phi and psi.
456  phi = std::atan2(sin_phi, cos_phi);
457  psi = 0.0;
458  }
459 
460  // Save the computed angles in the object.
461  euler_angles[0] = phi;
462  euler_angles[1] = theta;
463  euler_angles[2] = psi;
464 }
