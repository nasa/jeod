46 /**
47  * Contains data needed to construct a transformation matrix given a sequence
48  * of Euler angles and to extract a sequence of Euler angles from a matrix.
49  * See Orientation::compute_euler_angles_from_matrix for details.
50  */
51 struct EulerInfo {
52 
53  /**
54  * The axes about which the rotations are performed in the order in which
55  * the rotations are performed, with X=0, Y=1, Z=2. For example, an XYZ or
56  * roll pitch yaw sequence is {0,1,2} while a ZXZ sequence is {2,0,2}.
57  */
58  unsigned int indices[3]; //!< trick_units(--)
59 
60  /**
61  * The initial element of the sequence for aerodynamics sequences,
62  * but the index of the omitted axis for astronomical sequences.
63  * For example, the omitted axis in a ZXZ sequence is Y=1.
64  */
65  unsigned int alternate_x; //!< trick_units(--)
66 
67  /**
68  * The final element of the sequence for aerodynamics sequences,
69  * but the index of the omitted axis for astronomical sequences.
70  */
71  unsigned int alternate_z; //!< trick_units(--)
72 
73  /**
74  * Indicates whether the 3-axis rotation sequence generated by replacing the
75  * final element of the sequence with the one axis not specified by the
76  * first two elements of the sequence is an even (true) permutation or odd
77  * permutation (false) of XYZ.
78  * The alternative 3-axis sequence is identical to the original sequence in
79  * the case of aerodynamics sequences. The astronomical ZXZ sequence becomes
80  * ZXY via this replacement rule. Since ZXY is an even permutation of XYZ,
81  * the is_even_permutation member for a ZXZ sequence is true.
82  */
83  bool is_even_permutation; //!< trick_units(--)
84 
85  /**
86  * True if the sequence is an aerodynamics sequence such as XYZ;
87  * false for an astronomical sequence such as ZXZ.
88  */
89  bool is_aerodynamics_sequence; //!< trick_units(--)
90 };

116 /**
117  * Compute the left transformation quaternion from the Euler sequence.
118  * The quaternion is formed by generating a sequence of three simple
119  * quaternions corresponding to the three rotations. The composite
120  * quaternion is the reverse-order product of these three simple quaternions.
121  * \param[in] euler_sequence Euler sequence
122  * \param[in] euler_angles Euler angles\n Units: r
123  * \param[out] quat Resultant quaternion
124  */
125 void
126 Orientation::compute_quaternion_from_euler_angles (
127  EulerSequence euler_sequence,
128  const double euler_angles[3],
129  Quaternion & quat)
130 {
131  // Validate the value of the euler_sequence member.
132  // Note that an invalid value means the object is left unchanged.
133  if ((euler_sequence < EulerXYZ) || (euler_sequence > EulerZYZ)) {
134  MessageHandler::error (
135  __FILE__, __LINE__, OrientationMessages::invalid_enum,
136  "The euler_sequence data member has not been set or is invalid; "
137  "value=%d",
138  static_cast<int> (euler_sequence));
139  return;
140  }
141 
142  const unsigned int * axes = Euler_info[euler_sequence].indices;
143  Quaternion q[3], q21;
144 
145  for (int ii = 0; ii < 3; ii++) {
146  double htheta = 0.5*euler_angles[ii];
147  double cosht = std::cos (htheta);
148  double sinht = std::sin (htheta);
149  q[ii].scalar = cosht;
150  q[ii].vector[axes[ii]] = -sinht;
151  }
152 
153  q[2].multiply (q[1], q21);
154  q21.multiply (q[0], quat);
155  quat.normalize ();
156 
157  return;
158 }

161 /**
162  * Compute the transformation matrix from the Euler sequence.
163  * The matrix is formed by generating a sequence of three simple transformation
164  * matrices corresponding to the three rotations. The composite transformation
165  * matrix is the reverse-order product of these three simple matrices.
166  * \param[in] euler_sequence Euler sequence
167  * \param[in] euler_angles Euler angles\n Units: r
168  * \param[out] trans Resultant transformation matrix
169  */
170 void
171 Orientation::compute_matrix_from_euler_angles (
172  EulerSequence euler_sequence,
173  const double euler_angles[3],
174  double trans[3][3])
175 {
176  // Validate the value of the euler_sequence member.
177  // Note that an invalid value means the object is left unchanged.
178  if ((euler_sequence < EulerXYZ) || (euler_sequence > EulerZYZ)) {
179  MessageHandler::error (
180  __FILE__, __LINE__, OrientationMessages::invalid_enum,
181  "The euler_sequence data member has not been set or is invalid; "
182  "value=%d",
183  static_cast<int> (euler_sequence));
184  return;
185  }
186  const unsigned int * axes = Euler_info[euler_sequence].indices;
187  double m[3][3][3], m21[3][3];
188 
189  for (int ii = 0; ii < 3; ii++) {
190  Matrix3x3::initialize (m[ii]);
191  double sin_theta = std::sin (euler_angles[ii]);
192  double cos_theta = std::cos (euler_angles[ii]);
193  switch (axes[ii]) {
194  case 0:
195  m[ii][0][0] = 1.0;
196  m[ii][1][1] = cos_theta;
197  m[ii][1][2] = sin_theta;
198  m[ii][2][1] = -sin_theta;
199  m[ii][2][2] = cos_theta;
200  break;
201  case 1:
202  m[ii][1][1] = 1.0;
203  m[ii][0][0] = cos_theta;
204  m[ii][0][2] = -sin_theta;
205  m[ii][2][0] = sin_theta;
206  m[ii][2][2] = cos_theta;
207  break;
208  case 2:
209  m[ii][2][2] = 1.0;
210  m[ii][0][0] = cos_theta;
211  m[ii][0][1] = sin_theta;
212  m[ii][1][0] = -sin_theta;
213  m[ii][1][1] = cos_theta;
214  break;
215  default:
216  break;
217  }
218  }
219 
220  Matrix3x3::product (m[2], m[1], m21);
221  Matrix3x3::product (m21, m[0], trans);
222 
223  return;
224 }

227 /**
228  * Extract an Euler sequence from the transformation matrix.
229  * A transformation matrix constructed from an XYZ Euler sequence
230  * is of the form @f[
231  * \left[\array{ccc}
232  * \cos\psi\cos\theta & \cdots & \cdots \\
233  * -\sin\psi\cos\theta & \cdots & \cdots \\
234  * \sin\theta & -\cos\theta\sin\phi & \cos\theta\cos\phi
235  * \endarray\right]
236  * @f]
237  * Note that the [2][0] element of the matrix depends on theta only.
238  * The other two elements of the leftmost column are simple terms that depend on
239  * theta and psi only, and the other two elements of the bottommost row are
240  * simple terms that depend on theta and phi only.
241  * Those five elements are the key to extracting an XYZ Euler sequence from a
242  * transformation matrix.
243  * The same principle applies to all twelve of the Euler sequences:
244  * Five key elements contain all of the information needed to extract the
245  * desired sequence. The location and form of those key elements of course
246  * depends on the sequence.
247  *
248  * A problem arises in the above when cos(theta) is zero, or nearly so. This
249  * siutation is called 'gimbal lock'. Those four elements used to determine phi
250  * and psi are zero or nearly so. Fortunately That ugly stuff isn't so ugly in
251  * the case of gimbal lock. Once again looking at the matrix generated from an
252  * XYZ Euler sequence, when theta=pi/2 the matrix becomes @f[
253  * \left[\array{ccc}
254  * 0 & \sin(\phi+\psi) & -cos(\phi+\psi) \\
255  * 0 & \cos(\phi+\psi) & \sin(\phi+\psi) \\
256  * 1 & 0 & 0
257  * \endarray\right]
258  * @f]
259  * In this case there no way to determine both phi and psi; all that can be
260  * determined is their sum. One way to overcome this problem is to arbitrarily
261  * set one of those angles to an arbitrary value such as zero. That is the
262  * approach used in this method. This arbitrary setting enables an XYZ Euler
263  * sequence to be extracted from the matrix even in the case of gimbal lock.
264  * The same principle once again applies to all twelve sequences.
265  *
266  * In summary, for a transformation matrix corresponding to an XYZ sequence,
267  * - The [2][0] element of the matrix specifies theta.
268  * - The [1][0] and [0][0] elements of the matrix specify psi.
269  * - The [2][1] and [2][2] elements of the matrix specify phi.
270  * These psi and phi values are valid only when gimbal lock is not present.
271  * - The [1][2] and [1][1] elements of the matrix specify phi in the
272  * case of gimbal lock.
273  *
274  * Extending this analysis to the remaining eleven sequences provides the
275  * essential information needed to extract the desired Euler angles from a
276  * transformation matrix. This information is captured in the EulerInfo
277  * array Euler_info defined at the head of this file. With a
278  * reference <tt>info</tt> to the appropriate element of this array,
279  * - The [info.indices[2]][info.indices[0]] element of the matrix
280  * specifies the angle theta.
281  * - The [info.indices[1]][info.indices[0]] and
282  * [info.alternate_x][info.indices[0]] elements of the matrix
283  * specify the angle psi when gimbal lock is not present.
284  * - The [info.indices[2]][info.indices[1]] and
285  * [info.indices[2]][info.alternate_z] elements of the matrix
286  * specify the angle phi when gimbal lock is not present.
287  * - The [info.indices[1]][info.alternate_z] and
288  * [info.indices[1]][info.indices[1]] elements of the matrix
289  * specify angle phi when gimbal lock is present.
290  *
291  * \par Assumptions and Limitations
292  * - To within numerical accuracy, the transformation matrix in the
293  * Orientation object @e is a proper transformation matrix:
294  * - The magnitude of each row and column vector is nearly one.
295  * - The inner product of any two different rows / two different columns of
296  * the matrix nearly zero.
297  * - The determinant of the matrix is nearly one.
298  * - An element whose value is outside the range [-1,1] is only slightly
299  * outside that range and the deviation is numerical.
300  * \param[in] trans Transformation matrix
301  * \param[in] euler_sequence Euler sequence
302  * \param[out] euler_angles Resultant Euler angles\n Units: r
303  */
304 void
305 Orientation::compute_euler_angles_from_matrix (
306  const double trans[3][3],
307  EulerSequence euler_sequence,
308  double euler_angles[3])
309 {
310  // Validate the value of the euler_sequence member.
311  // Note that an invalid value means the object is left unchanged.
312  if ((euler_sequence < EulerXYZ) || (euler_sequence > EulerZYZ)) {
313  MessageHandler::error (
314  __FILE__, __LINE__, OrientationMessages::invalid_enum,
315  "The euler_sequence data member has not been set or is invalid; "
316  "value=%d",
317  static_cast<int> (euler_sequence));
318  return;
319  }
320  const EulerInfo & info = Euler_info[euler_sequence]; // See above
321 
322  double phi; // First Euler angle
323  double theta; // Second Euler angle
324  double psi; // Third Euler angle
325 
326  double cos_phi; // cos(phi) * extra stuff * sign
327  double sin_phi; // sin(phi) * extra stuff * sign
328  double cos_psi; // cos(psi) * extra stuff * sign
329  double sin_psi; // sin(psi) * extra stuff * sign
330  double theta_val; // sin(theta), -sin(theta) or cos(theta)
331  double alt_theta_val1; // Conceptually, 1-theta_val^2
332  double alt_theta_val2; // Conceptually, 1-theta_val^2
333  double alt_theta_val; // Conceptually, 1-theta_val^2
334 
335 
336  // Extract the key elements from the matrix assuming that this is not
337  // a gimbal lock situation.
338  // The trans[info.indices[2]][info.indices[0]] element is
339  // * sin(theta) for even permutation aerodynamics sequences,
340  // * -sin(theta) for odd permutation aerodynamics sequences, or
341  // * cos(theta) for all astronomical sequences.
342  theta_val = trans[info.indices[2]][info.indices[0]];
343 
344  // Get terms containing the sines and cosines of the first and third Euler
345  // angle times sin(theta) or cos(theta) (and sometimes negated).
346  sin_phi = trans[info.indices[2]][info.indices[1]];
347  cos_phi = trans[info.indices[2]][info.alternate_z];
348  sin_psi = trans[info.indices[1]][info.indices[0]];
349  cos_psi = trans[info.alternate_x][info.indices[0]];
350 
351  // Compute alternative theta values based on the above four terms.
352  alt_theta_val1 = std::sqrt (sin_phi*sin_phi + cos_phi*cos_phi);
353  alt_theta_val2 = std::sqrt (sin_psi*sin_psi + cos_psi*cos_psi);
354  alt_theta_val = 0.5 * (alt_theta_val1 + alt_theta_val2);
355 
356  // theta_val is -sin(theta) for odd permutation aerodynamics sequences.
357  // Negate to get rid of the minus sign.
358  if (info.is_aerodynamics_sequence && (! info.is_even_permutation)) {
359  theta_val = -theta_val;
360  }
361 
362 
363  // Compute theta.
364  if (alt_theta_val < std::fabs (theta_val)) {
365  double alt_theta = std::asin (alt_theta_val);
366 
367  if (info.is_aerodynamics_sequence) {
368  if (theta_val < 0.0) {
369  theta = -0.5*M_PI + alt_theta;
370  }
371  else {
372  theta = 0.5*M_PI - alt_theta;
373  }
374  }
375  else {
376  if (theta_val < 0.0) {
377  theta = M_PI - alt_theta;
378  }
379  else {
380  theta = alt_theta;
381  }
382  }
383  }
384  else {
385  if (info.is_aerodynamics_sequence) {
386  theta = std::asin (theta_val);
387  }
388  else {
389  theta = std::acos (theta_val);
390  }
391  }
392 
393 
394  // Not in a gimbal lock situation:
395  // Compute sin_phi, cos_phi, sin_psi, and cos_psi.
396  // - These are not the sine and cosine of the Euler angles phi and psi.
397  // Rather, they are sin(phi) etc. scaled by a common positive number.
398  // - Key elements of the matrix are of the form
399  // sign*cos(theta)*sin(phi) etc.
400  // - The trick then is to find these key elements and then ensure that
401  // sign*cos(theta) is positive for each of the four values.
402  // - The specifics of the sign correction are depend on whether the
403  // sequence is an aerodynamical or astronomical sequence.
404  if (alt_theta_val > gimbal_lock_threshold) {
405 
406  // Correct signs for aerodynamic sequences.
407  if (info.is_aerodynamics_sequence) {
408 
409  // The sine values have the wrong sign for right-handed aero sequences.
410  if (info.is_even_permutation) {
411  sin_phi = -sin_phi;
412  sin_psi = -sin_psi;
413  }
414  }
415 
416  // Correct signs for astronomical sequences.
417  else {
418 
419  // A cosine term has the wrong sign in the case of astro sequences.
420  // The term with the wrong sign is cos_phi for even permutations but
421  // cos_psi for odd permutations.
422  if (info.is_even_permutation) {
423  cos_phi = -cos_phi;
424  }
425  else {
426  cos_psi = -cos_psi;
427  }
428  }
429 
430  // Compute phi and psi.
431  phi = std::atan2 (sin_phi, cos_phi);
432  psi = std::atan2 (sin_psi, cos_psi);
433  }
434 
435 
436  // In a gimbal lock situation:
437  // All that can be determine is the difference between / sum of phi and psi.
438  // Arbitrarily setting psi to zero resolves this issue.
439  else {
440  // Compute sin_phi and cos_phi with the same constraint on the
441  // common scale factor as outlined above.
442  sin_phi = trans[info.indices[1]][info.alternate_z];
443  cos_phi = trans[info.indices[1]][info.indices[1]];
444 
445  // The sine value has the wrong sign for odd sequences.
446  if (! info.is_even_permutation) {
447  sin_phi = -sin_phi;
448  }
449 
450  // Compute phi and psi.
451  phi = std::atan2 (sin_phi, cos_phi);
452  psi = 0.0;
453  }
454 
455  // Save the computed angles in the object.
456  euler_angles[0] = phi;
457  euler_angles[1] = theta;
458  euler_angles[2] = psi;
459 
460  return;
461 }
