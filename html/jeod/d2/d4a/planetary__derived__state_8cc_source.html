62 /**
63  * Begin initialization of a PlanetaryDerivedState.
64  * The initialize method for all subclasses of DerivedState *must* pass
65  * the initialize call to their immediate parent class, which in turn
66  * must do the same, eventually invoking this method.
67  * \param[in,out] subject_body Subject body
68  * \param[in,out] dyn_manager Dynamics manager
69  */
70 void PlanetaryDerivedState::initialize(DynBody & subject_body, DynManager & dyn_manager)
71 {
72  // Initialize as a DerivedState.
73  DerivedState::initialize(subject_body, dyn_manager);
74 
75  // Find the planet.
76  // Note that find_planet doesn't return if the named planet is not found.
77  planet = find_planet(dyn_manager, reference_name, "reference_name");
78 
79  // Choose the planet fixed frame to be used
80  if(use_alt_pfix)
81  {
82  pfix_ptr = &(planet->alt_pfix);
83  }
84  else
85  {
86  pfix_ptr = &(planet->pfix);
87  }
88  pfix_ptr->subscribe();
89 
90  // Initialize the planet-fixed position.
91  state.initialize(planet);
92 }
