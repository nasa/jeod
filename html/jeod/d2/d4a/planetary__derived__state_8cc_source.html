81 /**
82  * Begin initialization of a PlanetaryDerivedState.
83  * The initialize method for all subclasses of DerivedState *must* pass
84  * the initialize call to their immediate parent class, which in turn
85  * must do the same, eventually invoking this method.
86  * \param[in,out] subject_body Subject body
87  * \param[in,out] dyn_manager Dynamics manager
88  */
89 void
90 PlanetaryDerivedState::initialize (
91  DynBody & subject_body,
92  DynManager & dyn_manager)
93 {
94 
95  // Initialize as a DerivedState.
96  DerivedState::initialize (subject_body, dyn_manager);
97 
98  // Find the planet.
99  // Note that find_planet doesn't return if the named planet is not found.
100  planet = find_planet (dyn_manager, reference_name, "reference_name");
101 
102  // Choose the planet fixed frame to be used
103  if (use_alt_pfix) {
104  pfix_ptr = &(planet->alt_pfix);
105  } else {
106  pfix_ptr = &(planet->pfix);
107  }
108  pfix_ptr->subscribe();
109 
110  // Initialize the planet-fixed position.
111  state.initialize (planet);
112 
113  return;
114 }
