100 /**
101  * Apply the initializer.
102  * \param[in,out] dyn_manager Dynamics manager
103  */
104 void DynBodyInitNedState::apply(DynManager & dyn_manager)
105 {
106  NorthEastDown ned_state; // -- North-East-Down frame vehicle state
107  RefFrameItems test_items(set_items); // -- The state components to be set
108 
109  // Initialize the N-E-D's planet reference to the specified planet.
110  ned_state.initialize(planet);
111 
112  // Populate the planet-fixed state.
113  // No reference body: Use the reference point as the basis for the state.
114  if(ref_body == nullptr)
115  {
116  // The point is stationary wrt the rotating planet.
117  Vector3::initialize(ned_state.ned_frame.state.trans.velocity);
118 
119  // Set the state with the supplied point, interpreted as geocentric
120  // or geodetic per the altlatlong_type enum.
121  switch(altlatlong_type)
122  {
123  case NorthEastDown::spherical:
124  ned_state.update_from_spher(ref_point);
125  break;
126 
127  case NorthEastDown::elliptical:
128  ned_state.update_from_ellip(ref_point);
129  break;
130 
131  case NorthEastDown::undefined:
132  default:
133  MessageHandler::fail(__FILE__,
134  __LINE__,
135  BodyActionMessages::illegal_value,
136  "%s failed:\n"
137  "AltLatLong type option not recognized",
138  action_identifier.c_str());
139 
140  // Not reached
141  return;
142  }
143 
144  // Set the frame name based on the reference body.
145  ned_state.ned_frame.set_name("ref_point", planet_name, "ned");
146  }
147 
148  // Reference body supplied: Use its translational state as the basis for
149  // the planet-fixed state.
150  else
151  {
152  RefFrameState rel_state;
153 
154  // Compute the state of the reference body to the planet fixed frame.
155  ref_body->composite_body.compute_relative_state(*pfix_ptr, rel_state);
156 
157  // Set the north-east-down frame's translational states.
158  ned_state.set_ned_trans_states(rel_state.trans.position, rel_state.trans.velocity);
159 
160  // Set the frame name based on the reference body.
161  ned_state.ned_frame.set_name(ref_body->name.get_name(), planet_name, "ned");
162  }
163 
164  // Construct the N-E-D frame's orientation.
165  ned_state.altlatlong_type = altlatlong_type;
166  ned_state.build_ned_orientation();
167 
168  /* (Temporarily) attach the north-east-down frame to the planet-centered,
169  planet-fixed frame. */
170  pfix_ptr->add_child(ned_state.ned_frame);
171 
172  // The position/velocity/attitude/rate data pertain to this newly
173  // constructed N-E-D frame.
174  reference_ref_frame = &ned_state.ned_frame;
175 
176  // Compute the state relative to the parent given the user inputs
177  // relative to the reference reference frame.
178  apply_user_inputs();
179 
180  /* Pass the message up the chain, with the reference frame for the
181  state set to this newly-constructed N-E-D frame */
182  DynBodyInitPlanetDerived::apply(dyn_manager);
183 
184  // Remove the temporary connection established above.
185  ned_state.ned_frame.remove_from_parent();
186 
187  // The N-E-D frame will go out of scope on returning from this function.
188  // The reference_ref_frame should no longer refer to this frame.
189  reference_ref_frame = nullptr;
190 }
