156 /**
157  * Runge-Kutta 4th order integration of the temperature variation
158  * @return void
159  */
160 double ThermalFacetRider::integrate()
161 {
162  if(!active)
163  {
164  power_emit = power_absorb; // Keep the physics consistent!
165  return dynamic_temperature;
166  }
167 
168 #define pow4(x) pow4_temp = (x) * (x), pow4_temp *= pow4_temp
169 
170  double pow4_temp; /* -- temporary value associated with T^4 macro, pow4 */
171  double T0; /* -- starting temperature */
172  double I1; /* -- intermediate value */
173  double cyc_cap = cycle_time / heat_capacity;
174  /* -- shorthand collection of values used in emission calcs*/
175  double Teq4; /* K*K*K*K
176  Equilibrium temperature to 4th power.*/
177  int dt_dir; /* --
178  direction in which temperature should go: +1 up, -1 down. */
179 
180  // assign temperature from previously calculated value.
181  dynamic_temperature = next_temperature;
182 
183  /* R-K 4 integration of energy flux to obtain new temperature */
184  T0 = dynamic_temperature;
185  Teq4 = power_absorb / rad_constant;
186 
187  pow4(T0);
188  if(pow4_temp > Teq4)
189  { // above equilibrium
190  dt_dir = -1;
191  }
192  else
193  {
194  dt_dir = 1;
195  }
196 
197  I1 = cyc_cap * (power_absorb - rad_constant * pow4_temp);
198  pow4(T0 + I1 / 2);
199 
200  double I2; /* intermediate value */
201  I2 = cyc_cap * (power_absorb - rad_constant * pow4_temp);
202 
203  if(I2 * dt_dir < 0)
204  { // jumped through asymptote.
205  MessageHandler::inform(__FILE__,
206  __LINE__,
207  ThermalMessages::invalid_integration_operation,
208  "\n"
209  "Temperature integration produced instability.\n"
210  "The integration step is sufficiently large that the predicted\n"
211  "temperature at the first stage of the integration is on the \n"
212  "wrong side of the asymptote. \n"
213  "The predicted temperature is being set to the equilibrium"
214  "temperature.\n");
215  next_temperature = sqrt(sqrt(Teq4));
216  d_temperature = next_temperature - dynamic_temperature;
217  power_emit = power_absorb - (heat_capacity * d_temperature / cycle_time);
218  return dynamic_temperature;
219  }
220 
221  double I3; /* intermediate value */
222  pow4(T0 + I2 / 2);
223  I3 = cyc_cap * (power_absorb - rad_constant * pow4_temp);
224 
225  double I4; /* intermediate value */
226  pow4(T0 + I3);
227  I4 = cyc_cap * (power_absorb - rad_constant * pow4_temp);
228 
229  if(dt_dir * I4 < 0)
230  {
231  I4 *= 0.5;
232  }
233 
234  d_temperature = (I1 + 2 * I2 + 2 * I3 + I4) / 6;
235 
236  if((d_temperature < -dynamic_temperature) || (d_temperature > 1E6))
237  {
238  MessageHandler::warn(__FILE__,
239  __LINE__,
240  ThermalMessages::invalid_integration_operation,
241  "\n"
242  "Temperature integration gone awry for unknown reason.\n"
243  "Resetting flag on facet and holding temperature constant.\n");
244  active = false;
245  return dynamic_temperature;
246  }
247 
248  if(dt_dir * d_temperature < 0)
249  {
250  MessageHandler::inform(__FILE__,
251  __LINE__,
252  ThermalMessages::invalid_integration_operation,
253  "\n"
254  "Temperature integration produced instability.\n"
255  "The integration step is sufficiently large that the predicted\n"
256  "value for temperature went through the asymptote; this led to\n"
257  "an invalid temperature prediction, that is farther from equilibrium \n"
258  "than when it started. Setting the predicted temperature to the \n"
259  "equilibrium temperature.");
260  next_temperature = sqrt(sqrt(Teq4));
261  d_temperature = next_temperature - dynamic_temperature;
262  }
263  else
264  { // temperature went right direction, but did it go too far?
265 
266  next_temperature = dynamic_temperature + d_temperature;
267  pow4(next_temperature);
268 
269  if(dt_dir * (Teq4 - pow4_temp) < 0)
270  { // overshot equilibrium value
271  MessageHandler::inform(__FILE__,
272  __LINE__,
273  ThermalMessages::invalid_integration_operation,
274  "\n"
275  "Temperature integration overshot target temperature.\n"
276  "Probably due to an excessively large integration step,\n"
277  "the temperature integrator produced a temperature on the wrong \n"
278  "side of the equilibrium temperature. \n"
279  "Setting the predicted temperature to the equilibrium temperature.");
280  next_temperature = sqrt(sqrt(Teq4));
281  d_temperature = next_temperature - dynamic_temperature;
282  }
283  }
284 
285  power_emit = power_absorb - (heat_capacity * d_temperature / cycle_time);
286 
287  return dynamic_temperature;
288 }
