179 /**
180  * Runge-Kutta 4th order integration of the temperature variation
181  * @return void
182  */
183 double
184 ThermalFacetRider::integrate (
185  void)
186 {
187 
188  if (!active) {
189  power_emit = power_absorb; // Keep the physics consistent!
190  return dynamic_temperature;
191  }
192 
193  #define pow4(x) pow4_temp = (x) * (x), pow4_temp *= pow4_temp
194 
195  double pow4_temp; /* -- temporary value associated with T^4 macro, pow4 */
196  double T0; /* -- starting temperature */
197  double I1; /* -- intermediate value */
198  double cyc_cap = cycle_time / heat_capacity;
199  /* -- shorthand collection of values used in emission calcs*/
200  double Teq4; /* K*K*K*K
201  Equilibrium temperature to 4th power.*/
202  int dt_dir; /* --
203  direction in which temperature should go: +1 up, -1 down. */
204 
205 
206  // assign temperature from previously calculated value.
207  dynamic_temperature = next_temperature;
208 
209  /* R-K 4 integration of energy flux to obtain new temperature */
210  T0 = dynamic_temperature;
211  Teq4 = power_absorb / rad_constant;
212 
213  pow4 (T0);
214  if (pow4_temp > Teq4) { // above equilibrium
215  dt_dir = -1;
216  }
217  else {
218  dt_dir = 1;
219  }
220 
221  I1 = cyc_cap * (power_absorb - rad_constant * pow4_temp);
222  pow4 (T0 + I1 / 2);
223 
224  double I2; /* intermediate value */
225  I2 = cyc_cap * (power_absorb - rad_constant * pow4_temp);
226 
227  if (I2 * dt_dir < 0) { // jumped through asymptote.
228  MessageHandler::inform (
229  __FILE__, __LINE__, ThermalMessages::invalid_integration_operation,
230  "\n"
231  "Temperature integration produced instability.\n"
232  "The integration step is sufficiently large that the predicted\n"
233  "temperature at the first stage of the integration is on the \n"
234  "wrong side of the asymptote. \n"
235  "The predicted temperature is being set to the equilibrium"
236  "temperature.\n");
237  next_temperature = sqrt (sqrt (Teq4));
238  d_temperature = next_temperature - dynamic_temperature;
239  power_emit = power_absorb -
240  (heat_capacity * d_temperature / cycle_time);
241  return dynamic_temperature;
242  }
243 
244 
245  double I3; /* intermediate value */
246  pow4 (T0 + I2 / 2);
247  I3 = cyc_cap * (power_absorb - rad_constant * pow4_temp);
248 
249  double I4; /* intermediate value */
250  pow4 (T0 + I3);
251  I4 = cyc_cap * (power_absorb - rad_constant * pow4_temp);
252 
253  if (dt_dir * I4 < 0) {
254  I4 *= 0.5;
255  }
256 
257  d_temperature = (I1 + 2 * I2 + 2 * I3 + I4) / 6;
258 
259  if ((d_temperature < -dynamic_temperature) || (d_temperature > 1E6)) {
260  MessageHandler::warn (
261  __FILE__, __LINE__, ThermalMessages::invalid_integration_operation,
262  "\n"
263  "Temperature integration gone awry for unknown reason.\n"
264  "Resetting flag on facet and holding temperature constant.\n");
265  active = false;
266  return dynamic_temperature;
267  }
268 
269 
270  if (dt_dir * d_temperature < 0) {
271  MessageHandler::inform (
272  __FILE__, __LINE__, ThermalMessages::invalid_integration_operation,
273  "\n"
274  "Temperature integration produced instability.\n"
275  "The integration step is sufficiently large that the predicted\n"
276  "value for temperature went through the asymptote; this led to\n"
277  "an invalid temperature prediction, that is farther from equilibrium \n"
278  "than when it started. Setting the predicted temperature to the \n"
279  "equilibrium temperature.");
280  next_temperature = sqrt (sqrt (Teq4));
281  d_temperature = next_temperature - dynamic_temperature;
282  }
283  else { // temperature went right direction, but did it go too far?
284 
285 
286  next_temperature = dynamic_temperature + d_temperature;
287  pow4 (next_temperature);
288 
289  if (dt_dir * (Teq4 - pow4_temp) < 0) { // overshot equilibrium value
290  MessageHandler::inform (
291  __FILE__, __LINE__, ThermalMessages::invalid_integration_operation,
292  "\n"
293  "Temperature integration overshot target temperature.\n"
294  "Probably due to an excessively large integration step,\n"
295  "the temperature integrator produced a temperature on the wrong \n"
296  "side of the equilibrium temperature. \n"
297  "Setting the predicted temperature to the equilibrium temperature.");
298  next_temperature = sqrt (sqrt (Teq4));
299  d_temperature = next_temperature - dynamic_temperature;
300  }
301  }
302 
303  power_emit = power_absorb - (heat_capacity * d_temperature / cycle_time);
304 
305  return dynamic_temperature;
306 }
