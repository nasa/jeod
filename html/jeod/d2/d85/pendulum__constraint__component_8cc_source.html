129 bool PendulumConstraintComponent::check_for_reset()
130 {
131  // Need to reset the integrators if this has been forced from the outside.
132  if(new_mode == PendulumConstraintMode::Reset)
133  {
134  return true;
135  }
136 
137  // Need to reset the integrators if the pendulum's integrated state is the
138  // cartesian state but it now would be better to integrate angular state.
139  else if(((mode == PendulumConstraintMode::Vertical) || (mode == PendulumConstraintMode::Inverted)) &&
140  (cos_sq_theta < cos_sq_theta_angular))
141  {
142  new_mode = PendulumConstraintMode::Horizontal;
143  return true;
144  }
145 
146  // Need to reset the integrators if the pendulum's integrated state is the
147  // angular state but it now would be better to integrate cartesian state.
148  else if((mode == PendulumConstraintMode::Horizontal) && (cos_sq_theta > cos_sq_theta_cartesian))
149  {
150  if(bob_position[2] > 0.0)
151  {
152  new_mode = PendulumConstraintMode::Vertical;
153  }
154  else
155  {
156  new_mode = PendulumConstraintMode::Inverted;
157  }
158  return true;
159  }
160 
161  else
162  {
163  new_mode = mode;
164  return false;
165  }

168 void PendulumConstraintComponent::transition_to_horizontal()
169 {
170  double rsq = bob_position[0] * bob_position[0] + bob_position[1] * bob_position[1];
171  double r = std::sqrt(rsq);
172  sin_theta = r / pendulum_length;
173  cos_theta = bob_position[2] / pendulum_length;
174  sin_phi = bob_position[1] / r;
175  cos_phi = bob_position[0] / r;
176  theta = std::acos(cos_theta);
177  phi = std::atan2(sin_phi, cos_phi);
178  construct_constraint_to_spherical_transform();
179  theta_dot = Vector3::dot(bob_velocity, constraint_to_spherical_transform[1]) / pendulum_length;
180  phi_dot = Vector3::dot(bob_velocity, constraint_to_spherical_transform[2]) / (pendulum_length * sin_theta);
181  compute_rho_hat_dot();
182 }

240 void PendulumConstraintComponent::construct_constraint_to_spherical_transform()
241 {
242  constraint_to_spherical_transform[0][0] = sin_theta * cos_phi;
243  constraint_to_spherical_transform[0][1] = sin_theta * sin_phi;
244  constraint_to_spherical_transform[0][2] = cos_theta;
245  constraint_to_spherical_transform[1][0] = cos_theta * cos_phi;
246  constraint_to_spherical_transform[1][1] = cos_theta * sin_phi;
247  constraint_to_spherical_transform[1][2] = -sin_theta;
248  constraint_to_spherical_transform[2][0] = -sin_phi;
249  constraint_to_spherical_transform[2][1] = cos_phi;

307 void PendulumConstraintComponent::compute_bob_state_cartesian()
308 {
309  // The above only integrated the x & y position and velocity.
310  // Compute the z position and velocity.
311  double rsq = bob_position[0] * bob_position[0] + bob_position[1] * bob_position[1];
312  double lsq = pendulum_length * pendulum_length;
313  double zsq = lsq - rsq;
314 
315  bob_position[2] = std::sqrt(zsq);
316  bob_velocity[2] = -(bob_position[0] * bob_velocity[0] + bob_position[1] * bob_velocity[1]) / bob_position[2];
317 
318  if(mode == PendulumConstraintMode::Inverted)
319  {
320  bob_position[2] = -bob_position[2];
321  bob_velocity[2] = -bob_velocity[2];
322  }
323 
324  cos_theta = bob_position[2] / pendulum_length;
325  cos_sq_theta = cos_theta * cos_theta;
326 
327  // Set the constraint direction for the next time around.
328  double bob_direction[3];
329  Vector3::scale(bob_position, 1.0 / pendulum_length, bob_direction);
330  set_constraint_direction(bob_direction);
331 }

333 void PendulumConstraintComponent::compute_constraint_response(const VehicleProperties &,
334  const VehicleNonGravState & non_grav_state)
335 {
336  constrained_mass->compute_non_grav_accel(non_grav_state);
337 
338  double tension_accel = constraint_force / constrained_mass->mass;
339  double damping_factor = pendulum_model->compute_damping_factor(
340  Vector3::vmag(constrained_mass->non_grav_accel_root));
341  // Or std::abs(tension_accel)
342 
343  /*
344  * Note: Tension accel could be calculated using the same expression
345  * used in setup_constraint(), but there's no point. The solution
346  * to the constraints system of equations for this constrained object
347  * is the tension force.
348  */
349 
350  Vector3::transform(constraint_frame->T_root_constraint, non_grav_state.omega_struct, omega_constraint_frame);
351  Vector3::transform(constraint_frame->T_root_constraint,
352  non_grav_state.omega_dot_struct,
353  omega_dot_constraint_frame);
354 
355  if(mode == PendulumConstraintMode::Horizontal)
356  {
357  double omega_rho = Vector3::dot(constraint_to_spherical_transform[0], omega_constraint_frame);
358  double accel_theta = Vector3::dot(constraint_to_spherical_transform[1],
359  constrained_mass->non_grav_accel_constraint_frame);
360  double omega_theta = Vector3::dot(constraint_to_spherical_transform[1], omega_constraint_frame);
361  double omega_dot_theta = Vector3::dot(constraint_to_spherical_transform[1], omega_dot_constraint_frame);
362  double accel_phi = Vector3::dot(constraint_to_spherical_transform[2],
363  constrained_mass->non_grav_accel_constraint_frame);
364  double omega_phi = Vector3::dot(constraint_to_spherical_transform[2], omega_constraint_frame);
365  double omega_dot_phi = Vector3::dot(constraint_to_spherical_transform[2], omega_dot_constraint_frame);
366 
367  theta_dot_dot = -(damping_factor * theta_dot + accel_theta / pendulum_length +
368  omega_rho * (omega_theta - 2.0 * phi_dot * sin_theta) + omega_dot_phi -
369  phi_dot * phi_dot * sin_theta * cos_theta);
370  phi_dot_dot = -(damping_factor * phi_dot +
371  (accel_phi / pendulum_length + omega_rho * (omega_phi + 2.0 * theta_dot) - omega_dot_theta +
372  2.0 * theta_dot * phi_dot * cos_theta) /
373  sin_theta);
374  }
375  else
376  {
377  double tension_accel_div_length = tension_accel / pendulum_length;
378  double omega_dot_cross_pos[3];
379  double omega_cross_omega_cross_pos[3];
380  double * non_grav_accel_constraint_frame = constrained_mass->non_grav_accel_constraint_frame;
381 
382  Vector3::cross(omega_dot_constraint_frame, bob_position, omega_dot_cross_pos);
383  Vector3::cross(omega_constraint_frame, omega_cross_pos, omega_cross_omega_cross_pos);
384 
385  for(int ii = 0; ii < 2; ++ii)
386  {
387  // Calculation is done in two steps for numerical reasons.
388  // The first part will typically be smaller than the second.
389  bob_accel[ii] = -(damping_factor * bob_velocity[ii] // Drag
390  + omega_dot_cross_pos[ii] // Euler
391  + 2.0 * omega_cross_vel[ii] // Coriolis
392  + omega_cross_omega_cross_pos[ii]) // Centrifugal
393  - (tension_accel_div_length * bob_position[ii] // Tension
394  + non_grav_accel_constraint_frame[ii]); // Inertial
395  }
396  }
397 
398  double damping_torque_constraint_frame[3];
399  double damping_torque_root[3];
400  Vector3::cross(bob_position, bob_velocity, damping_torque_constraint_frame);
401  Vector3::scale(constrained_mass->mass * damping_factor, damping_torque_constraint_frame);
402  Vector3::transform_transpose(constraint_frame->T_root_constraint,
403  damping_torque_constraint_frame,
404  damping_torque_root);
405  damping_wrench.set_torque(damping_torque_root);
406 }
