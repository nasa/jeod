111  * Recursively adds elements to the initialization tree.
112  * If the "parent" to a time-type is defined, adds the "parent" then

117  * change.

168  time_manager_init->set_status(index, 0);

171  // else
172  // If parent defined but not found, terminate

245 /**
246  * Calculate Gregorian calendar date and 24-hour clock representation
247  * from Truncated Julian date/time.
248  *
249  * \par Assumptions and Limitations
250  * - Coverage is from March 1, 1600 onward.
251  * - Produces a time in 24-hour clock format.
252  * - Assumes that the values year, month, day, hour, minute, second, and
253  * truncated_julian_time are all present in the same class.
254  */
255 void TimeStandard::calculate_calendar_values()
256 {
257  int julian_day; /* -- integral part of julian time */
258  double temp; /* -- temporary value */
259 
260  /*separate julian time into the integral and fractional parts */
261  julian_day = static_cast<int>(trunc_julian_time);
262  if(julian_day < 0)
263  {
264  julian_day -= 1;
265  }
266 
267  /* minutes in fractional day */
268  temp = 1440 * (trunc_julian_time - julian_day);
269  /* whole minutes in fractional day */
270  calendar_minute = static_cast<int>(temp);
271  /* seconds in final fractional minute */
272  calendar_second = 60 * (temp - calendar_minute);
273  if(calendar_second > 60 - clock_resolution)
274  {
275  calendar_minute++;
276  if(calendar_minute >= 1440)
277  {
278  julian_day++;
279  calendar_minute -= 1440;
280  }
281  calendar_second = 0;
282  }
283  /* integer division, multiples of 60 in total minutes */
284  calendar_hour = calendar_minute / 60;
285  /* subtract away the hour multiples of 60. */
286  calendar_minute -= (calendar_hour * 60);
287 
288  /* Note - could have done this with only one static_cast, but would have had
289  to go from trunc_julian_time direct to minutes, which would cause a
290  potential loss in precision. */
291 
292  /* convert the integral part into day, month,year format if day has changed */
293  if(julian_day != prev_julian_day)
294  {
295  prev_julian_day = julian_day;
296  // determine the number of whole 400-year periods since March 1, 1600, and
297  // the number of days since the "most recent" 400-year boundary
298  julian_day += 134493; /* 134493 is the offset from March 1, 1600 to the
299  start of Truncated Julian Time */
300  int n_400 = julian_day / 146097; // There are 146097 days in 400 years. (number of whole 400-year periods)
301  if(julian_day < 0)
302  {
303  n_400 -= 1;
304  }
305  int r_400 = julian_day - 146097 * n_400 + 1; /* -- number of years remaing after 400-year period */
306 
307  /* determine number of whole 100-year periods since the "most recent" 400-year
308  boundary, and number of days since the "most recent" 100-year boundary */
309  auto n_100 = static_cast<int>(r_400 / 36524.3); /* -- number of whole 100-year periods */
310  int r_100 = r_400 - 36524 * n_100 - 1; /* -- number of years remaing after 100-year period */
311 
312  /* ditto for 4-year periods */
313  int n_4 = r_100 / 1461; /* -- number of whole 4-year periods */
314  int r_4 = r_100 - 1461 * n_4 + 1; /* -- number of years remaing after 4-year period */
315 
316  /* and for 1-year periods, omitting remaining-days calculation */
317  auto n_1 = static_cast<int>(r_4 / 365.3); /* -- number of whole years */
318 
319  /* Modified r_4 for months calculation */
320  int r_4a = r_4 + 30 + 2 * n_1; /* -- modified value of r_4 */
321 
322  /* m = (Number of whole months +1) since the most recent 4-year period
323  (March=1, April=2, ... , February=12) */
324  auto m = static_cast<int>(r_4a / 30.585);
325 
326  /* Determine day of month, month number (Jan=1, etc.), and year number (AD) */
327  calendar_day = r_4a - static_cast<int>(30.585 * m);
328  calendar_month = m + 2 - 12 * ((m + 1) / 12);
329  calendar_year = 1600 + 400 * n_400 + 100 * n_100 + 4 * n_4 + ((m + 1) / 12);
330  }
331 }

359 /**
360  * Calculate Truncated Julian date/time from Gregorian calendar date and
361  * 24-hour clock representation.
362  *
363  * \par Assumptions and Limitations
364  * - Coverage s from March 1, 1600 onward.
365  * - Assumes that time is in 24-hour clock format; 1:00:00 pm cannot be read
366  * correctly, but 13:00:00 can.
367  * - Assumes that the values year, month, day, hour, minute, second, and
368  * truncated_julian_time are all present in the same class.
369  */
370 void TimeStandard::convert_from_calendar()
371 {
372  int m, /* adjusted month value */
373  n_1, /* -- number of whole years */
374  n_4, /* -- number of whole 4-year periods */
375  n_100, /* -- number of whole 100-year periods */
376  n_400, /* -- number of whole 400-year periods */
377  y, /* -- adjusted year value */
378  y_100, /* -- years remaining after subtracting off 100-yr multiples */
379  y_400; /* -- years remaining after subtracting off 400-yr multiples */
380 
381  /* adjust the year to represent the number of whole years since March 1, 1600 */
382  y = calendar_year - 1601 + (calendar_month + 9) / 12;
383 
384  /* Break down that number of years into a cumulative count of 400-year
385  periods, 100-year periods, 4-year periods, and 1-year periods. */
386  n_400 = y / 400;
387  y_400 = y - 400 * n_400;
388 
389  n_100 = y_400 / 100;
390  y_100 = y_400 - 100 * n_100;
391 
392  n_4 = y_100 / 4;
393  n_1 = y_100 - 4 * n_4;
394 
395  /* Calculate the number of (partial) months since the end of February in a
396  year that is divisible by 4 (even if it is not a leap year).
397  (March = 1,13,25,37; April = 2,14,26,38; ... ; February = 12,24,36,48) */
398  m = calendar_month - 2 + 12 * (1 - ((calendar_month + 9) / 12) + n_1);
399 
400  trunc_julian_time = calendar_day + floor(30.585 * m) - (2 * n_1) - 31 + (1461 * n_4) + (36524 * n_100) +
401  (146097 * n_400) - 134493 + (calendar_hour / 24.0) + (calendar_minute / 1440.0) +
402  (calendar_second / 86400.0);
403 
404  days = trunc_julian_time - tjt_at_epoch;
405  seconds = days * 86400;
406 }

672 /**
673  * Generate the number of seconds elapsed this year
674  *
675  * \par Assumptions and Limitations
676  * - Relies on the accuracy of the JEOD2.0 calendar.
677  * @return Current second of year.
678  */
679 double TimeStandard::seconds_of_year()
680 {
681  if(!Numerical::compare_exact(last_calendar_update, time_manager->simtime))
682  {
683  calculate_calendar_values();
684  last_calendar_update = time_manager->simtime;
685  }
686 
687  if(calendar_year == year_of_last_soy)
688  { // year has not changed since last update
689  return (seconds - seconds_at_year_start);
690  }
691  else if(((calendar_year - year_of_last_soy) == 1) && (calendar_month <= 2))
692  {
693  // year has ticked over, but less than 2 full months into new year
694  double seconds_of_year_value = ((calendar_month - 1) * 31 + (calendar_day - 1)) * 86400 + calendar_hour * 3600 +
695  calendar_minute * 60 + calendar_second;
696  seconds_at_year_start = seconds - seconds_of_year_value;
697  year_of_last_soy = calendar_year;
698  return seconds_of_year_value;
699  }
700 
701  // Get here is this is the first time calculating seconds_of_year, or if year
702  // has ticked over backward, or if it is on or after March 1 in the year
703  // immediately following the last seconds_of_year calculation.
704  int temp_c_month = calendar_month;
705  int temp_c_day = calendar_day;
706  int temp_c_hour = calendar_hour;
707  int temp_c_minute = calendar_minute;
708  double temp_c_second = calendar_second;
709  double temp_tjt = trunc_julian_time;
710  double temp_days = days;
711  double temp_seconds = seconds;
712 
713  // set the time to be at year-start
714  calendar_month = 1;
715  calendar_day = 1;
716  calendar_hour = 0;
717  calendar_minute = 0;
718  calendar_second = 0.0;
719 
720  // convert to decimal, save off appropriate values
721  convert_from_calendar();
722  seconds_at_year_start = seconds;
723  year_of_last_soy = calendar_year;
724 
725  // revert to proper time
726  calendar_month = temp_c_month;
727  calendar_day = temp_c_day;
728  calendar_hour = temp_c_hour;
729  calendar_minute = temp_c_minute;
730  calendar_second = temp_c_second;
731  trunc_julian_time = temp_tjt;
732  days = temp_days;
733  seconds = temp_seconds;
734 
735  return (seconds - seconds_at_year_start);
736 }
