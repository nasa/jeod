283 /**
284  * Calculate Gregorian calendar date and 24-hour clock representation
285  * from Truncated Julian date/time.
286  *
287  * \par Assumptions and Limitations
288  * - Coverage is from March 1, 1600 onward.
289  * - Produces a time in 24-hour clock format.
290  * - Assumes that the values year, month, day, hour, minute, second, and
291  * truncated_julian_time are all present in the same class.
292  */
293 void
294 TimeStandard::calculate_calendar_values (
295  void)
296 {
297  int julian_day; /* -- integral part of julian time */
298  double temp; /* -- temporary value */
299 
300  /*separate julian time into the integral and fractional parts */
301  julian_day = static_cast<int> (trunc_julian_time);
302  if (julian_day < 0) {
303  julian_day -= 1;
304  }
305 
306  /* minutes in fractional day */
307  temp = 1440 * (trunc_julian_time - julian_day);
308  /* whole minutes in fractional day */
309  calendar_minute = static_cast<int> (temp);
310  /* seconds in final fractional minute */
311  calendar_second = 60 * (temp - calendar_minute);
312  if (calendar_second > 60 - clock_resolution) {
313  calendar_minute++;
314  if (calendar_minute >= 1440) {
315  julian_day++;
316  calendar_minute -= 1440;
317  }
318  calendar_second = 0;
319  }
320  /* integer division, multiples of 60 in total minutes */
321  calendar_hour = calendar_minute / 60;
322  /* subtract away the hour multiples of 60. */
323  calendar_minute -= (calendar_hour * 60);
324 
325  /* Note - could have done this with only one static_cast, but would have had
326  to go from trunc_julian_time direct to minutes, which would cause a
327  potential loss in precision. */
328 
329  /* convert the integral part into day, month,year format if day has changed */
330  if (julian_day != prev_julian_day) {
331  prev_julian_day = julian_day;
332  // determine the number of whole 400-year periods since March 1, 1600, and
333  // the number of days since the "most recent" 400-year boundary
334  julian_day += 134493; /* 134493 is the offset from March 1, 1600 to the
335  start of Truncated Julian Time */
336  int n_400 = julian_day / 146097; // There are 146097 days in 400 years. (number of whole 400-year periods)
337  if (julian_day < 0) {
338  n_400 -= 1;
339  }
340  int r_400 = julian_day - 146097 * n_400 + 1; /* -- number of years remaing after 400-year period */
341 
342  /* determine number of whole 100-year periods since the "most recent" 400-year
343  boundary, and number of days since the "most recent" 100-year boundary */
344  int n_100 = static_cast<int> (r_400 / 36524.3); /* -- number of whole 100-year periods */
345  int r_100 = r_400 - 36524 * n_100 - 1; /* -- number of years remaing after 100-year period */
346 
347  /* ditto for 4-year periods */
348  int n_4 = r_100 / 1461; /* -- number of whole 4-year periods */
349  int r_4 = r_100 - 1461 * n_4 + 1; /* -- number of years remaing after 4-year period */
350 
351  /* and for 1-year periods, omitting remaining-days calculation */
352  int n_1 = static_cast<int> (r_4 / 365.3); /* -- number of whole years */
353 
354  /* Modified r_4 for months calculation */
355  int r_4a = r_4 + 30 + 2 * n_1; /* -- modified value of r_4 */
356 
357  /* m = (Number of whole months +1) since the most recent 4-year period
358  (March=1, April=2, ... , February=12) */
359  int m = static_cast<int> (r_4a / 30.585);
360 
361 /* Determine day of month, month number (Jan=1, etc.), and year number (AD) */
362  calendar_day = r_4a - static_cast<int> (30.585 * m);
363  calendar_month = m + 2 - 12 * ((m + 1) / 12);
364  calendar_year = 1600 + 400 * n_400 + 100 * n_100 + 4 * n_4 +
365  ((m + 1) / 12);
366  }
367 
368  return;
369 }

402 /**
403  * Calculate Truncated Julian date/time from Gregorian calendar date and
404  * 24-hour clock representation.
405  *
406  * \par Assumptions and Limitations
407  * - Coverage s from March 1, 1600 onward.
408  * - Assumes that time is in 24-hour clock format; 1:00:00 pm cannot be read
409  * correctly, but 13:00:00 can.
410  * - Assumes that the values year, month, day, hour, minute, second, and
411  * truncated_julian_time are all present in the same class.
412  */
413 void
414 TimeStandard::convert_from_calendar (
415  void)
416 {
417  int m, /* adjusted month value */
418  n_1, /* -- number of whole years */
419  n_4, /* -- number of whole 4-year periods */
420  n_100, /* -- number of whole 100-year periods */
421  n_400, /* -- number of whole 400-year periods */
422  y, /* -- adjusted year value */
423  y_100, /* -- years remaining after subtracting off 100-yr multiples */
424  y_400; /* -- years remaining after subtracting off 400-yr multiples */
425 
426  /* adjust the year to represent the number of whole years since March 1, 1600 */
427  y = calendar_year - 1601 + (calendar_month + 9) / 12;
428 
429  /* Break down that number of years into a cumulative count of 400-year
430  periods, 100-year periods, 4-year periods, and 1-year periods. */
431  n_400 = y / 400;
432  y_400 = y - 400 * n_400;
433 
434  n_100 = y_400 / 100;
435  y_100 = y_400 - 100 * n_100;
436 
437  n_4 = y_100 / 4;
438  n_1 = y_100 - 4 * n_4;
439 
440  /* Calculate the number of (partial) months since the end of February in a
441  year that is divisible by 4 (even if it is not a leap year).
442  (March = 1,13,25,37; April = 2,14,26,38; ... ; February = 12,24,36,48) */
443  m = calendar_month - 2 + 12 * (1 - ((calendar_month + 9) / 12) + n_1);
444 
445  trunc_julian_time = calendar_day + floor (30.585 * m) - (2 * n_1) - 31 +
446  (1461 *
447  n_4) + (36524 * n_100) + (146097 * n_400) - 134493 +
448  (calendar_hour / 24.0) + (calendar_minute / 1440.0) +
449  (calendar_second / 86400.0);
450 
451  days = trunc_julian_time - tjt_at_epoch;
452  seconds = days * 86400;
453 
454  return;
455 }

623  // already failed in create_init_tree. But just in case:

635  // The only type without a parent is the initializer, and that is not
636  // initialized from this routine.
637  if (!parent_ptr->is_initialized()) {
638  parent_ptr->initialize_from_parent(time_manager_init);

642  converter index in TimeManagerInit converter registry */

657  initialize_from_name.c_str(),
658  name.c_str());
659  }
660  int conv_ptr_index = time_manager_init->get_conv_ptr_index (
661  conv_index);
662  TimeConverter * converter_ptr =
663  time_manager->get_converter_ptr (conv_ptr_index);
664  if (!converter_ptr->is_initialized()) {
665  converter_ptr->initialize (parent_ptr,
666  this,
667  conv_dir);
668  }
669

672  }
673  else if (conv_dir == -1) {
674  converter_ptr->convert_b_to_a();
675  }

678  __FILE__, __LINE__, TimeMessages::memory_error, "\n"

680  "Conv_dir for the converter has an invalid value (<-1 or > 1)",
681  name.c_str(),
682  initialize_from_name.c_str());
683  }
684  initialized = true;
685  initial_value = seconds;
686  return;

690 /**
691  * Generate the number of seconds elapsed this year
692  *
693  * \par Assumptions and Limitations
694  * - Relies on the accuracy of the JEOD2.0 calendar.
695  * @return Current second of year.
696  */
697 double
698 TimeStandard::seconds_of_year (
699  void)
700 {
701  if (!Numerical::compare_exact(last_calendar_update,time_manager->simtime)) {
702  calculate_calendar_values();
703  last_calendar_update = time_manager->simtime;
704  }
705 
706  if (calendar_year == year_of_last_soy) { // year has not changed since last update
707  return (seconds - seconds_at_year_start);
708  }
709  else if (((calendar_year - year_of_last_soy) == 1) &&
710  (calendar_month <= 2)) {
711  // year has ticked over, but less than 2 full months into new year
712  double seconds_of_year_value =
713  ((calendar_month - 1) * 31 + (calendar_day - 1)) * 86400 +
714  calendar_hour * 3600 + calendar_minute * 60 + calendar_second;
715  seconds_at_year_start = seconds - seconds_of_year_value;
716  year_of_last_soy = calendar_year;
717  return seconds_of_year_value;
718  }
719 
720  // Get here is this is the first time calculating seconds_of_year, or if year
721  // has ticked over backward, or if it is on or after March 1 in the year
722  // immediately following the last seconds_of_year calculation.
723  int temp_c_month = calendar_month;
724  int temp_c_day = calendar_day;
725  int temp_c_hour = calendar_hour;
726  int temp_c_minute = calendar_minute;
727  double temp_c_second = calendar_second;
728  double temp_tjt = trunc_julian_time;
729  double temp_days = days;
730  double temp_seconds = seconds;
731 
732  // set the time to be at year-start
733  calendar_month = 1;
734  calendar_day = 1;
735  calendar_hour = 0;
736  calendar_minute = 0;
737  calendar_second = 0.0;
738 
739  // convert to decimal, save off appropriate values
740  convert_from_calendar();
741  seconds_at_year_start = seconds;
742  year_of_last_soy = calendar_year;
743 
744  // revert to proper time
745  calendar_month = temp_c_month;
746  calendar_day = temp_c_day;
747  calendar_hour = temp_c_hour;
748  calendar_minute = temp_c_minute;
749  calendar_second = temp_c_second;
750  trunc_julian_time = temp_tjt;
751  days = temp_days;
752  seconds = temp_seconds;
753 
754  return (seconds - seconds_at_year_start);
755 }
