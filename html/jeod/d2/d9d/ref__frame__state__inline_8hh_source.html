172 /**
173  * Compute the angular velocity unit vector.
174  *
175  * \par Assumptions and Limitations
176  * - Angular velocity magnitude has already been computed.
177  */
178 inline void RefFrameRot::compute_ang_vel_unit()
179 {
180  if(std::fpclassify(ang_vel_mag) != FP_ZERO)
181  {
182  Vector3::scale(ang_vel_this, 1.0 / ang_vel_mag, ang_vel_unit);
183  }
184  else
185  {
186  Vector3::initialize(ang_vel_unit);
187  }
188 }
