74 void RelativeKinematics::add_relstate(RelativeDerivedState & relstate)

101 /**
102  * Remove a relative state from the list of ones maintained by this model.
103  * \param[in] relstate Relstate to remove
104  */
105 void RelativeKinematics::remove_relstate(RelativeDerivedState & relstate)
106 {
107  auto it = std::find(relative_states.begin(), relative_states.end(), &relstate);
108 
109  if(it == relative_states.end())
110  {
111  MessageHandler::error(__FILE__,
112  __LINE__,
113  RelKinMessages::entry_not_found,
114  "Relative state %s is not in the RelKin list",
115  relstate.name.c_str());
116  return;
117  }
118 
119  relative_states.erase(it);
120  --num_rel_states;
121 }

123 /**
124  * Find a specific relative state maintained by this model.
125  * @return Void
126  * \param[in] relstate_name Relstate to find
127  */
128 RelativeDerivedState * RelativeKinematics::find_relstate(const std::string & relstate_name)
129 {
130  RelativeDerivedState * found_relstate = nullptr;
131  unsigned int n_relstates = num_rel_states;
132 
133  for(unsigned int ii = 0; ii < n_relstates; ++ii)
134  {
135  if(relstate_name == relative_states[ii]->name)
136  {
137  found_relstate = relative_states[ii];
138  break;
139  }
140  }
141 
142  return found_relstate;
143 }
