76 /**
77  * Calculation of force and torque due to radiation pressure
78  *
79  * \par Assumptions and Limitations
80  * - Only called when flux_mag > 0
81  * \param[in] flux_mag incident flux (per unit area)\n Units: N/m
82  * \param[in] flux_struct_hat the flux unit vector in structural frame
83  * \param[in] calculate_forces on/off flag for whether to calculate forces.
84  */
85 void FlatPlateRadiationFacet::incident_radiation(const double flux_mag,
86  const double flux_struct_hat[3],
87  const bool calculate_forces)
88 {
89  sin_theta = -(Vector3::dot(normal, flux_struct_hat));
90  if(sin_theta < 0)
91  { /* plate is not illuminated */
92  return;
93  }
94  cx_area = base_facet->area * sin_theta;
95 
96  areaxflux_e = cx_area * flux_mag;
97  thermal.power_absorb += (1.0 - albedo) * areaxflux_e;
98 
99  if(!calculate_forces)
100  {
101  return;
102  }
103 
104  // Convert to momentum
105  areaxflux = areaxflux_e / speed_of_light;
106 
107  double tempvec1[3];
108  double tempvec2[3];
109  // absorption
110  Vector3::scale(flux_struct_hat, areaxflux * (1.0 - albedo), tempvec1);
111  Vector3::incr(tempvec1, F_absorption);
112 
113  // reflection
114  double ref_flux = areaxflux * albedo; // total reflected flux
115 
116  // diffuse reflection
117  Vector3::scale(normal, two_thirds, tempvec1), Vector3::diff(flux_struct_hat,
118  tempvec1,
119  tempvec2); // tempvec2 = phi_hat - 2/3 n_hat
120  Vector3::scale(diffuse * ref_flux, tempvec2);
121  Vector3::incr(tempvec2, F_diffuse);
122 
123  // specular reflection */
124  Vector3::scale(normal,
125  2 * (diffuse - 1) * ref_flux * sin_theta,
126  tempvec1); // (diffuse-1) < 0, this makes F opposite to
127  // normal
128  Vector3::incr(tempvec1, F_specular);
129 }

131 /**
132  * Calculates the radiative emission force, accumulated force, and torque
133  * acting on a facet.
134  */
135 void FlatPlateRadiationFacet::radiation_pressure()
136 {
137  if(thermal.integrable_object.active)
138  {
139  thermal.integrable_object.compute_temp_dot();
140  }
141 
142  if(thermal.power_emit < 0)
143  {
144  MessageHandler::fail(__FILE__,
145  __LINE__,
146  RadiationMessages::unknown_numerical_error,
147  "\n"
148  "On facet(%s), the calculation of emitted power yielded negative emission"
149  ",\n which is a non-physical situation corresponding to emission \n"
150  "producing a net gain in thermal energy.\n",
151  base_facet->name.c_str());
152  }
153  else
154  {
155  // If there is some thermal emission, convert it to an emission force and
156  // add it to the total plate force to give the final plate force
157  Vector3::scale(normal, -two_thirds * thermal.power_emit / speed_of_light, F_emission);
158  }
159  // Combining, accumulating
160  Vector3::sum(F_absorption, F_specular, F_diffuse, force);
161  Vector3::incr(F_emission, force);
162 
163  // - Reminder crot_to_cp is the moment arm from cg (center of rotation) to cp
164  // (center of pressure - effective center of facet)
165  Vector3::cross(crot_to_cp, force, torque);
166 }
