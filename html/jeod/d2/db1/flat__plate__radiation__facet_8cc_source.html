104 /**
105  * Calculation of force and torque due to radiation pressure
106  *
107  * \par Assumptions and Limitations
108  * - Only called when flux_mag > 0
109  * \param[in] flux_mag incident flux (per unit area)\n Units: N/m
110  * \param[in] flux_struct_hat the flux unit vector in structural frame
111  * \param[in] calculate_forces on/off flag for whether to calculate forces.
112  */
113 void
114 FlatPlateRadiationFacet::incident_radiation (
115  const double flux_mag,
116  const double flux_struct_hat[3],
117  const bool calculate_forces)
118 {
119  sin_theta = -(Vector3::dot (normal, flux_struct_hat));
120  if (sin_theta < 0) { /* plate is not illuminated */
121  return;
122  }
123  cx_area = base_facet->area * sin_theta;
124 
125  areaxflux_e = cx_area * flux_mag;
126  thermal.power_absorb += (1.0 - albedo) * areaxflux_e;
127 
128  if (!calculate_forces) {
129  return;
130  }
131 
132  // Convert to momentum
133  areaxflux = areaxflux_e / speed_of_light;
134 
135  double tempvec1[3];
136  double tempvec2[3];
137  // absorption
138  Vector3::scale (flux_struct_hat,
139  areaxflux * (1.0 - albedo),
140  tempvec1);
141  Vector3::incr( tempvec1,
142  F_absorption);
143 
144  // reflection
145  double ref_flux = areaxflux * albedo; // total reflected flux
146 
147  // diffuse reflection
148  Vector3::scale ( normal,
149  two_thirds,
150  tempvec1),
151  Vector3::diff ( flux_struct_hat,
152  tempvec1,
153  tempvec2 ); // tempvec2 = phi_hat - 2/3 n_hat
154  Vector3::scale ( diffuse * ref_flux,
155  tempvec2);
156  Vector3::incr ( tempvec2,
157  F_diffuse);
158 
159  // specular reflection */
160  Vector3::scale (normal,
161  2 * (diffuse - 1) * ref_flux * sin_theta,
162  tempvec1); // (diffuse-1) < 0, this makes F opposite to
163  // normal
164  Vector3::incr ( tempvec1,
165  F_specular);
166 
167  return;
168 }

172 /**
173  * Calculates the radiative emission force, accumulated force, and torque
174  * acting on a facet.
175  */
176 void
177 FlatPlateRadiationFacet::radiation_pressure (
178  void)
179 {
180  if (thermal.integrable_object.active) {
181  thermal.integrable_object.compute_temp_dot();
182  }
183 
184  if (thermal.power_emit < 0) {
185  MessageHandler::fail (
186  __FILE__, __LINE__, RadiationMessages::unknown_numerical_error, "\n"
187  "On facet(%s), the calculation of emitted power yielded negative emission"
188  ",\n which is a non-physical situation corresponding to emission \n"
189  "producing a net gain in thermal energy.\n", base_facet->name.c_str());
190  }
191  else {
192  // If there is some thermal emission, convert it to an emission force and
193  // add it to the total plate force to give the final plate force
194  Vector3::scale (normal,
195  -two_thirds * thermal.power_emit / speed_of_light,
196  F_emission);
197  }
198  // Combining, accumulating
199  Vector3::sum ( F_absorption,
200  F_specular,
201  F_diffuse,
202  force);
203  Vector3::incr (F_emission,
204  force);
205 
206  // - Reminder crot_to_cp is the moment arm from cg (center of rotation) to cp
207  // (center of pressure - effective center of facet)
208  Vector3::cross (crot_to_cp,
209  force,
210  torque);
211 
212  return;
213 }
