85 /**

87  * in the form of a tree.

354  void attach(Links & new_parent)
355  {

410  /**
411  * Determine if a node is the progeny of another.
412  * @param target Target links object
413  * @return True if target is an ancestor of this node, false otherwise.
414  */
415  bool is_progeny_of(const Links & target) const
416  {
417  bool result = false;
418 
419  auto link_matcher = [&target](const Links * link)
420  {
421  return link == &target;
422  };
423 
424  if(std::any_of(path_to_node_.begin(), path_to_node_.end(), link_matcher))
425  {
426  result = true;
427  }
428 
429  return result;

432  /**
433  * Find the index of the node that represents the point of departure
434  * in the tree containing two nodes.
435  * @param target Some other node in the tree
436  * @return Index of the last common node
437  */
438  int find_last_common_index(const Links & target) const
439  {
440  int last_common;
441 
442  // Both links must have a path and must be part of the same tree. If this
443  // is not the case, return a negative value to indicate failure.
444  if((path_length() == 0) || (target.path_length() == 0) || (path_to_node_[0] != target.path_to_node_[0]))
445  {
446  last_common = -1;
447  }
448 
449  else
450  {
451  // Determine the maximum possible index as the shortest length - 1.
452 
453  int max_index = std::min(path_length(), target.path_length()) - 1;
454 
455  // Paths are common up to max_index: This is the last common node.
456  if(path_to_node_[max_index] == target.path_to_node_[max_index])
457  {
458  last_common = max_index;
459  }
460 
461  // Paths diverge before end: Find point of departure.
462  else
463  {
464  for(last_common = 0; path_to_node_[last_common + 1] == target.path_to_node_[last_common + 1];
465  ++last_common)
466  {
467  ; // Empty -- Loop will terminate at last common index.
468  }
469  }
470  }
471 
472  return last_common;
473  }
474 
475  /**
476  * Find the node that represents the point of departure in the tree
477  * containing two nodes.
478  * @param target Some other node in the tree
479  * @return Pointer to last common node
480  */
481  const Links * find_last_common_node(const Links & target) const
482  {
483  int last_common = find_last_common_index(target);
484  if(last_common < 0)
485  {
486  return nullptr;
487  }
488  else
489  {
490  return path_to_node_[last_common];
491  }
492  }

494 protected:
495  /**
496  * Recursively construct the path_to_node.
497  */
498  void construct_path_to_node()
499  {
500  // Non-null parent: Resize the path to the parent's size + 1 and
501  // copy the parent's path to form the start of the new path.
502  if(parent_ != nullptr)
503  {
504  set_path_size(parent_->path_length() + 1);
505  std::copy(parent_->path_to_node_.begin(), parent_->path_to_node_.end(), path_to_node_.begin());
506  }
507 
508  // Null parent (root): Resize path to one.
509  else
510  {
511  set_path_size(1u);
512  }
513 
514  // The path always ends with this object.
515  path_to_node_[path_to_node_.size() - 1] = static_cast<Links *>(this);
516 
517  // Reconstruct the path for each child.
518  for(auto * child : children_)
519  {
520  child->construct_path_to_node();
521  }
522  }

526  * Add a frame as a child of the frame containing these links.
527  * @param new_parent The node to which this object is to be attached.
528  */
529  void attach_internal(Links & new_parent)
530  {
531  // Attach the child to the parent.
532  parent_ = &new_parent;
533 
534  // Add the child as a child of the parent.
535  new_parent.children_.push_back(static_cast<Links *>(this));

538  /**

540  */
541  void detach_internal()
542  {
543  // Sever the links between child and parent.
544  if(parent_ != nullptr)
545  {
546  parent_->children_.erase(std::find(parent_->children_.begin(), parent_->children_.end(), this));
547  }
548 
549  // Break links between this and former parent_.
550  parent_ = nullptr;
551  }
