70 /**
71  * Modified from original DCFODE subroutine.
72  * calculate_integration_coefficients is called by dstode to set coefficients
73  * needed there.
74  * The coefficients for the current method, as given by the value of
75  * integration_method, are set for all orders and saved.
76  * The maximum order assumed here is 12 if integration_method =
77  * ImplicitAdamsNonStiff and 5 if integration_method = ImplicitBackDiffStiff.
78  *
79  * NOTE - A smaller value of the maximum order is also allowed and may be set by
80  * the user with the value control_data.max_order, which gets copied to the
81  * protected value max_order_internal.
82  *
83  * calculate_integration_coefficients is called once at the beginning of the
84  * problem, and again only if integration_method is changed.
85  *
86  * The coefficients are stored in two arrays:
87  * method_coeffs_complete is a 13x12 array that contains a complete set of
88  * coefficients for the method
89  * test_coeffs_complete is a 3x12 array that contains the coefficients for local
90  * error tests and selection of the step size and/or order.
91  * The 1st set of 12 coeffs is for order method_order_current - 1
92  * The 2nd set of 12 coeffs is for order method_order_current
93  * The 3rd set of 12 coeffs is for order method_order_current + 1
94  *
95  * The coefficients in method_coeffs_complete are computed by a genetrating
96  * polynomial. For a given order (note that order changes during the integration
97  * process up to the maximum allowable, and is identified in the integrator as the
98  * variable method_order_current), abbreviate
99  * method_coeffs_complete[i][order-1] to mcc[i]
100  *
101  * Then
102  * l(x) = mcc[0] + (mcc[1] * x) + (mcc[2] * x^2) + ... + mcc[order] * (x^order)
103  * For the implicit Adams methods, l(x) is given by
104  * dl/dx = (x+1)*(x+2)*...*(x+order-1)/(order-1)!, l(-1) = 0.;
105  * For the BDF methods, l(x) is given by
106  * l(x) = (x+1)*(x+2)* ... *(x+order) / ((order)! * (1 + 1/2 + ... + 1/order))
107  *
108  * (! represents factorial)
109  *
110  * Note that while method_coeffs_compelte is a rectangular array for convenience,
111  * it is effectively a triangular array since method_coeffs_complete[i][order-1]
112  * has no meaning for i >= ( order + 2)
113  * order=1: method_coeffs_complete[0-1][0]
114  * order=2: method_coeffs_complete[0-2][1]
115  * ...
116  * order=12: method_coeffs_complete[0-12][11]
117  * Hence a 13x12 array.
118  */
119 void LsodeFirstOrderODEIntegrator::calculate_integration_coefficients()
120 {
121  // NOTE poly_coeff is only a 12-array in the original fortran
122  // However, since:
123  // 1. indices in Fortran are 1-12, not 0-11, and
124  // 2. the index is integral to the calculation, and
125  // 3. I don't have time to refactor and validate,
126  // I'm going with a 13 array so that I can still have indices 1-12,
127  // and throw away index 0.
128  double poly_coeff[13] = {};
129  double rqfac, rq1fac;
130 
131  switch(control_data.integration_method)
132  {
133  case LsodeControlDataInterface::ImplicitAdamsNonStiff:
134 
135  // 100
136  // Reducing index on m_c_c and t_c_c, but not on poly_coeff.
137  method_coeffs_complete[0][0] = 1.0;
138  method_coeffs_complete[1][0] = 1.0;
139  test_coeffs_complete[0][0] = 0.0;
140  test_coeffs_complete[1][0] = 2.0;
141  test_coeffs_complete[0][1] = 1.0;
142  test_coeffs_complete[2][11] = 0.0;
143  poly_coeff[0] = 1.0;
144  rqfac = 1.0;
145 
146  // NOTE - NQ is used here as the index, but it is NOT passed in,
147  // and there is no common block in this method. NQ is this context
148  // does not appear to be associated with method_order_current.
149  // Leaving it as nq.
150  //
151  for(int nq = 2; nq <= 12; nq++) // do 140
152  {
153  // The oly_coeff array will contain the coefficients of the polynomial
154  // p(x) = (x+1)*(x+2)*...*(x+nq-1).;
155  // Initially, p(x) = 1.;
156  rq1fac = rqfac;
157  rqfac = rqfac / nq;
158 
159  // Form coefficients of p(x)*(x+nq-1). ------------------------
160  poly_coeff[nq - 1] = 0.0;
161  // In the original for loop, i ran from ((nq+1)-1) = nq down to
162  // ((nq+1)-(nq-1)) = 2. We are reducing the index by 1, so nq-1
163  // down to 1.
164  for(int ii = nq - 1; ii >= 1; ii--) // do 110
165  {
166  // 110
167  poly_coeff[ii] = poly_coeff[ii - 1] + (nq - 1) * poly_coeff[ii];
168  }
169  poly_coeff[0] = (nq - 1) * poly_coeff[0];
170 
171  // Compute integral, -1 to 0, of p(x) and x*p(x). ----------
172  double pint = poly_coeff[0];
173  double xpin = poly_coeff[0] / 2.0;
174  double tsign = 1.0;
175  for(int ii = 2; ii <= nq; ii++) // do 120
176  {
177  tsign = -tsign;
178  pint = pint + tsign * poly_coeff[ii - 1] / ii;
179  // 120
180  xpin = xpin + tsign * poly_coeff[ii - 1] / (ii + 1);
181  }
182  // Store coefficients in method_coeffs_complete (ELCO) and
183  // test_coeffs_complete (TESCO).
184  method_coeffs_complete[0][nq - 1] = pint * rq1fac;
185  method_coeffs_complete[1][nq - 1] = 1.0;
186  for(int ii = 2; ii <= nq; ii++) // do 130
187  {
188  // 130
189  method_coeffs_complete[ii][nq - 1] = rq1fac * poly_coeff[ii - 1] / ii;
190  }
191  double agamq = rqfac * xpin;
192  double ragq = 1.0 / agamq;
193  test_coeffs_complete[1][nq - 1] = ragq;
194  if(nq < 12)
195  {
196  test_coeffs_complete[0][nq] = ragq * rqfac / (nq + 1);
197  }
198  test_coeffs_complete[2][nq - 2] = ragq;
199  // 140
200  }
201  break;
202  case LsodeControlDataInterface::ImplicitBackDiffStiff:
203  // 200
204  poly_coeff[0] = 1.0;
205  rq1fac = 1.0;
206  for(int nq = 1; nq <= 5; nq++) // do 230
207  {
208  int nqp1 = nq + 1;
209  // Form coefficients of p(x)*(x+nq).
210  poly_coeff[nq] = 0.0;
211 
212  for(int ii = nq; ii >= 1; ii--) // do 210
213  {
214  // 210
215  poly_coeff[ii] = poly_coeff[ii - 1] + nq * poly_coeff[ii];
216  }
217  poly_coeff[0] = nq * poly_coeff[1];
218  // Store coefficients in method_coeffs_complete (ELCO) and
219  // test_coeffs_complete (TESCO).
220  for(int ii = 0; ii <= nq; ii++) // do 220
221  {
222  // 220
223  method_coeffs_complete[ii][nq - 1] = poly_coeff[ii] / poly_coeff[1];
224  }
225  method_coeffs_complete[1][nq - 1] = 1.0;
226  test_coeffs_complete[0][nq - 1] = rq1fac;
227  test_coeffs_complete[1][nq - 1] = nqp1 / method_coeffs_complete[0][nq - 1];
228  test_coeffs_complete[2][nq - 1] = (nq + 2) / method_coeffs_complete[0][nq - 1];
229  rq1fac = rq1fac / nq;
230  // 230
231  }
232  break;
233  default:
234  break;
235  }
236 }

321 /**
322  * Modified from DPREPJ.
323  * DPREPJ was called by DSTODE to compute and process the matrix
324  * P = I - h*el(1)*J , where J is an approximation to the Jacobian.
325  *
326  * NOTES
327  *
328  * DPREPJ has been split into 3 parts, book-ended by the external-calls.
329  * jacobian_prep_init contains the code that precededs the first external-call
330  * jacobian_prep_loop contains the code that continues to loop according to the
331  * limits as written is DPREPJ.
332  * jacobian_prep_wrap_up contains the code that follows successful completion of
333  * the looping section of DPREPJ.
334  *
335  * Note that the division in this implementation is not linear with that in the
336  * original Fortran. The external calls within the original fortran are embedded
337  * within switch-blocks and for loops; the return points - to go to the next
338  * routine will pick up from one of several locations in the Fortran code,
339  * depending on the configuration at the time the external call was made.
340  *
341  * For the ER7_Utils / JEOD / Trick implementation, the derivative/jac calls must
342  * be external to the integrate call, so we must fully back out and then reenter.
343  * Rentry goes to jacobian_prep_loop.
344  *
345  * Some variables that were local have been moved to the class so that their value
346  * is not lost in going from jacobian_prep_init to jacobian_prep_loop.
347  * These are typically identified with ***_dprepj to indicate that their sole
348  * purpose is within dprepj (the original name of the jacobian_prep_* routines).
349  *
350  * Here the jacobian is computed by the user-supplied routine JAC if
351  * corrector_method = NewtonIterUserJac or NewtonIterUserBandJac, or by finite
352  * differencing if corrector_method = NewtonIterInternalJac,
353  * JacobiNewtonInternalJac, or NewtonIterInternalBandJac.
354  *
355  * If corrector_method = JacobiNewtonInternalJac, a diagonal approximation to the
356  * Jacobian is used.
357  * The Jacobian is stored in arrays lin_alg
358  *
359  * If corrector_method != JacobiNewtonInternalJac, P is subjected to
360  * LU decomposition in preparation for later solution of linear systems with P as
361  * coefficient matrix.
362  * This is done by gauss_elim_factor (DGEFA) if
363  * corrector_method = NewtonIterUserJac or NewtonIterInternalJac,
364  * and by linear_solver (DGBFA) if
365  * corrector_method = NewtonIterUserBandJac or NewtonIterInternalBandJac.
366  *
367  * Note that the corrector_method using the banded Jacobians is not supported in
368  * this release, so linear_solver is not used.
369  *
370  * FTEM and ACOR were effectively the same, now arrays.accum_correction.
371  * SAVF is now arrays.save.
372  * WM is now arrays.lin_alg
373  */
