40 void ForceConstraintComponent::setup_constraint(const VehicleProperties & vehicle_properties,
41  const VehicleNonGravState &)
42 {
43  // Set the constraint to be a pure unit force at the constraint position
44  // and in the constraint direction.
45  constraint_wrench.set_force(constraint_direction_root, constrained_mass->constraint_position_root);
46 
47  // Compute the vehicle's translational acceleration in response to this
48  // constraint at the vehicle's center of mass.
49  Vector3::scale(constraint_direction_root, vehicle_properties.get_inverse_mass(), constraint_com_accel_root);
50 
51  // Compute the vehicle's rotational acceleration in response to this
52  // constraint.
53  double torque[3];
54  Vector3::cross(constrained_mass->constraint_position_com, constraint_direction_root, torque);
55  compute_rotational_response(vehicle_properties, torque);
