136 /**
137  * Calculate the torque generated on the vehicle by the facet. Assumes
138  * that the force is in the vehicle structural frame, but that close_point is not.
139  * \param[in] tmp_force force from one contact interaction.\n Units: N
140  */
141 void PointContactFacet::calculate_torque(double * tmp_force)
142 {
143  double cm[3];
144  double vec[3];
145  double tmp_torque[3];
146 
147  // find the vector from vehicle point to center of mass
148  Vector3::diff(vehicle_point->mass_point->position, vehicle_body->mass.composite_properties.position, cm);
149 
150  // rotate facet contact point to vehicle structure frame
151  Vector3::transform_transpose(vehicle_point->state.rot.T_parent_this, contact_point, vec);
152  Vector3::transform(vehicle_body->structure.state.rot.T_parent_this, vec, tmp_torque);
153  // find vector from cm to contact point
154  Vector3::sum(tmp_torque, cm, cm);
155  // cross cm vector with the rotated force to get the torque
156  Vector3::cross(cm, tmp_force, tmp_torque);
157  Vector3::sum(torque, tmp_torque, torque);
158 }
