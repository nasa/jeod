414 /**
415  * Initialize the De4xxEphemeris item data.
416  * \param[in,out] ephem_manager Ephemerides manager
417  */
418 void
419 De4xxEphemeris::initialize_items (
420  EphemeridesManager & ephem_manager)
421 {
422 
423  // Process the user-inputable selected_items array.
424  for (uint32_t ii = 0; ii < De4xxBase::number_jeod_items(file.file_spec.get_model_number()); ++ii) {
425  // Mark deselected items (and items that are not present) as deselected.
426  if ((! selected_items[ii]) || (item_data[ii].item == nullptr)) {
427  if (item_data[ii].item != nullptr) {
428  item_data[ii].item->disable ();
429  }
430  item_data[ii].status = De4xxEphemItem::Deselected;
431  }
432 
433  // Initially mark selected items as inactive.
434  // This status will be updated later.
435  else {
436  item_data[ii].status = De4xxEphemItem::Inactive;
437  }
438  }
439 
440  // Sanity check: The Earth and Moon must be set by the same model.
441  if (item_data[De4xxBase::De4xx_Ephem_Earth].status !=
442  item_data[De4xxBase::De4xx_Ephem_Moon].status) {
443  MessageHandler::fail (
444  __FILE__, __LINE__, EphemeridesMessages::inconsistent_setup,
445  "The Earth and Moon must be set by the same model.");
446  }
447 
448  // Tell the EphemeridesManager about each selected ephemeris item.
449  // While the constructor enables each modeled ephemeris item,
450  // these can be turned off using the selected_items data member.
451 
452  // Register the solar system and Earth-Moon barycenter frames with the
453  // EphemeridesManager.
454  if ((item_data[De4xxBase::De4xx_Ephem_SSbary].status !=
455  De4xxEphemItem::Deselected) &&
456  (ephem_manager.find_ref_frame (
457  solar_system_barycenter_frame.get_name()) == nullptr)) {
458  ephem_manager.add_integ_frame (solar_system_barycenter_frame);
459  }
460 
461  if ((item_data[De4xxBase::De4xx_Ephem_EMbary].status !=
462  De4xxEphemItem::Deselected) &&
463  (ephem_manager.find_ref_frame (
464  earth_moon_barycenter_frame.get_name()) == nullptr)) {
465  ephem_manager.add_integ_frame (earth_moon_barycenter_frame);
466  }
467 
468  // Tell the EphemeridesManager about the ephemeris points.
469  for ( uint32_t ii = 0;
470  ii < De4xxBase::number_trans_points(file.file_spec.get_model_number());
471  ++ii)
472  {
473  if (item_data[ii].status != De4xxEphemItem::Deselected) {
474  ephem_manager.add_ephem_item (points[ii]);
475  }
476  }
477 
478  // Finally register the lunar orientation ("lunar RNP").
479  if (item_data[De4xxBase::De4xx_Ephem_LLibration].status !=
480  De4xxEphemItem::Deselected) {
481  ephem_manager.add_ephem_item (lunar_orientation);
482  }
483 }

522  else {
523  item_data[ii].status = De4xxEphemItem::Deselected;
524  }

530  * Mark appropriate items in the model as active.
531  * @return Void
532  */
533 unsigned int
534 De4xxEphemeris::activate_nodes (
535  void)
536 {
537  unsigned int tot_active;
538 
539  // Get the basic status for each item.
540  tot_active = 0;
541  nactive_items = 0;
542  for (uint32_t ii = 0; ii < De4xxBase::number_jeod_items(file.file_spec.get_model_number()); ++ii) {
543  De4xxEphemItem & item = item_data[ii];
544  if (item.status != De4xxEphemItem::Deselected) {
545  item.enabled_item = item.item->get_enabled_item ();
546  if ((item.enabled_item == nullptr) ||
547  (! item.enabled_item->is_active ())) {
548  item.status = De4xxEphemItem::Inactive;
549  }
550  else if (item.enabled_item != item.item) {
551  item.status = De4xxEphemItem::InTree;
552  }
553  else {
554  item.status = De4xxEphemItem::Active;
555  nactive_items++;
556  }
557  }
558  else {
559  item.enabled_item = nullptr;
560  }
561 
562  if ((ii < De4xxBase::number_trans_points(file.file_spec.get_model_number())) &&
563  (item.status > De4xxEphemItem::Inactive)) {
564  tot_active++;
565  }
566  }
567 
568  return tot_active;

572 /**
573  * Adjust Earth, Moon, and Earth-Moon barycenter activity.
574  * @return Void
575  * \param[in] tot_active Number active translation nodes
576  */
577 unsigned int
578 De4xxEphemeris::activate_em_nodes (
579  unsigned int tot_active)
580 {
581  static const De4xxBase::De4xxEphemBodies embodies[3] = {
582  De4xxBase::De4xx_Ephem_Earth,
583  De4xxBase::De4xx_Ephem_Moon,
584  De4xxBase::De4xx_Ephem_EMbary
585  };
586  unsigned int embcount;
587 
588 
589  // Count how many of the Earth, Moon, and Earth-Moon barycenter are active.
590  embcount = 0;
591  for (unsigned int ii = 0; ii < 3; ++ii) {
592  De4xxEphemItem & item = item_data[embodies[ii]];
593  if (item.status > De4xxEphemItem::Inactive) {
594  embcount++;
595  }
596  }
597 
598  // Make all three of the Earth, Moon, and Earth-Moon barycenter active
599  // if any two of them are active OR
600  // if only the Earth or Moon is active and
601  // some other solar system body is active.
602  if ((embcount == 2) ||
603  ((embcount == 1) &&
604  (item_data[De4xxBase::De4xx_Ephem_EMbary].status <=
605  De4xxEphemItem::Inactive) &&
606  (tot_active > 1))) {
607 
608  for (unsigned int ii = 0; ii < 3; ++ii) {
609  De4xxEphemItem & item = item_data[embodies[ii]];
610  if (item.status == De4xxEphemItem::Deselected) {
611  MessageHandler::fail (
612  __FILE__, __LINE__, EphemeridesMessages::inconsistent_setup,
613  "Item '%s' needs to be activated but is deselected",
614  item.name);
615  }
616  else if (item.enabled_item == nullptr) {
617  item.enabled_item = item.item;
618  item.item->enable();
619  item.item->activate();
620  item.status = De4xxEphemItem::Active;
621  nactive_items++;
622  tot_active++;
623  }
624  else if (item.status == De4xxEphemItem::Inactive) {
625  item.enabled_item->activate();
626  if (item.enabled_item != item.item) {
627  item.status = De4xxEphemItem::InTree;
628  }
629  else {
630  item.status = De4xxEphemItem::Active;
631  nactive_items++;
632  }
633  tot_active++;
634  }
635  }
636  }
637 
638  return tot_active;
639 }

645 void
646 De4xxEphemeris::determine_root_node (
647  void)
648 {
649  int root_index;
650  unsigned int active_nodes[De4xxBase::number_trans_points(file.file_spec.get_model_number())];
651  unsigned int tot_active;
652 
653 
654  // Build a list of active planetary nodes.
655  tot_active = 0;
656  for (uint32_t ii = 0; ii < De4xxBase::number_trans_points(file.file_spec.get_model_number()); ++ii) {

662  // Determine the root point of the reference frame tree.
663  // Nothing is active: there is no root.
664  if (tot_active == 0) {
665  root_index = -1;
666  }
667 
668  // One item active: That active item is the root.
669  else if (tot_active == 1) {
670  root_index = active_nodes[0];
671  }
672 
673  // Activity is within the Earth-Moon barycenter: EMbary is the root.
674  else if ((tot_active == 3) &&
675  (item_data[De4xxBase::De4xx_Ephem_EMbary].status >
676  De4xxEphemItem::Inactive) &&
677  (item_data[De4xxBase::De4xx_Ephem_Earth].status >
678  De4xxEphemItem::Inactive) &&
679  (item_data[De4xxBase::De4xx_Ephem_Moon].status >
680  De4xxEphemItem::Inactive)) {
681  root_index = De4xxBase::De4xx_Ephem_EMbary;
682  }
683 
684  // All other cases: The root is the solar system barycenter.
685  else {
686  root_index = De4xxBase::De4xx_Ephem_SSbary;
687  }
688 
689 
690  // The root item does not need ephemeris.
691  // Its state is tautologically zero.
692  if (root_index >= 0) {
693  root_item = &item_data[root_index];
694  if (root_item->item == root_item->enabled_item) {
695  (dynamic_cast<EphemerisPoint*>(root_item->item))->initialize_state ();
696  nactive_items--;
697  }
698  root_item->status = De4xxEphemItem::IsRoot;
699  }
700 
701  else {
702  root_item = nullptr;
703  }
704 }

707 /**
708  * Mark appropriate items in the model as active.
709  * \param[in,out] ephem_manager Ephemerides manager
710  */
711 void
712 De4xxEphemeris::ephem_activate (
713  EphemeridesManager & ephem_manager JEOD_UNUSED)
714 {
715  unsigned int tot_active;
716 
717  // Nothing to do if the model is inactive.
718  if (! active) {
719  return;
720  }
721 
722  // Activate nodes based on the activity of the ephemeris points and
723  // based on the DE4xx reference frame tree.
724  tot_active = activate_nodes ();
725  activate_em_nodes (tot_active);
726 
727  // Determine the root point of the reference frame tree.
728  determine_root_node ();
729 
730 
731  // Count the number of active frames that belong to this model.
732  nactive_items = 0;
733  for (uint32_t ii = 0; ii < De4xxBase::number_jeod_items(file.file_spec.get_model_number()); ++ii) {
734  if (item_data[ii].status == De4xxEphemItem::Active) {
735  nactive_items++;
736  }
737  }
738 
739 
740  // Synchronize the file activity with this model's activity.
741  for (int ii = De4xxBase::De4xx_Ephem_Sun;
742  ii <= De4xxBase::De4xx_Ephem_EMbary;
743  ++ii) {
744  De4xxBase::De4xxFileEntries file_id = (De4xxBase::De4xxFileEntries)body_to_file_idx[ii];
745  if (file_id < file.io.metaData->number_file_items) {
746  file.item[file_id].active =
747  (item_data[ii].status == De4xxEphemItem::Active);
748  }
749  }
750 
751  // Librations = lunar orientation.
752  file.item[De4xxBase::De4xx_File_LLibration].active =
753  (item_data[De4xxBase::De4xx_Ephem_LLibration].status ==
754  De4xxEphemItem::Active);
755 
756  // This model doesn't use the nutations.
757  file.item[De4xxBase::De4xx_File_ENutation].active = false;
758 
759  // Force an update to reflect the change in status.
760  force_update = true;
761 
762  return;
763 }

766 /**
767  * Construct the ephemeris model portions of the reference frame tree.
768  * \param[in,out] ephem_manager Ephemerides manager
769  */
770 void
771 De4xxEphemeris::ephem_build_tree (
772  EphemeridesManager & ephem_manager)
773 {
774 
775  // Nothing to do if the model is inactive.
776  if (! active) {
777  return;
778  }
779 
780  // Remove all owned frames from the tree.
781  for ( uint32_t ii = 0;
782  ii < De4xxBase::number_trans_points(file.file_spec.get_model_number());
783  ++ii)
784  {
785  if ((item_data[ii].status != De4xxEphemItem::Deselected) &&
786  (item_data[ii].item == item_data[ii].enabled_item)) {
787  item_data[ii].frame->remove_from_parent ();
788  }
789  }
790 
791  // Set the root of the reference frame tree.
792  if ((root_item != nullptr) &&
793  (root_item->item == root_item->enabled_item)) {
794  ephem_manager.add_frame_to_tree (*(root_item->frame), nullptr);
795  }
796 
797  // Add non-root items to the tree.
798  for ( uint32_t ii = 0;
799  ii < De4xxBase::number_trans_points(file.file_spec.get_model_number());
800  ++ii)
801  {
802  De4xxEphemItem & item = item_data[ii];
803  if (item.status == De4xxEphemItem::Active) {
804  EphemerisRefFrame * parent_frame;
805  if ((ii == De4xxBase::De4xx_Ephem_Earth) ||
806  (ii == De4xxBase::De4xx_Ephem_Moon)) {
807  parent_frame = item_data[De4xxBase::De4xx_Ephem_EMbary].frame;
808  }
809  else {
810  parent_frame = item_data[De4xxBase::De4xx_Ephem_SSbary].frame;
811  }
812  if (parent_frame == nullptr) {
813  MessageHandler::fail (
814  __FILE__, __LINE__, EphemeridesMessages::inconsistent_setup,
815  "Unable to find parent frame for '%s'",
816  item.name);
817 
818  // Not reached
819  return;
820  }
821  ephem_manager.add_frame_to_tree (*(item.frame), parent_frame);
822  }
823  }
824 }

827 /**
828  * Update ephemerides for subscribed items.
829  */
830 void
831 De4xxEphemeris::ephem_update (
832  void)
833 {
834  static const double null_vec[3] = {0.0, 0.0, 0.0};
835 
836  // Nothing to do if
837  // - the model isn't active or
838  // - neither time nor the tree has changed.
839  if ((! active) ||
840  ((! force_update) && (Numerical::compare_exact(update_time,time_dyn->seconds)))) {
841  return;
842  }
843 
844  update_time = time_dyn->seconds;
845  force_update = false;
846 
847  // Update the root if it is owned by this model.
848  if ((root_item != nullptr) &&
849  (root_item->item == root_item->enabled_item)) {
850  EphemerisPoint * point = dynamic_cast<EphemerisPoint*>(root_item->item);
851  point->update (null_vec, null_vec, update_time);
852  }
853 
854  if (nactive_items > 0) {
855 
856  // Update the ephemeris file to reflect the current ephemeris time.
857  // FIXME: Switch to J2000
858  file.update (time_tt->trunc_julian_time * 86400.0);
859 
860  // Extract the requisite data directly represented in the ephemeris file
861  // from the ephemeris file object.
862  // Note: The Earth and Moon need special treatment (see below).
863  for (unsigned int ii = De4xxBase::De4xx_Ephem_Sun;
864  ii <= De4xxBase::De4xx_Ephem_EMbary;
865  ++ii) {
866 
867  // Point is directly represented and is needed:
868  // Update state based on the ephemeris file data for this object.
869  if ((ii != De4xxBase::De4xx_Ephem_Earth) &&
870  (ii != De4xxBase::De4xx_Ephem_Moon) &&
871  (item_data[ii].status == De4xxEphemItem::Active)) {
872  EphemerisPoint * point =
873  dynamic_cast<EphemerisPoint*>(item_data[ii].item);
874  De4xxFileItem & file_item = file.item[body_to_file_idx[ii]];
875  point->update (file_item.state[0], file_item.state[1], update_time);
876  }
877  }
878 
879  // Compute Earth and Moon states to reflect the ref frame tree structure.
880  // The Moon state is relative to the Earth in the ephemeris file.
881  if ((item_data[De4xxBase::De4xx_Ephem_Earth].status ==
882  De4xxEphemItem::Active) &&
883  (item_data[De4xxBase::De4xx_Ephem_Moon].status ==
884  De4xxEphemItem::Active)) {
885 
886  De4xxFileItem & moon_ephem =
887  file.item[body_to_file_idx[De4xxBase::De4xx_Ephem_Moon]];
888 
889  points[De4xxBase::De4xx_Ephem_Earth].update_scaled (
890  moon_ephem.state[0], moon_ephem.state[1],
891  -file.header.be_em_dist_ratio, update_time);
892 
893  points[De4xxBase::De4xx_Ephem_Moon].update_scaled (
894  moon_ephem.state[0], moon_ephem.state[1],
895  file.header.bm_em_dist_ratio, update_time);
896  }
897 
898  // Lunar orientation needed:
899  // Extract the lunar orientation Euler angles from the ephemeris file
900  // and update the lunar orientation.
901  if (item_data[De4xxBase::De4xx_Ephem_LLibration].status ==
902  De4xxEphemItem::Active) {
903  De4xxFileItem & file_angles =
904  file.item[De4xxBase::De4xx_File_LLibration];
905  lunar_orientation.update (file_angles.state[0], file_angles.state[1],
906  update_time);
907  }
908  }
909 }
