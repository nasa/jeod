320 /**
321  * Initialize the De4xxEphemeris item data.
322  * \param[in,out] ephem_manager Ephemerides manager
323  */
324 void De4xxEphemeris::initialize_items(EphemeridesManager & ephem_manager)
325 {
326  // Process the user-inputable selected_items array.
327  for(uint32_t ii = 0; ii < De4xxBase::number_jeod_items(file.file_spec.get_model_number()); ++ii)
328  {
329  // Mark deselected items (and items that are not present) as deselected.
330  if((!selected_items[ii]) || (item_data[ii].item == nullptr))
331  {
332  if(item_data[ii].item != nullptr)
333  {
334  item_data[ii].item->disable();
335  }
336  item_data[ii].status = De4xxEphemItem::Deselected;
337  }
338 
339  // Initially mark selected items as inactive.
340  // This status will be updated later.
341  else
342  {
343  item_data[ii].status = De4xxEphemItem::Inactive;
344  }
345  }
346 
347  // Sanity check: The Earth and Moon must be set by the same model.
348  if(item_data[De4xxBase::De4xx_Ephem_Earth].status != item_data[De4xxBase::De4xx_Ephem_Moon].status)
349  {
350  MessageHandler::fail(__FILE__,
351  __LINE__,
352  EphemeridesMessages::inconsistent_setup,
353  "The Earth and Moon must be set by the same model.");
354  }
355 
356  // Tell the EphemeridesManager about each selected ephemeris item.
357  // While the constructor enables each modeled ephemeris item,
358  // these can be turned off using the selected_items data member.
359 
360  // Register the solar system and Earth-Moon barycenter frames with the
361  // EphemeridesManager.
362  if((item_data[De4xxBase::De4xx_Ephem_SSbary].status != De4xxEphemItem::Deselected) &&
363  (ephem_manager.find_ref_frame(solar_system_barycenter_frame.get_name()) == nullptr))
364  {
365  ephem_manager.add_integ_frame(solar_system_barycenter_frame);
366  }
367 
368  if((item_data[De4xxBase::De4xx_Ephem_EMbary].status != De4xxEphemItem::Deselected) &&
369  (ephem_manager.find_ref_frame(earth_moon_barycenter_frame.get_name()) == nullptr))
370  {
371  ephem_manager.add_integ_frame(earth_moon_barycenter_frame);
372  }
373 
374  // Tell the EphemeridesManager about the ephemeris points.
375  for(uint32_t ii = 0; ii < De4xxBase::number_trans_points(file.file_spec.get_model_number()); ++ii)
376  {
377  if(item_data[ii].status != De4xxEphemItem::Deselected)
378  {
379  ephem_manager.add_ephem_item(points[ii]);
380  }
381  }
382 
383  // Finally register the lunar orientation ("lunar RNP").
384  if(item_data[De4xxBase::De4xx_Ephem_LLibration].status != De4xxEphemItem::Deselected)
385  {
386  ephem_manager.add_ephem_item(lunar_orientation);
387  }
388 }

390 /**
391  * Complete the initialization process.
392  * This method should be called after all other ephemeris models have
393  * completed their basic initialization and after all planets have
394  * registered themselves with the ephemeris manager.
395  * \param[in,out] ephem_manager Ephemerides manager

400  if(!active)
401  {

433  * Mark appropriate items in the model as active.
434  * @return Void
435  */
436 unsigned int De4xxEphemeris::activate_nodes()
437 {
438  unsigned int tot_active;
439 
440  // Get the basic status for each item.
441  tot_active = 0;
442  nactive_items = 0;
443  for(uint32_t ii = 0; ii < De4xxBase::number_jeod_items(file.file_spec.get_model_number()); ++ii)
444  {
445  De4xxEphemItem & item = item_data[ii];
446  if(item.status != De4xxEphemItem::Deselected)
447  {
448  item.enabled_item = item.item->get_enabled_item();
449  if((item.enabled_item == nullptr) || (!item.enabled_item->is_active()))
450  {
451  item.status = De4xxEphemItem::Inactive;
452  }
453  else if(item.enabled_item != item.item)
454  {
455  item.status = De4xxEphemItem::InTree;
456  }
457  else
458  {
459  item.status = De4xxEphemItem::Active;
460  nactive_items++;
461  }
462  }
463  else
464  {
465  item.enabled_item = nullptr;
466  }
467 
468  if((ii < De4xxBase::number_trans_points(file.file_spec.get_model_number())) &&
469  (item.status > De4xxEphemItem::Inactive))
470  {
471  tot_active++;
472  }
473  }
474 
475  return tot_active;

478 /**
479  * Adjust Earth, Moon, and Earth-Moon barycenter activity.
480  * @return Void
481  * \param[in] tot_active Number active translation nodes
482  */
483 unsigned int De4xxEphemeris::activate_em_nodes(unsigned int tot_active)
484 {
485  static const De4xxBase::De4xxEphemBodies embodies[3] = {De4xxBase::De4xx_Ephem_Earth,
486  De4xxBase::De4xx_Ephem_Moon,
487  De4xxBase::De4xx_Ephem_EMbary};
488  unsigned int embcount;
489 
490  // Count how many of the Earth, Moon, and Earth-Moon barycenter are active.
491  embcount = 0;
492  for(auto embodie : embodies)
493  {
494  De4xxEphemItem & item = item_data[embodie];
495  if(item.status > De4xxEphemItem::Inactive)
496  {
497  embcount++;
498  }
499  }
500 
501  // Make all three of the Earth, Moon, and Earth-Moon barycenter active
502  // if any two of them are active OR
503  // if only the Earth or Moon is active and
504  // some other solar system body is active.
505  if((embcount == 2) ||
506  ((embcount == 1) && (item_data[De4xxBase::De4xx_Ephem_EMbary].status <= De4xxEphemItem::Inactive) &&
507  (tot_active > 1)))
508  {
509  for(auto embodie : embodies)
510  {
511  De4xxEphemItem & item = item_data[embodie];
512  if(item.status == De4xxEphemItem::Deselected)
513  {
514  MessageHandler::fail(__FILE__,
515  __LINE__,
516  EphemeridesMessages::inconsistent_setup,
517  "Item '%s' needs to be activated but is deselected",
518  item.name.c_str());
519  }
520  else if(item.enabled_item == nullptr)
521  {
522  item.enabled_item = item.item;
523  item.item->enable();
524  item.item->activate();
525  item.status = De4xxEphemItem::Active;
526  nactive_items++;
527  tot_active++;
528  }
529  else if(item.status == De4xxEphemItem::Inactive)
530  {
531  item.enabled_item->activate();
532  if(item.enabled_item != item.item)
533  {
534  item.status = De4xxEphemItem::InTree;
535  }
536  else
537  {
538  item.status = De4xxEphemItem::Active;
539  nactive_items++;
540  }
541  tot_active++;
542  }
543  }
544  }
545 
546  return tot_active;
547 }

549 /**
550  * Determine which item should be the root of the ref frame tree.
551  */
552 void De4xxEphemeris::determine_root_node()
553 {
554  int root_index;
555  unsigned int active_nodes[De4xxBase::number_trans_points(file.file_spec.get_model_number())];
556  unsigned int tot_active;
557 
558  // Build a list of active planetary nodes.
559  tot_active = 0;
560  for(uint32_t ii = 0; ii < De4xxBase::number_trans_points(file.file_spec.get_model_number()); ++ii)
561  {
562  if(item_data[ii].status > De4xxEphemItem::Inactive)
563  {
564  active_nodes[tot_active++] = ii;
565  }
566  }
567 
568  // Determine the root point of the reference frame tree.
569  // Nothing is active: there is no root.
570  if(tot_active == 0)
571  {
572  root_index = -1;
573  }
574 
575  // One item active: That active item is the root.
576  else if(tot_active == 1)
577  {
578  root_index = active_nodes[0];
579  }
580 
581  // Activity is within the Earth-Moon barycenter: EMbary is the root.
582  else if((tot_active == 3) && (item_data[De4xxBase::De4xx_Ephem_EMbary].status > De4xxEphemItem::Inactive) &&
583  (item_data[De4xxBase::De4xx_Ephem_Earth].status > De4xxEphemItem::Inactive) &&
584  (item_data[De4xxBase::De4xx_Ephem_Moon].status > De4xxEphemItem::Inactive))
585  {
586  root_index = De4xxBase::De4xx_Ephem_EMbary;
587  }
588 
589  // All other cases: The root is the solar system barycenter.
590  else
591  {
592  root_index = De4xxBase::De4xx_Ephem_SSbary;
593  }
594 
595  // The root item does not need ephemeris.
596  // Its state is tautologically zero.
597  if(root_index >= 0)
598  {
599  root_item = &item_data[root_index];
600  if(root_item->item == root_item->enabled_item)
601  {
602  (dynamic_cast<EphemerisPoint *>(root_item->item))->initialize_state();
603  nactive_items--;
604  }
605  root_item->status = De4xxEphemItem::IsRoot;
606  }
607 
608  else
609  {
610  root_item = nullptr;
611  }
612 }

618 void De4xxEphemeris::ephem_activate(EphemeridesManager & ephem_manager JEOD_UNUSED)
619 {
620  unsigned int tot_active;
621 
622  // Nothing to do if the model is inactive.
623  if(!active)
624  {
625  return;
626  }
627 
628  // Activate nodes based on the activity of the ephemeris points and
629  // based on the DE4xx reference frame tree.
630  tot_active = activate_nodes();
631  activate_em_nodes(tot_active);
632 
633  // Determine the root point of the reference frame tree.
634  determine_root_node();
635 
636  // Count the number of active frames that belong to this model.
637  nactive_items = 0;
638  for(uint32_t ii = 0; ii < De4xxBase::number_jeod_items(file.file_spec.get_model_number()); ++ii)
639  {
640  if(item_data[ii].status == De4xxEphemItem::Active)
641  {
642  nactive_items++;
643  }
644  }
645 
646  // Synchronize the file activity with this model's activity.
647  for(int ii = De4xxBase::De4xx_Ephem_Sun; ii <= De4xxBase::De4xx_Ephem_EMbary; ++ii)
648  {
649  auto file_id = (De4xxBase::De4xxFileEntries)body_to_file_idx[ii];
650  if(file_id < file.io.metaData->number_file_items)
651  {
652  file.item[file_id].active = (item_data[ii].status == De4xxEphemItem::Active);
653  }
654  }
655 
656  // Librations = lunar orientation.
657  file.item[De4xxBase::De4xx_File_LLibration].active = (item_data[De4xxBase::De4xx_Ephem_LLibration].status ==
658  De4xxEphemItem::Active);
659 
660  // This model doesn't use the nutations.
661  file.item[De4xxBase::De4xx_File_ENutation].active = false;
662 
663  // Force an update to reflect the change in status.
664  force_update = true;
665 }

667 /**
668  * Construct the ephemeris model portions of the reference frame tree.
669  * \param[in,out] ephem_manager Ephemerides manager
670  */
671 void De4xxEphemeris::ephem_build_tree(EphemeridesManager & ephem_manager)
672 {
673  // Nothing to do if the model is inactive.
674  if(!active)
675  {
676  return;
677  }
678 
679  // Remove all owned frames from the tree.
680  for(uint32_t ii = 0; ii < De4xxBase::number_trans_points(file.file_spec.get_model_number()); ++ii)
681  {
682  if((item_data[ii].status != De4xxEphemItem::Deselected) && (item_data[ii].item == item_data[ii].enabled_item))
683  {
684  item_data[ii].frame->remove_from_parent();
685  }
686  }
687 
688  // Set the root of the reference frame tree.
689  if((root_item != nullptr) && (root_item->item == root_item->enabled_item))
690  {
691  ephem_manager.add_frame_to_tree(*(root_item->frame), nullptr);
692  }
693 
694  // Add non-root items to the tree.
695  for(uint32_t ii = 0; ii < De4xxBase::number_trans_points(file.file_spec.get_model_number()); ++ii)
696  {
697  De4xxEphemItem & item = item_data[ii];
698  if(item.status == De4xxEphemItem::Active)
699  {
700  EphemerisRefFrame * parent_frame;
701  if((ii == De4xxBase::De4xx_Ephem_Earth) || (ii == De4xxBase::De4xx_Ephem_Moon))
702  {
703  parent_frame = item_data[De4xxBase::De4xx_Ephem_EMbary].frame;
704  }
705  else
706  {
707  parent_frame = item_data[De4xxBase::De4xx_Ephem_SSbary].frame;
708  }
709  if(parent_frame == nullptr)
710  {
711  MessageHandler::fail(__FILE__,
712  __LINE__,
713  EphemeridesMessages::inconsistent_setup,
714  "Unable to find parent frame for '%s'",
715  item.name.c_str());
716 
717  // Not reached
718  return;
719  }
720  ephem_manager.add_frame_to_tree(*(item.frame), parent_frame);
721  }
722  }
723 }

725 /**
726  * Update ephemerides for subscribed items.
727  */
728 void De4xxEphemeris::ephem_update()
729 {
730  static const double null_vec[3] = {0.0, 0.0, 0.0};
731 
732  // Nothing to do if
733  // - the model isn't active or
734  // - neither time nor the tree has changed.
735  if((!active) || ((!force_update) && (Numerical::compare_exact(update_time, time_dyn->seconds))))
736  {
737  return;
738  }
739 
740  update_time = time_dyn->seconds;
741  force_update = false;
742 
743  // Update the root if it is owned by this model.
744  if((root_item != nullptr) && (root_item->item == root_item->enabled_item))
745  {
746  auto * point = dynamic_cast<EphemerisPoint *>(root_item->item);
747  point->update(null_vec, null_vec, update_time);
748  }
749 
750  if(nactive_items > 0)
751  {
752  // Update the ephemeris file to reflect the current ephemeris time.
753  // FIXME: Switch to J2000
754  file.update(time_tt->trunc_julian_time * 86400.0);
755 
756  // Extract the requisite data directly represented in the ephemeris file
757  // from the ephemeris file object.
758  // Note: The Earth and Moon need special treatment (see below).
759  for(unsigned int ii = De4xxBase::De4xx_Ephem_Sun; ii <= De4xxBase::De4xx_Ephem_EMbary; ++ii)
760  {
761  // Point is directly represented and is needed:
762  // Update state based on the ephemeris file data for this object.
763  if((ii != De4xxBase::De4xx_Ephem_Earth) && (ii != De4xxBase::De4xx_Ephem_Moon) &&
764  (item_data[ii].status == De4xxEphemItem::Active))
765  {
766  auto * point = dynamic_cast<EphemerisPoint *>(item_data[ii].item);
767  De4xxFileItem & file_item = file.item[body_to_file_idx[ii]];
768  point->update(file_item.state[0], file_item.state[1], update_time);
769  }
770  }
771 
772  // Compute Earth and Moon states to reflect the ref frame tree structure.
773  // The Moon state is relative to the Earth in the ephemeris file.
774  if((item_data[De4xxBase::De4xx_Ephem_Earth].status == De4xxEphemItem::Active) &&
775  (item_data[De4xxBase::De4xx_Ephem_Moon].status == De4xxEphemItem::Active))
776  {
777  De4xxFileItem & moon_ephem = file.item[body_to_file_idx[De4xxBase::De4xx_Ephem_Moon]];
778 
779  points[De4xxBase::De4xx_Ephem_Earth].update_scaled(moon_ephem.state[0],
780  moon_ephem.state[1],
781  -file.header.be_em_dist_ratio,
782  update_time);
783 
784  points[De4xxBase::De4xx_Ephem_Moon].update_scaled(moon_ephem.state[0],
785  moon_ephem.state[1],
786  file.header.bm_em_dist_ratio,
787  update_time);
788  }
789 
790  // Lunar orientation needed:
791  // Extract the lunar orientation Euler angles from the ephemeris file
792  // and update the lunar orientation.
793  if(item_data[De4xxBase::De4xx_Ephem_LLibration].status == De4xxEphemItem::Active)
794  {
795  De4xxFileItem & file_angles = file.item[De4xxBase::De4xx_File_LLibration];
796  lunar_orientation.update(file_angles.state[0], file_angles.state[1], update_time);
797  }
798  }
799 }
