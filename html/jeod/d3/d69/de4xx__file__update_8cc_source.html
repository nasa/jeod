49 /**
50  * Calcuate the position and velocity states of selected planetary bodies at
51  * some point in time. The EphemeridesState structure embedded in the Ephemerides
52  * structure contains an update indicator and state for each body. The state for
53  * a body is updated if the body's indicator indicates that a state update is
54  * needed.
55  *
56  * Body selection --
57  * The 'active_bodies' array in the EphemeridesState structure indicates which
58  * planets' states are to be updated. The function updates the position and
59  * velocity for the selected bodies.
60  *
61  * Time specification --
62  * Four input variables are available for specifying the time.
63  * - tt_offset is the offset between the remaining input times and Terrestrial
64  * Time (aka Terrestrial Dynamic Time, or TDT). Set this to zero if the
65  * other input times are already expressed in Terrestrial Time.
66  * - For highest precision, set epoch_time to the Julian date at midnight of
67  * the time point of interest and set either del_day or del_time to the
68  * difference between the time point of interest and the epoch_time.
69  * - An alternative that retains full precession is to pass the time at the
70  * start of the simulation in epoch_time and del_day and the time into the
71  * simulation in del_time. In this approach, epoch_time represents the
72  * Julian date at the midnight preceding the start of the simulation and
73  * del_day represents the time between the epoch time and simulation start.
74  * - For ease of use, set epoch_time to the Julian date representing the
75  * the time point of interest and set both del_day and del_time to 0.0.
76  * Note that this approach has a machine granularity of about 0.2 msec.
77  * - An intermediate alternative is to set the epoch_time to the start time
78  * of the simulation, del_day to zero and del_time to the simulation time
79  * in seconds. This approach will result in a small temporal bias due to
80  * the precision loss in the epoch time.
81  *
82  * Outputs --
83  * Positions and velocities are expressed in the ICRF coordinate system.
84  * The states of the Sun and planets, including Earth-Moon barycenter, are
85  * expressed with respect to the solar system barycenter. Lunar states are
86  * expressed with respect to the center of the Earth.
87  *
88  * NOTA BENE --
89  * The states of unselected bodies may or may not be changed.
90  *
91  * \par Assumptions and Limitations
92  * - Assumption 1.
93  * Ephemeris file is open.
94  * - Assumption 2.
95  * Offset times in del_day and del_time are small. See description above.
96  * - Assumption 3.
97  * The caller will not reference the states of bodies not requested in the
98  * active_bodies array. The unselected body states are fair game and may or
99  * may not be modified by this function.
100  * - Limitation 1.
101  * No light speed time-of-travel corrections.
102  * Such corrections must be made by the caller if needed.
103  * - Limitation 2.
104  * No relativistic time corrections for the difference between Terrestrial
105  * and Barycenter Dynamic Time.
106  * - Limitation 3.
107  * The states of the sun and the major planets are expressed in ICRF
108  * coordinates relative to the Solar System barycenter.
109  * The state of the Moon is expressed in ICRF coordinates relative
110  * to the center of the Earth.
111  * \param[in] time Time since reference\n Units: s
112  */

257 /**
258  * Calcuate the position and velocity states of selected planetary bodies at
259  * some point in time.
260  * \param[in] time Time since reference\n Units: s
261  * \param[in] fblk Fractional block
262  */
263 void De4xxFile::interpolate(double time, double fblk)
264 {
265  double fsub;
266  int subint;
267 
268  double chebyx, twox;
269  std::size_t nitems;
270  std::size_t nterms;
271  double dnpoly;
272  int item_offset;
273 
274  double pscale;
275  double vscale;
276 
277  std::size_t ii, jj;
278  int kk;
279 
280  /* Interpolate position and velocity. */
281  for(ii = 0; ii < io.metaData->number_file_items; ii++)
282  {
283  De4xxFileItem * item_ii = &(item[ii]);
284  if(item_ii->active)
285  {
286  EphemerisDataItemMeta & itemData = io.itemData[item_ii->item_idx];
287 
288  /* Get body-specific polynomial descriptors. */
289  pscale = item_ii->pscale;
290  nitems = item_ii->nitems;
291  nterms = itemData.nterms;
292  dnpoly = static_cast<double>(itemData.npoly);
293 
294  /* Compute the sub-interval of the loaded time span that contains
295  * the input epoch time. */
296  fsub = fblk * dnpoly;
297  subint = static_cast<int>(fsub);
298  fsub -= subint;
299 
300  /* Compute Chebychev coefficients T_k(x) and their derivatives
301  * dT_k/dx(x) for x = 2*fsub-1:
302  * T[0] = 1
303  * T[1] = x
304  * T[k] = 2 x T[k-1] - T[k-2]
305  * dT[0]/dx = 0
306  * dT[1]/dx = 1
307  * dT[k]/dx = 2 T[k-1] + 2 x dT[k-1]/dx - dT[k-2]/dx */
308  chebyx = fsub + fsub - 1.0;
309  if((coef.chebyterms < nterms) || (!Numerical::compare_exact(coef.chebyx, chebyx)))
310  {
311  coef.chebyterms = nterms;
312  coef.chebyx = chebyx;
313  twox = chebyx + chebyx;
314  coef.chebypoly[0] = 1.0;
315  coef.chebypoly[1] = chebyx;
316  coef.chebyderiv[0] = 0.0;
317  coef.chebyderiv[1] = 1.0;
318  for(jj = 2; jj < nterms; jj++)
319  {
320  coef.chebypoly[jj] = twox * coef.chebypoly[jj - 1] - coef.chebypoly[jj - 2];
321  coef.chebyderiv[jj] = coef.chebypoly[jj - 1] + coef.chebypoly[jj - 1] +
322  twox * coef.chebyderiv[jj - 1] - coef.chebyderiv[jj - 2];
323  }
324  }
325 
326  /* Compute the velocity scale factor.
327  * dr/dt = dr/dx dx/dfs dfs/dfb dfb/dd dd/dt
328  * dr/dx = value computed by applying coeffs vc
329  * x = 2fs-1 => dx/dfs = 2
330  * fs = fs0 + fb*dnpoly => dfs/dfb = dnpoly
331  * fb = fb0 + d / delta_epoch => dfb/dd = 1/delta_epoch
332  * d = t/86400.0 => dd/dt = 1/86400.0
333  * Thus
334  * vscale = 2*dnpoly / (delta_epoch*86400)
335  * Note: The JPL ephemeris file expresses distance in kilometers.
336  * Scale velocity scale by the pscale. */
337  vscale = 2.0 * dnpoly / (io.metaData->delta_epoch * 86400.0) * pscale;
338 
339  /* Compute the C-language offset to the coefficients for this
340  * sub-interval. */
341  item_offset = itemData.offset - 1 + nitems * nterms * subint;
342 
343  /* Interpolate to get position, velocity for each component. */
344  for(jj = 0; jj < nitems; jj++)
345  {
346  std::size_t coefs_starting_idx = item_offset + jj * nterms;
347  item_ii->state[0][jj] = 0.0;
348  item_ii->state[1][jj] = 0.0;
349  for(kk = nterms - 1; kk >= 0; kk--)
350  {
351  item_ii->state[0][jj] += coef.chebypoly[kk] * coef.coef[coefs_starting_idx + kk];
352  item_ii->state[1][jj] += coef.chebyderiv[kk] * coef.coef[coefs_starting_idx + kk];
353  }
354 
355  /* The JPL ephemeris file expresses distance in kilometers.
356  * Scale position, velocity to yield meters, meters/sec^2 */
357  item_ii->state[0][jj] *= pscale;
358  item_ii->state[1][jj] *= vscale;
359  }
360 
361  /* Timestamp the data. */
362  item_ii->update_time = time;
363  }
364  }
365 }
