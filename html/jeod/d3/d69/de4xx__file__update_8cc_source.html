50 /**
51  * Calcuate the position and velocity states of selected planetary bodies at
52  * some point in time. The EphemeridesState structure embedded in the Ephemerides
53  * structure contains an update indicator and state for each body. The state for
54  * a body is updated if the body's indicator indicates that a state update is
55  * needed.
56  *
57  * Body selection --
58  * The 'active_bodies' array in the EphemeridesState structure indicates which
59  * planets' states are to be updated. The function updates the position and
60  * velocity for the selected bodies.
61  *
62  * Time specification --
63  * Four input variables are available for specifying the time.
64  * - tt_offset is the offset between the remaining input times and Terrestrial
65  * Time (aka Terrestrial Dynamic Time, or TDT). Set this to zero if the
66  * other input times are already expressed in Terrestrial Time.
67  * - For highest precision, set epoch_time to the Julian date at midnight of
68  * the time point of interest and set either del_day or del_time to the
69  * difference between the time point of interest and the epoch_time.
70  * - An alternative that retains full precession is to pass the time at the
71  * start of the simulation in epoch_time and del_day and the time into the
72  * simulation in del_time. In this approach, epoch_time represents the
73  * Julian date at the midnight preceding the start of the simulation and
74  * del_day represents the time between the epoch time and simulation start.
75  * - For ease of use, set epoch_time to the Julian date representing the
76  * the time point of interest and set both del_day and del_time to 0.0.
77  * Note that this approach has a machine granularity of about 0.2 msec.
78  * - An intermediate alternative is to set the epoch_time to the start time
79  * of the simulation, del_day to zero and del_time to the simulation time
80  * in seconds. This approach will result in a small temporal bias due to
81  * the precision loss in the epoch time.
82  *
83  * Outputs --
84  * Positions and velocities are expressed in the ICRF coordinate system.
85  * The states of the Sun and planets, including Earth-Moon barycenter, are
86  * expressed with respect to the solar system barycenter. Lunar states are
87  * expressed with respect to the center of the Earth.
88  *
89  * NOTA BENE --
90  * The states of unselected bodies may or may not be changed.
91  *
92  * \par Assumptions and Limitations
93  * - Assumption 1.
94  * Ephemeris file is open.
95  * - Assumption 2.
96  * Offset times in del_day and del_time are small. See description above.
97  * - Assumption 3.
98  * The caller will not reference the states of bodies not requested in the
99  * active_bodies array. The unselected body states are fair game and may or
100  * may not be modified by this function.
101  * - Limitation 1.
102  * No light speed time-of-travel corrections.
103  * Such corrections must be made by the caller if needed.
104  * - Limitation 2.
105  * No relativistic time corrections for the difference between Terrestrial
106  * and Barycenter Dynamic Time.
107  * - Limitation 3.
108  * The states of the sun and the major planets are expressed in ICRF
109  * coordinates relative to the Solar System barycenter.
110  * The state of the Moon is expressed in ICRF coordinates relative
111  * to the center of the Earth.
112  * \param[in] time Time since reference\n Units: s
113  */

254 /**
255  * Calcuate the position and velocity states of selected planetary bodies at
256  * some point in time.
257  * \param[in] time Time since reference\n Units: s
258  * \param[in] fblk Fractional block
259  */
260 void De4xxFile::interpolate(double time, double fblk)
261 {
262  double fsub;
263  int subint;
264 
265  double chebyx, twox;
266  std::size_t nitems;
267  std::size_t nterms;
268  double dnpoly;
269  int item_offset;
270 
271  double pscale;
272  double vscale;
273 
274  std::size_t ii, jj;
275  int kk;
276 
277  /* Interpolate position and velocity. */
278  for (ii = 0; ii < io.metaData->number_file_items; ii++)
279  {
280  De4xxFileItem *item_ii = &(item[ii]);
281  if (item_ii->active)
282  {
283  EphemerisDataItemMeta &itemData = io.itemData[item_ii->item_idx];
284 
285  /* Get body-specific polynomial descriptors. */
286  pscale = item_ii->pscale;
287  nitems = item_ii->nitems;
288  nterms = itemData.nterms;
289  dnpoly = static_cast<double>(itemData.npoly);
290 
291  /* Compute the sub-interval of the loaded time span that contains
292  * the input epoch time. */
293  fsub = fblk * dnpoly;
294  subint = static_cast<int>(fsub);
295  fsub -= subint;
296 
297  /* Compute Chebychev coefficients T_k(x) and their derivatives
298  * dT_k/dx(x) for x = 2*fsub-1:
299  * T[0] = 1
300  * T[1] = x
301  * T[k] = 2 x T[k-1] - T[k-2]
302  * dT[0]/dx = 0
303  * dT[1]/dx = 1
304  * dT[k]/dx = 2 T[k-1] + 2 x dT[k-1]/dx - dT[k-2]/dx */
305  chebyx = fsub + fsub - 1.0;
306  if ((coef.chebyterms < nterms) || (!Numerical::compare_exact(coef.chebyx, chebyx)))
307  {
308  coef.chebyterms = nterms;
309  coef.chebyx = chebyx;
310  twox = chebyx + chebyx;
311  coef.chebypoly[0] = 1.0;
312  coef.chebypoly[1] = chebyx;
313  coef.chebyderiv[0] = 0.0;
314  coef.chebyderiv[1] = 1.0;
315  for (jj = 2; jj < nterms; jj++)
316  {
317  coef.chebypoly[jj] = twox * coef.chebypoly[jj - 1] - coef.chebypoly[jj - 2];
318  coef.chebyderiv[jj] = coef.chebypoly[jj - 1] + coef.chebypoly[jj - 1]
319  + twox * coef.chebyderiv[jj - 1] - coef.chebyderiv[jj - 2];
320  }
321  }
322 
323  /* Compute the velocity scale factor.
324  * dr/dt = dr/dx dx/dfs dfs/dfb dfb/dd dd/dt
325  * dr/dx = value computed by applying coeffs vc
326  * x = 2fs-1 => dx/dfs = 2
327  * fs = fs0 + fb*dnpoly => dfs/dfb = dnpoly
328  * fb = fb0 + d / delta_epoch => dfb/dd = 1/delta_epoch
329  * d = t/86400.0 => dd/dt = 1/86400.0
330  * Thus
331  * vscale = 2*dnpoly / (delta_epoch*86400)
332  * Note: The JPL ephemeris file expresses distance in kilometers.
333  * Scale velocity scale by the pscale. */
334  vscale = 2.0 * dnpoly / (io.metaData->delta_epoch * 86400.0) * pscale;
335 
336  /* Compute the C-language offset to the coefficients for this
337  * sub-interval. */
338  item_offset = itemData.offset - 1 + nitems * nterms * subint;
339 
340  /* Interpolate to get position, velocity for each component. */
341  for (jj = 0; jj < nitems; jj++)
342  {
343  std::size_t coefs_starting_idx = item_offset + jj * nterms;
344  item_ii->state[0][jj] = 0.0;
345  item_ii->state[1][jj] = 0.0;
346  for (kk = nterms - 1; kk >= 0; kk--)
347  {
348  item_ii->state[0][jj] += coef.chebypoly[kk] * coef.coef[coefs_starting_idx + kk];
349  item_ii->state[1][jj] += coef.chebyderiv[kk] * coef.coef[coefs_starting_idx + kk];
350  }
351 
352  /* The JPL ephemeris file expresses distance in kilometers.
353  * Scale position, velocity to yield meters, meters/sec^2 */
354  item_ii->state[0][jj] *= pscale;
355  item_ii->state[1][jj] *= vscale;
356  }
357 
358  /* Timestamp the data. */
359  item_ii->update_time = time;
360  }
361  }
362 }
