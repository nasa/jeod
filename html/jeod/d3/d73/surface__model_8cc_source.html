136 void SurfaceModel::initialize_mass_connections(BaseDynManager & manager)
137 {
138  NamedItem::validate_name(__FILE__, __LINE__, struct_body_name, "Structure Name", "struct_body_name");
139 
140  struct_body_ptr = manager.find_mass_body(struct_body_name);
141 
142  if(struct_body_ptr == nullptr)
143  {
144  MessageHandler::fail(__FILE__,
145  __LINE__,
146  SurfaceModelMessages::initialization_error,
147  "In SurfaceModel::initialize_mass_connections, no mass body was found, "
148  "in the given DynManager, with the name %s. "
149  "Initialization "
150  "has failed.\n",
151  struct_body_name.c_str());
152  }
153 
154  // Each facet is attached to a mass body. for articulation, we need the
155  // relative position and orientation of that mass body w.r.t. the
156  // user defined "main" mass body. If many facets are attached to
157  // the same mass body, we don't want to re-calculate the same
158  // relative states over and over again, so for each "attached to"
159  // mass body we see, create a new FacetStateInfo for it.
160 
161  // create an iterator to search the vector of FacetStateInfos.
162  std::list<FacetStateInfo>::iterator it;
163  for(unsigned int ii = 0; ii < facets.size(); ++ii)
164  {
165  facets[ii]->initialize_mass_connection(manager);
166 
167  // the facet should now have something in it's mass body ptr.
168  // search for it in the vector of FacetStateInfos. If it doesn't
169  // appear, create a new one for that mass body.
170 
171  it = find(articulation_states.begin(),
172  articulation_states.end(),
173  FacetStateInfo(facets[ii]->get_mass_body_ptr()));
174 
175  if(it == articulation_states.end())
176  {
177  articulation_states.push_back(FacetStateInfo(facets[ii]->get_mass_body_ptr()));
178  facets[ii]->mass_rel_struct = &articulation_states.back().mass_state;
179  }
180  else
181  {
182  facets[ii]->mass_rel_struct = &(it->mass_state);
183  }
184 
185  } // for(unsigned int ii)
186 }
