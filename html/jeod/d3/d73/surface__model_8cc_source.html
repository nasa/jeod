148 void
149 SurfaceModel::initialize_mass_connections (
150  BaseDynManager& manager)
151 {
152 
153  NamedItem::validate_name(__FILE__, __LINE__, struct_body_name,
154  "Structure Name", "struct_body_name");
155 
156  struct_body_ptr = manager.find_mass_body(struct_body_name);
157 
158  if(struct_body_ptr == nullptr) {
159  MessageHandler::fail (
160  __FILE__, __LINE__, SurfaceModelMessages::initialization_error,
161  "In SurfaceModel::initialize_mass_connections, no mass body was found, "
162  "in the given DynManager, with the name %s. "
163  "Initialization "
164  "has failed.\n", struct_body_name);
165  }
166 
167  // Each facet is attached to a mass body. for articulation, we need the
168  // relative position and orientation of that mass body w.r.t. the
169  // user defined "main" mass body. If many facets are attached to
170  // the same mass body, we don't want to re-calculate the same
171  // relative states over and over again, so for each "attached to"
172  // mass body we see, create a new FacetStateInfo for it.
173 
174  // create an iterator to search the vector of FacetStateInfos.
175  std::list<FacetStateInfo>::iterator it;
176  for(unsigned int ii = 0; ii < facets.size(); ++ii){
177  facets[ii]->initialize_mass_connection(manager);
178 
179  // the facet should now have something in it's mass body ptr.
180  // search for it in the vector of FacetStateInfos. If it doesn't
181  // appear, create a new one for that mass body.
182 
183  it = find(articulation_states.begin(),
184  articulation_states.end(),
185  FacetStateInfo(facets[ii]->get_mass_body_ptr() ) );
186 
187  if(it == articulation_states.end()) {
188  articulation_states.push_back(
189  FacetStateInfo(facets[ii]->get_mass_body_ptr() ));
190  facets[ii]->mass_rel_struct = &articulation_states.back().mass_state;
191  }
192  else {
193  facets[ii]->mass_rel_struct = &(it->mass_state);
194  }
195 
196  } // for(unsigned int ii)
197 
198 
199 
200  return;
201 
202 }
