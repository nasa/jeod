68 /**
69  * Detach the body from its parent.
70  * \param[in,out] dyn_manager Dynamics manager
71  */
72 void BodyDetachSpecific::apply(DynManager & dyn_manager)
73 {
74  bool succeeded = false;
75  std::string name_detach = "(?detach_from)";
76  std::string name_subject = "(?subject)";
77 
78  // Detach DynBody subject from its parent
79  if(dyn_subject != nullptr && dyn_detach_from != nullptr)
80  {
81  succeeded = dyn_subject->detach(*dyn_detach_from);
82  name_subject = dyn_subject->name.get_name();
83  name_detach = dyn_detach_from->name.get_name();
84  }
85  // Jettison MassBody from DynBody
86  else if(dyn_subject == nullptr && dyn_detach_from != nullptr)
87  {
88  succeeded = dyn_detach_from->remove_mass_body(*mass_subject);
89  name_subject = mass_subject->name.get_name();
90  name_detach = dyn_detach_from->name.get_name();
91  }
92  // Detach MassBody Subject from its parent
93  else if(dyn_subject == nullptr && dyn_detach_from == nullptr)
94  {
95  succeeded = mass_subject->detach(*mass_detach_from);
96  name_subject = mass_subject->name.get_name();
97  name_detach = mass_detach_from->name.get_name();
98  }
99  // Inconsistent setup not detected by BodyDetachSpecific::initialize()
100  else
101  {
102  succeeded = false;
103  }
104 
105  // Detachment succeeded: Debug.
106  if(succeeded)
107  {
108  MessageHandler::debug(__FILE__,
109  __LINE__,
110  BodyActionMessages::trace,
111  "%s: %s detached from %s.",
112  action_identifier.c_str(),
113  name_subject.c_str(),
114  name_detach.c_str());
115  }
116 
117  // Detachment failed: Terminate the sim if terminate_on_error is set.
118  else if(terminate_on_error)
119  {
120  MessageHandler::fail(__FILE__,
121  __LINE__,
122  BodyActionMessages::fatal_error,
123  "%s failed to detach %s from %s."
124  "The terminate_on_failure flag set and a detachment error occurred.\n"
125  "The detachment error described above is fatal per this setting.",
126  action_identifier.c_str(),
127  name_subject.c_str(),
128  name_detach.c_str());
129  }
130 
131  // Detachment failed, terminate_ not set: Tell the user about the problem.
132  else
133  {
134  MessageHandler::error(__FILE__,
135  __LINE__,
136  BodyActionMessages::not_performed,
137  "%s failed to detach %s from %s.",
138  action_identifier.c_str(),
139  name_subject.c_str(),
140  name_detach.c_str());
141  }
142 
143  // Forward the action up the chain.
144  BodyAction::apply(dyn_manager);
145 }
