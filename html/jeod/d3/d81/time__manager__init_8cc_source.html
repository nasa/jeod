413 /**
414  * Build and verify a "tree-like" structure to ensure that all time
415  * representations can be initialized from the single "initializer"
416  * representation.
417  * Create_init_tree builds the tree, using add_type_initialize to
418  * populate the tree recursively.
419  *
420  * \par Assumptions and Limitations
421  * - This is vastly improved if the user defines the parent type
422  * "initialize_from" for each time representation, except the
423  * top-level initializer type
424  * - Otherwise, the code will build the tree automatically, but it
425  * takes longer and may be less than ideal
426  */
427 void
428 TimeManagerInit::create_init_tree (
429  void)
430 {
431  int seeking_status = 0; /* -- for automated tree-builder routine
432  * looks for time-types with
433  * status = seeking_status
434  */
435 
436  // All status values = 0, already set in initialize_TMI_settings
437  status[dyn_time_index] = -2;
438  status[initializer_index] = 1;
439 
440  // Dyn time is not the time initialization source
441  if (initializer_index != dyn_time_index) {
442  // if initializer is a UDE, also need to place its "update_from" in the
443  // initialization tree with status = 2, because it will need to
444  // communicate with a standard time.
445  TimeUDE * init_ude_ptr = dynamic_cast<TimeUDE *>
446  (time_manager->time_vector[
447  initializer_index ]);
448  if (init_ude_ptr != nullptr) {
449  int upd_index =
450  time_manager->time_lookup (init_ude_ptr->update_from_name);
451  // If unable to tie the UDE to a standard: fatal error
452  if (upd_index < 0) {
453  MessageHandler::fail (
454  __FILE__, __LINE__, TimeMessages::invalid_setup_error, "\n"
455  "The inititializer (%s) update_from_name not registered.\n"
456  "The initializer is a UDE, so must have an update_from_name."
457  ,init_ude_ptr->update_from_name.c_str() );
458  } // else found the standard to convert the UDE to a nonarbitrary form
459  else {
460  status[ upd_index ] = 2; // from 1st gen initializer time, so 2
461  num_added_total = 3;
462  }
463  }
464  // else not initializing from a UDE
465  else {
466  // Since they are fundamental (root) times, don't need to place dyn_time
467  // or initializer in the tree, so consider these placed
468  num_added_total = 2;
469  }
470  }
471  // else dyn_time is the initializer
472  else {
473  num_added_total = 1; // avoid double-counting dyn_time and initializer
474  }
475 
476  num_added_pass = num_added_total; // Set to initialize while loop below
477 
478 
479 
480 
481  // Add types to the initialization tree
482  // until all types have been added or nothing else can be added.
483  while (num_added_total < time_manager->num_types) {
484  if (num_added_pass > 0) {
485  num_added_pass = 0;
486  seeking_status++;
487  for (int ii = 0; ii < time_manager->num_types; ++ii) {
488  if (status[ii] == 0) {
489  time_manager->time_vector[ ii ]->
490  add_type_initialize (seeking_status, this);
491  }
492  }
493  // add the number of types added in this pass to the total number of types
494  // added.
495  num_added_total += num_added_pass;
496  }
497  else {
498  // went through the last pass without adding anything, this pass will be
499  // no different. Tree-build has failed.
500  // Report the first culprit (only one is reported) and quit simulation.
501  for (int ii = 0; ii < time_manager->num_types;
502  ++ii) {
503  if (status[ii] == 0) {
504  MessageHandler::fail (
505  __FILE__, __LINE__, TimeMessages::invalid_setup_error, "\n"
506  "Could not get time-type %s into the tree structure for "
507  "initialization.\n"
508  "There must be insufficient time converters registered - that \n"
509  "are valid at initialization - to allow access to this time-type\n"
510  "from the initializer (%s).\n",
511  time_manager->time_vector[ii]->name.c_str(),
512  time_manager->time_vector[ initializer_index ]->name.c_str());
513  }
514  }
515  MessageHandler::fail (
516  __FILE__, __LINE__, TimeMessages::initialization_error, "\n"
517  "Error in generating the initialization tree.\n");
518  }
519  }
520 
521  return;
522 }

560 /**
561  * (To verify that the update procedures have a tree-like structure,
562  * and that all time representations can be updated from the dynamic time.)
563  * (Contains 3 functions - create_update_tree builds the tree, populated
564  * recursively by add_type_update. record_update records the update paths
565  * to facilitate runtime updates)
566  *
567  * \par Assumptions and Limitations
568  * - None
569  */
570 void
571 TimeManagerInit::create_update_tree (
572  void)
573 {
574  int seeking_status = 0; /* desired status value of next time when during the
575  * fully-automated building of the update tree. */
576 
577  for (int ii = 0; ii < time_manager->num_types; ++ii) {
578  status[ii] = 0;
579  }
580  status[dyn_time_index] = 1; // dyn_time is update source, always 1st gen.
581 
582  num_added_total = 1; // number of time-types placed in the update tree
583  // (Dyn-time has already been added at status 0).
584  num_added_pass = 1; /* counter for number of time-types that have
585  * been added into the update tree in any given
586  * pass. initial pass is only dyn_time, so set 1. */
587 
588 
589  // while there is something to add to the tree, and capability to add it:
590  while (num_added_total < time_manager->num_types) {
591  if (num_added_pass > 0) {
592  num_added_pass = 0;
593  seeking_status++;
594  for (int ii = 0; ii < time_manager->num_types; ++ii) {
595  if (status[ii] == 0) {
596  time_manager->time_vector[ii]->add_type_update (
597  seeking_status, this);
598  }
599  }
600  }
601  else {
602  // went through an entire without adding anything. This pass will be
603  // no different, because the auto-seek found nothing. Tree-build has
604  // failed. Report the first uninitialized culprit and quit
605  // the simulation.
606  for (int ii = 0; ii < time_manager->num_types; ++ii) {
607  if (status[ii] == 0) {
608  MessageHandler::fail (
609  __FILE__, __LINE__, TimeMessages::incomplete_setup_error,
610  "\n"
611  "Could not get time-type %s into the tree structure for "
612  "run-time updates.\n"
613  "There must be insufficient time converters registered for \n"
614  "run-time updates to allow access to this time-type from "
615  "TimeDyn.\n",
616  time_manager->time_vector[ii]->name.c_str());
617  }
618  }
619  MessageHandler::fail (
620  __FILE__, __LINE__, TimeMessages::initialization_error, "\n"
621  "Error in generating the update tree.\n");
622  }
623  }
624 
625  organize_update_list();
626 
627  return;

630 /**
631  * Reorganizes the update list according to initialization status
632  */
633 void
634 TimeManagerInit::organize_update_list()
635 {
636  // vector of finalized index positions
637  std::vector<unsigned> order;
638  order.reserve(time_manager->num_types);
639 
640  // Create index vector
641  for(int ii = 1; ii <= time_manager->num_types; ii++ )
642  {
643  for(int jj=0; jj < time_manager->num_types; jj++)
644  {
645  if( status[jj] == ii )
646  {
647  order.push_back(jj);
648  }
649  }
650  }
651 
652  // Apply index vector
653  for(unsigned ii = 0, jj = 0; ii < order.size()-1; ii++ )
654  {
655  if( ii == order[ii] ) // if properly placed, continue
656  {
657  continue;
658  }
659  for(jj = ii+1; jj < order.size(); jj++) // Find order swap
