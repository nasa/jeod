182  __LINE__,

217  time_manager->time_vector[ii]->name.c_str(),
218  time_manager->time_vector[jj]->name.c_str());
219  }

221  if(time_manager->time_vector[ii]->name == time_manager->time_vector[jj]->name)
222  {
223  MessageHandler::fail(__FILE__,
224  __LINE__,
225  TimeMessages::invalid_setup_error,
226  "\n"
227  "Two time types have been given the same name (%s). This will cause "
228  "confusion whenever a time-type has to be found by name only. \n",
229  time_manager->time_vector[ii]->name.c_str());

332  * To verify that there are no incompatibilities between specific
333  * converters
334  *
335  * \par Assumptions and Limitations
336  * - The instance of TimeTAI, if it exists, has name "TAI"
337  * - The instance of TimeUTC, if it exists, has name "UTC"
338  * - The instance of TimeUT1, if it exists, has name "UT1"
339  */
340 void TimeManagerInit::verify_converter_setup()
341 {
342  // check on the appropriate usage of "override_data_table" in
343  // TimeConverter_TAI_UTC and TimeConverter_TAI_UT1.
344  // first check whether the time types are registered
345  int tai_index = time_manager->time_lookup("TAI");
346  int utc_index = time_manager->time_lookup("UTC");
347  int ut1_index = time_manager->time_lookup("UT1");
348  if((tai_index > 0) && (utc_index > 0) && (ut1_index > 0))
349  {
350  // next check whether the converters are registered.
351  int tai_utc_conv_index = converter_ptrs_index[utc_index + tai_index * time_manager->num_types];
352  int tai_ut1_conv_index = converter_ptrs_index[ut1_index + tai_index * time_manager->num_types];
353  if((tai_utc_conv_index > 0) && tai_ut1_conv_index)
354  {
355  // finally check whether the flags are set differently in the two
356  // converter classes
357  auto * temp_conv1 = static_cast<TimeConverter_TAI_UTC *>(
358  time_manager->converter_vector[tai_utc_conv_index]);
359  auto * temp_conv2 = static_cast<TimeConverter_TAI_UT1 *>(
360  time_manager->converter_vector[tai_ut1_conv_index]);
361 
362  if(temp_conv1->override_data_table != temp_conv2->override_data_table)
363  {
364  MessageHandler::fail(__FILE__,
365  __LINE__,
366  TimeMessages::invalid_setup_error,
367  "\n"
368  "The UTC and UT1 time-types are usually calculated from look-up\n"
369  "tables of the offsets. These tables are tied very closely \n"
370  "together, and while the override_data_table flag provides the \n"
371  "functionality to override the data values in those tables, it \n"
372  "is required that this flag be set the same way for both tables.\n");
373  }
374  }

378 /**
379  * Build and verify a "tree-like" structure to ensure that all time
380  * representations can be initialized from the single "initializer"
381  * representation.
382  * Create_init_tree builds the tree, using add_type_initialize to
383  * populate the tree recursively.
384  *
385  * \par Assumptions and Limitations
386  * - This is vastly improved if the user defines the parent type
387  * "initialize_from" for each time representation, except the
388  * top-level initializer type
389  * - Otherwise, the code will build the tree automatically, but it
390  * takes longer and may be less than ideal
391  */
392 void TimeManagerInit::create_init_tree()
393 {
394  int seeking_status = 0; /* -- for automated tree-builder routine
395  * looks for time-types with
396  * status = seeking_status
397  */
398 
399  // All status values = 0, already set in initialize_TMI_settings
400  status[dyn_time_index] = -2;
401  status[initializer_index] = 1;
402 
403  if(initializer_index != dyn_time_index) // Dyn time is not the time initialization source
404  {
405  // if initializer is a UDE, also need to place its "update_from" in the
406  // initialization tree with status = 2, because it will need to
407  // communicate with a standard time.
408  auto * init_ude_ptr = dynamic_cast<TimeUDE *>(time_manager->time_vector[initializer_index]);
409  if(init_ude_ptr != nullptr)
410  {
411  int upd_index = time_manager->time_lookup(init_ude_ptr->update_from_name);
412  if(upd_index < 0) // If unable to tie the UDE to a standard: fatal error
413  {
414  MessageHandler::fail(__FILE__,
415  __LINE__,
416  TimeMessages::invalid_setup_error,
417  "\n"
418  "The inititializer (%s) update_from_name not registered.\n"
419  "The initializer is a UDE, so must have an update_from_name.",
420  init_ude_ptr->update_from_name.c_str());
421  }
422  else // else found the standard to convert the UDE to a nonarbitrary form
423  {
424  status[upd_index] = 2; // from 1st gen initializer time, so 2
425  num_added_total = 3;
426  }
427  }
428 
429  else // else not initializing from a UDE
430  {
431  // Since they are fundamental (root) times, don't need to place dyn_time
432  // or initializer in the tree, so consider these placed
433  num_added_total = 2;
434  }
435  }
436  else // else dyn_time is the initializer
437  {
438  num_added_total = 1; // avoid double-counting dyn_time and initializer
439  }
440 
441  num_added_pass = num_added_total; // Set to initialize while loop below
442 
443  // Add types to the initialization tree
444  // until all types have been added or nothing else can be added.
445  while(num_added_total < time_manager->num_types)
446  {
447  if(num_added_pass > 0)
448  {
449  num_added_pass = 0;
450  seeking_status++;
451  for(int ii = 0; ii < time_manager->num_types; ++ii)
452  {
453  if(status[ii] == 0)
454  {
455  time_manager->time_vector[ii]->add_type_initialize(seeking_status, this);
456  }
457  }
458  // add the number of types added in this pass to the total number of types
459  // added.
460  num_added_total += num_added_pass;
461  }
462  else
463  {
464  // went through the last pass without adding anything, this pass will be
465  // no different. Tree-build has failed.
466  // Report the first culprit (only one is reported) and quit simulation.
467  for(int ii = 0; ii < time_manager->num_types; ++ii)
468  {
469  if(status[ii] == 0)
470  {
471  MessageHandler::fail(__FILE__,
472  __LINE__,
473  TimeMessages::invalid_setup_error,
474  "\n"
475  "Could not get time-type %s into the tree structure for "
476  "initialization.\n"
477  "There must be insufficient time converters registered - that \n"
478  "are valid at initialization - to allow access to this time-type\n"
479  "from the initializer (%s).\n",
480  time_manager->time_vector[ii]->name.c_str(),
481  time_manager->time_vector[initializer_index]->name.c_str());
482  }
483  }
484  MessageHandler::fail(__FILE__,
485  __LINE__,
486  TimeMessages::initialization_error,
487  "\n"
488  "Error in generating the initialization tree.\n");
489  }
490  }
491 }

520 /**
521  * (To verify that the update procedures have a tree-like structure,
522  * and that all time representations can be updated from the dynamic time.)
523  * (Contains 3 functions - create_update_tree builds the tree, populated
524  * recursively by add_type_update. record_update records the update paths
525  * to facilitate runtime updates)
526  *
527  * \par Assumptions and Limitations
528  * - None
529  */
530 void TimeManagerInit::create_update_tree()
531 {
532  int seeking_status = 0; /* desired status value of next time when during the
533  * fully-automated building of the update tree. */
534 
535  for(int ii = 0; ii < time_manager->num_types; ++ii)
536  {
537  status[ii] = 0;
538  }
539  status[dyn_time_index] = 1; // dyn_time is update source, always 1st gen.
540 
541  num_added_total = 1; // number of time-types placed in the update tree
542  // (Dyn-time has already been added at status 0).
543  num_added_pass = 1; /* counter for number of time-types that have
544  * been added into the update tree in any given
545  * pass. initial pass is only dyn_time, so set 1. */
546 
547  // while there is something to add to the tree, and capability to add it:
548  while(num_added_total < time_manager->num_types)
549  {
550  if(num_added_pass > 0)
551  {
552  num_added_pass = 0;
553  seeking_status++;
554  for(int ii = 0; ii < time_manager->num_types; ++ii)
555  {
556  if(status[ii] == 0)
557  {
558  time_manager->time_vector[ii]->add_type_update(seeking_status, this);
559  }
560  }
561  }
562  else
563  {
564  // went through an entire without adding anything. This pass will be
565  // no different, because the auto-seek found nothing. Tree-build has
566  // failed. Report the first uninitialized culprit and quit
567  // the simulation.
568  for(int ii = 0; ii < time_manager->num_types; ++ii)
569  {
570  if(status[ii] == 0)
571  {
572  MessageHandler::fail(__FILE__,
573  __LINE__,
574  TimeMessages::incomplete_setup_error,
575  "\n"
576  "Could not get time-type %s into the tree structure for "
577  "run-time updates.\n"
578  "There must be insufficient time converters registered for \n"
579  "run-time updates to allow access to this time-type from "
580  "TimeDyn.\n",
581  time_manager->time_vector[ii]->name.c_str());
582  }
583  }
584  MessageHandler::fail(__FILE__,
585  __LINE__,
586  TimeMessages::initialization_error,
587  "\n"
588  "Error in generating the update tree.\n");
589  }
590  }

595 /**
596  * Reorganizes the update list according to initialization status
597  */
598 void TimeManagerInit::organize_update_list()
599 {
600  // vector of finalized index positions
601  std::vector<unsigned> order;
602  order.reserve(time_manager->num_types);
603 
604  // Create index vector
605  for(int ii = 1; ii <= time_manager->num_types; ii++)
606  {
607  for(int jj = 0; jj < time_manager->num_types; jj++)
608  {
609  if(status[jj] == ii)
610  {
611  order.push_back(jj);
612  }
613  }
614  }
615 
616  // Apply index vector
617  for(unsigned ii = 0, jj = 0; ii < order.size() - 1; ii++)
618  {
619  if(ii != order[ii])
620  {
621  for(jj = ii + 1; jj < order.size(); jj++) // Find order swap
622  {
623  if(order[jj] == ii) // found
624  {
625  break;
626  }
627  }
628  std::swap(time_manager->time_vector[ii], time_manager->time_vector[order[ii]]);
629  std::swap(status[ii], status[order[ii]]);
630  std::swap(order[ii], order[jj]);
631  }
632  }
633 }
