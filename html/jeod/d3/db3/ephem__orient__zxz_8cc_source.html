56 // EPSILON_TIME
57 // Limit designating an overly small time step, which typically result from
58 // numerical errors.
59 #define EPSILON_TIME 1e-12
60 
61 // TAYLOR_CUTOFF:
62 // Limit for approximating the quaternion exponential Q_rot = e^(phi*uhat) via a
63 // normalized two-term Taylor expansion. The error in this normalized expansion
64 // is bounded by phi^5/30.
65 // Because of finite-precision arithmetic, the error in the propagated
66 // quaternion Q_prop = Q_rot^T Q will be more than 1e-12 for even very small
67 // angles. Keeping the error in Q_rot smaller than 1 part in 1e-12 means
68 // that phi^5/30 <= 1e-12, or phi = 30e-12^0.2 = 0.00786.
69 #define TAYLOR_CUTOFF 0.00786

115 /**
116  * Propagate the orientation to the current time.
117  *
118  * \par Assumptions and Limitations
119  * - The orientation has been computed.
120  * - Rotation is constant during the update interval.
121  * \param[in] to_time Target dynamic time\n Units: s
122  */
123 void EphemerisZXZOrientation::propagate(double to_time)
124 {
125  RefFrameRot * ref_state = &(target_frame->state.rot);
126 
127  double deltat;
128  double hwdt;
129  double coshwdt, sinhwdt;
130  Quaternion q;
131 
132  // Calculate the time change and return if too small.
133  deltat = to_time - update_time;
134  if((deltat > -EPSILON_TIME) && (deltat < EPSILON_TIME))
135  {
136  return;
137  }
138 
139  // The propagated quaternion is
140  // Q_prop = exp(-wmag*dt/2*[0;what]) * Q
141  // The quaternion exponential is exactly
142  // exp(-wmag*dt/2*[0;what]) = [cos(wmag*dt/2), sin(wmag*dt/2)*what]
143  // Approximate cos(wmag*dt/2), sin(wmag*wdt/2) via 2-term Taylor expansions
144  // for small deltat.
145  hwdt = 0.5 * ref_state->ang_vel_mag * deltat;
146  if(hwdt < TAYLOR_CUTOFF)
147  {
148  double hwdtsq = hwdt * hwdt;
149  coshwdt = 1.0 - 0.5 * hwdtsq;
150  sinhwdt = hwdt * (1.0 - (1.0 / 6.0) * hwdtsq);
151  }
152 
153  // Over limit: Calculate cos(wmag*dt/2), sin(wmag*wdt/2) via math lib.
154  else
155  {
156  coshwdt = std::cos(hwdt);
157  sinhwdt = std::sin(hwdt);
158  }
159 
160  // Compute the quaternion exp(wmag*dt/2*what)
161  q.scalar = coshwdt;
162  Vector3::scale(ref_state->ang_vel_unit, sinhwdt, q.vector);
163 
164  // Propagate the ephemeris-based quaternion to the current time via
165  // Q(t+deltat) = exp(-wmag*dt/2*what)*Q(t)
166  q.conjugate_multiply(ref_state->Q_parent_this);
167  q.normalize_integ(ref_state->Q_parent_this);
168 
169  // Compute the corresponding transformation matrix.
170  ref_state->compute_transformation();
171 
172  // Set timestamps.
173  update_time = to_time;
174  target_frame->set_timestamp(update_time);
175 }

177 /**
178  * Compute a JEOD rotational state given a 3-1-3 inertial-to-planet-fixed
179  * Euler sequence and the time derivatives of the Euler angles.
180  * \param[in] angles zxz Euler angles\n Units: r
181  * \param[in] derivs zxz Euler angle time derivatives\n Units: r/s
182  * \param[in] time Update time\n Units: s
183  */
184 void EphemerisZXZOrientation::update(const double * angles, const double * derivs, double time)
185 {
186  RefFrameRot * ref_state = &(target_frame->state.rot);
187 
188  double phi, theta, psi;
189  double phidot, thetadot, psidot;
190  double cosphi, costheta, cospsi;
191  double sinphi, sintheta, sinpsi;
192  double htheta, hphi_psi_dif, hphi_psi_sum;
193  double coshtheta, sinhtheta;
194  double coshphi_psi_dif, sinhphi_psi_dif;
195  double coshphi_psi_sum, sinhphi_psi_sum;
196 
197  // Extract Euler angles and their rates from input vectors.
198  phi = euler_angle_313[0] = angles[0];
199  theta = euler_angle_313[1] = angles[1];
200  psi = euler_angle_313[2] = angles[2];
201  phidot = euler_rate_313[0] = derivs[0];
202  thetadot = euler_rate_313[1] = derivs[1];
203  psidot = euler_rate_313[2] = derivs[2];
204 
205  // Compute sines, cosines of the Euler angles.
206  cospsi = std::cos(psi);
207  sinpsi = std::sin(psi);
208  costheta = std::cos(theta);
209  sintheta = std::sin(theta);
210  cosphi = std::cos(phi);
211  sinphi = std::sin(phi);
212 
213  // Construct the inertial to planet-fixed transformation matrix as a
214  // yaw-roll-yaw Euler rotation:
215  // T_parent_this = T_yaw(psi)*T_roll(theta)*T_yaw(phi)
216  ref_state->T_parent_this[0][0] = cospsi * cosphi - sinpsi * costheta * sinphi;
217  ref_state->T_parent_this[0][1] = cospsi * sinphi + sinpsi * costheta * cosphi;
218  ref_state->T_parent_this[0][2] = sinpsi * sintheta;
219 
220  ref_state->T_parent_this[1][0] = -sinpsi * cosphi - cospsi * costheta * sinphi;
221  ref_state->T_parent_this[1][1] = -sinpsi * sinphi + cospsi * costheta * cosphi;
222  ref_state->T_parent_this[1][2] = cospsi * sintheta;
223 
224  ref_state->T_parent_this[2][0] = sintheta * sinphi;
225  ref_state->T_parent_this[2][1] = -sintheta * cosphi;
226  ref_state->T_parent_this[2][2] = costheta;
227 
228  // Construct the body's rotation rate vector (planet-fixed coordinates),
229  // derived by
230  // dT'/dt = T'*skew(w)
231  // where
232  // T' = transpose(T)
233  // skew(w) = skew-symmetric cross product matrix generated by w
234  // Thus
235  // skew(w) = T*dT'/dt
236  // = T*(psidot*dT/dsi + thetadot*dT/dtheta + phidot*dT/dphi)'
237  // From which the equations below follow.
238  ref_state->ang_vel_this[0] = phidot * sintheta * sinpsi + thetadot * cospsi;
239  ref_state->ang_vel_this[1] = phidot * sintheta * cospsi - thetadot * sinpsi;
240  ref_state->ang_vel_this[2] = phidot * costheta + psidot;
241 
242  // Compute data needed for propagating the lunar RNP between updates.
243  ref_state->ang_vel_mag = Vector3::vmag(ref_state->ang_vel_this);
244  Vector3::scale(ref_state->ang_vel_this, 1.0 / ref_state->ang_vel_mag, ref_state->ang_vel_unit);
245 
246  // Construct the inertial to planet-fixed left transformation quaternion as
247  // a * yaw-roll-yaw Euler rotation:
248  // Q_parent_this = Q_yaw(psi)*Q_roll(theta)*Q_yaw(phi)
249  htheta = 0.5 * theta;
250  hphi_psi_dif = 0.5 * (phi - psi);
251  hphi_psi_sum = 0.5 * (phi + psi);
252 
253  coshtheta = std::cos(htheta);
254  sinhtheta = std::sin(htheta);
255  coshphi_psi_dif = std::cos(hphi_psi_dif);
256  sinhphi_psi_dif = std::sin(hphi_psi_dif);
257  coshphi_psi_sum = std::cos(hphi_psi_sum);
258  sinhphi_psi_sum = std::sin(hphi_psi_sum);
259 
260  ref_state->Q_parent_this.scalar = coshphi_psi_sum * coshtheta;
261  ref_state->Q_parent_this.vector[0] = -coshphi_psi_dif * sinhtheta;
262  ref_state->Q_parent_this.vector[1] = -sinhphi_psi_dif * sinhtheta;
263  ref_state->Q_parent_this.vector[2] = -sinhphi_psi_sum * coshtheta;
264 
265  // Timestamp the reference state.
266  update_time = time;
267  target_frame->set_timestamp(update_time);
268 }
