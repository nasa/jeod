60 // Construct a DynBody.
61 DynBody::DynBody()
62  : mass(*this),
63  name(mass.name),
64  dyn_manager(mass.dyn_manager)
65 {
66  // Register the checkpointable items.
67  JEOD_REGISTER_CLASS(DynBody);
68  JEOD_REGISTER_CLASS(BodyRefFrame);
69  JEOD_REGISTER_CHECKPOINTABLE(this, trans_integrator);
70  JEOD_REGISTER_CHECKPOINTABLE(this, rot_integrator);

72  // This class integrates the composite body frame.
73  integrated_frame = &composite_body;
74 
75  // We want the inverse inertia tensor (MassBody member) calculated.
76  mass.compute_inverse_inertia = true;
77 
78  // Identify the mass points associated with the body ref frames.

84 // Destruct a DynBody.
85 DynBody::~DynBody()
86 {
87  // Detach this body from its parent.
88  if(dyn_parent != nullptr)
89  {
90  detach(*dyn_parent);
91  }
92 
93  // Detach each child DynBody from this body.
94  while(!dyn_children.empty())
95  {
96  DynBody * child = dyn_children.back();
97  child->detach(*this);
98  // If detach fails for some reason, need to remove from the container to avoid infinite loop
99  dyn_children.remove(child);
100  }
101  // Detach each sub MassBody from this body.
102  while(!mass_children.empty())
103  {
104  MassBody * sub = mass_children.back();
105  remove_mass_body(*sub);
106  // If detach fails for some reason, need to remove from the container to avoid infinite loop
107  mass_children.remove(sub);
108  }
109 
110  // Remove the body reference frames from the ref frame tree.
111  core_body.remove_from_parent();
112  composite_body.remove_from_parent();
113  structure.remove_from_parent();
114  if(dyn_manager)
115  {
116  dyn_manager->remove_ref_frame(core_body);
117  dyn_manager->remove_ref_frame(composite_body);
118  dyn_manager->remove_ref_frame(structure);
119  }
120 
121  // Remove the registrations of the checkpointable objects.
122  JEOD_DEREGISTER_CHECKPOINTABLE(this, trans_integrator);
123  JEOD_DEREGISTER_CHECKPOINTABLE(this, rot_integrator);
124 
125  // Delete the allocated vehicle points, in reverse order.
126  while(!vehicle_points.empty())
127  {
128  auto * point = vehicle_points.back();
129  if(dyn_manager != nullptr)
130  {
131  dyn_manager->remove_ref_frame(*point);
132  }
133  JEOD_DELETE_OBJECT(point);
134  vehicle_points.pop_back();
135  }
136 }
