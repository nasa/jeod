116 PURPOSE:
117  (Calculates the temperature at a specified altitude
118  using equation (10) for altitudes between 90 and 125 km and equation (13)
119  for altitudes greater than 125 km, from Smithsonian Astrophysical
120  Observatory Report 313, 1970.)
121 
122 NOTES: (This method is used to generate the temperature at the current

139  - rise to an inflection point at 125 km

141  ( The temperature T_125 and gradient T_grad_125 at the

147 double METAtmosphereThermal::compute_temperature(double altitude_km_in)
148 {
149  // temperature-offset is the difference between temperatures at the 125km and
150  // 90km altitude reference points:
151 
152  // For altitude in the range 90 km to 125 km, temperature is computed as
153  // a polynomial function of altitude difference from 125 km
154  // (see Jacchia(1971), p11, eq(10) or Jacchia(1970), p 9, eq(10)):
155  // T = T_125 + sum {i=1 to 4} (c_i * (z-125)^i)
156  // Satisfaction of the boundary conditions (see Jacchia(1971), p 12, eq(11))
157  // implies that:
158  // c_1 = (1.9 / 35) * (T_125 - T_90) see eq(11c)
159  // c_2 = 0 eq(11d)
160  // c_3 = ((4 - 3(1.9)) / (35)^3) * (T_125 - T_90) eq(11a,b)
161  // c_4 = ((3 - 2(1.9)) / (35)^4) * (T_125 - T_90) eq(11a,b)
162 
163  // Notes:
164  // - c_1 is the temperature gradient at 125km
165  // - the constants at the front of each c_i, e.g. (1.9 / 35) are stored
166  // in the class as values k_i:
167  // c_i = k_i * (T_125 - T_90)
168  // - the values c_i are not actually computed, the common factor
169  // (T_125 - T_90) is pulled out of the summation so that the
170  // values k_i can be used instead.
171 
172  // For altitudes > 125km, a different algorithm is used,
173  // (see Jacchia(1971), p 12, eq(13))
174 
175  // In both computations, the values (dz), (dz)^2, and dT are needed. so
176  // generate those first:
177  // base-altitude is delta-altitude above 125 km.
178  double dz = altitude_km_in - 125.0;
179  double dz_2 = dz * dz;
180  double dT = T_125 - T_90;
181 
182  if(dz <= 0.0)
183  {
184  // obtain the temperature as a polynomial in z:
185  double dz_3 = dz * dz_2;
186  double dz_4 = dz_2 * dz_2;
187  return T_125 + dT * ((k_1 * dz) + (k_3 * dz_3) + (k_4 * dz_4));
188  }
189  // else use equation 13, which is identical both Jacchia papers:
190  double dz_2_5 = dz_2 * sqrt(dz);
191  double coeff_A = 2 * (T_exosphere - T_125) / M_PI;
192  // magic number 4.5E-6 is empirical, identified as "B" in eq(13).
193  return T_125 + coeff_A * std::atan2(k_1 * dT * dz * (1.0 + (4.5E-6 * dz_2_5)), coeff_A);
194 }

347 void METAtmosphere::compute_solar_angles()
348 {
349  //****************************************************************************
350  // PART A - compute the necessary time representations
351  //****************************************************************************
352 
353  // Compute the fraction of day, year, and century:
354  // Truncated Julian Time (tjt, TJT) has epoch at 1957/05/24::00:00:00
355  // It is a principal clock representation in JEOD.
356  double tjt_prev_midnight = std::floor(trunc_julian_time);
357  double fraction_of_day = trunc_julian_time - tjt_prev_midnight;
358 
359  // day of year has value 1 throughout Jan01, etc
360  day_of_year = tjt_prev_midnight - tjt_year_start + 1.0;
361  // Note - this is going to churn on the first pass, so would be better
362  // done as part of initialization -- except this model does not have
363  // an initialize method.
364  // Class defaults kick this off at Jan01, 2000
365  while(day_of_year > max_days_this_year)
366  {
367  day_of_year -= max_days_this_year;
368  tjt_year_start += max_days_this_year;
369  year++;
370  if(year % 4 == 0)
371  {
372  max_days_this_year = 366;
373  }
374  else
375  {
376  max_days_this_year = 365;
377  }
378  }
379  while(day_of_year < 0)
380  {
381  year--;
382  if(year % 4 == 0)
383  {
384  max_days_this_year = 366;
385  }
386  else
387  {
388  max_days_this_year = 365;
389  }
390  day_of_year += max_days_this_year;
391  tjt_year_start -= max_days_this_year;
392  }
393  fraction_of_year = day_of_year / days_per_year;
394 
395  // Compute number of whole days since 1900/01/01::00:00:00.0
396  // trunc_julian_time gives the number of days since the TJT epoch
397  // (1957/05/24::00:00:00). Add 24980 to this value;
398  // 24980 is the number of days from 1900/01/01::00:00:00.0 to TJT epoch
399  double century_days = tjt_prev_midnight + 24980.0;
400  // What we really want, for today as a fraction of a century, is the number
401  // of days to noon today, so we need to add 0.5 to this value.
402  // (Formerly XJ)
403  double century_frac = (century_days + 0.5) / days_per_century;
404 
405  // Computation of celestial longitude requires something previously
406  // labeled FMJD (f? modified-julian-date?)
407  // FMJD = (XMJD - 2435839.0) + (GMT / 1440.0)
408  // XMJD is the modified Julian date at noon today
409  // 2435839.0 is the modified Julian date at 1956/12/31::12:00:00.0
410  // So the first part of that calculation is the number of days since that
411  // epoch to noon today. But then we add the fraction of this day to that
412  // value, so the first part has to represent a time ending at midnight.
413  // So it finishes up being the number of days since 1956/12/31::00:00:00.0
414  // to now. This may be an error.
415  // TODO Turner 10/2017
416  // Investigate whether the original implementation is correct.
417  // This looks suspicious.
418  // Findings: this is in neither of the Jacchia papers; the Jacchia
419  // papers just use the solar-hour-angle and fraction-of-year without
420  // specifying how to compute these angles.
421 
422  // 1956/12/31::00:00:00.0 epoch has a TJT of -4,162.0
423  // Thus fmjd represents days since 1956/12/31::0:0:0.0.
424  // It is not apparent why that date is significant.
425  double fmjd = 4162.0 + tjt_prev_midnight + fraction_of_day;
426 
427  //****************************************************************************
428  // PART B - compute the solar declination angle in radians:
429  //****************************************************************************
430 
431  // Solar declination angle requires celestial longitude.
432  // Compute celestial longitude in radians (formerly XLS).
433  // The rest of the celestial longitude computation requires empirically
434  // derived values:
435  const double B1 = 0.0172028; // i.e. 2pi / (365.25 days)
436  const double B2 = 0.0335;
437  const double B3 = 1.407;
438 
439  double celestial_longitude = std::fmod((B1 * fmjd + B2 * sin(0.017202 * (fmjd - 3.0)) - B3), two_pi);
440 
441  // Compute solar declination angle in radians, range [-pi/2, pi/2]:
442  const double dec_angle_const = (23.4523 - 0.013 * century_frac) * deg_to_rad;
443  solar_declination_angle = std::asin(std::sin(celestial_longitude) * std::sin(dec_angle_const));
444 
445  //****************************************************************************
446  // PART C - compute the solar hour angle in radians, range (-pi, pi]:
447  //****************************************************************************
448  // PART C-1 - compute the right ascension of Sun in radians, range [0, 2pi]:
449  //****************************************************************************
450 
451  // Note - this expression:
452  // ra = asin((tan(asin(sin(celestial_long)*sin(dec_angle_const)))) /
453  // (tan(dec_angle_const)))
454  // can be simplified but still requires computing
455  // - cos(dec_angle_const)
456  // - sqrt(1 - sin^2(cel_long)sin^2(dec_angle_const))
457  // which is 2 computations -- versus the 2 that are necessary to get the
458  // tan(*) values. So just go with the original.
459  double scratch1 = std::tan(solar_declination_angle);
460  double scratch2 = std::tan(dec_angle_const);
461  // protect against out-of-bounds on the next arcsin operation:
462  // If the ratio is out-of-bounds, assign to pi/2.
463  // The RA has to be put into the same quadrant as the celestial longitude,
464  // so for now generate RA in the first quadrant.
465  double solar_right_ascension = M_PI_2;
466  if(std::abs(scratch1) < std::abs(scratch2))
467  {
468  solar_right_ascension = std::abs(asin(scratch1 / scratch2));
469  }
470 
471  // Put Right-ascension in the same quadrant as celestial longitude via
472  // reflection operations. Right-ascension is currently in 1st quadrant.
473  // celestial_longitude is >= 0.0, resulting from the fmod calculation.
474  if(celestial_longitude > three_pi_two)
475  {
476  solar_right_ascension = two_pi - solar_right_ascension;
477  }
478  else if(celestial_longitude > M_PI)
479  {
480  solar_right_ascension = M_PI + solar_right_ascension;
481  }
482  else if(celestial_longitude > M_PI_2)
483  {
484  solar_right_ascension = M_PI - solar_right_ascension;
485  }
486 
487  //****************************************************************************
488  // PART C-2 - compute the solar hour angle in radians, range (-pi, pi]:
489  //****************************************************************************
490 
491  // Compute greenwich mean position in degrees.
492  // This algorithm requires some empirically derived constants and the
493  // number of minutes passed since the most recent midnight (00h).
494  double minutes_of_day = fraction_of_day * minutes_per_day;
495  const double A1 = 99.6909833;
496  const double A2 = 36000.76892;
497  const double A3 = 0.00038708;
498  const double A4 = 0.250684477;
499  double greenwich_mean_position = std::fmod((A1 + (A2 * century_frac) + (A3 * century_frac * century_frac) +
500  (A4 * minutes_of_day)),
501  360.0);
502 
503  // previous algorithm's application of constraints on right ascension
504  // point (RAP) was unnecessary because it is local, and constraint gets
505  // applied in the computation of solar-hour-angle anyway.
506  double right_ascension_point = greenwich_mean_position * deg_to_rad + longitude;
507  solar_hour_angle = right_ascension_point - solar_right_ascension;
508  while(solar_hour_angle > M_PI)
509  {
510  solar_hour_angle -= two_pi;
511  }
512  while(solar_hour_angle < -M_PI)
513  {
514  solar_hour_angle += two_pi;
515  }
516 }

558  // PART A - compute the solar-activity variation
559  //****************************************************************************
560  // Ci are solar activity variables
561  // TODO 1970/71 inconsistency
562  // 1970:
563  const double C1 = 383.0;
564  const double C2 = 3.32;

576  //****************************************************************************a
577  // angles beta = - 37 degrees
578  // gamma = 43 degrees
579  // p = 6 degrees

1149 double METAtmosphere::apply_gauss_quadrature(int altitude_index_start, double altitude_end)
1150 {
1151  // Gauss quadrature is used to approximate the integration over height of
1152  // either (mu g / T) or (g / T)
1153 
1154  // Between 90km and the barometric ceiling, (i.e. cell 0), the integration
1155  // uses the barometric equation, which requires the mean molecular weight
1156  // of the entire atmosphere. This is a variable, so it must be included
1157  // in the integrand.
1158  // Above the barometric-ceiling (i.e. cells 1 and higher), the integration
1159  // uses the diffusion equation for each species. In this case, each species
1160  // uses its own molecular weight and multiplies that by the integral of (g /
1161  // RT)
1162 
1163  // Cannot integrate both equations in a single call.
1164  if(altitude_index_start == 0 && altitude_end > barometric_equation_ceiling * 1.0001)
1165  {
1166  MessageHandler::error(__FILE__,
1167  __LINE__,
1168  "Invalid call.\n",
1169  "apply_gauss_quadrature was called for an altitude range starting in\n"
1170  "the barometric-equation range and ending in the diffusion-equation\n"
1171  "range. This requires two separate calls.\n");
1172  }
1173 
1174  bool include_mu = false;
1175  if(altitude_index_start == 0)
1176  {
1177  include_mu = true;
1178  }
1179 
1180  double integral = 0.0;
1181 
1182  // Divide the integration down into individual cells, compute the integral in
1183  // each cell, and add them together at the end to get the integral over the
1184  // whole range.
1185  for(unsigned int ii = altitude_index_start; ii < 8 && gauss_altitudes[ii] < altitude_end; ++ii)
1186  {
1187  unsigned int gauss_order = gauss_n[ii];
1188  double alt_lo = gauss_altitudes[ii];
1189  double alt_hi = std::min(altitude_end, gauss_altitudes[ii + 1]);
1190  double half_cell_height = 0.5 * (alt_hi - alt_lo);
1191 
1192  // - Now within each cell, apply a Gauss Quadrature integration of the
1193  // integrand (mu * g / (R * T)).
1194  // - mu, g, and T are functions of altitude, whereas R is constant. So we
1195  // really only need to integrate (mu * g / T) and then divide that by R.
1196  // - Gauss quadrature integration involves
1197  // computing the value of the function at several locations and summing
1198  // those.
1199  double integral_cell = 0.0;
1200  for(unsigned int kk = 0; kk < gauss_order; ++kk)
1201  {
1202  // The integrand is a function of altitude, so generate the altitude
1203  // for each of the Gauss Quadrature reference points in the interval.
1204  // The gauss reference-points are specified in a range [-1,1] where -1
1205  // corresponds to the lower bound and +1 to the upper bound of the
1206  // integration range. For this reference point, convert the "location"
1207  // value to a real altitude:
1208  double alt_eval_point = alt_lo + half_cell_height * (1.0 + GaussQuadrature::gauss_xvalues[gauss_order][kk]);
1209  // express this altitude as a radius as a multiple of a standard
1210  // earth-radius
1211  // r_G = r / r_E = (r_E + alt) / r_E = 1 + alt / r_E
1212  // Then, letting g_E be the gravitational accel at earth's surface:
1213  // g = GM/(r^2) = (g_E * r_E ^2) / r^2 = g_E / r_G^2
1214  //
1215  // Magic numbers 9.80665 is local g when r = 6.357666E3 km.
1216  double rad_eval_point = 1.0 + (alt_eval_point / 6.356766E3);
1217  double grav = 9.80665 / (rad_eval_point * rad_eval_point);
1218 
1219  // Now evaluate the function
1220  double value_eval_point = grav / thermal.compute_temperature(alt_eval_point);
1221  if(include_mu)
1222  {
1223  value_eval_point *= compute_mol_wt(alt_eval_point);
1224  }
1225  // weight this value and add it to the integral over the cell.
1226  integral_cell += GaussQuadrature::gauss_weights[gauss_order][kk] * value_eval_point;
1227  }
1228  // We now have a scaled version of the integration over the cell. The
1229  // Gauss quadrature assumes integration over [-1,1] and we need an
1230  // integration from alt_lo to alt_hi. So scale by
1231  // (real distance) / (theoretical distance), i.e. by
1232  // (alt_hi - alt_lo) /2, which we already have computed.
1233  integral_cell *= half_cell_height;
1234 
1235  // Now sum that to the total integral
1236  integral += integral_cell;
1237 
1238  // Continue onto the next cell.
1239  }
1240  return integral;
1241 }
