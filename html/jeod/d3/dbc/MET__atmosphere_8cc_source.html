190 /*****************************************************************************
191 compute_temperature
192 
193 PURPOSE:
194  (Calculates the temperature at a specified altitude
195  using equation (10) for altitudes between 90 and 125 km and equation (13)
196  for altitudes greater than 125 km, from Smithsonian Astrophysical
197  Observatory Report 313, 1970.)
198 
199 NOTES: (This method is used to generate the temperature at the current
200  altitude AND at other reference altitudes in support of other aspects
201  of the model. Therefore, it is necessary to have the altitude as a
202  passed argument. The argument should not be assumed to be the vehicle
203  altitude because it often is not.)
204 
205 REFERENCE:
206  (((Jacchia, L.G.) (New Static Models of the Thermosphere and
207  Exosphere with Empirical Temperature Profiles) (Smithsonian
208  Astrophysical Observatory Special Report No. 313) (--) (1970) (--))
209  ((Jacchia, L.G.) (Revised Static Models of the Termosphere
210  and Exosphere with Emperical Temperature Profiles)
211  (Smithsonian Astrophysical Observatory Report No. 332) (--)
212  (1971) (--)))
213 
214 Assumptions:(( All temperature profiles:
215  - start at height 90km with value 183K and gradient 0.
216  - rise to an inflection point at 125 km
217  - become asymptotic to the exospheric temperature above 125 km.)
218  ( The temperature T_125 and gradient T_grad_125 at the
219  inflection point are functions of the exospheric temperature.)
220  )
221 
222 
223 *****************************************************************************/
224 double
225 METAtmosphereThermal::compute_temperature(
226  double altitude_km_in)
227 {
228  // temperature-offset is the difference between temperatures at the 125km and
229  // 90km altitude reference points:
230 
231  // For altitude in the range 90 km to 125 km, temperature is computed as
232  // a polynomial function of altitude difference from 125 km
233  // (see Jacchia(1971), p11, eq(10) or Jacchia(1970), p 9, eq(10)):
234  // T = T_125 + sum {i=1 to 4} (c_i * (z-125)^i)
235  // Satisfaction of the boundary conditions (see Jacchia(1971), p 12, eq(11))
236  // implies that:
237  // c_1 = (1.9 / 35) * (T_125 - T_90) see eq(11c)
238  // c_2 = 0 eq(11d)
239  // c_3 = ((4 - 3(1.9)) / (35)^3) * (T_125 - T_90) eq(11a,b)
240  // c_4 = ((3 - 2(1.9)) / (35)^4) * (T_125 - T_90) eq(11a,b)
241 
242  // Notes:
243  // - c_1 is the temperature gradient at 125km
244  // - the constants at the front of each c_i, e.g. (1.9 / 35) are stored
245  // in the class as values k_i:
246  // c_i = k_i * (T_125 - T_90)
247  // - the values c_i are not actually computed, the common factor
248  // (T_125 - T_90) is pulled out of the summation so that the
249  // values k_i can be used instead.
250 
251  // For altitudes > 125km, a different algorithm is used,
252  // (see Jacchia(1971), p 12, eq(13))
253 
254  // In both computations, the values (dz), (dz)^2, and dT are needed. so
255  // generate those first:
256  // base-altitude is delta-altitude above 125 km.
257  double dz = altitude_km_in - 125.0;
258  double dz_2 = dz * dz;
259  double dT = T_125 - T_90;
260 
261  if (dz <= 0.0) {
262  // obtain the temperature as a polynomial in z:
263  double dz_3 = dz * dz_2;
264  double dz_4 = dz_2 * dz_2;
265  return T_125 + dT * ((k_1 * dz) +
266  (k_3 * dz_3) +
267  (k_4 * dz_4));
268  }
269  // else use equation 13, which is identical both Jacchia papers:
270  double dz_2_5 = dz_2 * sqrt (dz);
271  double coeff_A = 2 * (T_exosphere - T_125) / M_PI;
272  // magic number 4.5E-6 is empirical, identified as "B" in eq(13).
273  return T_125 + coeff_A *
274  std::atan2 ( k_1 * dT * dz * (1.0 + (4.5E-6 * dz_2_5)),
275  coeff_A);
276 }

411 
412 PURPOSE:
413  (Subroutine 'atmos_MET_TME' performs the calculations of the solar
414  declination angle and solar hour angle.)
415 
416 REFERENCE:
417  (((Jacchia, L.G.) (New Static Models of the Thermosphere and
418  Exosphere with Empirical Temperature Profiles) (Smithsonian
419  Astrophysical Observatory Special Report No. 313) (--) (1970) (--))
420  ((Jacchia, L.G.) (Revised Static Models of the Termosphere
421  and Exosphere with Emperical Temperature Profiles)
422  (Smithsonian Astrophysical Observatory Report No. 332) (--)
423  (1971) (--)))
424 
425 ASSUMPTIONS AND LIMITATIONS:
426  ((Too many to enumerate here.))
427 
428 
429 ******************************************************************************/
430 void
431 METAtmosphere::compute_solar_angles()
432 {
433  //****************************************************************************
434  // PART A - compute the necessary time representations
435  //****************************************************************************
436 
437  // Compute the fraction of day, year, and century:
438  // Truncated Julian Time (tjt, TJT) has epoch at 1957/05/24::00:00:00
439  // It is a principal clock representation in JEOD.
440  double tjt_prev_midnight = std::floor( trunc_julian_time);
441  double fraction_of_day = trunc_julian_time - tjt_prev_midnight;
442 
443  // day of year has value 1 throughout Jan01, etc
444  day_of_year = tjt_prev_midnight - tjt_year_start + 1.0;
445  // Note - this is going to churn on the first pass, so would be better
446  // done as part of initialization -- except this model does not have
447  // an initialize method.
448  // Class defaults kick this off at Jan01, 2000
449  while (day_of_year > max_days_this_year) {
450  day_of_year -= max_days_this_year;
451  tjt_year_start += max_days_this_year;
452  year++;
453  if (year % 4 == 0) {
454  max_days_this_year = 366;
455  }
456  else {
457  max_days_this_year = 365;
458  }
459  }
460  while (day_of_year < 0) {
461  year--;
462  if (year % 4 == 0) {
463  max_days_this_year = 366;
464  }
465  else {
466  max_days_this_year = 365;
467  }
468  day_of_year += max_days_this_year;
469  tjt_year_start -= max_days_this_year;
470  }
471  fraction_of_year = day_of_year / days_per_year;
472 
473  // Compute number of whole days since 1900/01/01::00:00:00.0
474  // trunc_julian_time gives the number of days since the TJT epoch
475  // (1957/05/24::00:00:00). Add 24980 to this value;
476  // 24980 is the number of days from 1900/01/01::00:00:00.0 to TJT epoch
477  double century_days = tjt_prev_midnight + 24980.0;
478  // What we really want, for today as a fraction of a century, is the number
479  // of days to noon today, so we need to add 0.5 to this value.
480  // (Formerly XJ)
481  double century_frac = (century_days + 0.5) / days_per_century;
482 
483  // Computation of celestial longitude requires something previously
484  // labeled FMJD (f? modified-julian-date?)
485  // FMJD = (XMJD - 2435839.0) + (GMT / 1440.0)
486  // XMJD is the modified Julian date at noon today
487  // 2435839.0 is the modified Julian date at 1956/12/31::12:00:00.0
488  // So the first part of that calculation is the number of days since that
489  // epoch to noon today. But then we add the fraction of this day to that
490  // value, so the first part has to represent a time ending at midnight.
491  // So it finishes up being the number of days since 1956/12/31::00:00:00.0
492  // to now. This may be an error.
493  // TODO Turner 10/2017
494  // Investigate whether the original implementation is correct.
495  // This looks suspicious.
496  // Findings: this is in neither of the Jacchia papers; the Jacchia
497  // papers just use the solar-hour-angle and fraction-of-year without
498  // specifying how to compute these angles.
499 
500  // 1956/12/31::00:00:00.0 epoch has a TJT of -4,162.0
501  // Thus fmjd represents days since 1956/12/31::0:0:0.0.
502  // It is not apparent why that date is significant.
503  double fmjd = 4162.0 + tjt_prev_midnight + fraction_of_day;
504 
505  //****************************************************************************
506  // PART B - compute the solar declination angle in radians:
507  //****************************************************************************
508 
509  // Solar declination angle requires celestial longitude.
510  // Compute celestial longitude in radians (formerly XLS).
511  // The rest of the celestial longitude computation requires empirically
512  // derived values:
513  const double B1 = 0.0172028; // i.e. 2pi / (365.25 days)
514  const double B2 = 0.0335;
515  const double B3 = 1.407;
516 
517  double celestial_longitude = std::fmod( (B1 * fmjd +
518  B2 * sin(0.017202 * (fmjd - 3.0)) -
519  B3),
520  two_pi);
521 
522  // Compute solar declination angle in radians, range [-pi/2, pi/2]:
523  const double dec_angle_const = (23.4523 - 0.013*century_frac) * deg_to_rad;
524  solar_declination_angle = std::asin( std::sin(celestial_longitude) *
525  std::sin(dec_angle_const));
526 
527  //****************************************************************************
528  // PART C - compute the solar hour angle in radians, range (-pi, pi]:
529  //****************************************************************************
530  // PART C-1 - compute the right ascension of Sun in radians, range [0, 2pi]:
531  //****************************************************************************
532 
533  // Note - this expression:
534  // ra = asin((tan(asin(sin(celestial_long)*sin(dec_angle_const)))) /
535  // (tan(dec_angle_const)))
536  // can be simplified but still requires computing
537  // - cos(dec_angle_const)
538  // - sqrt(1 - sin^2(cel_long)sin^2(dec_angle_const))
539  // which is 2 computations -- versus the 2 that are necessary to get the
540  // tan(*) values. So just go with the original.
541  double scratch1 = std::tan( solar_declination_angle);
542  double scratch2 = std::tan( dec_angle_const);
543  // protect against out-of-bounds on the next arcsin operation:
544  // If the ratio is out-of-bounds, assign to pi/2.
545  // The RA has to be put into the same quadrant as the celestial longitude,
546  // so for now generate RA in the first quadrant.
547  double solar_right_ascension = M_PI_2;
548  if ( std::abs(scratch1) < std::abs(scratch2)) {
549  solar_right_ascension = std::abs( asin( scratch1 / scratch2));
550  }
551 
552  // Put Right-ascension in the same quadrant as celestial longitude via
553  // reflection operations. Right-ascension is currently in 1st quadrant.
554  // celestial_longitude is >= 0.0, resulting from the fmod calculation.
555  if (celestial_longitude > three_pi_two) {
556  solar_right_ascension = two_pi - solar_right_ascension;
557  }
558  else if (celestial_longitude > M_PI) {
559  solar_right_ascension = M_PI + solar_right_ascension;
560  }
561  else if (celestial_longitude > M_PI_2) {
562  solar_right_ascension = M_PI - solar_right_ascension;
563  }
564 
565  //****************************************************************************
566  // PART C-2 - compute the solar hour angle in radians, range (-pi, pi]:
567  //****************************************************************************
568 
569  // Compute greenwich mean position in degrees.
570  // This algorithm requires some empirically derived constants and the
571  // number of minutes passed since the most recent midnight (00h).
572  double minutes_of_day = fraction_of_day * minutes_per_day;
573  const double A1 = 99.6909833;
574  const double A2 = 36000.76892;
575  const double A3 = 0.00038708;
576  const double A4 = 0.250684477;
577  double greenwich_mean_position =
578  std::fmod (( A1 +
579  (A2 * century_frac) +
580  (A3 * century_frac * century_frac) +
581  (A4 * minutes_of_day)),
582  360.0);

780  // rho_1 / rho_0 = ((M_1 / T_1) / (M_0 / T_0)) * exp(int{ -(Mg/RT) dz})

784  // rho_1 = rho_0 * (M_1 * T_0) / (M_0 * T_1) * exp(int{ -(Mg/RT) dz})

788  // 2.1926E-8 (g K

814  // not constant, so this does not have a constant value at 105km, so even
815  // if the integration ceiling is 105km, it still has to be integrated.
816  double integral_Mg_RT = apply_gauss_quadrature ( 0,
817  integration_ceiling_A) /
818  R_gas_constant;

820  // Now put it all together:
821 
822  // Magic number:
823  // FIXME Turner 2018-03

825  // 2.1926E-5 was previously identified as being the product of:

835  // 28.82678 (1971)

895  }

899  species.num_density[4] *= std::pow( temp_ratio, -0.38);

904  // already set to 1.0
905  //****************************************************************************

907  double temperature_500 = thermal.compute_temperature(500.0);
908  double log_temperature_500 = log10( temperature_500);

910  double integral_g_RT = apply_gauss_quadrature ( 6, altitude_km) /

912

915  // 79.13 = documented 73.13 + 6 to go from cm^-3 to m^-3

917  // 5.5 as documented
918  // documented values from Kockarts and Nicolet (1962, 1963)

921  (79.13 - (39.4 * log_temperature_500) +

924  std::exp (-species.mol_weight[5] * integral_g_RT);
925  }
926

928  if (species.num_density[ii] < 1.0) {
929  species.num_density[ii] = 1.0;
930  }

1234 METAtmosphere::apply_gauss_quadrature(
1235  int altitude_index_start,
1236  double altitude_end)
1237 {
1238  // Gauss quadrature is used to approximate the integration over height of
1239  // either (mu g / T) or (g / T)
1240 
1241  // Between 90km and the barometric ceiling, (i.e. cell 0), the integration
1242  // uses the barometric equation, which requires the mean molecular weight
1243  // of the entire atmosphere. This is a variable, so it must be included
1244  // in the integrand.
1245  // Above the barometric-ceiling (i.e. cells 1 and higher), the integration
1246  // uses the diffusion equation for each species. In this case, each species
1247  // uses its own molecular weight and multiplies that by the integral of (g /
1248  // RT)
1249 
1250  // Cannot integrate both equations in a single call.
1251  if (altitude_index_start == 0 &&
1252  altitude_end > barometric_equation_ceiling * 1.0001){
1253  MessageHandler::error(
1254  __FILE__,__LINE__,"Invalid call.\n",
1255  "apply_gauss_quadrature was called for an altitude range starting in\n"
1256  "the barometric-equation range and ending in the diffusion-equation\n"
1257  "range. This requires two separate calls.\n");
1258  }
1259  bool include_mu = (altitude_index_start == 0) ? true : false;
1260  double integral = 0.0;
1261 
1262  // Divide the integration down into individual cells, compute the integral in
1263  // each cell, and add them together at the end to get the integral over the
1264  // whole range.
1265  for (unsigned int ii = altitude_index_start;
1266  ii < 8 && gauss_altitudes[ii] < altitude_end;
1267  ++ii) {
1268  unsigned int gauss_order = gauss_n[ii];
1269  double alt_lo = gauss_altitudes[ii];
1270  double alt_hi = std::min( altitude_end, gauss_altitudes[ii+1]);
1271  double half_cell_height = 0.5 * (alt_hi - alt_lo);
1272 
1273  // - Now within each cell, apply a Gauss Quadrature integration of the
1274  // integrand (mu * g / (R * T)).
1275  // - mu, g, and T are functions of altitude, whereas R is constant. So we
1276  // really only need to integrate (mu * g / T) and then divide that by R.
1277  // - Gauss quadrature integration involves
1278  // computing the value of the function at several locations and summing
1279  // those.
1280  double integral_cell = 0.0;
1281  for (unsigned int kk = 0; kk < gauss_order; ++kk) {
1282  // The integrand is a function of altitude, so generate the altitude
1283  // for each of the Gauss Quadrature reference points in the interval.
1284  // The gauss reference-points are specified in a range [-1,1] where -1
1285  // corresponds to the lower bound and +1 to the upper bound of the
1286  // integration range. For this reference point, convert the "location"
1287  // value to a real altitude:
1288  double alt_eval_point = alt_lo + half_cell_height * (1.0 +
1289  GaussQuadrature::gauss_xvalues[gauss_order][kk]);
1290  // express this altitude as a radius as a multiple of a standard
1291  // earth-radius
1292  // r_G = r / r_E = (r_E + alt) / r_E = 1 + alt / r_E
1293  // Then, letting g_E be the gravitational accel at earth's surface:
1294  // g = GM/(r^2) = (g_E * r_E ^2) / r^2 = g_E / r_G^2
1295  //
1296  // Magic numbers 9.80665 is local g when r = 6.357666E3 km.
1297  double rad_eval_point = 1.0 + (alt_eval_point / 6.356766E3);
1298  double grav = 9.80665 / (rad_eval_point * rad_eval_point);
1299 
1300  // Now evaluate the function
1301  double value_eval_point =
1302  grav / thermal.compute_temperature(alt_eval_point);
1303  if (include_mu) {
1304  value_eval_point *= compute_mol_wt(alt_eval_point);
1305  }
1306  // weight this value and add it to the integral over the cell.
1307  integral_cell += GaussQuadrature::gauss_weights[gauss_order][kk] *
1308  value_eval_point;
1309  }
1310  // We now have a scaled version of the integration over the cell. The
1311  // Gauss quadrature assumes integration over [-1,1] and we need an
1312  // integration from alt_lo to alt_hi. So scale by
1313  // (real distance) / (theoretical distance), i.e. by
1314  // (alt_hi - alt_lo) /2, which we already have computed.
1315  integral_cell *= half_cell_height;
1316 
1317  // Now sum that to the total integral
1318  integral += integral_cell;
1319 
1320  // Continue onto the next cell.
1321  }
1322  return integral;
1323 }
