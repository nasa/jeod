494 /**
495  * Abstract class for describing a type without necessarily needing
496  * to create a JeodMemoryTypeDescriptor of that type. The intent is to
497  * avoid creating a type descriptor for a type if the type is already
498  * represented in the type table.
499  *
500  * Usage of a JeodMemoryTypePreDescriptor is highly constrained.
501  * There are two simple rules:
502  * - Never cache a pointer or reference to a JeodMemoryTypeDescriptor
503  * in long-term memory.
504  * - Never cache a pointer or reference to a JeodMemoryTypeDescriptor obtained
505  * by calling the JeodMemoryTypeDescriptor's get_descriptor method.
506  */
507 class JeodMemoryTypePreDescriptor
508 {
509 public:
510  virtual ~JeodMemoryTypePreDescriptor() = default;
511 
512  /**
513  * Get the type info for the type.
514  * @return Type info
515  */
516  virtual const std::type_info & get_typeid() const = 0;
517 
518  /**
519  * Get a type descriptor for the type.
520  * The returned value should not be cached in a permanent store.
521  * The reference has a lifespan limited to that of the
522  * JeodMemoryTypePreDescriptor object.
523  * @return Type descriptor.
524  */
525  virtual const JeodMemoryTypeDescriptor & get_descriptor() = 0;
526 };

531 template<typename Type> class JeodMemoryTypePreDescriptorDerived : public JeodMemoryTypePreDescriptor
532 {
533 public:
534  // Typedefs
535 
536  /**
537  * The type descriptor this class describes.
538  */
539  using TypeDescriptor = JeodMemoryTypeDescriptorDerived<Type>;
540 
541  /**
542  * Default constructor.
543  */
544  explicit JeodMemoryTypePreDescriptorDerived(bool exportable = true)
545  : is_exportable(exportable)
546  {
547  }
548 
549  /**
550  * Copy constructor.
551  */
552  JeodMemoryTypePreDescriptorDerived(const JeodMemoryTypePreDescriptorDerived & src)
553  : is_exportable(src.exportable)
554  {
555  if(src.descriptor)
556  {
557  descriptor = new TypeDescriptor(*src.descriptor);
558  }
559  }
560 
561  /**
562  * Destructor.
563  */
564  ~JeodMemoryTypePreDescriptorDerived() override
565  {
566  delete descriptor;
567  }
568 
569  /**
570  * Get a reference to this object.
571  *
572  * This is an utter hack. Because the descriptor is created after the fact,
573  * a function that receives a JeodMemoryTypePreDescriptor must either take
574  * a copy or a non-const reference as input. A reference is preferred.
575  * The problem: Non-const references cannot be bound to rvalues. They can
576  * however be bound to other references, and hence this method.
577  *
578  * Note well: The returned reference has a lifespan limited to that of this
579  * object. Use with great care. This is not intended for general consumption.
580  *
581  * @return Reference to this object.
582  */
583  JeodMemoryTypePreDescriptor & get_ref()
584  {
585  return *this;
586  }
587 
588  /**
589  * Get the type info for the type.
590  * @return Type info
591  */
592  const std::type_info & get_typeid() const override
593  {
594  return typeid(Type);
595  }
596 
597  /**
598  * Get a type descriptor for the type.
599  *
600  * Note well: The referenced value has a lifespan limited to that of this
601  * object. The returned value must not be cached in a permanent store.
602  * Use new in conjunction with the copy constructor instead.
603  *
604  * @return Type descriptor.
605  */
606  const JeodMemoryTypeDescriptor & get_descriptor() override
607  {
608  if(!descriptor)
609  {
610  descriptor = new TypeDescriptor(is_exportable);
611  }
612  return *descriptor;
613  }
614 
615 private:
616  TypeDescriptor * descriptor{}; // trick_io(**)
617  bool is_exportable{true}; // trick_io(**)
618 };
