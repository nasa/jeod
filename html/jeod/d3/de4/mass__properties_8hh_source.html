70 /**
71  * Defines mass properties -- mass and inertia tensor.
72  * A MassBody contains two MassProperties objects that represent the body's
73  * core and composite mass properties.
74  * A MassProperties object is-a MassPoint and hence a MassPointState
75  * by inheritance. The parent point of a MassProperties object is always
76  * the structural frame of the MassBody that 'owns' the MassProperties
77  * object.
78  *
79  * Note that a MassProperties does not explicitly define the center of mass
80  * location and body frame orientation. These properties are defined by
81  * elements inherited from MassPointState:
82  * - position defines the center of mass in the structural frame.
83  * - Q_parent_this/T_parent_this define the orientation of the body frame
84  * with respect to the structural frame.
85  */
86 class MassProperties : public MassPoint
87 {
88  JEOD_MAKE_SIM_INTERFACES(jeod, MassProperties)
89 
90  // Member data
91 public:
92  /**
93  * Mass of the subject mass element (always >= 0). The core mass element's
94  * mass is set externally while the composite mass element's mass is computed
95  * by the mass model.
96  */
97  double mass{}; //!< trick_units(kg)
98 
99  /**
100  * Inertia tensor of the subject mass element about the subject mass element's
101  * center of mass, expressed in subject mass element body coordinates.
102  * Note: The diagonal elements of the inertia tensor are positive moments
103  * of inertial while the off-diagonal elements are megative
104  * products of inertia.
105  */
106  double inertia[3][3]{}; //!< trick_units(kg*m2)
107 
108  /**
109  * The inverse of mass.
110  * Used only for root dynamic bodies composite props.
111  */
112  double inverse_mass{}; //!< trick_units(1/kg)
113 
114  /**
115  * The inverse of the composite inertia.
116  * Used only for root dynamic bodies composite props.
117  */
118  double inverse_inertia[3][3]{}; //!< trick_units(1/kg/m2)
119 
120  // Member functions
121 public:
122  MassProperties() = default;
123  MassProperties(const MassProperties &) = delete;
124  MassProperties & operator=(const MassProperties &) = delete;
125 };
