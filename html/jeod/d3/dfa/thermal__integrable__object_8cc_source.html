91 /**
92  * Integrate the thermal state
93  * @param[in] dyn_dt Dynamic time step, in dynamic time seconds.
94  * @param[in] target_stage The stage of the integration process
95  * that the integrator should try to attain.
96  * @return The status (time advance, pass/fail status) of the integration.
97  */
98 er7_utils::IntegratorResult ThermalIntegrableObject::integrate(double dyn_dt, unsigned int target_stage)
99 {
100  double t_eq_pow4 = rider->power_absorb / rider->rad_constant;
101  er7_utils::IntegratorResult result = integrator.integrate(dyn_dt, target_stage, &temp_dot, &temp);
102 
103  t_pow4 = temp * temp;
104  t_pow4 *= t_pow4;
105 
106  if(temp_dot * (t_eq_pow4 - t_pow4) < 0.0)
107  { // overshot the asymtotic value
108  MessageHandler::inform(__FILE__,
109  __LINE__,
110  ThermalMessages::invalid_integration_operation,
111  "\n"
112  "Temperature integration produced instability.\n"
113  "The integration step is sufficiently large that the predicted\n"
114  "temperature at stage %u of the integration is on the \n"
115  "wrong side of the asymptote. \n"
116  "The predicted temperature is being set to the equilibrium"
117  "temperature.\n",
118  target_stage);
119  t_pow4 = t_eq_pow4;
120  temp = std::sqrt(std::sqrt(t_pow4));
121  }
122 
123  return result;
124 }
