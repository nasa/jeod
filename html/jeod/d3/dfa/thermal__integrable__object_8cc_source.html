105 /**
106  * Integrate the thermal state
107  * @param[in] dyn_dt Dynamic time step, in dynamic time seconds.
108  * @param[in] target_stage The stage of the integration process
109  * that the integrator should try to attain.
110  * @return The status (time advance, pass/fail status) of the integration.
111  */
112 er7_utils::IntegratorResult
113 ThermalIntegrableObject::integrate (
114  double dyn_dt,
115  unsigned int target_stage)
116 {
117  double t_eq_pow4 = rider->power_absorb/rider->rad_constant;
118  er7_utils::IntegratorResult result =
119  integrator.integrate (dyn_dt, target_stage, &temp_dot, &temp);
120 
121  t_pow4 = temp*temp;
122  t_pow4*=t_pow4;
123 
124  if (temp_dot*(t_eq_pow4-t_pow4) < 0.0) { // overshot the asymtotic value
125  MessageHandler::inform (
126  __FILE__, __LINE__, ThermalMessages::invalid_integration_operation,
127  "\n"
128  "Temperature integration produced instability.\n"
129  "The integration step is sufficiently large that the predicted\n"
130  "temperature at stage %u of the integration is on the \n"
131  "wrong side of the asymptote. \n"
132  "The predicted temperature is being set to the equilibrium"
133  "temperature.\n", target_stage);
134  t_pow4 = t_eq_pow4;
135  temp = std::sqrt(std::sqrt(t_pow4));
136  }
137 
138  return result;
139 }
