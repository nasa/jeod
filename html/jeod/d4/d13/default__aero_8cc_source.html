104

106 
107  if (std::fpclassify(BC) == FP_ZERO) {

116  }
117  drag = -(aero_drag_param_ptr->dynamic_pressure * mass) / BC;
118  break;
119 
120  case DRAG_OPT_CONST:
121

123  // so nothing further must be done by this algorithm until after
124  // the case statement
125

134  drag = 0.0;

138
