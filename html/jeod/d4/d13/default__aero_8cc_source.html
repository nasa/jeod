59 void DefaultAero::aerodrag_force(const double,
60  const double rel_vel_hat[3],
61  AeroDragParameters * aero_drag_param_ptr,
62  double mass,
63  double force[3],
64  double torque[3])
65 {
66  switch(option)
67  {
68  case DRAG_OPT_CD:
69 
70  drag = -aero_drag_param_ptr->dynamic_pressure * area * Cd;
71  break;
72 
73  case DRAG_OPT_BC:
74 
75  if(std::fpclassify(BC) == FP_ZERO)
76  {
77  MessageHandler::fail(__FILE__,
78  __LINE__,
79  AerodynamicsMessages::runtime_error,
80  "The Aerodynamics object was told to use ballistic "
81  "coefficient, but no coefficient was set. This is "
82  "a fatal error");
83  }
84  drag = -(aero_drag_param_ptr->dynamic_pressure * mass) / BC;
85  break;
86 
87  case DRAG_OPT_CONST:
88 
89  break; // The drag has been set in the variable "drag" by the user,
90  // so nothing further must be done by this algorithm until after
91  // the case statement
92 
93  default:
94 
95  MessageHandler::warn(__FILE__,
96  __LINE__,
97  AerodynamicsMessages::runtime_warns,
98  "No option was sent to the Aerodynamics object for calculating "
99  "drag. Drag has been set to zero.");
100 
101  drag = 0.0;
102 
103  break;
104  }
105 
106  Vector3::scale(rel_vel_hat, drag, force);
107  Vector3::initialize(torque);
108 }
