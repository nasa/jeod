42 /**
43  * Update composite mass properties for a mass tree marked for update.
44  * The properties are updated from the bottom up to generate correct results.
45  *
46  * \par Assumptions and Limitations
47  * - Rigid bodies
48  * - Tree attachment structure
49  */
50 void MassBody::update_mass_properties()
51 {
52  // If this body is not marked for update then just return
53  if(!needs_update)
54  {
55  return;
56  }
57 
58  // The core and composite properties are the same for an atomic body.
59  if(links.is_atomic())
60  {
61  composite_properties.mass = core_properties.mass;
62  if(composite_properties.mass > 0.0)
63  {
64  composite_properties.inverse_mass = 1.0 / composite_properties.mass;
65  }
66  else
67  {
68  composite_properties.inverse_mass = 0.0;
69  }
70 
71  Vector3::copy(core_properties.position, composite_properties.position);
72  Vector3::initialize(core_wrt_composite.position);
73 
74  Matrix3x3::copy(core_properties.inertia, composite_properties.inertia);
75  }
76 
77  // Calculate the composite properties for a composite body.
78  else
79  {
80  // First update all of the attached bodies' mass properties, if needed.
81  for(auto * link : TreeLinksChildrenRange<MassBodyLinks>(links))
82  {
83  MassBody * child = &(link->container());
84  if(child->needs_update)
85  {
86  child->update_mass_properties();
87  }
88  }
89 
90  // Calculate the composite mass and center of mass.
91  calc_composite_cm();
92 
93  // Calculate the location of the child's composite CoM wrt this body's
94  // composite CoM in this body's body frame.
95  double r_cm_cm_str[3]; // M CoM to CoM in structural coords
96  for(auto * link : TreeLinksChildrenRange<MassBodyLinks>(links))
97  {
98  MassBody & child = link->container();
99 
100  Vector3::diff(child.composite_wrt_pstr.position, composite_properties.position, r_cm_cm_str);
101  Vector3::transform(composite_properties.T_parent_this, r_cm_cm_str, child.composite_wrt_pbdy.position);
102  }
103 
104  // Calculate the location of the parent's (i.e. this body) core CoM
105  // wrt its new composite CoM in this body's body frame.
106  Vector3::diff(core_properties.position, composite_properties.position, r_cm_cm_str);
107  Vector3::transform(composite_properties.T_parent_this, r_cm_cm_str, core_wrt_composite.position);
108 
109  // Calculate the composite mass, CoM, and inertia for this body.
110  calc_composite_inertia();
111  }
112 
113  // For a root body, calculate the inverse inertia tensor (this is only
114  // done for root bodies) and set the rather meaningless locations of
115  // of various points with respect to a parent point to zero.
116  if(links.is_root())
117  {
118  if(compute_inverse_inertia && (composite_properties.mass > 0.0))
119  {
120  Matrix3x3::invert_symmetric(composite_properties.inertia, composite_properties.inverse_inertia);
121  }
122  else
123  {
124  Matrix3x3::initialize(composite_properties.inverse_inertia);
125  }
126 
127  Vector3::initialize(structure_point.position);
128  Vector3::initialize(composite_wrt_pstr.position);
129  Vector3::initialize(composite_wrt_pbdy.position);
130  }
131 
132  // For non-root bodies compute the location of the composite CoM wrt the
133  // the parent structural origin in the parent structural frame.
134  // Note that the location of the composite CoM wrt the parent composite CoM
135  // is calculated above for child bodies using this element of the child
136  // structure.
137  else
138  {
139  Vector3::transform_transpose(structure_point.T_parent_this,
140  composite_properties.position,
141  composite_wrt_pstr.position);
142  Vector3::incr(structure_point.position, composite_wrt_pstr.position);
143  }
144 
145  // Clear the update flag
146  needs_update = false;
147 }
