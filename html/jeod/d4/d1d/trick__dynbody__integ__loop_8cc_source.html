96 // Initialize an integration loop object.
97 void JeodDynbodyIntegrationLoop::initialize_integ_loop()
98 {
99  // The integrator_constructor should have been populated at this point.
100  if(*integ_constructor == nullptr)

103  __LINE__,
104  SimInterfaceMessages::integration_error,
105  "The integrator constructor for a JeodDynbodyIntegrationLoop "
106  "has not been initialized.");
107  return;
108  }
109 
110  // Create the loop's integration group object by
111  // using the provided integration group as a factory.
112  integ_group = integ_group_factory->create_group(*this,
113  **integ_constructor,
114  integ_interface,

117 
118  // Let the dynamics manager know about the group.
119  dyn_manager->add_integ_group(*integ_group);
120 
121  // Construct this loops derivative/integration/etc. job queues.
122  rebuild_jobs();

125 // Find the sim object that contains the specified integrable object.
126 Trick::SimObject * JeodDynbodyIntegrationLoop::find_containing_sim_object(
127  er7_utils::IntegrableObject & integrable_object)
128 {
129  // Get the simulation engine's name for the specified integrable object.
130  std::string obj_name = JeodSimulationInterface::get_name_at_address(&integrable_object,
131  JeodMemoryManager::get_type_descriptor(
132  typeid(er7_utils::IntegrableObject)));
133 
134  if(obj_name[0] == '&')
135  {
136  obj_name = obj_name.substr(1);
137  }
138 
139  Trick::SimObject * sim_object = nullptr;
140 
141  // Find the sim object using the integrable object's name, starting with the
142  // outermost parent name and proceeding to more basic names.
143  // For example, if the integrable_object's name is foo.bar.baz, look for
144  // a SimObject named foo.bar, and if that fails, one named foo.
145  for(std::size_t pos = obj_name.find_last_of('.'); pos < obj_name.npos; pos = obj_name.find_last_of('.'))
146  {
147  obj_name.erase(pos);
148  sim_object = manager.find_sim_object(JeodSimulationInterface::get_address_at_name(obj_name));
149  if(sim_object != nullptr)
150  {
151  break;
152  }
153  }
154 
155  return sim_object;
156 }

186 // Integrate over the specified time span.
187 int JeodDynbodyIntegrationLoop::integrate_dt(double beg_sim_time, double del_sim_time)
188 {
189  int ipass = 0;
190  bool need_derivs = integ_group->get_first_step_derivs_flag() || get_first_step_deriv_from_integrator();
191  bool first_pass = true;
192 
193  trick_curr_integ = integ_ptr;
194 
195  // Integrate until the integrators say "we're done" by returning zero.
196  do
197  {
198  if(need_derivs)
199  {
200  call_deriv_jobs();
201  }
202  need_derivs = true;
203 
204  // The bodies themselves are not represented by an integration job.
205  ipass = integ_group->integrate_group(beg_sim_time, del_sim_time);
206 
207  // Perform any integration class jobs associated with this integrator.
208  integ_jobs.reset_curr_index();
209  Trick::JobData * curr_job;
210  while((curr_job = integ_jobs.get_next_job()) != nullptr)
211  {
212  // Get the integrator object associated with this job.
213  trick_curr_integ = *(static_cast<Trick::Integrator **>(curr_job->sup_class_data));
214  if(trick_curr_integ == nullptr)
215  {
216  MessageHandler::error(__FILE__,
217  __LINE__,
218  SimInterfaceMessages::integration_error,
219  "Integrate job has no associated Integrator.");
220  return 1;
221  }
222 
223  // Set the integrator's time, first pass only.
224  if(first_pass)
225  {
226  trick_curr_integ->time = beg_sim_time;
227  trick_curr_integ->dt = del_sim_time;
228  }
229 
230  // Integrate, checking for agreement with ipass.
231  int result = curr_job->call();
232 
233  if(result != ipass)
234  {
235  MessageHandler::error(__FILE__,
236  __LINE__,
237  SimInterfaceMessages::integration_error,
238  "JEOD and Trick integrators are not in sync.");
239  return 1;
240  }
241  }
242 
243  first_pass = false;
244  } while(ipass != 0);
245 
246  return 0;
247 }
