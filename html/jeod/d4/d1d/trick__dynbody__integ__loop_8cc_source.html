144 // Find the sim object that contains the specified integrable object.
145 Trick::SimObject *
146 JeodDynbodyIntegrationLoop::find_containing_sim_object (
147  er7_utils::IntegrableObject & integrable_object)
148 {
149  // Get the simulation engine's name for the specified integrable object.
150  std::string obj_name =
151  JeodSimulationInterface::get_name_at_address (
152  &integrable_object,
153  JeodMemoryManager::get_type_descriptor (
154  typeid(er7_utils::IntegrableObject)));
155 
156  if (obj_name[0] == '&') {
157  obj_name = obj_name.substr(1);
158  }
159 
160  Trick::SimObject * sim_object = nullptr;
161 
162  // Find the sim object using the integrable object's name, starting with the
163  // outermost parent name and proceeding to more basic names.
164  // For example, if the integrable_object's name is foo.bar.baz, look for
165  // a SimObject named foo.bar, and if that fails, one named foo.
166  for (std::size_t pos = obj_name.find_last_of ('.');
167  pos < obj_name.npos;
168  pos = obj_name.find_last_of ('.')) {
169  obj_name.erase (pos);
170  sim_object =
171  manager.find_sim_object (
172  JeodSimulationInterface::get_address_at_name (obj_name));
173  if (sim_object != nullptr) {
174  break;
175  }
176  }
177 
178  return sim_object;
179 }

220 // Integrate over the specified time span.
221 int
222 JeodDynbodyIntegrationLoop::integrate_dt (
223  double beg_sim_time,
224  double del_sim_time)
225 {
226  int ipass = 0;
227  bool need_derivs =
228  integ_group->get_first_step_derivs_flag() ||
229  get_first_step_deriv_from_integrator();
230  bool first_pass = true;
231 
232  trick_curr_integ = integ_ptr;
233 
234  // Integrate until the integrators say "we're done" by returning zero.
235  do {
236  if (need_derivs) {
237  call_deriv_jobs ();
238  }
239  need_derivs = true;
240 
241  // The bodies themselves are not represented by an integration job.
242  ipass = integ_group->integrate_group (beg_sim_time, del_sim_time);
243 
244  // Perform any integration class jobs associated with this integrator.
245  integ_jobs.reset_curr_index();
246  Trick::JobData * curr_job;
247  while ((curr_job = integ_jobs.get_next_job()) != nullptr) {
248 
249  // Get the integrator object associated with this job.
250  trick_curr_integ =
251  *(static_cast<Trick::Integrator**>(curr_job->sup_class_data));
252  if (trick_curr_integ == nullptr) {
253  MessageHandler::error (
254  __FILE__, __LINE__, SimInterfaceMessages::integration_error,
255  "Integrate job has no associated Integrator.");
256  return 1;
257  }
258 
259  // Set the integrator's time, first pass only.
260  if (first_pass) {
261  trick_curr_integ->time = beg_sim_time;
262  trick_curr_integ->dt = del_sim_time;
263  }
264 
265  // Integrate, checking for agreement with ipass.
266  int result = curr_job->call();
267 
268  if (result != ipass) {
269  MessageHandler::error (
270  __FILE__, __LINE__, SimInterfaceMessages::integration_error,
271  "JEOD and Trick integrators are not in sync.");
272  return 1;
273  }
274  }
275 
276  first_pass = false;
277  } while (ipass != 0);
278 
279  return 0;
280 }
