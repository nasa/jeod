95 /**
96  * Construct the skew symmetric cross product matrix:
97  * mat[i][k] = epsilon_ijk vec[j], epsilon_ijk is the Levi-Cevita symbol
98  * \param[in] vec Vector
99  * \param[out] cross_mat Cross product matrix
100  */
101 inline void Matrix3x3::cross_matrix(const double vec[3], double cross_mat[3][3])
102 {
103  cross_mat[0][0] = cross_mat[1][1] = cross_mat[2][2] = 0.0;
104 
105  // ijk = 012, 210
106  cross_mat[0][2] = vec[1];
107  cross_mat[2][0] = -vec[1];
108 
109  // ijk = 120, 021
110  cross_mat[1][0] = vec[2];
111  cross_mat[0][1] = -vec[2];
112 
113  // ijk = 201, 102
114  cross_mat[2][1] = vec[0];
115  cross_mat[1][2] = -vec[0];
116 }
