44 /**
45  * Determine if contact has occurred between the facets of the pair.
46  */
47 void PointContactPair::in_contact()
48 {
49  double subject_contact_point[3];
50  double target_contact_point[3];
51  double vec[3], nvec[3];
52  double target_mag;
53 
54  // find direction vector to target from relstate
55  Vector3::normalize(rel_state.rel_state.trans.position, vec);
56  // pass direction vector to the subject so it can calculate the closest point.
57  point_subject->calculate_contact_point(vec);
58 
59  // rotate direction vector to target frame
60  Vector3::transform(rel_state.rel_state.rot.T_parent_this, rel_state.rel_state.trans.position, nvec);
61  Vector3::negate(nvec, nvec);
62  Vector3::normalize(nvec, vec);
63  point_target->calculate_contact_point(vec);
64 
65  // get the contact points calculated above
66  Vector3::copy(point_subject->contact_point, subject_contact_point);
67  Vector3::copy(point_target->contact_point, target_contact_point);
68 
69  // convert target point into subject frame through rotation vector addition
70  Vector3::transform_transpose(rel_state.rel_state.rot.T_parent_this, target_contact_point, vec);
71 
72  Vector3::sum(rel_state.rel_state.trans.position, vec, target_contact_point);
73  // find the distance to the target_contact_point
74  target_mag = Vector3::vmag(target_contact_point);
75 
76  // compare length to target point with subject radius
77  if(point_subject->radius > target_mag)
78  {
79  double rel_velocity[3];
80  // find the vector that characterizes the interpenetration
81  Vector3::diff(target_contact_point, subject_contact_point, vec);
82  // find the relative velocity of the target in subject frame
83  Vector3::cross(rel_state.rel_state.rot.ang_vel_this, subject_contact_point, rel_velocity);
84  Vector3::diff(rel_velocity, rel_state.rel_state.trans.velocity, rel_velocity);
85  // calculate the forces on the facets
86  interaction->calculate_forces(point_subject, point_target, &rel_state, vec, rel_velocity);
87  }
88 }
