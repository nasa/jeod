75 void LvlhFrame::initialize(DynManager & dyn_manager)
76 {
77  // Cache a pointer to the dyn manager. We will need it for the destructor.
78  local_dm = &dyn_manager;
79 
80  if(subject_frame == nullptr)
81  {
82  // Check that subject_name has something to it.
83  if(subject_name.empty())
84  {
85  MessageHandler::fail(__FILE__,
86  __LINE__,
87  LvlhFrameMessages::invalid_configuration,
88  "Either subject_name or subject_frame must be specified.");
89 
90  // Not reached
91  return;
92  }
93 
94  // Find the subject frame in the dynamics manager's list.
95  subject_frame = dyn_manager.find_ref_frame(subject_name);
96 
97  // Ensure the above worked.
98  if(subject_frame == nullptr)
99  {
100  MessageHandler::fail(__FILE__,
101  __LINE__,
102  LvlhFrameMessages::invalid_name,
103  "Invalid subject frame '%s' for LvlhFrame",
104  subject_name.c_str());
105 
106  // Not reached
107  return;
108  }
109  }
110  else
111  {
112  subject_name = subject_frame->get_name();
113  }
114 
115  // (Try to) ensure the subject frame is computed.
116  subject_frame->subscribe();
117 
118  if(planet_centered_inertial == nullptr)
119  {
120  // Check that some kind of planet name was provided.
121  if(planet_name.empty())
122  {
123  MessageHandler::fail(__FILE__,
124  __LINE__,
125  LvlhFrameMessages::invalid_configuration,
126  "Either planet_centered_inertial frame or planet_name must be specified.");
127 
128  // Not reached
129  return;
130  }
131 
132  // Find the planet in the dynamics manager's list.
133  BasePlanet * planet;
134  planet = dyn_manager.find_base_planet(planet_name);
135 
136  // Ensure the above worked.
137  if(planet == nullptr)
138  {
139  MessageHandler::fail(__FILE__,
140  __LINE__,
141  LvlhFrameMessages::invalid_name,
142  "Invalid planet name '%s' for LvlhFrame",
143  planet_name.c_str());
144 
145  // Not reached
146  return;
147  }
148  planet_centered_inertial = &planet->inertial;
149  }
150  else
151  {
152  planet_name = planet_centered_inertial->get_name();
153  planet_name = planet_name.substr(0, planet_name.rfind(".inertial"));
154  }
155 
156  // Name the LVLH frame as <subject_frame>.<planet>.lvlh
157  frame.set_name(subject_name, planet_name, "lvlh");
158

160  // connect it to the planet's inertial frame, and subscribe
161  // to that frame.
162  dyn_manager.add_ref_frame(frame);
163  planet_centered_inertial->add_child(frame);
164  planet_centered_inertial->subscribe();
165 
166  // Initialize the LVLH frame's angular velocity.
167  // Note: Only the y-component of the angular velocity will be changed
168  // via update(). The x and z components are identically zero.
169  Vector3::initialize(frame.state.rot.ang_vel_this);
170  Vector3::initialize(frame.state.rot.ang_vel_unit);
171  frame.state.rot.ang_vel_unit[1] = -1;
172 }

174 /**
175  * Update the state.
176  */
177 void LvlhFrame::update()
178 {
179  // Compute the LVLH frame directly from the subject frame's position and
180  // velocity if the planet centered inertial frame is the direct parent
181  // of the subject frame.
182  if(subject_frame->get_parent() == planet_centered_inertial)
183  {
184  compute_lvlh_frame(subject_frame->state.trans);
185  }
186 
187  // Otherwise, compute the subject frame's translational state wrt the
188  // planet and use that to determine the LVLH frame.
189  else
190  {
191  RefFrameState subject_rel_state;
192  subject_frame->compute_relative_state(*planet_centered_inertial, subject_rel_state);
193  compute_lvlh_frame(subject_rel_state.trans);
194  }
195 
196  // Timestamp the frame per the vehicle timestamp.
197  frame.set_timestamp(subject_frame->timestamp());
198 }

236 /**
237  * Update the state of the LVLH frame wrt its parent.
238  * \param[in] rel_trans Planet relative state
239  */
240 void LvlhFrame::compute_lvlh_frame(const RefFrameTrans & rel_trans)
241 {
242  double angmom[3];
243  double hmag;
244  double rmagsq;
245  double rmag;
246  double wmag;
247  RefFrameTrans & lvlh_trans = frame.state.trans;
248  RefFrameRot & lvlh_rot = frame.state.rot;
249 
250  // The origin of the LVLH frame is co-located and co-moving with the vehicle.
251  Vector3::copy(rel_trans.position, lvlh_trans.position);
252  Vector3::copy(rel_trans.velocity, lvlh_trans.velocity);
253 
254  // Compute the orbital parameters related to angular momentum.
255  Vector3::cross(rel_trans.position, rel_trans.velocity, angmom);
256  hmag = Vector3::vmag(angmom);
257  rmagsq = Vector3::vmagsq(rel_trans.position);
258  rmag = std::sqrt(rmagsq);
259  wmag = hmag / rmagsq;
260 
261  // Compute the orientation of the LVLH frame wrt inertial:
262  // zhat = -r/rmag
263  // yhat = -h/hmag
264  // xhat = yhat cross zhat
265  Vector3::scale(rel_trans.position, -1.0 / rmag, lvlh_rot.T_parent_this[2]);
266  Vector3::scale(angmom, -1.0 / hmag, lvlh_rot.T_parent_this[1]);
267  Vector3::cross(lvlh_rot.T_parent_this[1], lvlh_rot.T_parent_this[2], lvlh_rot.T_parent_this[0]);
268  Vector3::normalize(lvlh_rot.T_parent_this[0]);
269 
270  // Compute the quaternion from the transformation matrix.
271  frame.state.rot.compute_quaternion();
272 
273  // Update the angular velocity.
274  lvlh_rot.ang_vel_mag = wmag;
275  lvlh_rot.ang_vel_this[1] = -wmag;
276 }
