72  * Swap the negative ones in the supplied config with the default values,

75 static GaussJacksonConfig set_default_config_values(const GaussJacksonConfig & config)
76 {
77  GaussJacksonConfig result = config;
78 
79  // Set the initial order if it is at the default.
80  if(result.initial_order == -1u)
81  {
82  // The initial order must not exceed the final order, so a final order
83  // of 2 dictates that the initial order be 2.
84  if(result.final_order == 2)
85  {
86  result.initial_order = 2;
87  }
88 
89  // The initial order is set to 8 if this is consistent with the final
90  // order and if this is compatible with the priming technique.
91  else if((result.final_order >= 8) && ((result.priming_technique == er7_utils::Integration::Unspecified) ||
92  (result.priming_technique == er7_utils::Integration::RKFehlberg78)))
93  {
94  result.initial_order = 8;
95  }
96 
97  // Use 4 if neither of the above cases applies.
98  else
99  {
100  result.initial_order = 4;
101  }
102  }
103 
104  // Set the final order if it is at the default.
105  // Use 14 if the initial order is 14, 12 otherwise.
106  if(result.final_order == -1u)
107  {
108  if(result.initial_order == 14)
109  {
110  result.final_order = 14;
111  }
112  else
113  {
114  result.final_order = 12;
115  }
116  }
117 
118  // Set the priming technique if it is at the default.
119  // Use RK2 if the initial order is 2, RKFehlberg78 if the initial order is
120  // 8 or more, RK4 otherwise.
121  if(result.priming_technique == er7_utils::Integration::Unspecified)
122  {
123  if(result.initial_order == 2)
124  {
125  result.priming_technique = er7_utils::Integration::RK2Midpoint;
126  }
127  else if(result.initial_order >= 8)
128  {
129  result.priming_technique = er7_utils::Integration::RKFehlberg78;
130  }
131  else
132  {
133  result.priming_technique = er7_utils::Integration::RungeKutta4;
134  }
135  }
136 
137  // Set the number of doubling steps if it is at the default,
138  // but only if it makes sense to do so.
139  if((result.ndoubling_steps == -1u) && (result.final_order >= result.initial_order) && (result.final_order <= 14))
140  {
141  result.ndoubling_steps = std::min(1u, (result.final_order - result.initial_order) / 2u);
142  }
143 
144  // Set the maximum corrections allowed if it is at the default.
145  if(result.max_correction_iterations == -1u)
146  {
147  result.max_correction_iterations = 10;
148  }
149 
150  // Set the tolerances if they are at the defaults.
151  if(Numerical::compare_exact(result.relative_tolerance, -1.0))
152  {
153  result.relative_tolerance = 1e-14;
154  }
155 
156  if(Numerical::compare_exact(result.absolute_tolerance, -1.0))
157  {
158  result.absolute_tolerance = 1e-10;
159  }
160 
161  return result;
162 }

167 static unsigned int validate_config(const GaussJacksonConfig & config)
168 {
169  unsigned int error_count = 0;
170 
171  // The initial order must be an even number between 2 and 14.
172  if(((config.initial_order & 1) != 0) || (config.initial_order < 2) || (config.initial_order > 14))
173  {
174  er7_utils::MessageHandler::error(__FILE__,
175  __LINE__,
176  er7_utils::IntegrationMessages::invalid_request,
177  "Illegal Gauss-Jackson configuration:\n"
178  "The initial order %u must be an even integer between 2 and 14.",
179  config.initial_order);
180  ++error_count;
181  }
182 
183  // The final order must be an even number between initial_order and 14.
184  if(((config.final_order & 1) != 0) || (config.final_order < config.initial_order) || (config.final_order > 14))
185  {
186  er7_utils::MessageHandler::error(__FILE__,
187  __LINE__,
188  er7_utils::IntegrationMessages::invalid_request,
189  "Illegal Gauss-Jackson configuration:\n"
190  "The final order %u must be an even integer between\n"
191  "The initial order %u and 14.",
192  config.final_order,
193  config.initial_order);
194  ++error_count;
195  }
196 
197  // The number of doubling steps must be 20 or less.
198  if(config.ndoubling_steps > 20)
199  {
200  er7_utils::MessageHandler::error(__FILE__,
201  __LINE__,
202  er7_utils::IntegrationMessages::invalid_request,
203  "Illegal Gauss-Jackson configuration:\n"
204  "The number of doubling steps %u is not 20 or less.",
205  config.ndoubling_steps);
206  ++error_count;
207  }
208 
209  // The relative tolerance must be between 0 and 1.
210  if((config.relative_tolerance < 0.0) || (config.relative_tolerance > 1.0))
211  {
212  er7_utils::MessageHandler::error(__FILE__,
213  __LINE__,
214  er7_utils::IntegrationMessages::invalid_request,
215  "Illegal Gauss-Jackson configuration:\n"
216  "The relative tolerance %g must be between 0 and 1, inclusive.",
217  config.relative_tolerance);
218  ++error_count;
219  }
220 
221  // The absolute tolerance must be non-negative.
222  if(config.absolute_tolerance < 0.0)
223  {
224  er7_utils::MessageHandler::error(__FILE__,
225  __LINE__,
226  er7_utils::IntegrationMessages::invalid_request,
227  "Illegal Gauss-Jackson configuration:\n"
228  "The absolute tolerance %g must be non-negative.",
229  config.relative_tolerance);
230  ++error_count;
231  }
232 
233  return error_count;
234 }

236 GaussJacksonConfig GaussJacksonConfig::validate_configuration(const GaussJacksonConfig & config)
237 {
238  // Set defaults.
239  GaussJacksonConfig result = set_default_config_values(config);
240 
241  // Check for violations.
242  unsigned int error_count = validate_config(result);
243 
244  // Any violations means to reject the input and use the standard instead.
245  if(error_count > 0)
246  {
247  const char errorStrings[2][7] = {"errors", "error"};
248  int index = 1;
249  if(error_count > 1)
250  {
251  index = 0;
252  }
253  er7_utils::MessageHandler::error(__FILE__,
254  __LINE__,
255  er7_utils::IntegrationMessages::invalid_request,
256  "The default Gauss-Jackson configuration "
257  "will be used instead of the supplied configuration\n"
258  "because of the above %s.",
259  errorStrings[index]);
260  result = standard_configuration();
261  }
262 
263  return result;
264 }
