255  /**
256  * Issue a reset to each member of a container.
257  * @tparam T The container type.
258  * @param[in,out] container The container to be reset.
259  */
260  template<typename T> void reset_container(const T & container)
261  {
262  // Reset each registered integrable object.
263  for(auto iter = container.begin(); iter != container.end(); ++iter)
264  {
265  (*iter)->reset_integrators();
266  }

269  /**
270  * Integrate each member of a collection.
271  * @tparam T The container type.
272  * @param[in] dyn_dt Dynamic time step, in dynamic time seconds.
273  * @param[in] target_stage The stage of the integration process
274  * that the integrator should try to attain.
275  * @param[in,out] container The container to be integrated.
276  */
277  template<typename T>
278  er7_utils::IntegratorResult integrate_container(double dyn_dt, unsigned int target_stage, const T & container)
279  {
280  er7_utils::IntegratorResult status(false);
281 
282  // Integrate each registered integrable object.
283  for(auto iter = container.begin(); iter != container.end(); ++iter)
284  {
285  integ_merger.merge_integrator_result((*iter)->integrate(dyn_dt, target_stage), status);
286  }
287  return status;
288  }
