67 /**
68  * A constrained point mass is a point mass that is constrained to more or
69  * less move with a DynBody object. The "more or less" means that this
70  * object needs to be modeled as a constrained object rather than as a
71  * rigidly attached object.
72  */
73 class ConstrainedPointMass
74 {
75  JEOD_MAKE_SIM_INTERFACES(jeod, ConstrainedPointMass)
76 
77 public:
78  // Instances of this class are assumed to be protected members of
79  // a DynBodyConstraint-derived object, so all member data are public.
80 
81  // Member data.
82 
83  /**
84  * The frame in which the constraint position is most naturally expressed.
85  */
86  const ConstraintFrame * constraint_frame;
87 
88  /**
89  * The effective mass of this constrained object.
90  */
91  double mass{}; //!< trick_units(kg)
92 
93  /**
94  * The position at which the constraint equation applies, expressed in the
95  * constraint frame. Note well: This may or may not be the position of the
96  * of the constrained mass. For example, the constraint position for a
97  * pendulum whose mass is concentrated at the end of a massless rod is the
98  * hinge point position rather than the bob position.
99  */
100  double constraint_position_constraint_frame[3]{}; //!< trick_units(m)
101 
102  /**
103  * The constraint position, expressed in root structural frame coordinates.
104  */
105  double constraint_position_root[3]{}; //!< trick_units(m)
106 
107  /**
108  * The constraint position relative to the root vehicle center of mass,
109  * expressed in root structural frame coordinates.
110  */
111  double constraint_position_com[3]{}; //!< trick_units(m)
112 
113  /**
114  * The non-gravitational acceleration at the constraint point in root
115  * structural coordinates.
116  */
117  double non_grav_accel_root[3]{}; //!< trick_units(m/s^2)
118 
119  /**
120  * The acceleration at the constraint point due to vehicular rotation and
121  * external forces and torques, in constraint frame coordinates.
122  */
123  double non_grav_accel_constraint_frame[3]{}; //!< trick_units(m/s^2)
124 
125  /**
126  * The force at the constraint point; mass * non_grav_accel_constraint_frame.
127  */
128  double force_constraint_frame[3]{}; //!< trick_units(N)

140  /**

149  /**
150  * Prepare this object for solving a constraint problem.
151  * In this case, that means computing constraint_position_com.
152  * @param vehicle_properties Various vehicle properties.
153  * @param non_grav_state Non-gravitational State.
154  * @note This function is called prior to the calls to set_self_coeff(),
155  * set_cross_coeff(), and set_r_h_s().
156  */
157  virtual void setup_constraint(const VehicleProperties & vehicle_properties,
158  const VehicleNonGravState & non_grav_state)
159  {
160  compute_constraint_position_root();
161  compute_constraint_position_com(vehicle_properties);
162  compute_non_grav_accel(non_grav_state);
163  }
