81 /**
82  * Class template to construct a simulation engine attributes object
83  * that represents some type.
84  * All partial template instantiations of this template define a
85  * class with a single static function named attributes.
86  * This default implementation is for a primitive type.
87  * Subsequent partial instantiations will address other types.
88  * @tparam Type The type for which an attributes is to be constructed.
89  * @tparam is_class True if the type is a class, false otherwise.
90  */

93 public:
94 
95  /**

97  * @return Constructed attributes object.

100  {

102  JeodSimulationInterface::get_memory_interface();
