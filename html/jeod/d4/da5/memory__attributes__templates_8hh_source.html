81 /**
82  * Class template to construct a simulation engine attributes object
83  * that represents some type.
84  * All partial template instantiations of this template define a
85  * class with a single static function named attributes.
86  * This default implementation is for a primitive type.
87  * Subsequent partial instantiations will address other types.
88  * @tparam Type The type for which an attributes is to be constructed.
89  * @tparam is_class True if the type is a class, false otherwise.
90  */
91 template<typename Type, bool is_class> class JeodSimEngineAttributes
92 {
93 public:
94  /**
95  * Construct a JEOD_ATTRIBUTES_TYPE that represents a primitive type.
96  * @return Constructed attributes object.
97  */
98  static JEOD_ATTRIBUTES_TYPE attributes(bool)
99  {
100  const JeodMemoryInterface & memory_interface = JeodSimulationInterface::get_memory_interface();
101  return memory_interface.primitive_attributes(typeid(Type));
102  }
103 };
104 
105 /**
106  * Partial template instantiation of JeodSimEngineAttributes for
107  * a pointer type.
108  * @tparam Type The pointed-to type.
109  */
110 template<typename Type> class JeodSimEngineAttributes<Type *, false>
111 {
112 public:
113  /**
114  * Construct a JEOD_ATTRIBUTES_TYPE that represents a pointer type.
115  * @param is_exportable True => type is exportable.
116  * @return Constructed attributes object.
117  */
118  static JEOD_ATTRIBUTES_TYPE attributes(bool is_exportable = true)
119  {
120  const JeodMemoryInterface & memory_interface = JeodSimulationInterface::get_memory_interface();
121  const bool type_is_class = std::is_class<Type>::value;
122  return memory_interface.pointer_attributes(

148 template<typename Type> class JeodSimEngineAttributes<Type, true>

150 public:
151  /**
152  * Construct a JEOD_ATTRIBUTES_TYPE that represents a structured type.

154  * @return Constructed attributes object.
