98 /**
99  * Accumulate the total of the get_wrench calls to the active constraints.
100  * @return True if the sum is non-zero.
101  * @tparam T The class that defines the get_wrench call.
102  & @param get_wrench Pointer to member function that returns a Wrench.
103  * @param active_constraints Vector of active constraints.
104  * @param point The vector offset for computing the wrenching torque.
105  * @param sum The sum to be accumulated.
106  */
107 template<typename T>
108 bool accumulate_wrenches(const Wrench & (T::*get_wrench)() const,
109  const DynBodyConstraintsSolver::ConstraintsVectorT & active_constraints,
110  const double point[3],
111  Wrench & sum)
112 {
113  bool have_non_trivial_sum = false;
114 
115  sum.reset_force_and_torque();
116  sum.set_point(point);
117 
118  // Accumulate the wrench from the calls to get_wrench.
119  for(auto constraint : active_constraints)
120  {
121  const Wrench & wrench = (constraint->*get_wrench)();
122  if(wrench.is_active())
123  {
124  have_non_trivial_sum = true;
125  sum += wrench;
126  }
127  }
128 
129  return have_non_trivial_sum && !(is_zero_vector(sum.get_force()) && is_zero_vector(sum.get_torque()));
130 }

444 void DynBodyConstraintsSolver::solve(const VehicleProperties & vehicle_properties, VehicleNonGravState & non_grav_state)
445 {
446  unsigned n_constraints = active_constraints.size();
447  if(n_constraints == 0)
448  {
449  return;
450  }
451 
452  // Accumulate the pre-constraint wrenches and update the
453  // non_grav_state to reflect the vehicle's response to these wrenches.
454  if(accumulate_wrenches<DynBodyConstraint>(&DynBodyConstraint::get_effector_wrench,
455  active_constraints,
456  vehicle_properties.get_structure_to_body_offset(),
457  effector_wrench))
458  {
459  update_vehicle_response(effector_wrench, vehicle_properties, non_grav_state);
460  }
461 
462  // Build the constraints system of equations.
463  constraint_indices.clear();
464  constraint_indices.reserve(n_constraints);
465  build_system_of_equations(vehicle_properties, non_grav_state, n_constraints, constraint_indices);
466 
467  // Solve the constraints system of equations and broadcast the solution.
468  solver->solve(x_vector);
469  send_solution(n_constraints, constraint_indices);
470 
471  // Accumulate the constraint wrenches and update the
472  // non_grav_state to reflect the vehicle's response to these wrenches.
473  if(accumulate_wrenches<DynBodyConstraint>(&DynBodyConstraint::get_constraint_wrench,
474  active_constraints,
475  vehicle_properties.get_structure_to_body_offset(),
476  constraints_wrench))
477  {
478  update_vehicle_response(constraints_wrench, vehicle_properties, non_grav_state);
479  }
480 
481  // Tell the constraints to set up the differential equations
482  // due to the vehicle response that the constraints need to integrate.
483  compute_constraint_responses(vehicle_properties, non_grav_state);
484 
485  // Accumulate the nonlinear response wrenches and update the
486  // non_grav_state to reflect the vehicle's response to these wrenches.
487  if(accumulate_wrenches<DynBodyConstraint>(&DynBodyConstraint::get_nonlinear_response_wrench,
488  active_constraints,
489  vehicle_properties.get_structure_to_body_offset(),
490  nonlinear_responses_wrench))
491  {
492  update_vehicle_response(nonlinear_responses_wrench, vehicle_properties, non_grav_state);
493  }
494 }
495 
496 // Construct A and b in the system of equations A*x=b, where
497 // - A is the matrix of partial derivatives, dx_i/dx_j,
498 // - x is the vector of constraint values (not referenced here), and
499 // - b is the vector of right hand side values.
500 void DynBodyConstraintsSolver::build_system_of_equations(const VehicleProperties & vehicle_properties,
501  const VehicleNonGravState & non_grav_state,
502  unsigned n_constraints,
503  IndexPairVectorT & constraint_indices)
504 {
505  // Tell each constraint to set its internal state in preparation
506  // for constructing the system of equations.
507  for(auto constraint : active_constraints)
508  {
509  constraint->setup_constraint(vehicle_properties, non_grav_state);
510  }
511 
512  // Build the mapping from constraints to indices.
513  // Note: This could be combined with the above loop, but that would
514  // require writing "and".
515  unsigned prev_end = 0u;
516 
517  constraint_indices.clear();
518  for(auto constraint : active_constraints)
519  {
520  unsigned new_end = prev_end + constraint->n_dimensions;
521  constraint_indices.push_back(IndexPairT(prev_end, new_end));
522  prev_end = new_end;
523  }
524 
525  // Set the overall dimensionality of the problem.
526  unsigned n_dims = constraint_indices.back().second;
527  solver->set_n_dimensions(n_dims);
528  x_vector.resize(n_dims);
529 
530  // Populate the A matrix and b vector by querying each constraint.
531  for(unsigned ii = 0; ii < n_constraints; ++ii)
532  {
533  auto & range_ii = constraint_indices[ii];
534 
535  // This constraint is not truly active if its dimensionality is zero.
536  if(range_ii.second != range_ii.first)
537  {
538  auto & constraint_ii = *active_constraints[ii];
539  // Set the right hand side element for this constraint.
540  constraint_ii.set_r_h_s(solver->make_b_vector_view(range_ii));
541 
542  // Set the elements of the A matrix for this constraint vs itself.
543  constraint_ii.set_self_coeff(vehicle_properties, solver->make_a_matrix_view(range_ii, range_ii));
544 
545  // Set the elements of the A matrix for this constraint vs others.
546  for(unsigned jj = 0; jj < n_dims; ++jj)
547  {
548  if(jj != ii)
549  {
550  auto & range_jj = constraint_indices[jj];
551  if(range_jj.second != range_jj.first)
552  {
553  auto & constraint_jj = *active_constraints[jj];
554  constraint_ii.set_cross_coeff(vehicle_properties,
555  constraint_jj,
556  solver->make_a_matrix_view(range_ii, range_jj));
557  }
558  }
559  }
560  }
561  }
562 }

578 // Compute the total of the get_wrench calls and the vehicle's response to it.
579 void DynBodyConstraintsSolver::update_vehicle_response(const Wrench & sum,
580  const VehicleProperties & vehicle_properties,
581  VehicleNonGravState & non_grav_state)
582 {
583  // Transform the constraints wrench to the vehicle center of mass.
584  const Wrench & wrench_co_m = sum.transform_to_point(vehicle_properties.get_structure_to_body_offset());
585 
586  // Increment the vehicle's center of mass non-grav accel (struct coords)
587  // by the translational response to the constraint wrench at the CoM:
588  // accel += force / mass.
589  Vector3::scale_incr(wrench_co_m.get_force(), vehicle_properties.get_inverse_mass(), non_grav_state.accel_struct);
590 
591  // Increment the vehicle's rotational accel (body coordinates) by the
592  // rotational response to the constraint wrench:
593  // \dot{omega_{body}} += I^{-1} * (R * torque)
594  // where R is the structure-to-body transformation and I is the
595  // body-referenced inertia tensor.
596  double torque_body[3];
597  Vector3::transform_incr(vehicle_properties.get_inverse_inertia(),
598  Vector3::transform(vehicle_properties.get_structure_to_body_transform(),
599  wrench_co_m.get_torque(),
600  torque_body),
601  non_grav_state.omega_dot_body);
602 
603  // Transform the angular acceleration to structural.
604  Vector3::transform_transpose(vehicle_properties.get_structure_to_body_transform(),
605  non_grav_state.omega_dot_body,
606  non_grav_state.omega_dot_struct);
607 }
