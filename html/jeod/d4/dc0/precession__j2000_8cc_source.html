49 void PrecessionJ2000::update_rotation()
50 {
51  // define the three parameters, eta theta and z, that are needed
52  // for the precession matrix. Also define time2 and time3, time
53  // squared and cubed
54 
55  double zeta = 0.0;
56  double theta = 0.0;
57  double z = 0.0;
58  double time = current_time;
59  double time2 = time * time;
60  double time3 = time2 * time;
61 
62  // based on time as centuries since julian epoch (J2000), calculate
63  // those three params
64  zeta = 2306.2181 * time + 0.30188 * time2 + 0.017998 * time3;
65  theta = 2004.3109 * time - 0.42665 * time2 - 0.041833 * time3;
66  z = 2306.2181 * time + 1.09468 * time2 + 0.018203 * time3;
67 
68  // convert these from seconds to degrees
69  // these are * 1/DEGTOSEC in an effort to numerically match 1.52
70  zeta = (zeta / DEGTOSEC) * DEGTORAD;
71  theta = (theta / DEGTOSEC) * DEGTORAD;
72  z = (z / DEGTOSEC) * DEGTORAD;
73  // take trigs of the three parameters for inclusion in the matrix
74 
75  double c_zeta = cos(zeta);
76  double s_zeta = sin(zeta);
77  double c_theta = cos(theta);
78  double s_theta = sin(theta);
79  double c_z = cos(z);
80  double s_z = sin(z);
81 
82  // now populate the rotation matrix with the trig functions,
83  // as follows:
84  // celestial frame = rot_z(zeta) * rot_y(-theta) * rot_z(z) * terrest.
85 
86  rotation[0][0] = c_theta * c_z * c_zeta - s_z * s_zeta;
87  rotation[0][1] = s_z * c_theta * c_zeta + s_zeta * c_z;
88  rotation[0][2] = c_zeta * s_theta;
89 
90  rotation[1][0] = -s_zeta * c_theta * c_z - c_zeta * s_z;
91  rotation[1][1] = -s_z * s_zeta * c_theta + c_z * c_zeta;
92  rotation[1][2] = -s_zeta * s_theta;
93 
94  rotation[2][0] = -s_theta * c_z;
95  rotation[2][1] = -s_theta * s_z;
96  rotation[2][2] = c_theta;
97 }
