70  const GravityManager & gravity_manager JEOD_UNUSED)
71 {
72  unsigned int n_integ_frames = integ_frames.size();
73 
74  // Allocate an array of frame information, one element per integration frame.
75  frames = JEOD_ALLOC_CLASS_ARRAY(n_integ_frames, GravityIntegFrame);
76 
77  // Initialize each element in the frames array.
78  // NOTE: The nth element in the frames array corresponds to the nth
79  // integration frame.
80  for(unsigned int ii = 0; ii < n_integ_frames; ++ii)
81  {
82  const EphemerisRefFrame & integ_frame = *integ_frames[ii];
83 
84  frames[ii].ref_frame = &integ_frame;
85 
86  // Determine if the gravitational body acts as a primary body (Newton's
87  // law of gravitation) or as a third body (Newton's law less frame accel
88  // toward the gravitational body).
89  // ASSUMPTION: Gravity is direct if the planet inertial frame is a child
90  // of the integration frame, 3rd body otherwise.
91  // CONSTRAINT: The reference frame tree must be constructed to ensure
92  // the above constraint is satisfied.
93  // This constraint applies to the tree construction.
94  if(inertial->is_progeny_of(integ_frame))
95  {
96  frames[ii].is_third_body = false;
97  }
98  else
99  {
100  frames[ii].is_third_body = true;
101  }
102 
103  // Initialize the frame data (position, acceleration, and bogus time).
104  Vector3::initialize(frames[ii].pos);

107  }
