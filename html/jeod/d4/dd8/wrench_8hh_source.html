173  /**
174  * Increment this wrench by the other, but only if both are active.
175  * The other wrench is effectively reseated to this wrench's point
176  * prior to incrementing.
177  * @param other Wrench with which this wrench is to be incremented.
178  * @return *this.
179  */
180  Wrench& operator+= (const Wrench& other)
181  {
182  if ((active) && (other.active))
183  {
184  double delta_pos[3];
185  Vector3::diff (other.point, point, delta_pos);
186  Vector3::incr (other.force, force);
187  Vector3::incr (other.torque, torque);
188  Vector3::cross_incr (delta_pos, other.force, torque);
189  }
190 
191  return *this;
192  }

399  /**
400  * Construct an equivalent Wrench about the specified point.
401  * @param new_point The point about which this is to be represented.
402  * @return Equivalent wrench about the specified point.
403  */
404  Wrench transform_to_point (
405  const double new_point[3]) const
406  {
407  Wrench result{torque, force, new_point, active};
408  double delta_pos[3];
409  Vector3::diff (point, new_point, delta_pos);
410  Vector3::cross_incr (delta_pos, force, result.torque);
411  return result;

414  /**
415  * Construct an equivalent Wrench about the current point, but
416  * in a different reference frame.
417  * @param point_state Contains the position and orientation of the current
418  * frame in the parent frame.
419  * @return Equivalent wrench in the parent frame.
420  */
421  Wrench transform_to_parent (
422  const MassPointState& point_state) const
423  {
424  Wrench result;
425  Vector3::transform_transpose (point_state.T_parent_this, torque,
426  result.torque);
427  Vector3::transform_transpose (point_state.T_parent_this, force,
428  result.force);
429  Vector3::transform_transpose (point_state.T_parent_this, point,
430  result.point);
431  Vector3::incr (point_state.position, result.point);
432  return result;
433  }

440  /**
441  * The torque exerted on the DynBody by the force/torque agent, expressed
442  * in structural coordinates.
443  *
444  * This torque should not include the torque that results from the force
445  * not passing through the center of mass. A typical thruster, for example,
446  * should have the torque set to zero. On the other hand, a Hall effect
447  * thruster will have a non-zero torque due to the swirling of the exhaust.
448  */
