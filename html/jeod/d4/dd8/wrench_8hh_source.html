138  /**
139  * Increment this wrench by the other, but only if both are active.
140  * The other wrench is effectively reseated to this wrench's point
141  * prior to incrementing.
142  * @param other Wrench with which this wrench is to be incremented.
143  * @return *this.
144  */
145  Wrench & operator+=(const Wrench & other)
146  {
147  if((active) && (other.active))
148  {
149  double delta_pos[3];
150  Vector3::diff(other.point, point, delta_pos);
151  Vector3::incr(other.force, force);
152  Vector3::incr(other.torque, torque);
153  Vector3::cross_incr(delta_pos, other.force, torque);
154  }
155 
156  return *this;
157  }

335  /**
336  * Construct an equivalent Wrench about the specified point.
337  * @param new_point The point about which this is to be represented.
338  * @return Equivalent wrench about the specified point.
339  */
340  Wrench transform_to_point(const double new_point[3]) const
341  {
342  Wrench result{torque, force, new_point, active};
343  double delta_pos[3];
344  Vector3::diff(point, new_point, delta_pos);
345  Vector3::cross_incr(delta_pos, force, result.torque);
346  return result;

349  /**
350  * Construct an equivalent Wrench about the current point, but
351  * in a different reference frame.
352  * @param point_state Contains the position and orientation of the current
353  * frame in the parent frame.
354  * @return Equivalent wrench in the parent frame.
355  */
356  Wrench transform_to_parent(const MassPointState & point_state) const
357  {
358  Wrench result;
359  Vector3::transform_transpose(point_state.T_parent_this, torque, result.torque);
360  Vector3::transform_transpose(point_state.T_parent_this, force, result.force);
361  Vector3::transform_transpose(point_state.T_parent_this, point, result.point);
362  Vector3::incr(point_state.position, result.point);
363  return result;
364  }
