105 /**
106  * Changeing time direction and/or scale factor
107  * @return Void
108  */
109 bool TimeDyn::update_offset()
110 {
111  /* The scale-factor can be changed mid-sim.
112  TimeDyn uses this value to determine the rate and direction in which time
113  propagates with respect to simtime.
114  ref_scale is the recorded scale, testing scale_factor against ref-scale
115  identifies whether scale_factor has been externally modified. */
116  if(!Numerical::compare_exact(ref_scale, scale_factor))
117  {
118  offset = seconds - (scale_factor * time_manager->simtime);
119  if(ref_scale * scale_factor < 0)
120  { // if there is a direction change
121  time_manager->verify_table_lookup_ends(); // adjusts end-of-table settings
122  }
123  ref_scale = scale_factor;
124  return true;
125  }
126  else
127  {
128  return false;
129  }
130 }
