119 /**
120  * Changeing time direction and/or scale factor
121  * @return Void
122  */
123 bool
124 TimeDyn::update_offset (
125  void)
126 {
127  /* The scale-factor can be changed mid-sim.
128  TimeDyn uses this value to determine the rate and direction in which time
129  propagates with respect to simtime.
130  ref_scale is the recorded scale, testing scale_factor against ref-scale
131  identifies whether scale_factor has been externally modified. */
132  if (!Numerical::compare_exact(ref_scale,scale_factor)) {
133  offset = seconds - (scale_factor * time_manager->simtime);
134  if (ref_scale * scale_factor < 0) { // if there is a direction change
135  time_manager->verify_table_lookup_ends(); // adjusts end-of-table settings
136  }
137  ref_scale = scale_factor;
138  return true;
139  }
140  else {
141  return false;
142  }
143 }
