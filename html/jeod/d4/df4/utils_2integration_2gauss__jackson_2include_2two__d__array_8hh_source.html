72 /**
73  * RAII template class that implements a rectangular two dimensional array.
74  * The implementation provides two special-purpose features that are needed by
75  * some of the ER7 utilities and JEOD integrators. The rows of the array can
76  * be rotated and downsampled.
77  * @tparam T Type of each element of the array.
78  */
79 template<typename T> class TwoDArray
80 {
81 public:
82  /**
83  * Default constructor.
84  */
85  TwoDArray() = default;
86 
87  /**
88  * Copy constructor.
89  * @param src Item to be copied.
90  */
91  TwoDArray(const TwoDArray<T> & src)
92  {
93  // Allocate new storage based on the source size.
94  allocate(src.n, src.m);
95 
96  // Copy the source data.
97  // Maintenance note:
98  // Do not use memcpy to copy the whole data_array in one shot. This must
99  // be done on a row by row basis because of rotation and downsampling.
100  for(int ii = 0; ii < n; ++ii)
101  {
102  std::memcpy(row_array[ii], src.row_array[ii], m * sizeof(T));
103  }
104  }
105 
106  /**
107  * Destructor.
108  */
109  ~TwoDArray()

111  deallocate_internal();
112  }

115  * Copy and swap assignment constructor.
116  * @param src Item to be copied.
117  */
118  TwoDArray<T> & operator=(TwoDArray<T> src)
119  {
120  swap(*this, src);
121  return *this;
122  }

246  /**
247  * Allocate the array.
248  * @param N Number of rows in the array.
249  * @param M Number of columns in the array.
250  * @throw std::domain_error N and/or M won't fit in a signed int.
251  */
252  void allocate(std::size_t N, std::size_t M)
253  {
254  if((N > std::size_t(std::numeric_limits<int>::max())) || (M > std::size_t(std::numeric_limits<int>::max())))
255  {
256  JEOD_THROW(std::domain_error("Sizes too large"));
257  }
258 
259  if(data_array != nullptr)
260  {
261  deallocate_internal();
262  }
263 
264  n = static_cast<int>(N);
265  m = static_cast<int>(M);
266  allocate_internal();
267 
268  for(int ii = 0; ii < n; ++ii)
269  {
270  row_array[ii] = data_array + ii * m;
271  }
272  }
273 
274  /**
275  * Rotate elements 0 to limit downward, with array element 0 moved to
276  * array element limit.
277  * @param limit Index of last element participating in the rotation.
278  * @throw std::out_of_range If @a limit is an invalid index.
279  */
280  void rotate_down(int limit)
281  {
282  if((limit < 0) || (limit >= n))
283  {
284  JEOD_THROW(std::out_of_range("Index out of range"));
285  }
286 
287  T * zeroth = row_array[0];
288  for(int ii = 0; ii < limit; ++ii)
289  {
290  row_array[ii] = row_array[ii + 1];
291  }
292  row_array[limit] = zeroth;
293  }
294 
295  /**
296  * Rotate elements 0 to limit upward, with array element limit moved to
297  * array element 0.
298  * @param limit Index of last element participating in the rotation.
299  * @throw std::out_of_range If @a limit is an invalid index.
300  */
301  void rotate_up(int limit)
302  {
303  if((limit < 0) || (limit >= n))
304  {
305  JEOD_THROW(std::out_of_range("Index out of range"));
306  }
307 
308  T * last = row_array[limit];
309  for(int ii = limit; ii > 0; --ii)
310  {
311  row_array[ii] = row_array[ii - 1];
312  }
313  row_array[0] = last;
314  }
315 
316  /**
317  * Downsample the array by swapping pointers.
318  * @param limit Number of usable rows after downsample.
319  * @throw std::out_of_range If @a limit represents an invalid index.
320  */
321  void downsample(int limit)
322  {
323  if((limit < 0) || (2 * (limit - 1) >= n))
324  {
325  JEOD_THROW(std::out_of_range("Index out of range"));
326  }
327 
328  for(int ii = 1; ii < limit; ++ii)
329  {
330  std::swap(row_array[ii], row_array[2 * ii]);
331  }
332  }
333 
334  /**
335  * Swap the contents of *this with the other.
336  * @param other Other array.
337  */
338  void swap(TwoDArray<T> & other)
339  {
340  std::swap(n, other.n);
341  std::swap(m, other.m);
342  std::swap(data_array, other.data_array);
343  std::swap(row_array, other.row_array);
344  }
345 
346  /**
347  * Swap the contents of the two provided arrays.
348  * @param first First array.
349  * @param second Second array.
350  */
351  friend void swap(TwoDArray<T> & first, TwoDArray<T> & second)
352  {
353  first.swap(second);
354  }
355 
356 protected:
357  // Member data.
358  // These are protected rather than private so that Trick can see them.
359 
360  /**

362  */
363  int n{}; //!< trick_units(--)

369

373  T * data_array{}; //!< trick_units(--)
374

378  T ** row_array{}; //!< trick_units(--)
379 
380 private:
381  /**
382  * Allocate memory for the array.
383  */
384  void allocate_internal()
385  {
386  data_array = er7_utils::alloc::allocate_array<T>(n * m);
387  row_array = er7_utils::alloc::allocate_array<T *>(n);
388  }
389 
390  /**
391  * Deallocate memory for the array.
392  */
393  void deallocate_internal()
394  {
395  er7_utils::alloc::deallocate_array<T>(data_array);
396  er7_utils::alloc::deallocate_array<T *>(row_array);
397  }
