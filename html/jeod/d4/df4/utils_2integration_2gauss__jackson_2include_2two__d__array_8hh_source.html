75 /**
76  * RAII template class that implements a rectangular two dimensional array.
77  * The implementation provides two special-purpose features that are needed by
78  * some of the ER7 utilities and JEOD integrators. The rows of the array can
79  * be rotated and downsampled.
80  * @tparam T Type of each element of the array.
81  */
82 template <typename T>
83 class TwoDArray
84 {
85 public:
86 
87  /**
88  * Default constructor.
89  */
90  TwoDArray ()
91  :
92  n (0),
93  m (0),
94  data_array (0),
95  row_array (0)
96  {}
97 
98  /**
99  * Copy constructor.
100  * @param src Item to be copied.
101  */
102  TwoDArray (const TwoDArray<T>& src)
103  :
104  n (0),
105  m (0),
106  data_array (0),
107  row_array (0)
108  {
109  // Allocate new storage based on the source size.
110  allocate (src.n, src.m);
111 
112  // Copy the source data.
113  // Maintenance note:
114  // Do not use memcpy to copy the whole data_array in one shot. This must
115  // be done on a row by row basis because of rotation and downsampling.
116  for (int ii = 0; ii < n; ++ii) {
117  std::memcpy (row_array[ii], src.row_array[ii], m*sizeof(T));
118  }
119  }

267  /**
268  * Allocate the array.
269  * @param N Number of rows in the array.
270  * @param M Number of columns in the array.
271  * @throw std::domain_error N and/or M won't fit in a signed int.
272  */
273  void allocate (std::size_t N, std::size_t M)
274  {
275  if ((N > std::size_t(std::numeric_limits<int>::max())) ||
276  (M > std::size_t(std::numeric_limits<int>::max()))) {
277  throw std::domain_error ("Sizes too large");
278  }
279 
280  if (data_array != 0) {
281  deallocate_internal ();
282  }
283 
284  n = static_cast<int> (N);
285  m = static_cast<int> (M);
286  allocate_internal ();
287 
288  for (int ii = 0; ii < n; ++ii) {
289  row_array[ii] = data_array + ii*m;
290  }
291  }
292 
293 
294  /**
295  * Rotate elements 0 to limit downward, with array element 0 moved to
296  * array element limit.
297  * @param limit Index of last element participating in the rotation.
298  * @throw std::out_of_range If @a limit is an invalid index.
299  */
300  void rotate_down (int limit)
301  {
302  if ((limit < 0) || (limit >= n)) {
303  throw std::out_of_range ("Index out of range");
304  }
305 
306  T* zeroth = row_array[0];
307  for (int ii = 0; ii < limit; ++ii) {
308  row_array[ii] = row_array[ii+1];
309  }
310  row_array[limit] = zeroth;
311  }
312 
313  /**
314  * Rotate elements 0 to limit upward, with array element limit moved to
315  * array element 0.
316  * @param limit Index of last element participating in the rotation.
317  * @throw std::out_of_range If @a limit is an invalid index.
318  */
319  void rotate_up (int limit)
320  {
321  if ((limit < 0) || (limit >= n)) {
322  throw std::out_of_range ("Index out of range");
323  }
324 
325  T* last = row_array[limit];
326  for (int ii = limit; ii > 0; --ii) {
327  row_array[ii] = row_array[ii-1];
328  }
329  row_array[0] = last;
330  }
331 
332 
333  /**
334  * Downsample the array by swapping pointers.
335  * @param limit Number of usable rows after downsample.
336  * @throw std::out_of_range If @a limit represents an invalid index.
337  */
338  void downsample (int limit)
339  {
340  if ((limit < 0) || (2*(limit-1) >= n)) {
341  throw std::out_of_range ("Index out of range");
342  }
343 
344  for (int ii = 1; ii < limit; ++ii) {
345  std::swap (row_array[ii], row_array[2*ii]);
346  }
347  }

350  /**
351  * Swap the contents of *this with the other.
352  * @param other Other array.
353  */
354  void swap(TwoDArray<T>& other)
355  {
356  std::swap (n, other.n);
357  std::swap (m, other.m);
358  std::swap (data_array, other.data_array);
359  std::swap (row_array, other.row_array);
360  }

363  * Swap the contents of the two provided arrays.
364  * @param first First array.
365  * @param second Second array.
366  */
367  friend void swap(TwoDArray<T>& first, TwoDArray<T>& second)
368  {
369  first.swap (second);
370  }
