43 /**
44  * Complete initialization of the JEOD manager model.
45  */
46 void DynManager::initialize_simulation()
47 {
48  // Sanity check:
49  // There should be an overarching Gravity Manager for the simulation for all
50  // but non-empty space simulations and planet-based sims with no gravity.
51  // (That there is only one Gravity Manager is enforced elsewhere.)
52  if((mode != DynManagerInit::EphemerisMode_EmptySpace) && (gravity_manager == nullptr) && (!gravity_off))
53  {
54  MessageHandler::warn(__FILE__,
55  __LINE__,
56  DynManagerMessages::inconsistent_setup,
57  "A Gravity Manager has not been registered with the dynamics manager\n"
58  "for a simulation that involves planets and dynamic bodies.\n"
59  "Gravity will not be calculated for this simulation.\n"
60  "Please check the order in which initialization jobs are called.");
61  }
62 
63  // Initialize the ephemerides.
64  initialize_ephemerides();
65 
66  // Initialize the gravity controls for each dynamic body.
67  if(gravity_manager != nullptr)
68  {
69  initialize_gravity_controls();
70  }
71 
72  // Check that each subscribed frame has an owner.
73  check_ref_frame_ownership();
74 
75  // Compute ephemerides for active planets.
76  activate_ephemerides();
77 
78  // Update the ephemerides.
79  update_ephemerides();
80 
81  // Initialize the gravitational bodies.
82  if(gravity_manager != nullptr)
83  {
84  gravity_manager->initialize_state(*this);
85  }
86 
87  // Initialize the integration groups.
88  initialize_integ_groups();
89 
90  // Initialize the dynamic bodies.
91  initialize_dyn_bodies();
92 
93  // Indicate that initialization has been completed.
94  initialized = true;
95 }

100 void DynManager::initialize_integ_groups()
101 {
102  // Initialize the integration groups.
103  // Monolithic mode (no external groups registered):
104  // Initialize the default group.
105  if(integ_groups.empty())
106  {
107  default_integ_group->initialize_group(*this);
108  }
109 
110  // Multiple integrators mode:
111  // Initialize each integration group.
112  // The default group is left uninitialized.
113  else
114  {
115  for(auto group : integ_groups)
116  {
117  group->initialize_group(*this);
118  }
119  }
120 }

122 /**
123  * Add DynBody objects to the default integration group.
124  * @param[in,out] group Group to be updated
125  */
126 void DynManager::update_integration_group(JeodIntegrationGroup & group)
127 {
128  // The group should be the default integration group.
129  // This is one of those "this should never happen" kind of errors.
130  if(&group != default_integ_group)
131  {
132  MessageHandler::fail(__FILE__, __LINE__, DynManagerMessages::inconsistent_setup, "Internal error.");
133  }
134 
135  // Add each body that doesn't have an integrator to the default group.
136  for(std::vector<DynBody *>::const_iterator it = dyn_bodies.begin(); it != dyn_bodies.end(); ++it)
137  {
138  DynBody * body = *it;
139  if(body->get_integration_group() == nullptr)
140  {
141  default_integ_group->add_dyn_body(*body);
142  }
143  }
144 }
