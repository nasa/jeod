46  * Complete initialization of the JEOD manager model.
47  */
48 void
49 DynManager::initialize_simulation (
50  void)
51 {
52 
53  // Sanity check:
54  // There should be an overarching Gravity Manager for the simulation for all
55  // but non-empty space simulations and planet-based sims with no gravity.
56  // (That there is only one Gravity Manager is enforced elsewhere.)
57  if ((mode != DynManagerInit::EphemerisMode_EmptySpace) &&
58  (gravity_manager == nullptr) &&
59  (! gravity_off)) {
60  MessageHandler::warn (
61  __FILE__, __LINE__, DynManagerMessages::inconsistent_setup,
62  "A Gravity Manager has not been registered with the dynamics manager\n"
63  "for a simulation that involves planets and dynamic bodies.\n"
64  "Gravity will not be calculated for this simulation.\n"
65  "Please check the order in which initialization jobs are called.");
66  }
67 
68 
69  // Initialize the ephemerides.
70  initialize_ephemerides ();
71 
72  // Initialize the gravity controls for each dynamic body.
73  if (gravity_manager != nullptr) {
74  initialize_gravity_controls ();
75  }
76 
77  // Check that each subscribed frame has an owner.
78  check_ref_frame_ownership ();
79 
80  // Compute ephemerides for active planets.
81  activate_ephemerides ();
82 
83  // Update the ephemerides.
84  update_ephemerides ();
85 
86  // Initialize the gravitational bodies.
87  if (gravity_manager != nullptr) {
88  gravity_manager->initialize_state (*this);
89  }
90 
91  // Initialize the integration groups.
92  initialize_integ_groups ();
93 
94  // Initialize the dynamic bodies.
95  initialize_dyn_bodies ();
96 
97  // Indicate that initialization has been completed.
98  initialized = true;
99 
100  return;

107 void
108 DynManager::initialize_integ_groups (
109  void)
110 {
111  // Initialize the integration groups.
112  // Monolithic mode (no external groups registered):
113  // Initialize the default group.
114  if (integ_groups.empty()) {
115  default_integ_group->initialize_group (*this);
116  }
117 
118  // Multiple integrators mode:
119  // Initialize each integration group.
120  // The default group is left uninitialized.
121  else {
122  for (std::vector<DynamicsIntegrationGroup *>::iterator it =
123  integ_groups.begin();
124  it != integ_groups.end();
125  ++it) {
126  DynamicsIntegrationGroup * group = *it;
127  group->initialize_group (*this);
128  }
129  }
130 }

138 DynManager::update_integration_group (

142  // This is one of those "this should never happen" kind of errors.
143  if (&group != default_integ_group) {
144  MessageHandler::fail (

146  "Internal error.");
147  }
148 
149  // Add each body that doesn't have an integrator to the default group.
150  for (std::vector<DynBody *>::const_iterator it = dyn_bodies.begin();
151  it != dyn_bodies.end();
152  ++it) {
153  DynBody * body = *it;
154  if (body->get_integration_group() == nullptr) {
155  default_integ_group->add_dyn_body (*body);
156  }
157  }
