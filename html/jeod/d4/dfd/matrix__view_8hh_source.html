66 /**
67  * Encapsulates a collection of values of the underlying type that can
68  * conceptually be indexed by a pair of integers (unsigned).
69  *
70  * Future: For a 1xN or Nx1 matrix, add a vector-like view.
71  *
72  * Future: Add python-like referencing, where SomeView[-1][-1] references
73  * the last value.
74  *
75  * @tparam ElemType The matrix element type, assumed to be a primitive type.
76  */
77 template<typename ElemType, typename RefType> class MatrixView
78 {
79 public:
80  /**
81  * Non-default constructor.
82  * @param n_rows_in The number of rows in the view.
83  * @param n_cols_in The number of columns in the view.
84  */
85  MatrixView(unsigned n_rows_in, unsigned n_cols_in)
86  : n_rows(n_rows_in),
87  n_cols(n_cols_in)
88  {
89  }
90 
91  /**
92  * Destructor. There's nothing to destruct here.
93  */
94  virtual ~MatrixView() = default;
95 
96  /**
97  * Copy assignment.
98  * The default implementation does an item-for-item copy,
99  * which might be suboptimal (e.g., memcpy might be faster).
100  * @param src The view whose data are to be copied into this.
101  * @throw std::length_error if this view and the source view are
102  * incompatibly sized.
103  */
104  virtual MatrixView & operator=(MatrixView & src)
105  {
106  if((src.n_rows != n_rows) || (src.n_cols != n_cols))
107  {
108  JEOD_THROW(std::length_error("Incompatible sizes"));
109  }
110 
111  // To make cppcheck happy
112  n_rows = src.n_rows;
113  n_cols = src.n_cols;
114 
115  for(unsigned ii = 0; ii < n_rows; ++ii)
116  {
117  for(unsigned jj = 0; jj < n_cols; ++jj)
118  {
119  (*this)(ii, jj) = src(ii, jj);
120  }
121  }
122 
123  return *this;
124  }
125 
126  /**
127  * Non-const function call operator, as an index operator.
128  * @param i_row Row number of element to be indexed.
129  * @param i_col Column number of element to be indexed.
130  * @return Reference to the indexed value.
131  */
132  virtual RefType operator()(unsigned int i_row, unsigned int i_col) = 0;
133 
134  /**
135  * Const function call operator, as an index operator.
136  * @param i_row Row number of element to be indexed.
137  * @param i_col Column number of element to be indexed.
138  * @return The indexed value.
139  */
140  virtual ElemType operator()(unsigned int i_row, unsigned int i_col) const
141  {
142  return const_cast<MatrixView &>(*this)(i_row, i_col);
143  }
144 
145  /**
146  * Non-const safe (bounds-checked) index operator.
147  * @param i_row Row number of element to be indexed.
148  * @param i_col Column number of element to be indexed.
149  * @return Reference to the indexed value.
150  * @throw std::out_of_range if index is outside the range of the view.
151  */
152  virtual RefType at(unsigned int i_row, unsigned int i_col)
153  {
154  if((i_row >= n_rows) || (i_col >= n_cols))
155  {
156  JEOD_THROW(std::out_of_range("Illegal element access"));
157  }
158  return (*this)(i_row, i_col);
159  }
160 
161  /**
162  * Const safe (bounds-checked) index operator.
163  * @param i_row Row number of element to be indexed.
164  * @param i_col Column number of element to be indexed.
165  * @return The indexed value.
166  * @throw std::out_of_range if index is outside the range of the view.
167  */
168  virtual ElemType at(unsigned int i_row, unsigned int i_col) const
169  {
170  return const_cast<MatrixView &>(*this).at(i_row, i_col);
171  }

173  /**
174  * Size function.

179  return SolverTypes::IndexPairT(n_rows, n_cols);
180  }
