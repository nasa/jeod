75 // Collect forces and torques acting on the vehicle.
76 void
77 StructureIntegratedDynBody::collect_forces_and_torques ()
78 {
79 
80  // Collect forces and torques that directly act on this body.
81  collect_local_forces_and_torques();
82 
83  // Collect forces and torques on attached bodies and add them to the
84  // force and torque collections for this body.
85  for (auto child : dyn_children)
86  {
87  child->collect_forces_and_torques ();
88  }
89 
90  // At this point forces and torques have been accumulated with the body
91  // and the method will have been recursively invoked on all child bodies.
92  // The remaining actions depend on whether this body is a child body
93  // or a root body.
94 
95  // Child body: Propagate forces, torques to the parent body.
96  if (dyn_parent != nullptr)
97  {
98  PropagateForcesAndTorques();
99  }
100 
101  // Root body: Compute total forces and torques and resultant accelerations.
102  else
103  {
104  Wrench effector_wrench_co_m;
105 
106  // Translational and rotational dynamics on:
107  // Translate the structure-referenced wrench to the center of mass.
108  if (translational_dynamics && rotational_dynamics)
109  {
110  effector_wrench_co_m = effector_wrench.transform_to_point (
111  mass.composite_properties.position);
112  }
113  // No else; the default constructor did the right thing.
114 
115  // Rotational dynamics is on:
116  if (rotational_dynamics)
117  {
118  // Compute the body-referenced inertial torque (omega \cross L).
119  compute_inertial_torque();
120 
121  // Compute the total torque, structural referenced.
122  Vector3::sum (collect.effector_torq,
123  collect.environ_torq,
124  collect.no_xmit_torq,
125  collect.extern_torq_struct);
126  Vector3::incr (effector_wrench_co_m.get_torque(),
127  collect.extern_torq_struct);
128 
129  // Compute the rotational acceleration of the body.
130  compute_rotational_acceleration ();
131  }
132 
133  // Rotational dynamics is off:
134  // Zero out the collected torques and the rotational acceleration.
135  else
136  {
137  Vector3::initialize (collect.extern_torq_struct);
138  Vector3::initialize (collect.extern_torq_body);
139  Vector3::initialize (collect.inertial_torq);
140  Vector3::initialize (derivs.rot_accel);
141  Vector3::initialize (struct_derivs.rot_accel);
142  }
143 
144  // Translational dynamics is on:
145  // Compute total force on the body, structural and inertial referenced,
146  // and from this, compute translational accelerations.
147  // Note: The gravitational acceleration must have already been computed.
148  if (translational_dynamics) {
149 
150  // Compute the total force, structural referenced.
151  Vector3::sum (collect.effector_forc,
152  collect.environ_forc,
153  collect.no_xmit_forc,
154  collect.extern_forc_struct);
155  Vector3::incr (effector_wrench_co_m.get_force(),
156  collect.extern_forc_struct);
157 
158  // Compute the inertial-referenced translational acceleration
159  // at the structure origin.
160  compute_translational_acceleration ();
161  }
162 
163  // Translational dynamics is off:
164  // Zero out the collected forces and the translational accelerations.
165  else
166  {
167  Vector3::initialize (collect.extern_forc_struct);
168  Vector3::initialize (collect.extern_forc_inrtl);
169  Vector3::initialize (derivs.non_grav_accel);
170  Vector3::initialize (derivs.trans_accel);
171  Vector3::initialize (struct_derivs.non_grav_accel);
172  Vector3::initialize (struct_derivs.trans_accel);
173  }
174  }
175 }

178 // Collect the local forces and torques acting on the vehicle.
179 void
180 StructureIntegratedDynBody::collect_local_forces_and_torques ()
181 {
182 
183  // Translational and rotational dynamics on:
184  // Accumulate the external wrenches on this body about the structure origin.
185  if (translational_dynamics && rotational_dynamics)
186  {
187  effector_wrench_collection.accumulate (effector_wrench);
188  }
189  else
190  {
191  effector_wrench.reset_force_and_torque();
192  }
193 
194  // Translational dynamics on: Accumulate the external forces on this body.
195  if (translational_dynamics) {
196  accumulate_forces (
197  collect.collect_effector_forc, collect.effector_forc);
198  accumulate_forces (
199  collect.collect_environ_forc, collect.environ_forc);
200  accumulate_forces (
201  collect.collect_no_xmit_forc, collect.no_xmit_forc);
202  }
203 
204  // Translational dynamics off: Zero-out force accumulators.
205  else
206  {
207  Vector3::initialize (collect.effector_forc);
208  Vector3::initialize (collect.environ_forc);
209  Vector3::initialize (collect.no_xmit_forc);
210  }
211 
212 
213  // Rotational dynamics on: Accumulate the external torques on this body.
214  if (rotational_dynamics)
215  {
216  accumulate_torques (
217  collect.collect_effector_torq, collect.effector_torq);
218  accumulate_torques (
219  collect.collect_environ_torq, collect.environ_torq);
220  accumulate_torques (
221  collect.collect_no_xmit_torq, collect.no_xmit_torq);
222  }
223 
224  // Rotational dynamics off: Zero-out torque accumulators.
225  else
226  {
227  Vector3::initialize (collect.effector_torq);
228  Vector3::initialize (collect.environ_torq);
229  Vector3::initialize (collect.no_xmit_torq);
230  }

234 // Propagate forces and torques up the kinematic chain.
235 void
236 StructureIntegratedDynBody::PropagateForcesAndTorques ()
237 {
238  double effector_force_parent_structure[3];
239  double environ_force_parent_structure[3];
240 
241  // Translational and rotational dynamics on:
242  // Transmit accumulated wrenches to the parent,
243  // but in the parent's structural frame.
244  if (translational_dynamics && rotational_dynamics)
245  {
246  StructureIntegratedDynBody* struct_parent =
247  dynamic_cast<StructureIntegratedDynBody*>(dyn_parent);
248  struct_parent->effector_wrench +=
249  effector_wrench.transform_to_parent(mass.structure_point);
250  }
251 
252  // Translational dynamics is on:
253  // Transmit accumulated forces to the parent,
254  // but in the parent's structural frame.
255  if (translational_dynamics) {
256 
257  // Transform transmittable forces to parent structural.
258  // The inherited (Mass) structure_point object provides the
259  // transformation from parent structural to child structural.
260  // The transpose is needed to transform from child to parent.
261  Vector3::transform_transpose (
262  mass.structure_point.T_parent_this, collect.effector_forc,
263  effector_force_parent_structure);
264  Vector3::transform_transpose (
265  mass.structure_point.T_parent_this, collect.environ_forc,
266  environ_force_parent_structure);
267 
268  // Transmit these forces to the parent dyn body.
269  Vector3::incr (effector_force_parent_structure,
270  dyn_parent->collect.effector_forc);
271  Vector3::incr (environ_force_parent_structure,
272  dyn_parent->collect.environ_forc);
273  }
274 
275  // Translational dynamics is off:
276  // Zero out transmitted forces as these become torques in the parent.
277  else
278  {
279  Vector3::initialize (effector_force_parent_structure);
280  Vector3::initialize (environ_force_parent_structure);
281  }
282 
283  // Rotational dynamics is on:
284  // Transmit accumulated torques to the parent,
285  // but in the parent's structural frame.
286  if (rotational_dynamics) {
287  double effector_torque_parent_structure[3];
288  double environ_torque_parent_structure[3];
289  double parent_co_m_to_child_co_m[3];
290  // Reconstruct composite relative state
291  RefFrameState composite_wrt_pstr;
292  composite_body.compute_relative_state(dyn_parent->structure, composite_wrt_pstr );
293  // Transform transmittable torques to parent structural.
294  Vector3::transform_transpose (
295  mass.structure_point.T_parent_this, collect.effector_torq,
296  effector_torque_parent_structure);
297  Vector3::transform_transpose (
298  mass.structure_point.T_parent_this, collect.environ_torq,
299  environ_torque_parent_structure);
300 
301  // Compute the parent cm to child cm offset in parent structural.
302  Vector3::diff (composite_wrt_pstr.trans.position,
303  dyn_parent->mass.composite_properties.position,
304  parent_co_m_to_child_co_m);
305 
306  // Compute the torque contributions of the child forces.
307  Vector3::cross_incr (
308  parent_co_m_to_child_co_m, effector_force_parent_structure,
309  effector_torque_parent_structure);
310  Vector3::cross_incr (
311  parent_co_m_to_child_co_m, environ_force_parent_structure,
312  environ_torque_parent_structure);
313 
314  // Transmit the torques to the parent dyn body.
315  Vector3::incr (effector_torque_parent_structure,
316  dyn_parent->collect.effector_torq);
317  Vector3::incr (environ_torque_parent_structure,
318  dyn_parent->collect.environ_torq);
319  }
320 
321  // There is nothing to do here if rotational dynamics is off.
322  else
323  {
324  }
325 }

328 // Compute the inertial torque.
329 void
330 StructureIntegratedDynBody::compute_inertial_torque ()
331 {
332  double ang_vel_body[3]; // Body referenced angular velocity
333  double ang_mom_body[3]; // Body referenced angular momentum
334 
335  // Transform the structural-referenced angular velocity to body.
336  // The inherited (Mass) composite_properties member provides the
337  // transformation from structure to body.
338  Vector3::transform( mass.composite_properties.T_parent_this,
339  structure.state.rot.ang_vel_this,
340  ang_vel_body);
341 
342  // Compute body-referenced angular momentum L and inertial torque
343  // w cross L, where w is the body-referenced angular velocity.
344  Vector3::transform( mass.composite_properties.inertia,
345  ang_vel_body,
346  ang_mom_body);
347  Vector3::cross( ang_vel_body,
348  ang_mom_body,
349  collect.inertial_torq);
350 }

353 // Compute the body- and structure-referenced rotational acceleration vectors.
354 void
355 StructureIntegratedDynBody::compute_rotational_acceleration ()
356 {
357  double torque_body[3]; // Body referenced total torque
358 
359  // Transform the structural-referenced torque to body.
360  Vector3::transform ( mass.composite_properties.T_parent_this,
361  collect.extern_torq_struct,
362  collect.extern_torq_body );
363 
364  // Compute the body-referenced apparent torque via
365  // \dot L_body = \dot L_inertial - omega \cross L.
366  Vector3::diff (
367  collect.extern_torq_body, collect.inertial_torq,
368  torque_body);
369 
370  // Solve the rotational EOM for body accelerations.
371  Vector3::transform( mass.composite_properties.inverse_inertia,
372  torque_body,
373  derivs.rot_accel );
374 
375  // Truncate small rotational accels to avoid numerical problems.
376  Vector3::zero_small( 1e-20, derivs.rot_accel);
377 
378  // We're integrating the structural frame; transform to structural.
379  Vector3::transform_transpose( mass.composite_properties.T_parent_this,
380  derivs.rot_accel,
381  struct_derivs.rot_accel);
382 }

385 // Compute the inertial-referenced translational acceleration vector.
386 void
387 StructureIntegratedDynBody::compute_translational_acceleration()
388 {
389  // Transform the total external force to inertial.
390  // The structure member provides the transformation from inertial to
391  // structural. The transpose is needed to transform to inertial.
392  Vector3::transform_transpose (
393  structure.state.rot.T_parent_this, collect.extern_forc_struct,
394  collect.extern_forc_inrtl);
395 
396  // Compute accelerations at the composite center of mass.
397  Vector3::scale( collect.extern_forc_inrtl,
398  mass.composite_properties.inverse_mass,
399  derivs.non_grav_accel);
400 
401  // Compute fictitious acceleration due to rotational velocity.
402  double omega_cross_r[3];
403 
404  Vector3::cross (
405  mass.composite_properties.position, structure.state.rot.ang_vel_this,
406  omega_cross_r);
407  Vector3::cross (
408  structure.state.rot.ang_vel_this, omega_cross_r,
409  inertial_accel_struct_omega);
410 
411  complete_translational_acceleration();
412 }

415 // Compute the inertial-referenced translational acceleration vector.
416 void
417 StructureIntegratedDynBody::complete_translational_acceleration ()
418 {
419 
420  // Compute translational acceleration at the composite center of mass.
421  Vector3::sum (
422  derivs.non_grav_accel, grav_interaction.grav_accel,
423  derivs.trans_accel);
424 
425  // Compute acceleration due to rotation.
426  // Note that this ignores the Coriolis effect due to movement of
427  // the center of mass inside the vehicle.
428  Vector3::cross (
429  mass.composite_properties.position, struct_derivs.rot_accel,
430  inertial_accel_struct_omega_dot);
431  Vector3::sum (
432  inertial_accel_struct_omega, inertial_accel_struct_omega_dot,
433  inertial_accel_struct);
434  Vector3::transform_transpose (
435  structure.state.rot.T_parent_this, inertial_accel_struct,
436  inertial_accel_inrtl);
437 
438  // Compute accelerations at the structural origin.
439  Vector3::sum (
440  derivs.non_grav_accel, inertial_accel_inrtl,
441  struct_derivs.non_grav_accel);
442  Vector3::sum (
443  struct_derivs.non_grav_accel, grav_interaction.grav_accel,
444  struct_derivs.trans_accel);
445 }
