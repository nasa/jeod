68 // Collect forces and torques acting on the vehicle.
69 void StructureIntegratedDynBody::collect_forces_and_torques()
70 {
71  // Collect forces and torques that directly act on this body.
72  collect_local_forces_and_torques();
73 
74  // Collect forces and torques on attached bodies and add them to the
75  // force and torque collections for this body.
76  for(auto child : dyn_children)
77  {
78  child->collect_forces_and_torques();
79  }
80 
81  // At this point forces and torques have been accumulated with the body
82  // and the method will have been recursively invoked on all child bodies.
83  // The remaining actions depend on whether this body is a child body
84  // or a root body.
85 
86  // Child body: Propagate forces, torques to the parent body.
87  if(dyn_parent != nullptr)
88  {
89  PropagateForcesAndTorques();
90  }
91 
92  // Root body: Compute total forces and torques and resultant accelerations.
93  else
94  {
95  Wrench effector_wrench_co_m;
96 
97  // Translational and rotational dynamics on:
98  // Translate the structure-referenced wrench to the center of mass.
99  if(translational_dynamics && rotational_dynamics)
100  {
101  effector_wrench_co_m = effector_wrench.transform_to_point(mass.composite_properties.position);
102  }
103  // No else; the default constructor did the right thing.
104 
105  // Rotational dynamics is on:
106  if(rotational_dynamics)
107  {
108  // Compute the body-referenced inertial torque (omega \cross L).
109  compute_inertial_torque();
110 
111  // Compute the total torque, structural referenced.
112  Vector3::sum(collect.effector_torq, collect.environ_torq, collect.no_xmit_torq, collect.extern_torq_struct);
113  Vector3::incr(effector_wrench_co_m.get_torque(), collect.extern_torq_struct);
114 
115  // Compute the rotational acceleration of the body.
116  compute_rotational_acceleration();
117  }
118 
119  // Rotational dynamics is off:
120  // Zero out the collected torques and the rotational acceleration.
121  else
122  {
123  Vector3::initialize(collect.extern_torq_struct);
124  Vector3::initialize(collect.extern_torq_body);
125  Vector3::initialize(collect.inertial_torq);
126  Vector3::initialize(derivs.rot_accel);
127  Vector3::initialize(struct_derivs.rot_accel);
128  }
129 
130  // Translational dynamics is on:
131  // Compute total force on the body, structural and inertial referenced,
132  // and from this, compute translational accelerations.
133  // Note: The gravitational acceleration must have already been computed.
134  if(translational_dynamics)
135  {
136  // Compute the total force, structural referenced.
137  Vector3::sum(collect.effector_forc, collect.environ_forc, collect.no_xmit_forc, collect.extern_forc_struct);
138  Vector3::incr(effector_wrench_co_m.get_force(), collect.extern_forc_struct);
139 
140  // Compute the inertial-referenced translational acceleration
141  // at the structure origin.
142  compute_translational_acceleration();
143  }
144 
145  // Translational dynamics is off:
146  // Zero out the collected forces and the translational accelerations.
147  else
148  {
149  Vector3::initialize(collect.extern_forc_struct);
150  Vector3::initialize(collect.extern_forc_inrtl);
151  Vector3::initialize(derivs.non_grav_accel);
152  Vector3::initialize(derivs.trans_accel);
153  Vector3::initialize(struct_derivs.non_grav_accel);
154  Vector3::initialize(struct_derivs.trans_accel);
155  }
156  }
157 }

159 // Collect the local forces and torques acting on the vehicle.
160 void StructureIntegratedDynBody::collect_local_forces_and_torques()
161 {
162  // Translational and rotational dynamics on:
163  // Accumulate the external wrenches on this body about the structure origin.
164  if(translational_dynamics && rotational_dynamics)
165  {
166  effector_wrench_collection.accumulate(effector_wrench);
167  }
168  else
169  {
170  effector_wrench.reset_force_and_torque();
171  }
172 
173  // Translational dynamics on: Accumulate the external forces on this body.
174  if(translational_dynamics)
175  {
176  accumulate_forces(collect.collect_effector_forc, collect.effector_forc);
177  accumulate_forces(collect.collect_environ_forc, collect.environ_forc);
178  accumulate_forces(collect.collect_no_xmit_forc, collect.no_xmit_forc);
179  }
180 
181  // Translational dynamics off: Zero-out force accumulators.
182  else
183  {
184  Vector3::initialize(collect.effector_forc);
185  Vector3::initialize(collect.environ_forc);
186  Vector3::initialize(collect.no_xmit_forc);
187  }
188 
189  // Rotational dynamics on: Accumulate the external torques on this body.
190  if(rotational_dynamics)
191  {
192  accumulate_torques(collect.collect_effector_torq, collect.effector_torq);
193  accumulate_torques(collect.collect_environ_torq, collect.environ_torq);
194  accumulate_torques(collect.collect_no_xmit_torq, collect.no_xmit_torq);
195  }
196 
197  // Rotational dynamics off: Zero-out torque accumulators.
198  else
199  {
200  Vector3::initialize(collect.effector_torq);
201  Vector3::initialize(collect.environ_torq);
202  Vector3::initialize(collect.no_xmit_torq);
203  }
204 }

206 // Propagate forces and torques up the kinematic chain.
207 void StructureIntegratedDynBody::PropagateForcesAndTorques()
208 {
209  double effector_force_parent_structure[3];
210  double environ_force_parent_structure[3];
211 
212  // Translational and rotational dynamics on:
213  // Transmit accumulated wrenches to the parent,
214  // but in the parent's structural frame.
215  if(translational_dynamics && rotational_dynamics)
216  {
217  auto * struct_parent = dynamic_cast<StructureIntegratedDynBody *>(dyn_parent);
218  struct_parent->effector_wrench += effector_wrench.transform_to_parent(mass.structure_point);
219  }
220 
221  // Translational dynamics is on:
222  // Transmit accumulated forces to the parent,
223  // but in the parent's structural frame.
224  if(translational_dynamics)
225  {
226  // Transform transmittable forces to parent structural.
227  // The inherited (Mass) structure_point object provides the
228  // transformation from parent structural to child structural.
229  // The transpose is needed to transform from child to parent.
230  Vector3::transform_transpose(mass.structure_point.T_parent_this,
231  collect.effector_forc,
232  effector_force_parent_structure);
233  Vector3::transform_transpose(mass.structure_point.T_parent_this,
234  collect.environ_forc,
235  environ_force_parent_structure);
236 
237  // Transmit these forces to the parent dyn body.
238  Vector3::incr(effector_force_parent_structure, dyn_parent->collect.effector_forc);
239  Vector3::incr(environ_force_parent_structure, dyn_parent->collect.environ_forc);
240  }
241 
242  // Translational dynamics is off:
243  // Zero out transmitted forces as these become torques in the parent.
244  else
245  {
246  Vector3::initialize(effector_force_parent_structure);
247  Vector3::initialize(environ_force_parent_structure);
248  }
249 
250  // Rotational dynamics is on:
251  // Transmit accumulated torques to the parent,
252  // but in the parent's structural frame.
253  if(rotational_dynamics)
254  {
255  double effector_torque_parent_structure[3];
256  double environ_torque_parent_structure[3];
257  double parent_co_m_to_child_co_m[3];
258  // Reconstruct composite relative state
259  RefFrameState composite_wrt_pstr;
260  composite_body.compute_relative_state(dyn_parent->structure, composite_wrt_pstr);
261  // Transform transmittable torques to parent structural.
262  Vector3::transform_transpose(mass.structure_point.T_parent_this,
263  collect.effector_torq,
264  effector_torque_parent_structure);
265  Vector3::transform_transpose(mass.structure_point.T_parent_this,
266  collect.environ_torq,
267  environ_torque_parent_structure);
268 
269  // Compute the parent cm to child cm offset in parent structural.
270  Vector3::diff(composite_wrt_pstr.trans.position,
271  dyn_parent->mass.composite_properties.position,
272  parent_co_m_to_child_co_m);
273 
274  // Compute the torque contributions of the child forces.
275  Vector3::cross_incr(parent_co_m_to_child_co_m,
276  effector_force_parent_structure,
277  effector_torque_parent_structure);
278  Vector3::cross_incr(parent_co_m_to_child_co_m, environ_force_parent_structure, environ_torque_parent_structure);
279 
280  // Transmit the torques to the parent dyn body.
281  Vector3::incr(effector_torque_parent_structure, dyn_parent->collect.effector_torq);
282  Vector3::incr(environ_torque_parent_structure, dyn_parent->collect.environ_torq);
283  }
284 
285  // There is nothing to do here if rotational dynamics is off.
286  else
287  {
288  }
289 }

291 // Compute the inertial torque.
292 void StructureIntegratedDynBody::compute_inertial_torque()
293 {
294  double ang_vel_body[3]; // Body referenced angular velocity
295  double ang_mom_body[3]; // Body referenced angular momentum
296 
297  // Transform the structural-referenced angular velocity to body.
298  // The inherited (Mass) composite_properties member provides the
299  // transformation from structure to body.
300  Vector3::transform(mass.composite_properties.T_parent_this, structure.state.rot.ang_vel_this, ang_vel_body);
301 
302  // Compute body-referenced angular momentum L and inertial torque
303  // w cross L, where w is the body-referenced angular velocity.
304  Vector3::transform(mass.composite_properties.inertia, ang_vel_body, ang_mom_body);
305  Vector3::cross(ang_vel_body, ang_mom_body, collect.inertial_torq);
306 }

308 // Compute the body- and structure-referenced rotational acceleration vectors.
309 void StructureIntegratedDynBody::compute_rotational_acceleration()
310 {
311  double torque_body[3]; // Body referenced total torque
312 
313  // Transform the structural-referenced torque to body.
314  Vector3::transform(mass.composite_properties.T_parent_this, collect.extern_torq_struct, collect.extern_torq_body);
315 
316  // Compute the body-referenced apparent torque via
317  // \dot L_body = \dot L_inertial - omega \cross L.
318  Vector3::diff(collect.extern_torq_body, collect.inertial_torq, torque_body);
319 
320  // Solve the rotational EOM for body accelerations.
321  Vector3::transform(mass.composite_properties.inverse_inertia, torque_body, derivs.rot_accel);
322 
323  // Truncate small rotational accels to avoid numerical problems.
324  Vector3::zero_small(1e-20, derivs.rot_accel);
325 
326  // We're integrating the structural frame; transform to structural.
327  Vector3::transform_transpose(mass.composite_properties.T_parent_this, derivs.rot_accel, struct_derivs.rot_accel);
328 }

330 // Compute the inertial-referenced translational acceleration vector.
331 void StructureIntegratedDynBody::compute_translational_acceleration()
332 {
333  // Transform the total external force to inertial.
334  // The structure member provides the transformation from inertial to
335  // structural. The transpose is needed to transform to inertial.
336  Vector3::transform_transpose(structure.state.rot.T_parent_this,
337  collect.extern_forc_struct,
338  collect.extern_forc_inrtl);
339 
340  // Compute accelerations at the composite center of mass.
341  Vector3::scale(collect.extern_forc_inrtl, mass.composite_properties.inverse_mass, derivs.non_grav_accel);
342 
343  // Compute fictitious acceleration due to rotational velocity.
344  double omega_cross_r[3];
345 
346  Vector3::cross(mass.composite_properties.position, structure.state.rot.ang_vel_this, omega_cross_r);
347  Vector3::cross(structure.state.rot.ang_vel_this, omega_cross_r, inertial_accel_struct_omega);
348 
349  complete_translational_acceleration();
350 }

352 // Compute the inertial-referenced translational acceleration vector.
353 void StructureIntegratedDynBody::complete_translational_acceleration()
354 {
355  // Compute translational acceleration at the composite center of mass.
356  Vector3::sum(derivs.non_grav_accel, grav_interaction.grav_accel, derivs.trans_accel);
357 
358  // Compute acceleration due to rotation.
359  // Note that this ignores the Coriolis effect due to movement of
360  // the center of mass inside the vehicle.
361  Vector3::cross(mass.composite_properties.position, struct_derivs.rot_accel, inertial_accel_struct_omega_dot);
362  Vector3::sum(inertial_accel_struct_omega, inertial_accel_struct_omega_dot, inertial_accel_struct);
363  Vector3::transform_transpose(structure.state.rot.T_parent_this, inertial_accel_struct, inertial_accel_inrtl);
364 
365  // Compute accelerations at the structural origin.
366  Vector3::sum(derivs.non_grav_accel, inertial_accel_inrtl, struct_derivs.non_grav_accel);
367  Vector3::sum(struct_derivs.non_grav_accel, grav_interaction.grav_accel, struct_derivs.trans_accel);
368 }
