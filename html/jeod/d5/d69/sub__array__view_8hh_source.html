69 /**
70  * Represents a subarray view into an array.
71  * The function call operator provides unprotected access to an element
72  * of the subarray. The member function at() ensures that the indexed
73  * element is valid.
74  */
75 template <class ParentType, typename ElemType, typename RefType>
76 class SubArrayView : public MatrixView<ElemType, RefType>
77 {
78 public:
79 
80  /**
81  * Non-default constructor.
82  * @param array_in The array for which a view is to be constructed.
83  * @param row_range The topmost and bottommost row of the view, plus one.
84  * @param col_range The leftmost and rightmost column of the view, plus one.
85  * @throw std::out_of_range if sub-array bounds are invalid.
86  */
87  SubArrayView (
88  ParentType& array_in,
89  const SolverTypes::IndexPairT& row_range,
90  const SolverTypes::IndexPairT& col_range)
91  :
92  SubArrayView(
93  array_in,
94  row_range.first, row_range.second,
95  col_range.first, col_range.second)
96  { }
97 
98  /**
99  * Non-default constructor.
100  * @param array_in The array for which a view is to be constructed.
101  * @param beg_row_in The topmost row of the view.
102  * @param end_row_in The bottommost row of the view, plus one.
103  * @param beg_col_in The leftmost column of the view.
104  * @param end_col_in The rightmost column of the view, plus one.
105  * @throw std::out_of_range if sub-array bounds are invalid.
106  */
107  SubArrayView (
108  ParentType& array_in,
109  unsigned beg_row_in,
110  unsigned end_row_in,
111  unsigned beg_col_in,
112  unsigned end_col_in)
113  :
114  MatrixView<ElemType, RefType>(
115  end_row_in-beg_row_in, end_col_in-beg_col_in),
116  array(array_in),
117  beg_row(beg_row_in),
118  beg_col(beg_col_in)
119  {
120  const auto& array_size = array_in.size();
121  if ((end_row_in < beg_row_in) ||
122  (end_row_in > array_size.first) ||
123  (end_col_in < beg_col_in) ||
124  (end_col_in > array_size.second))
125  {
126  throw std::out_of_range(
127  "Attempt to create illegal SubArrayView");
128  }
129  }
130 
131  /**
132  * Get a modifiable (L-value) reference to the i_row, i_col element
133  * of the sub-array. No bounds checking is performed.
134  * @param i_row The row number, zero based.
135  * @param i_col The column number, zero based.
136  * @return L-value reference to the specified element.
137  */
138  virtual RefType operator() (unsigned i_row, unsigned i_col) final
139  {
140  return array(beg_row+i_row, beg_col+i_col);
141  }
