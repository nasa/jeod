47  * force calculation for a simple spring based contact
48  * dynamics model, takes in geometry information from the appropriate
49  * ContactFacet::calculate_forces but doesn't know about specific type of
50  * ContactFacet
51  * \param[in,out] subject subject frame of the relative state
52  * \param[in,out] target target frame of the relative state
53  * \param[in] rel_state relative state between subject and target in subject frame
54  * \param[in] penetration_vector vector that characterises the interpenetration of the subject and the target
55  * \param[in] rel_velocity relative velocity of the subject and the target in the subject frame
56  */
57 void SpringPairInteraction::calculate_forces(ContactFacet * subject,
58  ContactFacet * target,
59  RelativeDerivedState * rel_state,
60  double * penetration_vector,
61  double * rel_velocity)
62 {
63  double friction_vec[3]; /* Friction directional vector */
64  double damping_force[3]; /* Damping force */
65  double friction_force[3]; /* Friction force */
66  double nvec[3], vec[3]; /* temp vectors */
67  double mag, mag_force, damping_mag; /* vector magnitudes */
68  double force[3], tmp_force[3]; /* forces */
69 
70  // normalize penetration vector and avoid really small values
71  Vector3::zero_small(1.0E-10, penetration_vector);
72  Vector3::normalize(penetration_vector, nvec);
73 
74  /* Spring Constant Force */
75  // scale penetract vector with the spring constant
76  Vector3::scale(penetration_vector, this->spring_k, force);
77 
78  /* Damping Force */
79  // mag of velocity in penetration direction
80  mag = Vector3::dot(rel_velocity, nvec);
81  // calculate magnitude of damping force
82  damping_mag = mag * this->damping_b;
83  // scale nvec with force magnitue to get damping force
84  Vector3::scale(nvec, -damping_mag, damping_force);
85 
86  // add damping force to total force on facet
87  Vector3::sum(force, damping_force, force);
88 
89  /* Friction Force */
90  Vector3::normalize(rel_velocity, friction_vec);
91  // cross velocity with penetration vector
92  Vector3::cross(friction_vec, nvec, vec);
93  // cross penetration vector with result to get vector along subject surface
94  Vector3::cross(nvec, vec, friction_vec);
95 
96  // calcualte the magnitude of the spring force
97  mag_force = Vector3::vmag(force);
98  mag = mag_force * this->mu;
99  // scale surface vector with magnitude
100  Vector3::scale(friction_vec, -mag, friction_force);
101  Vector3::sum(force, friction_force, force);
102  friction_mag = Vector3::vmag(friction_force);
103 
104  /* add Force and Torque to Subject Facet */
105  // rotate force into vehicle structure frame
106  Vector3::transform_transpose(subject->vehicle_point->state.rot.T_parent_this, force, vec);
107  Vector3::transform(subject->vehicle_body->structure.state.rot.T_parent_this, vec, tmp_force);
108  // calculate torque and add to facet total
109  subject->calculate_torque(tmp_force);
110  // add force to facet total
111  Vector3::sum(subject->force, tmp_force, subject->force);
112 
113  /* add Force and Torque to Target Facet */
114  if(target != nullptr)
115  {
116  // negate force to make it equal and opposite to force on subject
117  Vector3::negate(force, force);
118  // rotate force to target frame
119  Vector3::transform(rel_state->rel_state.rot.T_parent_this, force, nvec);
120  // rotate force into vehicle structure frame
121  Vector3::transform_transpose(target->vehicle_point->state.rot.T_parent_this, nvec, vec);
122  Vector3::transform(target->vehicle_body->structure.state.rot.T_parent_this, vec, tmp_force);
123  // calculate torque and add to facet total
124  target->calculate_torque(tmp_force);
125  // add force to facet total
126  Vector3::sum(target->force, tmp_force, target->force);
127  }
128 }
