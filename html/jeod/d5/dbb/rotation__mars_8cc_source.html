62 void RotationMars::update_rotation()
63 {
64  // If full RNP not requested, just take the current time and multiply it
65  // by the rotational velocity to get the updated Mars spin angle
66  if(!use_full_rnp)
67  {
68  // Phi(t) = Phi_o + Phi_dot*time
69  phi_spin = 0.0;
70  phi_spin = current_time * planet_rotational_velocity;
71  }
72  else
73  {
74  // Request nutation correction from nutation
75  if(nutation == nullptr)
76  {
77  MessageHandler::fail(__FILE__,
78  __LINE__,
79  RNPMessages::setup_error,
80  "RotationMars is not currently pointing "
81  "to a nutation object.");
82  return;
83  }
84  double psi_nut = 0.0;
85  psi_nut = nutation->nutation_in_longitude;
86 
87  // Set up the relevant rotation equation from pg. 41 of Konopliv 2006:
88  // Phi(t) = Phi_o + Phi_dot*time - Psi_nut * cos(obliquity angle)
89  phi_spin = phi_at_j2000 + planet_rotational_velocity * current_time - psi_nut * cos(nutation->obliquity_angle);
90 
91  if(phi_spin < 0.0)
92  {
93  phi_spin += 2.0 * M_PI;
94  }
95  }
96 
97  // Take trigs of the rotation angle to build the rotation matrix
98  double cos_phi = cos(phi_spin);
99  double sin_phi = sin(phi_spin);
100 
101  // Now populate the rotation matrix, as follows:
102  // celestial frame = rot_z(-Phi) * mars frame
103  rotation[0][0] = cos_phi;
104  rotation[0][1] = -sin_phi;
105  rotation[0][2] = 0.0;
106 
107  rotation[1][0] = sin_phi;
108  rotation[1][1] = cos_phi;
109  rotation[1][2] = 0.0;
110 
111  rotation[2][0] = 0.0;
112  rotation[2][1] = 0.0;
113  rotation[2][2] = 1.0;
114 }
