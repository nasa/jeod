83  * RotationMars specific implementation of update_rotation, for
84  * axial rotation.
85  */
86 void
87 RotationMars::update_rotation (
88  void)
89 {
90 
91  // If full RNP not requested, just take the current time and multiply it
92  // by the rotational velocity to get the updated Mars spin angle
93  if (!use_full_rnp) {
94 
95  // Phi(t) = Phi_o + Phi_dot*time
96  phi_spin = 0.0;
97  phi_spin = current_time * planet_rotational_velocity;
98  }
99  else {
100 
101  // Request nutation correction from nutation
102  if (nutation == nullptr) {
103  MessageHandler::fail (
104  __FILE__, __LINE__, RNPMessages::setup_error,
105  "RotationMars is not currently pointing "
106  "to a nutation object.");
107  return;
108  }
109  double psi_nut = 0.0;
110  psi_nut = nutation->nutation_in_longitude;
111 
112  // Set up the relevant rotation equation from pg. 41 of Konopliv 2006:
113  // Phi(t) = Phi_o + Phi_dot*time - Psi_nut * cos(obliquity angle)
114  phi_spin = phi_at_j2000 + planet_rotational_velocity * current_time
115  - psi_nut * cos(nutation->obliquity_angle);
116 
117  if (phi_spin < 0.0) {
118  phi_spin += 2.0 * M_PI;
119  }
120  }
121 
122  // Take trigs of the rotation angle to build the rotation matrix
123  double cos_phi = cos (phi_spin);
124  double sin_phi = sin (phi_spin);
125 
126  // Now populate the rotation matrix, as follows:
127  // celestial frame = rot_z(-Phi) * mars frame
128  rotation[0][0] = cos_phi;
129  rotation[0][1] = -sin_phi;
130  rotation[0][2] = 0.0;
131 
132  rotation[1][0] = sin_phi;
133  rotation[1][1] = cos_phi;
134  rotation[1][2] = 0.0;
135 
136  rotation[2][0] = 0.0;
137  rotation[2][1] = 0.0;
138  rotation[2][2] = 1.0;
139 
140  return;
141 }
