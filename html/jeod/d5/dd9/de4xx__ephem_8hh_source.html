106  /**
107  * Enumerates the status values of a De4xEphemItem.
108  */
109  enum Status {
110  Deselected = 0, /**< The item is marked as deselected or the corresponding
111  reference frame is not present in the simulation.
112  The corresponding ephemeris item is permanently
113  disabled under such circumstances. */
114  Inactive = 1, /**< The item has not been deselected and the corresponding
115  reference frame is present but is inactive. */
116  IsRoot = 2, /**< The item has not been deselected and the corresponding
117  reference frame is present and active. However, the
118  item in question is the root of the reference frame
119  tree and hence its state is the trivial state. */
120  InTree = 3, /**< The item has not been deselected and the corresponding
121  reference frame is present and active. However, the
122  ephemeris item that updates this frame is not a
123  part of this model. */
124  Active = 4, /**< The item has not been deselected, the corresponding
125  reference frame is present and active, and the
126  ephemeris item that updates this frame belongs to
127  this model. */
128  };
