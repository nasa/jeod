106  /**
107  * Enumerates the status values of a De4xEphemItem.
108  */
109  enum Status
110  {
111  Deselected = 0, /**< The item is marked as deselected or the corresponding
112  reference frame is not present in the simulation.
113  The corresponding ephemeris item is permanently
114  disabled under such circumstances. */
115  Inactive = 1, /**< The item has not been deselected and the corresponding
116  reference frame is present but is inactive. */
117  IsRoot = 2, /**< The item has not been deselected and the corresponding
118  reference frame is present and active. However, the
119  item in question is the root of the reference frame
120  tree and hence its state is the trivial state. */
121  InTree = 3, /**< The item has not been deselected and the corresponding
122  reference frame is present and active. However, the
123  ephemeris item that updates this frame is not a
124  part of this model. */
125  Active = 4, /**< The item has not been deselected, the corresponding
126  reference frame is present and active, and the
127  ephemeris item that updates this frame belongs to
128  this model. */
129  };
