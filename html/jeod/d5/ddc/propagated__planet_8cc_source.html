77  * Copy rotational state from/to the body reference frame.
78  * \param[in] dyn_time Dynamic time seconds\n Units: s
79  */
80 void PropagatedEphemerisPlanet::update(double dyn_time)
81 {
82  RefFrameState rel_state;
83 
84  // Item is enabled =>
85  // The body reference frame state is the source of the planetary state.
86  if(enabled)
87  {
88  body_ref_frame.compute_relative_state(*(target_frame->get_parent()), rel_state);
89  target_frame->state.trans = rel_state.trans;
90  target_frame->set_timestamp(dyn_time);
91  }
92 
93  // This object is still 'active' when the ephemeris item is disabled.
94  // Enabled/disabled dictates which item updates the planet state.
95  // In this case, that the object is disabled means that the body ref frame
96  // state is driven by the planet state.
97  else
98  {
99  target_frame->compute_relative_state(*(body_ref_frame.get_parent()), rel_state);
100  body.set_state(RefFrameItems::Pos_Vel, rel_state, body_ref_frame);
101  body_ref_frame.set_timestamp(dyn_time);
102  }
103 
104  update_time = dyn_time;
105 }

216 /**
217  * Initialize a PropagatedPlanet model.
218  *
219  * \par Assumptions and Limitations
220  * - This method must be called after the ephemeris model that defines

222  * \param[in] time_manager Time manager
223  * \param[in,out] dyn_manager_ref Dynamics manager

225 void PropagatedPlanet::initialize_model(const TimeManager & time_manager, DynManager & dyn_manager_ref)
226 {
227  // Nothing to do if the model is inactive.
228  if(!active)
229  {
230  return;
231  }
232 
233  // Construct the identifier for this model.
234  ident = NamedItem::construct_name("PropagatedPlanet", planet_name);
235 
236  // Save the dyn_manager object pointer.
237  dyn_manager = &dyn_manager_ref;
238 
239  // Get pointers to the ephemeris time and dynamic time time objects.
240  time_dyn = dynamic_cast<const TimeDyn *>(time_manager.get_time_ptr("Dyn"));
241  if(time_dyn == nullptr)
242  {
243  MessageHandler::fail(__FILE__,
244  __LINE__,
245  EphemeridesMessages::inconsistent_setup,
246  "Could not find the Dynamic Time time object.");
247 
248  // Not reached
249  return;
250  }
251 
252  // Find the parent frame, which must exist.
253  parent_frame = dyn_manager->find_integ_frame(parent_name);
254 
255  if(parent_frame == nullptr)
256  {
257  MessageHandler::fail(__FILE__,
258  __LINE__,
259  EphemeridesMessages::inconsistent_setup,
260  "Frame '%s' has not been registered with the manager.",
261  parent_name.c_str());
262  // Not reached
263  return;
264  }
265 
266  // Tell the dynamics manager that this is an active ephemeris model.
267  dyn_manager->add_ephemeris(*this);
268 
269  // Tell the EphemeridesManager about the ephemeris items that manage
270  // the planet's inertial and planet-fixed frames.
271  ephem_planet.set_name(planet_name, "inertial");
272  ephem_orient.set_name(planet_name, "pfix");
273  dyn_manager->add_ephem_item(ephem_planet);
274  dyn_manager->add_ephem_item(ephem_orient);
275 
276  // Name the body and its parent frame if not already done by the user.
277  if(body.name.size() == 0)
278  {
279  body.name.set_name(planet_name);
280  }
281  if(body.integ_frame_name.empty())
282  {
283  body.integ_frame_name = parent_name;
284  }
285 
286  // Initialize the body and register it with the dynamics manager.
287  body.initialize_model(*dyn_manager);

294 void PropagatedPlanet::ephem_initialize(EphemeridesManager & ephem_manager)
295 {
296  // Nothing to do if the model is inactive.
297  if(!active)
298  {
299  return;
300  }
301 
302  // Find the planet, which must exist.
303  planet = ephem_manager.find_base_planet(planet_name);
304 
305  if(planet == nullptr)
306  {
307  MessageHandler::fail(__FILE__,
308  __LINE__,
309  EphemeridesMessages::inconsistent_setup,
310  "Planet '%s' has not been registered with the manager.",
311  planet_name.c_str());
312  // Not reached
313  return;
314  }
315 
316  ephem_manager.set_target_frame(planet->inertial);
317  ephem_manager.set_target_frame(planet->pfix);
318 
319  // The ephemeris items must target the proper frames.
320  if((ephem_planet.get_target_frame() != &planet->inertial) || (ephem_orient.get_target_frame() != &planet->pfix))
321  {
322  MessageHandler::fail(__FILE__,
323  __LINE__,
324  EphemeridesMessages::inconsistent_setup,
325  "Propagated planet '%s' does not target the planet frames.",
326  planet_name.c_str());
327  }
328 
329  // Dynamic bodies are always active, so the planetary frames must also always
330  // be active. Subscribe to the planet frames to ensure this is the case.
331  planet->inertial.subscribe();
332  planet->pfix.subscribe();
333 
334  // Enable/disable the ephemeris objects per the mode.
335  // Enabled means that the planet frame depends on the body frame;
336  // disabled means the body frame depends on the planet frame.
337  set_mode();
338 }

349 /**
350  * Change the behavior of a PropagatedPlanet.
351  */
352 void PropagatedPlanet::set_mode()
353 {
354  // Switching from the body being the source to the planet is illegal.
355  if((((mode == TransFromBody_RotFromPlanet) || (mode == TransFromBody_RotFromBody)) &&
356  ((commanded_mode == TransFromPlanet_RotFromPlanet) || (commanded_mode == TransFromPlanet_RotFromBody))) ||
357  (((mode == TransFromPlanet_RotFromBody) || (mode == TransFromBody_RotFromBody)) &&
358  ((commanded_mode == TransFromPlanet_RotFromPlanet) || (commanded_mode == TransFromBody_RotFromPlanet))))
359  {
360  MessageHandler::fail(__FILE__,
361  __LINE__,
362  EphemeridesMessages::inconsistent_setup,
363  "Illegal attempt to switch PropagatedPlanet to ephemeris mode.");
364  return;
365  }
366 
367  // Enable/disable the ephemeris objects per the mode.
368  // Enabled means that the planet frame depends on the body frame;
369  // disabled means the body frame depends on the planet frame.
370  if((commanded_mode == TransFromBody_RotFromPlanet) || (commanded_mode == TransFromBody_RotFromBody))
371  {
372  ephem_planet.enable();
373  }
374  else
375  {
376  ephem_planet.disable();
377  }
378 
379  if((commanded_mode == TransFromPlanet_RotFromBody) || (commanded_mode == TransFromBody_RotFromBody))
380  {
381  ephem_orient.enable();
382  }
383  else
384  {
385  ephem_orient.disable();
386  }
387 
388  mode = commanded_mode;
389  dyn_manager->ephem_note_tree_status_change();
390 }

426 /**
427  * Update ephemerides for subscribed items.
428  */
429 void PropagatedPlanet::ephem_update()
430 {
431  // Nothing to do if the model is inactive.
432  if(!active)
433  {
434  return;
435  }
436 
437  // Initialize the dyn body if this hasn't been done yet.
438  if(!initialized)
439  {
440  dyn_manager->initialize_dyn_body(body);
441  initialized = true;
442  }
443 
444  // Get the time of this update.
445  update_time = time_dyn->seconds;
446 
447  // Update the dependent translational and rotational states from the sources.
448  ephem_planet.update(update_time);
449  ephem_orient.update(update_time);
450 
451  // Not in full body mode => need to propagate state throughout the body.
452  if(mode != TransFromBody_RotFromBody)
453  {
454  body.propagate_state();
455  }
456 
457  // Finally, change the mode to the commanded mode if needed.
458  // This must be done *after* the two update calls above to accommodate
459  // a change from FromPlanet to FromBody at post-initialization time.
460  if(commanded_mode != mode)
461  {
462  set_mode();
463  }
464 }
