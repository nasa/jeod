62  const DynBody * dyn_body,
63  const char * file,

77 // Set the parts of the specified reference frame as indicated by set_items.
78 void DynBody::set_state(RefFrameItems::Items set_items, const RefFrameState & state, BodyRefFrame & subject_frame)
79 {
80  DynBody * root_body = get_root_body_internal();
81 
82  // The subject frame must be owned by this dynamic body.
83  check_frame_ownership(subject_frame, this, __FILE__, __LINE__);
84 
85  // Copy the state items specified by set_items to the reference frame
86  // specified by frame_id.
87 
88  // Position:
89  // Copy the position vector.
90  if((set_items & RefFrameItems::Pos) != 0)
91  {
92  root_body->set_state_source_internal(RefFrameItems::Pos, subject_frame);
93  Vector3::copy(state.trans.position, subject_frame.state.trans.position);
94  }
95 
96  // Velocity:
97  // Copy the velocity vector.
98  if((set_items & RefFrameItems::Vel) != 0)
99  {
100  root_body->set_state_source_internal(RefFrameItems::Vel, subject_frame);
101  Vector3::copy(state.trans.velocity, subject_frame.state.trans.velocity);
102  }
103 
104  // Attitude:
105  // Copy the attitude quaternion and compute the transformation matrix.
106  if((set_items & RefFrameItems::Att) != 0)
107  {
108  root_body->set_state_source_internal(RefFrameItems::Att, subject_frame);
109  subject_frame.state.rot.Q_parent_this = state.rot.Q_parent_this;
110  state.rot.Q_parent_this.left_quat_to_transformation(subject_frame.state.rot.T_parent_this);
111  }
112 
113  // Attitude rate:
114  // Copy the angular velocity vector and compute derived items.
115  if((set_items & RefFrameItems::Rate) != 0)
116  {
117  root_body->set_state_source_internal(RefFrameItems::Rate, subject_frame);
118  Vector3::copy(state.rot.ang_vel_this, subject_frame.state.rot.ang_vel_this);
119  subject_frame.state.rot.compute_ang_vel_products();
120  }
121 }

123 // Set the source of aspects of the state.
124 void DynBody::set_state_source(RefFrameItems::Items items, BodyRefFrame & frame)
125 {
126  DynBody * root_body; // This body's root body
127  const DynBody * source_body; // The body that owns the source frame
128 
129  // Handle errors.
130 
131  // 1. The supplied frame must be owned by a DynBody.
132  source_body = dynamic_cast<const DynBody *>(frame.get_owner());
133  if(source_body == nullptr)
134  {
135  MessageHandler::fail(__FILE__,
136  __LINE__,
137  DynBodyMessages::invalid_frame,
138  "Reference frame '%s' is not a valid body reference frame.",
139  frame.get_name().c_str());
140 
141  // Not reached
142  return;
143  }
144 
145  // 2. Minimally, the source body must be in the same tree as is this body.
146  // Because this is a public method, a more restrictive check is made:
147  // The source body or its root body must be this body.
148  // In other words, the method can be invoked in one of two ways:
149  // - Sending the message to the owner of the specified frame, or
150  // - Sending the message to the root body.
151  if((source_body != this) && (source_body->get_root_body() != this))
152  {
153  MessageHandler::fail(__FILE__,
154  __LINE__,
155  DynBodyMessages::invalid_frame,
156  "Reference frame '%s', is owned by body '%s', "
157  "which is not a child body of '%s'.",
158  frame.get_name().c_str(),
159  source_body->name.c_str(),
160  name.c_str());
161 
162  // Not reached
163  return;
164  }
165 
166  // Use the internal equivalent to do the bulk of the work.
167  if(dyn_parent == nullptr)
168  {
169  root_body = this;
170  }
171  else
172  {
173  root_body = get_root_body_internal();
174  }
175  root_body->set_state_source_internal(items, frame);
176 }

260 // Set the source of aspects of the state.
261 void DynBody::set_state_source_internal(RefFrameItems::Items items, BodyRefFrame & frame)
262 {
263  // For each of the state elements specified by the input 'items',
264  // Denote that the state element in question is not initialized and
265  // set the source of that state element to the supplied frame.
266 
267  // Position.
268  if((items & RefFrameItems::Pos) == RefFrameItems::Pos)
269  {
270  initialized_states.remove(RefFrameItems::Pos);
271  position_source = &frame;
272  }
273 
274  // Velocity.
275  if((items & RefFrameItems::Vel) == RefFrameItems::Vel)
276  {
277  initialized_states.remove(RefFrameItems::Vel);
278  velocity_source = &frame;
279  }
280 
281  // Attitude.
282  if((items & RefFrameItems::Att) == RefFrameItems::Att)
283  {
284  initialized_states.remove(RefFrameItems::Att);
285  attitude_source = &frame;
286  }
287 
288  // Attitude rate.
289  if((items & RefFrameItems::Rate) == RefFrameItems::Rate)
290  {
291  initialized_states.remove(RefFrameItems::Rate);
292  rate_source = &frame;
293  }
294 
295  // Denote that the specified items are set in the specified frame.
296  frame.initialized_items.add(items);
297 }
