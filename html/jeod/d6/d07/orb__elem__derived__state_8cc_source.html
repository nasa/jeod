83 /**
84  * Begin initialization of a OrbElemDerivedState.
85  * The initialize method for all subclasses of DerivedState *must* pass
86  * the initialize call to their immediate parent class, which in turn
87  * must do the same, eventually invoking this method.
88  * \param[in,out] subject_body Subject body
89  * \param[in,out] dyn_manager Dynamics manager
90  */
91 void
92 OrbElemDerivedState::initialize (
93  DynBody & subject_body,
94  DynManager & dyn_manager)
95 {
96 
97  // Initialize as a DerivedState.
98  DerivedState::initialize (subject_body, dyn_manager);
99 
100  // Find the planet.
101  // Note that find_planet doesn't return if the named planet is not found.
102  planet = find_planet (dyn_manager, reference_name, "reference_name");
103 
104  // Choose the planet inertial frame to be used
105  if (use_alt_inertial) {
106  inertial_ptr = &(planet->alt_inertial);
107  } else {
108  inertial_ptr = &(planet->inertial);
109  }
110  inertial_ptr->subscribe();
111 
112  // Copy the orbital object body name.
113  elements.set_object_name (subject_body.name.c_str());
114 
115  // Set the planet name about which the object orbits.
116  elements.set_planet_name (planet->name.c_str());
117 
118  return;
119 }

122 /**
123  * Update the state.
124  */
125 void
126 OrbElemDerivedState::update (
127  void)
128 {
129  // Invoke the parent class update method.
130  DerivedState::update(); // This really doesn't do anything!
131 
132  // Check to see if the integration state is planet centered inertial.
133  if (subject->composite_body.get_parent() == inertial_ptr) {
134  compute_orbital_elements (subject->composite_body.state.trans);
135  }
136  else {
137  // If not planet centered inertial, compute it.
138  subject->composite_body.compute_relative_state (*inertial_ptr,
139  rel_state);
140  compute_orbital_elements (rel_state.trans);
141  }
142 
143  return;
144 }
