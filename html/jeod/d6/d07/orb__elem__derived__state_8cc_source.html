63 /**
64  * Begin initialization of a OrbElemDerivedState.
65  * The initialize method for all subclasses of DerivedState *must* pass
66  * the initialize call to their immediate parent class, which in turn
67  * must do the same, eventually invoking this method.
68  * \param[in,out] subject_body Subject body
69  * \param[in,out] dyn_manager Dynamics manager
70  */
71 void OrbElemDerivedState::initialize(DynBody & subject_body, DynManager & dyn_manager)
72 {
73  // Initialize as a DerivedState.
74  DerivedState::initialize(subject_body, dyn_manager);
75 
76  // Find the planet.
77  // Note that find_planet doesn't return if the named planet is not found.
78  planet = find_planet(dyn_manager, reference_name, "reference_name");
79 
80  // Choose the planet inertial frame to be used
81  if(use_alt_inertial)
82  {
83  inertial_ptr = &(planet->alt_inertial);
84  }
85  else
86  {
87  inertial_ptr = &(planet->inertial);
88  }
89  inertial_ptr->subscribe();
90 
91  // Copy the orbital object body name.
92  elements.set_object_name(subject_body.name.c_str());
93 
94  // Set the planet name about which the object orbits.
95  elements.set_planet_name(planet->name.c_str());
96 }

98 /**
99  * Update the state.
100  */
101 void OrbElemDerivedState::update()
102 {
103  // Invoke the parent class update method.
104  DerivedState::update(); // This really doesn't do anything!
105 
106  // Check to see if the integration state is planet centered inertial.
107  if(subject->composite_body.get_parent() == inertial_ptr)
108  {
109  compute_orbital_elements(subject->composite_body.state.trans);
110  }
111  else
112  {
113  // If not planet centered inertial, compute it.
114  subject->composite_body.compute_relative_state(*inertial_ptr, rel_state);
115  compute_orbital_elements(rel_state.trans);
116  }
117 }
