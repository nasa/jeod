249 /**
250  * Process spk objects and store array of loaded spk IDs
251  */
252 void SpiceEphemeris::process_spk()
253 {
254  // Get 'count', the number of SPICE spk kernels.
255  int count = 0;
256  ktotal_c("spk", &count);
257 
258  // If list of files to load was empty, notify user and exit.
259  if(count == 0)
260  {
261  MessageHandler::fail(__FILE__,
262  __LINE__,
263  EphemeridesMessages::item_not_in_file,
264  "File of SPICE kernels to load is empty or incorrectly formatted.\n");
265 
266  // Not reached.
267  return;
268  }
269 
270  // Declare 'found_ids', a so-called cell -- a SPICE-defined variable type --
271  // of integers which will contain an unsorted, cumulative list of all SPICE
272  // IDs loaded. The IDs will then be extracted and sorted for use by JEOD.
273  SPICEINT_CELL(found_ids, MAX_IDS);
274 
275  // Walk the list of all loaded SPICE files and from it populate
276  // spk_object_ids, a sorted array of loaded spk IDs.
277  std::list<int> spk_object_ids;
278  unsigned int num_spk_objs_loaded = 0; // Will contain num of loaded objects
279  int file_handle;
280  int found;
281  char file_name[MAX_PATH_LENGTH];
282  char file_type[MAX_NAME_LENGTH];
283  char loaded_from[MAX_PATH_LENGTH];
284 
285  for(int which = 0; which < count; which++)
286  {
287  // Get the next spk file in the set of loaded ones.
288  kdata_c(which,
289  "spk",
290  MAX_PATH_LENGTH,
291  MAX_NAME_LENGTH,
292  MAX_PATH_LENGTH,
293  file_name,
294  file_type,
295  loaded_from,
296  &file_handle,
297  &found);
298 
299  // Accumulate the list of all available ephemeris objects across all
300  // spk files; spkobj_c() automatically appends to found_ids if it
301  // already contains data.
302  spkobj_c(file_name, &found_ids);
303  }
304 
305  // Check how many objects were loaded from the kernels.
306  num_spk_objs_loaded = card_c(&found_ids);
307 
308  // Extract the loaded IDs into a sortable list.
309  for(unsigned int ii = 0; ii < num_spk_objs_loaded; ++ii)
310  {
311  SpiceInt which_id = SPICE_CELL_ELEM_I(&found_ids, ii);
312  spk_object_ids.push_back(which_id);
313 
314  // Print extracted ID if debugging messages requested.
315  MessageHandler::debug(__FILE__,
316  __LINE__,
317  EphemeridesMessages::debug,
318  "Found object %d in SPICE file.\n",
319  which_id);
320  }
321 
322  spk_object_ids.sort();
323 
324  // Now check the list of planet names to make sure each has a corresponding
325  // spk ID.
326 
327  // Print number of given planet names if debugging requested.
328  MessageHandler::debug(__FILE__,
329  __LINE__,
330  EphemeridesMessages::debug,
331  "Number of names to load: %lu\n",
332  planet_names.size());
333 
334  // Walk list of specified objects to be loaded and try to find them in the
335  // loaded SPICE files. If successful, load them.
336  for(unsigned int ii = 0; ii < planet_names.size(); ++ii)
337  {
338  // Convert current name to uppercase for searching SPICE object list.
339  std::string search_name = planet_names[ii];
340  std::transform(search_name.begin(), search_name.end(), search_name.begin(), toupper);
341 
342  // Have SPICE convert the name to a standard SPICE ID number.
343  int obj_id = 0; // The ID found, if one is found.
344  int bool_found = 0; // Flag indicating success or failure of the search.
345  bodn2c_c(search_name.c_str(), &obj_id, &bool_found);
346 
347  // Check whether SPICE recognized the specified object.
348  if(!bool_found)
349  {
350  MessageHandler::fail(__FILE__,
351  __LINE__,
352  EphemeridesMessages::item_not_in_file,
353  "Name %s is not known to SPICE\n",
354  search_name.c_str());
355 
356  // Not reached.
357  return;
358  }
359 
360  // Now try to find the SPICE-provided ID in the list of loaded objects.
361  bool found_id_in_list = false;
362  for(std::list<int>::const_iterator it = spk_object_ids.begin(); it != spk_object_ids.end(); ++it)
363  {
364  if(obj_id == *it)
365  {
366  found_id_in_list = true;
367  break;
368  }
369  }
370 
371  // Report failure and exit sim if ID not found.
372  if(!found_id_in_list)
373  {
374  MessageHandler::fail(__FILE__,
375  __LINE__,
376  EphemeridesMessages::item_not_in_file,
377  "ID %d corresponding to name %s was never loaded\n",
378  obj_id,
379  search_name.c_str());
380 
381  // Not reached.
382  return;
383  }
384 
385  // Print the object found if debugging messages requested.
386  MessageHandler::debug(__FILE__,
387  __LINE__,
388  EphemeridesMessages::debug,
389  "Current object to load: %s\n",
390  planet_names[ii].c_str());
391 
392  // Found it! Create a corresponding SpiceEphemPoint and add it
393  // to this model's list of points.
394  loaded_spk.push_back(create_new_ephem_point(planet_names[ii], search_name));
395  }
396 }

623 /**
624  * Check what if any barycenters need creating and do so. Any new
625  * barycenters are added to the all_loaded_items list and registered
626  * with the dynamics manager.
627  */
628 void SpiceEphemeris::create_barycenters()
629 {
630  // Handle the trivial case in which zero or one object(s) have been loaded,
631  // meaning that no barycenters need to be created and the sim will run
632  // in either single-planet or empty-space mode.
633  if(loaded_spk.size() < 2)
634  {
635  return;
636  }
637 
638  // Assemble an std::set<int> of parent IDs from the loaded items.
639  std::set<int> parents;
640 
641  for(unsigned int ii = 0; ii < loaded_spk.size(); ++ii)
642  {
643  parents.insert(loaded_spk[ii]->get_parent_id());
644  }
645 
646  // Do we need SSBary?
647  auto f = parents.find(0);
648 
649  if(f != parents.end())
650  {
651  add_barycenter(0);
652  parents.erase(f);
653  }
654  else
655  {
656  if(parents.size() > 1)
657  {
658  // SSBary needed because there is more than 1 planetary system.
659  add_barycenter(0);
660  }
661  }
662 
663  // Add everything else.
664  for(int parent : parents)
665  {
666  add_barycenter(parent);
667  }
668 }

670 /**
671  * Determine which item should be the root of the ref frame tree.
672  */
673 void SpiceEphemeris::determine_root_node()
674 {
675  // Determine the root point of the reference frame tree.
676 
677  // Nothing is extant: there is no root.
678  if(loaded_spk.size() == 0)
679  {
680  root_item = nullptr;
681  return;
682  }
683 
684  // One item exists: That item is the root.
685  else if(loaded_spk.size() == 1)
686  {
687  root_item = loaded_spk[0];
688  }
689 
690  // Non-trivial root, i.e., a barycenter; find it.
691  else
692  {
693  // Based on how function create_barycenters() works, if SSBary
694  // exists in the list it has to be the root. If it doesn't, and
695  // there is more than one loaded item (which resulted in a
696  // barycenter having been created), then the single barycenter
697  // extant has to be the root. Since the barycenters have SPICE IDs
698  // 0 through 9, with SSBary being zero, the search can be neatly
699  // accomplished by a single search loop which stops on the first
700  // barycenter it finds.
701  SpiceEphemPoint * found_bary = nullptr;
702  for(int ii = 0; ii < 10; ++ii)
703  {
704  found_bary = find_spice_id(ii);
705  if(found_bary != nullptr)
706  {
707  root_item = found_bary;
708  break;
709  }
710  }

786  * \param[in,out] ephem_manager Ephemerides manager
787  */
788 void SpiceEphemeris::ephem_build_tree(EphemeridesManager & ephem_manager)
789 {
790  // Nothing to do if the model is inactive.
791  if(inactive)
792  {
793  return;
794  }
795 
796  // Remove all owned frames from the tree.
797  for(unsigned int ii = 0; ii < loaded_spk.size(); ++ii)
798  {
799  EphemerisRefFrame * target_frame = loaded_spk[ii]->get_target_frame();
800  target_frame->remove_from_parent();
801  }
802 
803  // Determine root of the reference frame tree if don't already know it.
804  // Note that should already know it.
805  determine_root_node();

808  // been loaded, so no reference frame tree can be built.

812  __LINE__,

816  // Not reached.
817  return;
818  }

821  MessageHandler::debug(__FILE__,

824  "Adding root %s.\n",
825  root_item->get_target_frame()->get_name().c_str());
826

828  ephem_manager.add_frame_to_tree(*(root_item->get_target_frame()), nullptr);

871 /**
872  * Find the ID of the parent of a given SPICE object. Here "parent"
873  * means both:
874  * 1. the frame to which the object is connected in the JEOD reference
875  * frame tree (the "parent" frame).
876  * 2. the object that will be used as the "observer" in the calls to
877  * SPICE to obtain the state of the given object.
878  * @return of parent\n Units: ID
879  * \param[in] obj_id child ID of which parent is to be found
880  */
881 int SpiceEphemeris::find_parent_id(int obj_id)
882 {
883  // Objects with SPICE IDs greater than 999 are minor planets,
884  // comets, asteroids, etc. Their parent should be SSBary.
885  if(obj_id > 999)
886  {
887  return 0;
888  }
889 
890  // Mercury and Venus are moonless, so their planetary systems have no
891  // distinct barycenter. Thus, their parent in the JEOD ref frame tree
892  // sense is the SSBary.
893  if(obj_id == 199 || obj_id == 299)
894  {
895  return 0;
896  }
897 
898  // Everything else must be part of a non-trivial planetary system. Return
899  // the planetary barycenter. This also works for the sun.
900  return obj_id / 100;
901 }

903 /**
904  * Update ephemerides of inertial frames supplied by spk files.
905  */
906 void SpiceEphemeris::update_trans()
907 {
908  const double null_vec[3] = {0.0, 0.0, 0.0};
909 
910  // Update the root if it is owned by this model.
911  if(root_item != nullptr)
912  {
913  root_item->update(null_vec, null_vec, update_time);
914  }
915 
916  // Perform updates of nodes with respect to their parents.
917  for(unsigned int ii = 0; ii < loaded_spk.size(); ++ii)
918  {
919  // Skip root
920  if(loaded_spk[ii] != root_item)
921  {
922  // Call to SPICE update function.
923  double state[6], position[3], velocity[3], light_time;
924  spkez_c(loaded_spk[ii]->get_spice_id(),
925  *tdb_seconds,
926  "J2000",
927  "NONE",
928  loaded_spk[ii]->get_parent_id(),
929  state,
930  &light_time);
931 
932  // Check whether SPICE returned an error message.
933  if(failed_c())
934  {
935  // Error message returned; obtain it from SPICE.
936  char err_msg[MAX_MSG_LENGTH];
937  getmsg_c("long", MAX_MSG_LENGTH, err_msg);
938 
939  MessageHandler::fail(__FILE__,
940  __LINE__,
941  EphemeridesMessages::item_not_in_file,
942  "Regarding ref frame %s, spkez_c reports the following error: %s\n",
943  loaded_spk[ii]->get_target_frame()->get_name().c_str(),
944  err_msg);
945 
946  // Not reached
947  return;
948  }
949 
950  // Store off state for reference frame update.
951  for(unsigned int jj = 0; jj < 3; ++jj)
952  {
953  position[jj] = state[jj];
954  velocity[jj] = state[3 + jj];
955  }
956 
957  // Tell ephemeris item to update its target frame. Note that the scale
958  // factor is to convert the output from SPICE into meters.
959  loaded_spk[ii]->update_scaled(position, velocity, 1000.0, update_time);
960  }
961  }
962 }
