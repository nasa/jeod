74  lvlh_state.set_subject_frame(subject->composite_body);
75  lvlh_state.set_planet_name(reference_name);
76  lvlh_state.initialize(dyn_manager);
77 
78  lvlh_frame.set_name(subject_body.name.get_name(), reference_name, "lvlh");
79  planet_centered_inertial = const_cast<RefFrame *>(lvlh_state.frame.get_parent());
80  planet_centered_inertial->add_child(lvlh_frame);

82  // If requested, register the frame with the dynamics manager.
