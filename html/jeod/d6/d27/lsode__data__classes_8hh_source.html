133  // Breakdown of RWORK:
134  // NOTE: LYH, LEWT, LACOR, LSAVF, LWM all provide base address in a larger
135  // array (RWORK) for he vectors YH, EWT, ACOR, SAVF, and WM.
136  // I am dviding the RWORK array into these individual arrays, so do not
137  // need the indices any more.*/
138 
139  // RWORK[1-21] is divided into multiple variables now.
140 
141  /**
142  * was RWORK[LYH:LYH+NYH*(MAXORD+1)-1]. LYH = 21
143  * First index is to "i" in y_i, second index is to history order.
144  * history[i,j] = rwork[LYH + j*nyh + i], with lyh = 21 typically.
145  */
146  double ** history{}; //!< trick_units(--)
147  double lin_alg_1{};
148  double lin_alg_2{};
149  /**
150  * was RWORK[LWM:LWM+LENWM-1]. LWM = LYH + (NYH*(MAXORD+1))
151  * lin_alg_1 = rwork[lwm]
152  * lin_alg_2 = rwork[lwm + 1]
153  * lin_alg[i,j] = rwork[lwm+ j*n + i + 2].
154  * The first two elements are treated differently, then it goes to an
155  * array that is sized based on the correction_method.
156  * The array sizes are as follows, ordered by value of correction_method:
157  * 0: 0
158  * 1,2: n x n
159  * 3: 1 x n
160  * 4,5: (2*ml+mu+1) x n.
161  */
162  double ** lin_alg{}; //!< trick_units(--)
163  /**
164  * was RWORK[LEWT:LEWT+N-1]. LEWT = LWM + LENWM
165  * error_weight[i] = rwork[lewt+i].
166  */
167  double * error_weight{}; //!< trick_units(--)
168  /**
169  * was RWORK[LSAVF:LSAVF+N-1]. LSAVF = LEWT + N
170  * save[i] = rwork[lsavf+i].
171  */
172  double * save{}; //!< trick_units(--)
173  /**
174  * was RWORK[LACOR:LACOR+N-1]. LACOR = LSAVF + N
