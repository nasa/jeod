56 /**
57  * Recursively adds elements to the update tree.
58  * If the "parent" to a time-type is defined, adds the "parent" then
59  * returns to adding the "child" type.
60  * If the "parent" is not defined it searches for a suitable "parent"
61  * from the types already in the tree. If that search is successful,
62  * it adds the "child" to the tree, otherwise it returns without
63  * change.
64  *
65  * \par Assumptions and Limitations
66  * - None
67  * \param[in] seeking_status status-value for auto-seek.
68  * \param[in] time_manager_init The TM initializer.
69  */

71 {

73  int conv_dir = 0; // -- direction of application of converter

75  int parent_index = -1; // -- index-value of the parent time

77  time_manager_init->set_status(index, -1);
78  // used to check for loops; resets before exit
79 
80  parent_index = time_manager->time_lookup(update_from_name);

82  // If parent name not found in registered times: failure
83  if(parent_index == -1)
84  {
85  MessageHandler::fail(__FILE__,
86  __LINE__,

89  "An attempt was made to define the source"

95  update_from_name.c_str(),

98  }
99  // else if parent name is blank, do an auto-seek.
100  else if(parent_index == -2)
101  {
102  for(int ii = 0; ii < time_manager->num_types; ++ii)
103  {
104  if(time_manager_init->get_status(ii) == seeking_status)
105  {
106  // TimeManagerInit converter unique index formula:
107  conv_index = ii * time_manager->num_types + index;
108  conv_dir = time_manager_init->get_conv_dir_upd(conv_index);
109  if(conv_dir != 0)
110  { // found a parent.
111  parent_index = ii;
112  parent_ptr = time_manager->get_time_ptr(ii);
113  add_parent(*parent_ptr);
114  MessageHandler::inform(__FILE__,
115  __LINE__,
116  TimeMessages::incomplete_setup_error,
117  "\n"
118  "The update tree was not fully defined by the user: \n"
119  "Time-type %s has no defined update path.\n"
120  "It will be updated from time-type %s.\n",
121  name.c_str(),
122  parent_ptr->name.c_str());
123 
124  break;
125  }
126  }
127  }
128  // if through all registered times and have not found any parent:
129  if(parent_index == -2)
130  {
131  // cannot be added to the tree at this time.
132  // Remain uninitialized for now.
133  time_manager_init->set_status(index, 0);
134  return;
135  }
136  }
137  else
138  { // parent defined and found
139  // Link it in
140  int parent_status = time_manager_init->get_status(parent_index);
141  // 1. verify that this is not a loop.
142  if(parent_status == -1)

153  }
154

156  conv_index = parent_index * time_manager->num_types + index;
157  conv_dir = time_manager_init->get_conv_dir_upd(conv_index);
158  if(conv_dir == 0)
159  {
160  MessageHandler::fail(__FILE__,
161  __LINE__,
162  TimeMessages::invalid_setup_error,
163  "\n"
164  "An attempt was made to define time-type '%s'"
165  "as the source for updating %s, but no converter was provided to"
166  " convert from %s to %s. \n",
167  time_manager->get_time_ptr(parent_index)->name.c_str(),
168  name.c_str(),
169  time_manager->get_time_ptr(parent_index)->name.c_str(),
170  name.c_str());
171  }

173  // 3. if the parent is not in the tree, add it
174  // Two possibilities here - build the update tree by shoots (recursive
175  // addition), or in layers. Recursive addition is more efficient (so add
176  // the ..., then the grandparent, then the parent, then this). This was
177  // the original intention, but it has the following drawback:
178  // Suppose we have a serial/linear tree with one object at each of level
179  // 1,2,3,4, and one additional (poorly defined) type that should go in
180  // level 5. Suppose we are now in level 3, the other types have already
181  // been added in the level-2 sweep, the autoseek looks for the parent to
182  // the remaining type at level 2, finds none, and cannot add anything,
183  // so the code fails out. The only prevention of this is to maintain a
184  // value for the maximum depth of the tree so that the code only fails
185  // when the autoseek has tested the deepest levels. But this eliminates
186  // the efficiency of building the tree recursively.
187  // The alternative, building by layers, would attempt to add the
188  // parent to the current level, and then not attempt to add this type,
189  // because this type will be added at the next level.

191  // If first loop, construct links

193  {
194  add_parent(*parent_ptr);

205  "re-parent to time (%s). "

207  name.c_str(),
208  links.parent()->name.c_str(),
209  parent_ptr->name.c_str());
210  }

217  // }
218  // 4. if parent could not be added, reset status of this type to 0
219  // (indicating failed placement) and exit.
220  // if (parent_status == 0) {

235  if(converter_ptr == nullptr)
