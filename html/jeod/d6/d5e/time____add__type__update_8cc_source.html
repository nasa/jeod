57 /**
58  * Recursively adds elements to the update tree.
59  * If the "parent" to a time-type is defined, adds the "parent" then
60  * returns to adding the "child" type.
61  * If the "parent" is not defined it searches for a suitable "parent"
62  * from the types already in the tree. If that search is successful,
63  * it adds the "child" to the tree, otherwise it returns without
64  * change.
65  *
66  * \par Assumptions and Limitations
67  * - None
68  * \param[in] seeking_status status-value for auto-seek.
69  * \param[in] time_manager_init The TM initializer.
70  */

77  int conv_dir = 0; // -- direction of application of converter
78  JeodBaseTime * parent_ptr = nullptr; // -- pointer to the parent time.
79  int parent_index = -1; // -- index-value of the parent time
80 
81  time_manager_init->set_status (index, -1);
82  // used to check for loops; resets before exit
83 
84  parent_index = time_manager->time_lookup (update_from_name);
85 
86  // If parent name not found in registered times: failure
87  if (parent_index == -1) {
88  MessageHandler::fail (
89  __FILE__, __LINE__, TimeMessages::invalid_setup_error, "\n"
90  "An attempt was made to define the source"
91  " of the update of %s as being '%s', but '%s' was not"
92  " recognized by the time manager as a valid name.\n"
93  " Error likely in Time %s.update_from_name \n",
94  name.c_str(),
95  update_from_name.c_str(),
96  update_from_name.c_str(),
97  name.c_str());
98  }
99  // else if parent name is blank, do an auto-seek.
100  else if (parent_index == -2) {
101  for (int ii = 0; ii < time_manager->num_types; ++ii) {
102  if (time_manager_init->get_status (ii) == seeking_status) {
103  // TimeManagerInit converter unique index formula:
104  conv_index = ii * time_manager->num_types + index;
105  conv_dir = time_manager_init->get_conv_dir_upd (conv_index);
106  if (conv_dir != 0) { // found a parent.
107  parent_index = ii;
108  parent_ptr = time_manager->get_time_ptr (ii);
109  add_parent(*parent_ptr);
110  MessageHandler::inform (
111  __FILE__, __LINE__, TimeMessages::incomplete_setup_error,
112  "\n"
113  "The update tree was not fully defined by the user: \n"
114  "Time-type %s has no defined update path.\n"
115  "It will be updated from time-type %s.\n",
116  name.c_str(),
117  parent_ptr->name.c_str());
118 
119  break;
120  }
121  }
122  }
123  // if through all registered times and have not found any parent:
124  if (parent_index == -2) {
125  // cannot be added to the tree at this time.
126  // Remain uninitialized for now.
127  time_manager_init->set_status (index, 0);
128  return;
129  }
130  }
131  else { // parent defined and found
132  // Link it in
133  int parent_status = time_manager_init->get_status (parent_index);
134  // 1. verify that this is not a loop.
135  if (parent_status == -1) {
136  MessageHandler::fail (
137  __FILE__, __LINE__, TimeMessages::invalid_setup_error, "\n"
138  "In defining the tree for updating the time-type, an infinite loop\n"
139  "was encountered with the pre-defined 'time.time_XXX.update_from'\n"
140  "values. Time-type '%s' depends on time-type '%s', and vice-versa. \n",
141  update_from_name.c_str(),
142  name.c_str());
143  }
144 
145  // 2. check that a converter exists to do this task
146  conv_index = parent_index * time_manager->num_types + index;
147  conv_dir = time_manager_init->get_conv_dir_upd (conv_index);
148  if (conv_dir == 0) {
149  MessageHandler::fail (
150  __FILE__, __LINE__, TimeMessages::invalid_setup_error, "\n"
151  "An attempt was made to define time-type '%s'"
152  "as the source for updating %s, but no converter was provided to"
153  " convert from %s to %s. \n",
154  time_manager->get_time_ptr (parent_index)->name.c_str(),
155  name.c_str(),
156  time_manager->get_time_ptr (parent_index)->name.c_str(),
157  name.c_str());
158  }
159 
160  // 3. if the parent is not in the tree, add it
161  // Two possibilities here - build the update tree by shoots (recursive
162  // addition), or in layers. Recursive addition is more efficient (so add
163  // the ..., then the grandparent, then the parent, then this). This was
164  // the original intention, but it has the following drawback:
165  // Suppose we have a serial/linear tree with one object at each of level
166  // 1,2,3,4, and one additional (poorly defined) type that should go in
167  // level 5. Suppose we are now in level 3, the other types have already
168  // been added in the level-2 sweep, the autoseek looks for the parent to
169  // the remaining type at level 2, finds none, and cannot add anything,
170  // so the code fails out. The only prevention of this is to maintain a
171  // value for the maximum depth of the tree so that the code only fails
172  // when the autoseek has tested the deepest levels. But this eliminates
173  // the efficiency of building the tree recursively.
174  // The alternative, building by layers, would attempt to add the
175  // parent to the current level, and then not attempt to add this type,
176  // because this type will be added at the next level.
177  parent_ptr = time_manager->get_time_ptr (parent_index);
178  // If first loop, construct links
179  if( links.parent() == nullptr )
180  {
181  add_parent(*parent_ptr);
182  }
183  // if subsequent recursion, check consistent
184  else if( links.parent() != parent_ptr )
185  {
186  MessageHandler::fail (
187  __FILE__, __LINE__, TimeMessages::invalid_node, "\n"
188  "Time converter attempting to double parent."
189  "Time (%s) has parent time (%s), but attempting to \n"
190  "re-parent to time (%s). "
191  "Unable to reconcile time update tree.\n",
192  name.c_str(), links.parent()->name.c_str(),
193  parent_ptr->name.c_str() );
194  }
195 
196  if (parent_status == 0) {
197  parent_ptr->add_type_update (seeking_status, time_manager_init);
198  //parent_status = time_manager_init->get_status (parent_index);
199  // DO NOT RESET if building by layers.
200  // }
201  // 4. if parent could not be added, reset status of this type to 0
202  // (indicating failed placement) and exit.
203  // if (parent_status == 0) {
204  // Do not need to test anymore, it must be = 0 since it hasn't changed
205  // in previous if-block. Also, if adding parent, do not want to try
206  // adding this type now when working on a layered approach, it
207  // must bounce out.
208  time_manager_init->set_status (index, 0);
209  return;

211  }

214  // placed in the tree ( status[parent_index] > 0).
215 
216  int mgr_conv_ix = time_manager_init->get_conv_ptr_index (
217  conv_index);
218  TimeConverter * converter_ptr = time_manager->get_converter_ptr (mgr_conv_ix);
219  if (converter_ptr == nullptr) {
220  MessageHandler::fail (

222  "No converter found at index %i in the time_manager's converter registry"
223  ", mapped from \n"
224  "time_manager_init, derived from the time types %s and %s,\n"
225  "with indices %i and %i respectively.\n",
226  conv_index, name.c_str(), parent_ptr->name.c_str(), index, parent_index);
227  return;
228  }
229 
230  if (!converter_ptr->is_initialized()) {
231  converter_ptr->initialize (
232  parent_ptr,
233  this,
234  conv_dir);
235  }
236  update_converter_ptr = converter_ptr;
237  update_converter_direction = conv_dir;

239  time_manager_init->increment_status (index, parent_index);

242  return;
