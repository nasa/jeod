83 /**
84  * Initialize Gottlieb gravity coefficients.
85  */
86 void SphericalHarmonicsGravitySource::initialize_body()
87 {
88  // If degree > 0 then create and fill Gottlieb coefficient arrays.
89  // Otherwise, only spherical gravity can be used.
90  if(degree > 0)
91  {
92  double num1;
93  double den1;
94  double num2;
95  double den2;
96  double ** Pnm;
97 
98  Pnm = JEOD_ALLOC_PRIM_ARRAY(degree + 1, double *);
99 
100  a_by_rad = JEOD_ALLOC_PRIM_ARRAY(degree + 1, double);
101  alpha = JEOD_ALLOC_PRIM_ARRAY(degree + 1, double);
102  beta = JEOD_ALLOC_PRIM_ARRAY(degree + 1, double);
103  xi = JEOD_ALLOC_PRIM_ARRAY(degree + 1, double *);
104  eta = JEOD_ALLOC_PRIM_ARRAY(degree + 1, double *);
105  zeta = JEOD_ALLOC_PRIM_ARRAY(degree + 1, double *);
106  upsilon = JEOD_ALLOC_PRIM_ARRAY(degree + 1, double *);
107  nrdiag = JEOD_ALLOC_PRIM_ARRAY(degree + 1, double);
108  int_to_double = JEOD_ALLOC_PRIM_ARRAY(degree + 2, double);
109 
110  for(unsigned int ii = 0; ii <= degree; ++ii)
111  {
112  Pnm[ii] = JEOD_ALLOC_PRIM_ARRAY(ii + 3, double);
113  xi[ii] = JEOD_ALLOC_PRIM_ARRAY(ii + 1, double);
114  eta[ii] = JEOD_ALLOC_PRIM_ARRAY(ii + 1, double);
115  zeta[ii] = JEOD_ALLOC_PRIM_ARRAY(ii + 1, double);
116  upsilon[ii] = JEOD_ALLOC_PRIM_ARRAY(ii + 1, double);
117  int_to_double[ii] = static_cast<double>(ii);
118  }
119 
120  int_to_double[degree + 1] = static_cast<double>(degree + 1);
121 
122  // In the code below, the equation numbers and page numbers refer to
123  // the Gottlieb 1993 paper.
124 
125  // Bottom of page 47 and page 48, and see equation (7-8)
126  Pnm[0][0] = 1.0;
127  Pnm[0][1] = 0.0;
128  Pnm[0][2] = 0.0;
129  Pnm[1][1] = sqrt(3.0);
130  Pnm[1][2] = 0.0;
131  Pnm[1][3] = 0.0;
132 
133  // Pages 46-47
134  for(unsigned int ii = 2; ii <= degree; ++ii)
135  {
136  for(unsigned int jj = 0; jj <= (ii - 1); ++jj)
137  {
138  // Equation (7-10)
139  num1 = (2.0 * int_to_double[ii] - 1.0) * (2.0 * int_to_double[ii] + 1.0);
140  den1 = (int_to_double[ii] + int_to_double[jj]) * (int_to_double[ii] - int_to_double[jj]);
141  xi[ii][jj] = sqrt(num1 / den1);
142 
143  // Equation (7-10)
144  num2 = (2.0 * int_to_double[ii] + 1.0) * (int_to_double[ii] + int_to_double[jj] - 1.0) *
145  (int_to_double[ii] - int_to_double[jj] - 1.0);
146  den2 = (int_to_double[ii] + int_to_double[jj]) * (int_to_double[ii] - int_to_double[jj]) *
147  (2.0 * int_to_double[ii] - 3.0);
148  if(std::fpclassify(num2) == FP_ZERO)
149  {
150  eta[ii][jj] = 0.0;
151  }
152  else
153  {
154  eta[ii][jj] = sqrt(num2 / den2);
155  }
156  }
157 
158  for(unsigned int jj = 0; jj <= ii; ++jj)
159  {
160  if(ii == jj)
161  {
162  zeta[ii][jj] = 0.0;
163  upsilon[ii][jj] = 0.0;
164  }
165  else if(jj == 0)
166  {
167  // Equation (7-19)
168  zeta[ii][0] = sqrt(int_to_double[ii] * (int_to_double[ii] + 1.0) / 2.0);
169  // Equation (7-22)
170  upsilon[ii][0] = sqrt(int_to_double[ii] * (int_to_double[ii] - 1.0) * (int_to_double[ii] + 1.0) *
171  (int_to_double[ii] + 2.0) / 2.0);
172  }
173  else
174  {
175  // Equation (7-19)
176  zeta[ii][jj] = sqrt((int_to_double[ii] - int_to_double[jj]) *
177  (int_to_double[ii] + int_to_double[jj] + 1.0));
178  // Equation (7-22)
179  upsilon[ii][jj] = sqrt(
180  (int_to_double[ii] - int_to_double[jj]) * (int_to_double[ii] + int_to_double[jj] + 1.0) *
181  (int_to_double[ii] - int_to_double[jj] - 1.0) * (int_to_double[ii] + int_to_double[jj] + 2.0));
182  }
183  }
184 
185  // P(n,n) term, equation (7-8)
186  Pnm[ii][ii] = sqrt((2.0 * int_to_double[ii] + 1.0) / (2.0 * int_to_double[ii])) * Pnm[ii - 1][ii - 1];
187 
188  // P(n,n+1) and P(n,n+2) terms, table 1 (p. 14)
189  Pnm[ii][ii + 1] = 0.0;
190  Pnm[ii][ii + 2] = 0.0;
191 
192  // Equation (7-15) and (7-16)
193  nrdiag[ii] = sqrt(2.0 * int_to_double[ii] + 1.0) * Pnm[ii - 1][ii - 1];
194 
195  // Equation (7-13)
196  alpha[ii] = sqrt((2.0 * int_to_double[ii] + 1.0) * (2.0 * int_to_double[ii] - 1.0)) / int_to_double[ii];
197  beta[ii] = sqrt((2.0 * int_to_double[ii] + 1.0) / (2.0 * int_to_double[ii] - 3.0)) *
198  (int_to_double[ii] - 1.0) / int_to_double[ii];
199  }
200 
201  // Clean up local variable Pnm (needed to init nrdiag above)
202  JEOD_DELETE_2D(Pnm, degree + 1, true);
203  }
204 }

206 /**
207  * Find the given variational gravity effect if already exists.
208  * @return Index number of delta-coeff; -1 if not found
209  * \param[in] delta_coeff delta-coeff to be found
210  */
211 int SphericalHarmonicsGravitySource::find_deltacoeff(const SphericalHarmonicsDeltaCoeffs & delta_coeff) const
212 {
213  int deltacoeff_idx = 0;
214  unsigned int found_deltacoeff = 0; // a "flag" for when idx = 0 on purpose
215  unsigned int n_deltacoeffs = delta_coeffs.size();
216 
217  for(unsigned int ii = 0; ii < n_deltacoeffs; ++ii)
218  {
219  if(std::strcmp(typeid(delta_coeff).name(), typeid(delta_coeffs[ii]).name()) == 0)
220  {
221  found_deltacoeff = 1;
222  deltacoeff_idx = ii;
223  break;
224  }
225  }
226  if(!found_deltacoeff)
227  {
228  deltacoeff_idx = -1;
229  }
230  return deltacoeff_idx;
231 }

233 /**
234  * Add a gravitational variation effect (i.e., a delta coeffs) to the
235  * vector of effects.
236  * \param[in] var_init Effect init structure
237  * \param[in] dyn_manager Dynamics manager
238  * \param[in] var_effect Delta coeff to be added
239  */
240 void SphericalHarmonicsGravitySource::add_deltacoeff(SphericalHarmonicsDeltaCoeffsInit & var_init,
241  BaseDynManager & dyn_manager,
242  SphericalHarmonicsDeltaCoeffs & var_effect)
243 {
244  // Variational effect must be unique for each grav body. Check for conflicts.
245  if(find_deltacoeff(var_effect) != -1)
246  {
247  MessageHandler::fail(__FILE__,
248  __LINE__,
249  GravityMessages::duplicate_entry,
250  "Duplicate gravitational variation effect '%s'",
251  typeid(var_effect).name());
252  return;
253  }
254 
255  // Save the new effect in the delta_coeffs container.
256  delta_coeffs.push_back(&var_effect);
257 
258  // Populate the grav_source ptr inside the given delta_coeffs
259  var_effect.grav_source = this;
260 
261  // Send the init structure to the delta-coeff for it to set itself up with
262  var_effect.initialize(var_init, dyn_manager);
263 }
