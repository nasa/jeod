68 /**
69  * PolarMotionJ2000 specific implementaiton of update_rotation from
70  * PlanetRotation
71  */
72 void PolarMotionJ2000::update_rotation()
73 {
74  // This implements an interpolation table, based on the time in
75  // modified julian date in UT1 being the independent variable and
76  // the xp and yp values being the dependent variables
77  double time = current_time;
78  if(!override_table)
79  {
80  unsigned int index_in_table = 0;
81  if(time >= polar_mjd[last_table_index])
82  {
83  // Julian date is beyond the table, set to last value
84  xp = xp_tbl[last_table_index];
85  yp = yp_tbl[last_table_index];
86 
87  if(warn_table != true)
88  {
89  MessageHandler::warn(__FILE__,
90  __LINE__,
91  RNPMessages::polar_motion_table_warning,
92  "Julian date is beyond values in polar motion \n tables."
93  "The polar motion is being set to\n the last value of the tables.");
94  warn_table = true;
95  }
96  }
97 
98  else if(time <= polar_mjd[0])
99  {
100  xp = xp_tbl[0];
101  yp = yp_tbl[0];
102 
103  if(warn_table != true)
104  {
105  MessageHandler::warn(__FILE__,
106  __LINE__,
107  RNPMessages::polar_motion_table_warning,
108  "Julian date is before values in polar motion \n tables. "
109  "The polar motion is being set to\n the first value of the tables.");
110  warn_table = true;
111  }
112  }
113 
114  else
115  { // need to interpolate xp and yp out of the tables
116  warn_table = false;
117  for(unsigned int i = 0; i < last_table_index; ++i)
118  {
119  if(time < polar_mjd[i + 1])
120  {
121  index_in_table = i;
122  break;
123  } // if(time < polar_mjd)
124  } // for(unsigned int i = 0)
125 
126  xp = xp_tbl[index_in_table] +
127  (xp_tbl[index_in_table + 1] - xp_tbl[index_in_table]) *
128  ((time - polar_mjd[index_in_table]) / (polar_mjd[index_in_table + 1] - polar_mjd[index_in_table]));
129 
130  yp = yp_tbl[index_in_table] +
131  (yp_tbl[index_in_table + 1] - yp_tbl[index_in_table]) *
132  ((time - polar_mjd[index_in_table]) / (polar_mjd[index_in_table + 1] - polar_mjd[index_in_table]));
133 
134  } // else
135  } // if(!override_table)
136 
137  /* The original version which used small angle approximations
138  has been replaced by the exact trigonometric solution per #713 */
139  //
140  // rotation[0][0] = 1.0;
141  // rotation[0][1] = 0.0;
142  // rotation[0][2] = -xp;
143  //
144  // rotation[1][0] = 0.0;
145  // rotation[1][1] = 1.0;
146  // rotation[1][2] = yp;
147  //
148  // rotation[2][0] = xp;
149  // rotation[2][1] = -yp;
150  // rotation[2][2] = 1.0;
151 
152  double cos_xp = cos(xp);
153  double sin_xp = sin(xp);
154  double cos_yp = cos(yp);
155  double sin_yp = sin(yp);
156 
157  rotation[0][0] = cos_xp;
158  rotation[0][1] = 0.0;
159  rotation[0][2] = -sin_xp;
160 
161  rotation[1][0] = sin_xp * sin_yp;
162  rotation[1][1] = cos_yp;
163  rotation[1][2] = cos_xp * sin_yp;
164 
165  rotation[2][0] = sin_xp * cos_yp;
166  rotation[2][1] = -sin_yp;
167  rotation[2][2] = cos_xp * cos_yp;
168 }

176 {

178 
179  if(pm_init == nullptr)

181  MessageHandler::fail(__FILE__,

188

201  {
