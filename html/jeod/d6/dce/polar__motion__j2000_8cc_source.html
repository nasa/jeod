99 /**
100  * PolarMotionJ2000 specific implementaiton of update_rotation from
101  * PlanetRotation
102  */
103 void
104 PolarMotionJ2000::update_rotation (
105  void)
106 {
107 
108  // This implements an interpolation table, based on the time in
109  // modified julian date in UT1 being the independent variable and
110  // the xp and yp values being the dependent variables
111  double time = current_time;
112  if (!override_table) {
113 
114  unsigned int index_in_table = 0;
115  if (time >= polar_mjd[last_table_index]) {
116 
117  // Julian date is beyond the table, set to last value
118  xp = xp_tbl[last_table_index];
119  yp = yp_tbl[last_table_index];
120 
121  if (warn_table != true) {
122  MessageHandler::warn (
123  __FILE__, __LINE__, RNPMessages::polar_motion_table_warning,
124  "Julian date is beyond values in polar motion \n tables."
125  "The polar motion is being set to\n the last value of the tables.");
126  warn_table = true;
127  }
128  }
129 
130 
131  else if (time <= polar_mjd[0]) {
132 
133  xp = xp_tbl[0];
134  yp = yp_tbl[0];
135 
136  if (warn_table != true) {
137  MessageHandler::warn (
138  __FILE__, __LINE__, RNPMessages::polar_motion_table_warning,
139  "Julian date is before values in polar motion \n tables. "
140  "The polar motion is being set to\n the first value of the tables.");
141  warn_table = true;
142  }
143 
144  }
145 
146  else { // need to interpolate xp and yp out of the tables
147  warn_table = false;
148  for (unsigned int i = 0; i < last_table_index; ++i) {
149  if (time < polar_mjd[i + 1]) {
150  index_in_table = i;
151  break;
152  } // if(time < polar_mjd)
153  } // for(unsigned int i = 0)
154 
155  xp = xp_tbl[index_in_table] +
156  (xp_tbl[index_in_table + 1] - xp_tbl[index_in_table]) *
157  ((time - polar_mjd[index_in_table]) /
158  (polar_mjd[index_in_table + 1] - polar_mjd[index_in_table]));
159 
160 
161  yp = yp_tbl[index_in_table] +
162  (yp_tbl[index_in_table + 1] - yp_tbl[index_in_table]) *
163  ((time - polar_mjd[index_in_table]) /
164  (polar_mjd[index_in_table + 1] - polar_mjd[index_in_table]));
165 
166  } // else
167  } // if(!override_table)
168 
169  /* The original version which used small angle approximations
170  has been replaced by the exact trigonometric solution per #713 */
171 //
172 // rotation[0][0] = 1.0;
173 // rotation[0][1] = 0.0;
174 // rotation[0][2] = -xp;
175 //
176 // rotation[1][0] = 0.0;
177 // rotation[1][1] = 1.0;
178 // rotation[1][2] = yp;
179 //
180 // rotation[2][0] = xp;
181 // rotation[2][1] = -yp;
182 // rotation[2][2] = 1.0;
183 
184  double cos_xp = cos(xp); double sin_xp = sin(xp);
185  double cos_yp = cos(yp); double sin_yp = sin(yp);
186 
187  rotation[0][0] = cos_xp;
188  rotation[0][1] = 0.0;
189  rotation[0][2] = -sin_xp;
190 
191  rotation[1][0] = sin_xp * sin_yp;
192  rotation[1][1] = cos_yp;
193  rotation[1][2] = cos_xp * sin_yp;
194 
195  rotation[2][0] = sin_xp * cos_yp;
196  rotation[2][1] = -sin_yp;
197  rotation[2][2] = cos_xp * cos_yp;
198 
199 
200  return;
201 }
