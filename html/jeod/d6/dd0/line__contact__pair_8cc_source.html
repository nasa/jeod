67  * Determine if contact has occurred between the facets of the pair.

69 void
70 LineContactPair::in_contact(
71  void)
72 {
73  double subject_contact_point[3]; // subject contact point
74  double target_contact_point[3]; // target contact point
75  double sub_p1[3], sub_p2[3]; // subject line segment end points
76  double tar_p3[3], tar_p4[3]; // target line segment end points
77  double vec[3]; // temp vectors
78  double subject_mag, target_mag; // distace vector magnitudes
79 
80  // initialize the end points of the line segments
81  Vector3::initialize(sub_p1);
82  Vector3::initialize(sub_p2);
83  Vector3::initialize(tar_p3);
84  Vector3::initialize(tar_p4);
85 
86  // set end points of line segments assuming cylinder aligned with x-axis
87  sub_p1[0] = - line_subject->length / 2.0;
88  sub_p2[0] = line_subject->length / 2.0;
89  tar_p3[0] = - line_target->length / 2.0;
90  tar_p4[0] = line_target->length / 2.0;
91 
92  // convert end points into subject frame through rotation vector addition
93  Vector3::transform_transpose (
94  rel_state.rel_state.rot.T_parent_this,
95  tar_p3, vec);
96  Vector3::sum(rel_state.rel_state.trans.position, vec, tar_p3);
97  Vector3::transform_transpose (
98  rel_state.rel_state.rot.T_parent_this,
99  tar_p4, vec);
100  Vector3::sum(rel_state.rel_state.trans.position, vec, tar_p4);
101 
102  // find the closest point on the two line segments.
103  ContactUtils::dist_line_segments(
104  sub_p1, sub_p2,
105  tar_p3, tar_p4,
106  subject_contact_point, target_contact_point);
107 
108  // put contact point calculated above into the subject facet
109  Vector3::copy(subject_contact_point, line_subject->contact_point);
110 
111  // find vector between the two close points
112  Vector3::diff(target_contact_point, subject_contact_point, vec);
113  //Vector3::normalize(vec, nvec);
114 
115  // find the contact point on the surface of the subject
116  line_subject->calculate_contact_point(vec);
117 
118  // convert subject and target vectors into target frame
119  // subject line closest point
120  Vector3::diff(subject_contact_point, rel_state.rel_state.trans.position, vec);
121  Vector3::transform (rel_state.rel_state.rot.T_parent_this,
122  vec, subject_contact_point);
123  // target line closest point
124  Vector3::diff(target_contact_point, rel_state.rel_state.trans.position, vec);
125  Vector3::transform (rel_state.rel_state.rot.T_parent_this,
126  vec, target_contact_point);
127 
128  // put contact point calculated above into the target facet
129  Vector3::copy(target_contact_point, line_target->contact_point);
130 
131  // find vector between the two close points
132  Vector3::diff(subject_contact_point, target_contact_point, vec);
133  //Vector3::normalize(vec, nvec);
134 
135  // find the contact point on the surface of the target
136  line_target->calculate_contact_point(vec);
137 
138  // get the contact points back from the facets
139  Vector3::copy(line_subject->contact_point, subject_contact_point);
140  Vector3::copy(line_target->contact_point, target_contact_point);
141 
142  // convert target point into subject frame through rotation and vector addition
143  Vector3::transform_transpose (rel_state.rel_state.rot.T_parent_this,
144  target_contact_point, vec);
145  Vector3::sum(rel_state.rel_state.trans.position, vec, target_contact_point);
146 
147  // find the distances to the contact points
148  subject_mag = Vector3::vmag(subject_contact_point);
149  target_mag = Vector3::vmag(target_contact_point);
150 
151  // compare length to target point with subject radius
152  if (subject_mag > target_mag) {
153  double rel_velocity[3]; // relative velocity in subject frame
154  // find the vector that characterizes the interpenetration
155  Vector3::diff(target_contact_point, subject_contact_point, vec);
156  // find the relative velocity of the target in subject frame
157  Vector3::cross (rel_state.rel_state.rot.ang_vel_this, subject_contact_point, rel_velocity);
158  Vector3::diff (rel_velocity, rel_state.rel_state.trans.velocity, rel_velocity);
159  // calculate the forces on the facets
160  interaction->calculate_forces(line_subject, line_target, &rel_state, vec, rel_velocity);
161  }
162 
163  return;
164 }
