48 void LineContactPair::in_contact()
49 {
50  double subject_contact_point[3]; // subject contact point
51  double target_contact_point[3]; // target contact point
52  double sub_p1[3], sub_p2[3]; // subject line segment end points
53  double tar_p3[3], tar_p4[3]; // target line segment end points
54  double vec[3]; // temp vectors
55  double subject_mag, target_mag; // distace vector magnitudes
56 
57  // initialize the end points of the line segments
58  Vector3::initialize(sub_p1);
59  Vector3::initialize(sub_p2);
60  Vector3::initialize(tar_p3);
61  Vector3::initialize(tar_p4);
62 
63  // set end points of line segments assuming cylinder aligned with x-axis
64  sub_p1[0] = -line_subject->length / 2.0;
65  sub_p2[0] = line_subject->length / 2.0;
66  tar_p3[0] = -line_target->length / 2.0;
67  tar_p4[0] = line_target->length / 2.0;
68 
69  // convert end points into subject frame through rotation vector addition
70  Vector3::transform_transpose(rel_state.rel_state.rot.T_parent_this, tar_p3, vec);
71  Vector3::sum(rel_state.rel_state.trans.position, vec, tar_p3);
72  Vector3::transform_transpose(rel_state.rel_state.rot.T_parent_this, tar_p4, vec);
73  Vector3::sum(rel_state.rel_state.trans.position, vec, tar_p4);
74 
75  // find the closest point on the two line segments.
76  ContactUtils::dist_line_segments(sub_p1, sub_p2, tar_p3, tar_p4, subject_contact_point, target_contact_point);
77 
78  // put contact point calculated above into the subject facet
79  Vector3::copy(subject_contact_point, line_subject->contact_point);
80 
81  // find vector between the two close points
82  Vector3::diff(target_contact_point, subject_contact_point, vec);
83  // Vector3::normalize(vec, nvec);
84 
85  // find the contact point on the surface of the subject
86  line_subject->calculate_contact_point(vec);
87 
88  // convert subject and target vectors into target frame
89  // subject line closest point
90  Vector3::diff(subject_contact_point, rel_state.rel_state.trans.position, vec);
91  Vector3::transform(rel_state.rel_state.rot.T_parent_this, vec, subject_contact_point);
92  // target line closest point
93  Vector3::diff(target_contact_point, rel_state.rel_state.trans.position, vec);
94  Vector3::transform(rel_state.rel_state.rot.T_parent_this, vec, target_contact_point);
95 
96  // put contact point calculated above into the target facet
97  Vector3::copy(target_contact_point, line_target->contact_point);
98 
99  // find vector between the two close points
100  Vector3::diff(subject_contact_point, target_contact_point, vec);
101  // Vector3::normalize(vec, nvec);
102 
103  // find the contact point on the surface of the target
104  line_target->calculate_contact_point(vec);
105 
106  // get the contact points back from the facets
107  Vector3::copy(line_subject->contact_point, subject_contact_point);
108  Vector3::copy(line_target->contact_point, target_contact_point);
109 
110  // convert target point into subject frame through rotation and vector addition
111  Vector3::transform_transpose(rel_state.rel_state.rot.T_parent_this, target_contact_point, vec);
112  Vector3::sum(rel_state.rel_state.trans.position, vec, target_contact_point);
113 
114  // find the distances to the contact points
115  subject_mag = Vector3::vmag(subject_contact_point);
116  target_mag = Vector3::vmag(target_contact_point);
117 
118  // compare length to target point with subject radius
119  if(subject_mag > target_mag)
120  {
121  double rel_velocity[3]; // relative velocity in subject frame
122  // find the vector that characterizes the interpenetration
123  Vector3::diff(target_contact_point, subject_contact_point, vec);
124  // find the relative velocity of the target in subject frame
125  Vector3::cross(rel_state.rel_state.rot.ang_vel_this, subject_contact_point, rel_velocity);
126  Vector3::diff(rel_velocity, rel_state.rel_state.trans.velocity, rel_velocity);
127  // calculate the forces on the facets
128  interaction->calculate_forces(line_subject, line_target, &rel_state, vec, rel_velocity);
129  }
130 }
