92  * Convert from cartesian to spherical position
93  */
94 void PlanetFixedPosition::cart_to_spher()
95 {
96  // Compute the radial distance from planet center
97  double r_local = Vector3::vmag(cart_coords);
98 
99  // Protect against division by zero.
100  if(r_local < planet->r_eq * Small_radius_limit)
101  {
102  // Check for a NaN error having crept in before determining which
103  // error to send.
104  if(std::isnan(cart_coords[0]))
105  {
106  MessageHandler::fail(__FILE__,
107  __LINE__,
108  PlanetFixedMessages::domain_error,
109  "Prior computation of coordinates has introduced a NaN "
110  "(not-a-number) error.\n"
111  "Likely a problem with vehicle instability\n. Exiting.\n");
112  }
113  else
114  {
115  MessageHandler::fail(__FILE__,
116  __LINE__,
117  PlanetFixedMessages::domain_error,
118  "Cartesian coordinates are nearly zero.");
119  }
120  return;
121  }
122 
123  // Compute the latitude from the cartesian Z position
124  sphere_coords.latitude = asin(cart_coords[2] / r_local);
125 
126  // Compute the longitude from X and Y positions.
127  sphere_coords.longitude = atan2(cart_coords[1], cart_coords[0]);
128 
129  // Compute the altitude from the radius and the mean equtorial radius
130  sphere_coords.altitude = r_local - planet->r_eq;

133 /**
134  * Convert from cartesian to elliptical position
135  */
136 void PlanetFixedPosition::cart_to_ellip()
137 {
138  // Compute the magnitude of the projection onto the equatorial plane.
139  double x_ellipse_sq = (cart_coords[0] * cart_coords[0]) + (cart_coords[1] * cart_coords[1]);
140  double x_ellipse = sqrt(x_ellipse_sq);
141 
142  // Store locally the Z position in plane of ellipse and its square.
143  double z_ellipse = cart_coords[2];
144  double z_ellipse_sq = z_ellipse * z_ellipse;
145 
146  // Compute the magnitude of the position.
147  double r_ellipse_sq = x_ellipse_sq + z_ellipse_sq;
148  double r_ellipse = sqrt(r_ellipse_sq);
149 
150  // Protect against division by zero.
151  if(r_ellipse < planet->r_eq * Small_radius_limit)
152  {
153  // Check for a NaN error having crept in before determining which
154  // error to send.
155  if(std::isnan(cart_coords[0]))
156  {
157  MessageHandler::fail(__FILE__,
158  __LINE__,
159  PlanetFixedMessages::domain_error,
160  "Prior computation of coordinates has introduced a NaN "
161  "(not-a-number) error.\n"
162  "Likely a problem with vehicle instability\n. Exiting.\n");
163  }
164  else
165  {
166  MessageHandler::fail(__FILE__,
167  __LINE__,
168  PlanetFixedMessages::domain_error,
169  "Cartesian coordinates are nearly zero.");
170  }
171  return;
172  }
173 
174  // Solve for elliptic parameters
175  get_elliptic_parameters(x_ellipse, z_ellipse, ellip_coords.latitude, ellip_coords.altitude);
176 
177  // Check for being directly over the pole
178  if(std::fpclassify(x_ellipse) != FP_ZERO)
179  {
180  // Compute the longitude from the X and Y components
181  ellip_coords.longitude = atan2(cart_coords[1], cart_coords[0]);
182  } // end if
183 }

185 /**
186  * Convert from spherical to cartesian position
187  */
188 void PlanetFixedPosition::spher_to_cart()
189 {
190  // Compute the Sine & Cosine functions of spherical latitude
191  double sin_lat = sin(sphere_coords.latitude);
192  double cos_lat = cos(sphere_coords.latitude);
193 
194  // Compute the Sine & Cosine functions of spherical longitude
195  double sin_lon = sin(sphere_coords.longitude);
196  double cos_lon = cos(sphere_coords.longitude);
197 
198  // Compute the planet fixed cartesian coordinates
199  double radius = planet->r_eq + sphere_coords.altitude;
200  cart_coords[0] = radius * cos_lat * cos_lon;
201  cart_coords[1] = radius * cos_lat * sin_lon;
202  cart_coords[2] = radius * sin_lat;
203 }

205 /**
206  * Convert from elliptical to cartesian position
207  */
208 void PlanetFixedPosition::ellip_to_cart()
209 {
210  // Compute the Sine function of elliptical latitude
211  double sin_lat = sin(ellip_coords.latitude);
212  double cos_lat = cos(ellip_coords.latitude);
213 
214  // Compute the radius of curvature for the local vertical
215  double e = planet->e_ellipsoid; // Planet's ellipsoid eccentricity
216  double radius_eq = planet->r_eq; // Equatorial radius
217  double e_sq = e * e; // Square of elliptic eccentricity
218  double rc_ellipse = radius_eq / (sqrt(1.0 - (e_sq * sin_lat * sin_lat)));
219 
220  // Compute the X position in the plane of the ellipse
221  double x_ellipse = (rc_ellipse + ellip_coords.altitude) * cos_lat;
222 
223  // Compute the planet fixed cartesian coordinates
224  cart_coords[0] = x_ellipse * cos(ellip_coords.longitude);
225  cart_coords[1] = x_ellipse * sin(ellip_coords.longitude);
226  cart_coords[2] = ((rc_ellipse * (1.0 - e_sq)) + ellip_coords.altitude) * sin_lat;
227 }

229 /*******************************************************************************
230 Function: PlanetFixedPosition::get_elliptic_parameters
231 Purpose: Calculate latitude and altitude of a Cartesian point relative to
232 an oblate ellipsoid.
233 References:
234 (((Borkowski, K.m)
235 (Accurate Algorithms To Transform Geocentric To Geodetic Coordinates)
236 (Bull. God., 63 (1989), pp. 5056.)))
237 *******************************************************************************/
238 int PlanetFixedPosition::get_elliptic_parameters( /* Return: --
239  Number of iterations */
240  double r, // In: M Equatorial position, sqrt(x^2+y^2)
241  double z, // In: M Polar position
242  double & lat, // Out: r Latitude
243  double & alt, // Out: M Altitude
244  int maxIters) // In: -- Maximum number of iterations
245 {
246  int numIters = 0;
247  double a = planet->r_eq;
248  double b = planet->r_pol;
249  double y = 0.0;
250 
251  if(r > 0.0)
252  {
253  double y0 = atan(a * z / (b * r));
254  double ar = a * r;
255  double bz = b * z;
256  double w = atan(bz / ar);
257  double c = (a * a - b * b) / sqrt(ar * ar + bz * bz);
258 
259  for(numIters = 0; numIters < maxIters; numIters++)
260  {
261  double d = 2.0 * (cos(y0 - w) - c * cos(2.0 * y0));
262  y = y0 - (2.0 * sin(y0 - w) - c * sin(2.0 * y0)) / d;
263  if(fabs(y - y0) < 1.0e-12)
264  {
265  break;
266  } // end if
267  y0 = y;
268  } // end for numIters
269  lat = atan(a * sin(y) / (b * cos(y)));
270  } // end if
271  else
272  {
273  y = lat = 0.5 * z * M_PI / fabs(z);
274  } // end else
275  alt = (r - a * cos(y)) * cos(lat) + (z - b * sin(y)) * sin(lat);
276  return numIters;
277 } // end get_elliptic_parameters
