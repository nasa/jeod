49 /**
50  * Update the rotational state of the target frame
51  * \param[in] time_tdb Ephemeris time (TDB)\n Units: s
52  * \param[in] time_dyn dyn time for timestamp\n Units: s
53  */
54 void SpiceEphemOrientation::update(double time_tdb, double time_dyn)
55 {
56  // Call the wrapper function around sxform_c(), the SPICE function
57  // that attempts to populate the 6 x 6 matrix that will be used to
58  // update the target frame's state. get_spice_transformation() contains
59  // the safety check which confirms whether the frame is covered by the
60  // currently loaded set of SPICE kernels.
61  double trans6x6[6][6];
62  get_spice_transformation(time_tdb, trans6x6);
63 
64  // Extract the upper left and lower left 3 x 3 submatrices of the 6 x 6
65  // matrix returned by get_spice_transformation(). In this case, these are
66  // respectively the transformation from J2000 to planet-fixed, and its
67  // time derivative.
68  double t_dot[3][3]; // derivative of T_parent_this
69 
70  for(unsigned ii = 0; ii < 3; ++ii)
71  {
72  for(unsigned jj = 0; jj < 3; ++jj)
73  {
74  target_frame->state.rot.T_parent_this[ii][jj] = trans6x6[ii][jj];
75  t_dot[ii][jj] = trans6x6[3 + ii][jj];
76  }
77  }
78 
79  // Calculate and store the target frame's angular velocity
80  double omega_b_wrt_a_in_b[3][3]; // skew-symmetric ang-vel matrix
81 
82  Matrix3x3::product_right_transpose(target_frame->state.rot.T_parent_this, t_dot, omega_b_wrt_a_in_b);
83 
84  target_frame->state.rot.ang_vel_this[0] = -omega_b_wrt_a_in_b[1][2];
85  target_frame->state.rot.ang_vel_this[1] = omega_b_wrt_a_in_b[0][2];
86  target_frame->state.rot.ang_vel_this[2] = -omega_b_wrt_a_in_b[0][1];
87 
88  target_frame->state.rot.compute_quaternion();
89 
90  // Update timestamps for both this model and the target frame
91  update_time = time_dyn;
92  target_frame->set_timestamp(time_dyn);
93 }

110 /**
111  * Populate the SPICE 6 x 6 matrix via sxform_c().
112  * \param[in] time_tdb Ephem time (TDB)
113  * \param[out] trans6x6 Spice matrix
114  */
115 void SpiceEphemOrientation::get_spice_transformation(double time_tdb, double trans6x6[6][6])
116 {
117  // sxform_c() is a SPICE library function that populates a 6 x 6 matrix.
118  // The matrix is populated as follows: [[R, 0]; [dR/dt, R]], where R
119  // is the 3 x 3 transformation matrix from parent to child, and
120  // dR/dt is its time derivative. Note that only the upper left (R) and lower
121  // left (dR/dt) submatrices will be needed to perform the target frame state
122  // update.
123  sxform_c("J2000", spice_frame_name.c_str(), time_tdb, trans6x6);
124 
125  // Check for successful call of sxform_c()
126  if(failed_c())
127  {
128  MessageHandler::fail(__FILE__,
129  __LINE__,
130  EphemeridesMessages::item_not_in_file,
131  "The reference frame %s was not in the loaded kernels.",
132  spice_frame_name.c_str());
133 
134  // Not reached
135  return;
136  }
