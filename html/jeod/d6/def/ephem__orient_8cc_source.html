92 void EphemerisOrientation::note_frame_status_change(RefFrame * frame)

95

98  {
99  MessageHandler::error(__FILE__,

102  "Unexpected frame passed to note_frame_status_change");
103  return;
104  }
105 
106  // Planet-centered frame is active:
107  // Activate this object and issue a subscription if needed.
108  if(target_frame->is_active())
109  {
110  activate();
111 
112  if(!subscribed_to_inertial)
113  {
114  subscribed_to_inertial = true;
115  inertial->subscribe();
116  }
117  }
118 
119  // Planet-centered frame is not active:
120  // Deactivate and retract subscription.
121  else
122  {
123  deactivate();
124 
125  if(subscribed_to_inertial)
126  {
127  subscribed_to_inertial = false;
128  inertial->unsubscribe();
129  }
