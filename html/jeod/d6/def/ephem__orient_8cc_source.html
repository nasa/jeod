121 void
122 EphemerisOrientation::note_frame_status_change (
123  RefFrame * frame)
124 {
125  RefFrame * inertial = const_cast <RefFrame *> (frame->get_parent());
126 
127  // The frame had better be the planet-fixed frame.
128  if (frame != target_frame) {
129  MessageHandler::error (
130  __FILE__, __LINE__, EphemeridesMessages::internal_error,
131  "Unexpected frame passed to note_frame_status_change");
132  return;
133  }
134 
135  // Planet-centered frame is active:
136  // Activate this object and issue a subscription if needed.
137  if (target_frame->is_active()) {
138 
139  activate ();
140 
141  if (! subscribed_to_inertial) {
142  subscribed_to_inertial = true;
143  inertial->subscribe();
144  }
145  }
146 
147  // Planet-centered frame is not active:
148  // Deactivate and retract subscription.
149  else {
150 
151  deactivate ();
152 
153  if (subscribed_to_inertial) {
154  subscribed_to_inertial = false;
155  inertial->unsubscribe();
156  }
157  }
