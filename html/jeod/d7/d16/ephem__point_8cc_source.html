59 void EphemerisPoint::note_frame_status_change(RefFrame * frame)
60 {
61  // The frame had better be the inertial frame.
62  if(frame != target_frame)
63  {
64  MessageHandler::error(__FILE__,
65  __LINE__,
66  EphemeridesMessages::internal_error,
67  "Unexpected frame passed to note_frame_status_change");
68  return;
69  }
70 
71  // Make this point have the same activity level as the inertial frame
72  // associated with the point.
73  if(target_frame->is_active())
74  {
75  activate();
76  }
77  else
78  {
79  deactivate();
80  }
81 }
