76 /**
77  * Begin initialization of a LvlhDerivedState.
78  * The initialize method for all subclasses of DerivedState *must* pass
79  * the initialize call to their immediate parent class, which in turn
80  * must do the same, eventually invoking this method.
81  * \param[in,out] subject_body Subject body
82  * \param[in,out] dyn_manager Dynamics manager
83  */
84 void NedDerivedState::initialize(DynBody & subject_body, DynManager & dyn_manager)
85 {
86  // Initialize as a DerivedState.
87  DerivedState::initialize(subject_body, dyn_manager);
88 
89  // Find the planet.
90  // Note that find_planet doesn't return if the named planet is not found.
91  planet = find_planet(dyn_manager, reference_name, "reference_name");
92 
93  // Choose the planet fixed frame to be used
94  if(use_alt_pfix)
95  {
96  pfix_ptr = &(planet->alt_pfix);
97  }
98  else
99  {
100  pfix_ptr = &(planet->pfix);
101  }
102  pfix_ptr->subscribe();
103 
104  // Name the frame as <vehicle>.<planet>.ned
105  ned_state.ned_frame.set_name(subject_body.name.get_name(), reference_name, "ned");
106 
107  // Connect the frame to the planet's planet-fixed frame
108  // and subscribe to that frame.
109  pfix_ptr->add_child(ned_state.ned_frame);
110 
111  // If requested, register the frame with the dynamics manager.
112  if(register_frame)
113  {
114  dyn_manager.add_ref_frame(ned_state.ned_frame);
115  local_dm = &dyn_manager;
116  }
117 
118  // Initialize the NED state.
119  ned_state.initialize(planet);
120 }

122 /**
123  * Update the state.
124  */
125 void NedDerivedState::update()
126 {
127  // Update the NED frame based on the vehicle's state relative to the planet.
128  subject->composite_body.compute_relative_state(*pfix_ptr, pfix_rel_state);
129  compute_ned_frame(pfix_rel_state.trans);
130 
131  // Timestamp the frame per the vehicle timestamp.
132  ned_state.ned_frame.set_timestamp(subject->composite_body.timestamp());
133 }
