80 /**
81  * A JeodMemoryTable maps strings to values with a coordinated map/vector pair.
82  * \tparam ValueType The underlying type of the values maintained in the table.
83  * The stored values are pointers to this underlying type.
84  *
85  * \details
86  * A JeodMemoryTable contains two data members: a std::map and a std::vector.
87  * The map data member maps keys to integers. The integer mapped by a key is the
88  * index into the vector where the value associated with the key is stored.
89  *
90  * So why not just use a map? The reason is that storing an integer requires
91  * less memory than storing a string or a pointer to a string, particularly on
92  * 64 bit machines. In the application at hand, keeping track of memory
93  * allocations, the number of data types is relatively small compared to the
94  * to the number of allocated chunks of data. The extra overhead of maintaining
95  * a map and a vector is small compared to the savings that results from storing
96  * thousands of integers rather than pointers or strings.
97  *
98  * \par Principal Operations
99  * - add()\n
100  * Returns the integer value associated with a key in the table's map.
101  * In the case of a new key/value pair, a new key/vector size entry is added
102  * to the map and the value is added to the end of the vector.
103  * Note well: The value is ignored when the key is already in the map.
104  * - del()\n
105  * Deletes the key from the table's map and deletes the cloned value at the
106  * corresponding index. The vector itself is modified (truncated) only in
107  * the special case of deleting the last-added entry. This ensures that
108  * stored indices will remain valid.
109  * - get()\n
110  * Returns the value in the table's vector at the specified index.
111  *
112  * \par Assumptions and Limitations
113  * - The value is ignored for duplicate key entries. The underlying assumption
114  * is that all of the values for those duplicate entries are somehow
115  * equal to one another.
116  * - As-is, the table is not thread-safe. Calls to add() and get() made in a
117  * multi-threaded environment should be protected by a mutex. This protection
118  * is the responsibility of the (programmatic) users.
119  * - JEOD reserves index 0 for internal use. Valid indices are positive.
120  * - The del() method should be used only if the (programmatic) user @e knows
121  * that no other references to the to-be-deleted entry exist.
122  */
123 template<typename ValueType> class JeodMemoryTable
124 {
125  // Types
126 public:
127  /**
128  * Maps strings to an index number.
129  */
130  using NameIndex = std::map<const std::string, unsigned int>;
131 
132  /**
133  * Maps index numbers to key values.
134  */
135  using ValueList = std::vector<const ValueType *>;
136 
137  /**
138  * Const iterator over values.
139  */
140  using const_value_iterator = typename ValueList::const_iterator;
141 
142  // Constructors, destructors, etc.
143 public:
144  /**
145  * Default constructor.
146  * Note that JEOD reserves table index 0 as meaning nothing.
147  */
148  JeodMemoryTable()
149  : string_to_index(),
150  value_list()
151  {
152  // Reserve the zero index by pushing a null pointer on the vector.
153  value_list.push_back(nullptr);
154  }
155 
156  /**
157  * Destructor.
158  * The contents of the vector are clones created by add() and hence must
159  * be deleted to avoid a leak.
160  */
161  virtual ~JeodMemoryTable()
162  {
163  // Delete the contents of the identifier list.
164  // Note that element zero is null, so loop starts at 1.
165  for(unsigned int ii = 1; ii < value_list.size(); ++ii)
166  {
167  delete value_list[ii];
168  value_list[ii] = nullptr;
169  }
170 
171  string_to_index.clear();
172  value_list.clear();
173  }
174 
175  // The copy constructor and assignment operator are deleted.
176  JeodMemoryTable(const JeodMemoryTable &) = delete;
177  JeodMemoryTable & operator=(const JeodMemoryTable &) = delete;
178 
179  // Implemented interfaces
180 public:
181  /**
182  * Find the index number at which key/value pair is stored in the table.
183  * \return Index number mapped by the key
184  * \param[in] key Key
185  */
186  unsigned int find(const std::string & key) const
187  {
188  unsigned int found_idx;
189 
190  // Find the key in the map.
191  auto iter(string_to_index.find(key));
192 
193  // Not found: Set the found index to zero to indicate so.
194  // (The zeroth element is reserved just for this purpose.)
195  if(iter == string_to_index.end())
196  {
197  found_idx = 0;
198  }
199 
200  // Found: The mapped index is the return value.
201  else
202  {
203  found_idx = iter->second;
204  }
205 
206  return found_idx;
207  }
208 
209  /**
210  * Returns a const iterator that points to the first element of the list.
211  */
212  const_value_iterator begin() const
213  {
214  return value_list.begin() + 1;
215  }
216 
217  /**
218  * Returns a const iterator that points past the last element of the list.
219  */
220  const_value_iterator end() const
221  {
222  return value_list.end();
223  }
224 
225  /**
226  * Add a key/value pair to the table.
227  * \return Index number mapped by the key
228  * \param[in] key Key
229  * \param[in] val Value
230  */
231  unsigned int add(const std::string & key, const ValueType & val)
232  {
233  unsigned int found_idx;
234 
235  // Find the key in the map.
236  auto iter(string_to_index.find(key));
237 
238  // Not found: Create new entries in both the map and the vector.
239  if(iter == string_to_index.end())
240  {
241  found_idx = value_list.size();
242  string_to_index.insert(NameIndex::value_type(key, found_idx));
243  value_list.push_back(clone(val));
244  }
245 
246  // Found: The mapped index is the return value.
247  else
248  {
249  found_idx = iter->second;
250  }
251 
252  return found_idx;
253  }
254 
255  /**
256  * Delete the key and associated data from the table.
257  * Use with care.
258  * \param[in] key Key
259  * \throw std::invalid_argument on attempting to delete an element that
260  * is not in the table.
261  */
262  void del(const std::string & key)
263  {
264  // Find the key in the map.
265  auto iter(string_to_index.find(key));
266 
267  // Found: Delete. Note: This assumes something is in value_list[idx].
268  if(iter != string_to_index.end())
269  {
270  unsigned int idx = iter->second;
271  delete value_list[idx];
272  value_list[idx] = nullptr;
273 
274  // Deleting last element added: Shrink the vector.
275  if(idx == value_list.size() - 1)
276  {
277  value_list.pop_back();
278  }
279 
280  // Finally, delete the key/index pair from the map.
281  string_to_index.erase(iter);
282  }
283 
284  // Not found: (programmatic) user error.
285  else
286  {
287  JEOD_THROW(std::invalid_argument("key not in table"));
288  }
289  }
290 
291  /**
292  * Retrieve the value for the specified index from the list.
293  * \return Value for specified index.
294  * \param[in] idx Table index whose value is to be retrieved.
295  * \throw std::out_of_range for an index of zero or for an index beyond
296  * the range of the vector.
297  * \throw std::invalid_argument when the index is in range but the
298  * value is null. This only happens when the item in question has
299  * previously been deleted.
300  */
301  const ValueType * get(unsigned int idx) const
302  {
303  const ValueType * result;
304 
305  if((idx == 0) || (idx >= value_list.size()))
306  {
307  JEOD_THROW(std::out_of_range("invalid index"));
308  }
309 
310  result = value_list[idx];
311 
312  if(result == nullptr)
313  {
314  JEOD_THROW(std::invalid_argument("deleted entry"));
315  }
316 
317  return result;
318  }
319 
320  // Virtual interfaces
321 protected:
322  /**
323  * (Somehow) clone the input value.
324  * \return Clone of input value.
325  * \param[in] value Value to be cloned.
326  */
327  virtual const ValueType * clone(const ValueType & value) const = 0;
328 
329  // Member data
330 private:
331  /**
332  * Maps keys to indices in the value_list.
333  */
334  NameIndex string_to_index; //!< trick_io(**)
335 
336  /**
337  * Vector of values.
338  */
339  ValueList value_list; //!< trick_io(**)
340 };
