97  /**
98  * Specifies whether the is_ready() method is to look for the vehicle
99  * entering (SwitchOnApproach) the new integration frame's sphere of
100  * influence versus leaving (SwitchOnDeparture) the current integration
101  * frame's sphere of influence.
102  */
103  enum SwitchSense {
104  SwitchOnApproach = 0, /*
105  The subject DynBody's integration frame is to be switched to the
106  specified integration frame when the distance between the DynBody
107  and the origin of the new integration frame becomes less than
108  the specified switch distance. */
109 
110  SwitchOnDeparture = 1 /*
111  The subject DynBody's integration frame is to be switched to the
112  specified integration frame when the distance between the DynBody
113  and the origin of the current integration frame becomes greater than
114  the specified switch distance. */
115  };
