83 void AerodynamicDrag::aero_drag(double inertial_velocity[3],
84  AtmosphereState * atmos_ptr,
85  double T_inertial_struct[3][3],
86  double mass,
87  double center_grav[3])
88 {
89  unsigned int i_p; // -- plate index value
90  double relative_vel_cm[3]; /* M/s
91  velocity of vehicle center of mass relative to ambient atmosphere in
92  inertial Reference Frame */
93  double rel_vel_cm_struct[3]; /* M/s
94  velocity of vehicle center of mass relative to ambient atmosphere in
95  Structural Reference Frame */
96  double rel_vel_struct_hat[3]; /* M/s
97  unit vector associated with rel_vel_cm_struct */
98  double rel_vel_mag; /* M/s
99  magnitude of rel_vel_cm_struct */
100 
101  // If aerodynamics is not needed, get out
102  if(!active)
103  {
104  return;
105  }
106 
107  // Safety: initialize values to zero
108  Vector3::initialize(aero_force);
109  Vector3::initialize(aero_torque);
110 
111  Vector3::diff(inertial_velocity, atmos_ptr->wind, relative_vel_cm);
112  // Transform relative velocity to structural reference frame and calculate the
113  // magnitude and unit vector of the velocity in the structural frame
114  Vector3::transform(T_inertial_struct, relative_vel_cm, rel_vel_cm_struct);
115 
116  rel_vel_mag = Vector3::vmag(rel_vel_cm_struct);
117  if(std::fpclassify(rel_vel_mag) != FP_ZERO)
118  {
119  double inv_vel = 1 / rel_vel_mag; /* s/M inverse of rel_vel_mag */
120  Vector3::scale(rel_vel_cm_struct, inv_vel, rel_vel_struct_hat);
121  }
122  else
123  {
124  Vector3::unit(1, rel_vel_struct_hat);
125  }
126  // dynamic pressure is the kinetic energy density of the free-moelcular flow
127 
128  if(constant_density == false)
129  {
130  density = atmos_ptr->density;
131  }
132  param.dynamic_pressure = 0.5 * density * rel_vel_mag * rel_vel_mag;
133 
134  // decide if you will be using the full plate model, or the simple default
135  // behavior
136  if(use_default_behavior == true)
137  {
138  default_behavior->aerodrag_force(rel_vel_mag, rel_vel_struct_hat, &param, mass, aero_force, aero_torque);
139  return;
140  }
141 
142  // If we are to this point, we NEED to have a pointer to the aero_surface
143  if(aero_surface_ptr == nullptr)
144  {
145  MessageHandler::fail(__FILE__,
146  __LINE__,
147  AerodynamicsMessages::runtime_error,
148  "use_default_behavior was set to false, but no aero_surface_ptr "
149  "was supplied. Please supply an aero_surface_ptr.");
150  return;
151  }
152 
153  for(i_p = 0; i_p < aero_surface_ptr->facets_size; ++i_p)
154  {
155  Vector3::initialize(aero_surface_ptr->aero_facets[i_p]->force);
156  Vector3::initialize(aero_surface_ptr->aero_facets[i_p]->torque);
157  }
158  // Compute the aerodynamic forces on each plate.
159  for(i_p = 0; i_p < aero_surface_ptr->facets_size; ++i_p)
160  {
161  aero_surface_ptr->aero_facets[i_p]->aerodrag_force(rel_vel_mag, rel_vel_struct_hat, &param, center_grav);
162  // sum individual plate forces over all plates.
163  Vector3::incr(aero_surface_ptr->aero_facets[i_p]->force, aero_force);
164  Vector3::incr(aero_surface_ptr->aero_facets[i_p]->torque, aero_torque);
165  }
166 }
