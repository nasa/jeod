42  * drag force, based on the given parameters
43  * \param[in] rel_vel_mag The magnitude of the relative velocity\n Units: M/s
44  * \param[in] rel_vel_struct_hat The unit vector of the total relative velocity, in the structural frame
45  * \param[in] aero_drag_param_ptr The aerodynamic drag parameters used for drag calculation
46  * \param[in] center_grav The center of gravity of the vehicle, in the structural frame\n Units: M

49 void FlatPlateAeroFacet::aerodrag_force(const double rel_vel_mag,
50  const double rel_vel_struct_hat[3],
51  AeroDragParameters * aero_drag_param_ptr,
52  double center_grav[3])
53 {
54  double sin_alpha; /* --
55  alpha is the angle of attack, the angle between the plate and the relative
56  velocity vector of the palte w.r.t. the ambient gas; this is the sine of
57  that angle. */
58  double force_base; /* N
59  common force calculation, before scaling with the appropriate drag
60  coefficient ( = dynamic pressure * area) */
61  double vec_force_n[3]; /* N vector force normal to the plate */
62  double vec_force_t[3]; /* N
63  vector force tangential to the plate, or parallel with the velocity vector,
64  depending on application */
65  double tangent[3]; /* --
66  the projection of the velocity vector onto the plate, tangential to the
67  plate */
68  Vector3::initialize(tangent);
69 
70  // The base facet's temperature can be updated from dynamically, so copy
71  // that over.
72 
73  temperature = base_facet->temperature;
74 
75  // alpha is the angle of attack, the angle between the plate and the velocity
76  // vector (w.r.t. wind).
77 
78  sin_alpha = Vector3::dot(normal, rel_vel_struct_hat);
79  // If the plate is not windward-facing, it has no aerodynamic drag on it.
80  if(sin_alpha <= 0.0)
81  {
82  Vector3::initialize(force);
83  force_n = 0.0;
84  force_t = 0.0;
85  return;
86  }
87 
88  // consistency check
89  if((calculate_drag_coef == false) && (coef_method == AeroDragEnum::Calc_coef))
90  {
91  MessageHandler::warn(__FILE__,
92  __LINE__,
93  AerodynamicsMessages::runtime_warns,
94  "Must have calculate_drag_coef set to 'true' if \n"
95  "using the calc_coef method of obtaining the drag coefficient.\n"
96  "Resetting the value of calculate_drag_coef.\n");
97  calculate_drag_coef = true;
98  }
99 
100  if(calculate_drag_coef == true)
101  {
102  if(std::fpclassify(aero_drag_param_ptr->gas_const) == FP_ZERO ||
103  std::fpclassify(aero_drag_param_ptr->temp_free_stream) == FP_ZERO)
104  {
105  MessageHandler::fail(__FILE__,
106  __LINE__,
107  AerodynamicsMessages::runtime_error,
108  "Either the gas_const or temp_free_stream field(s) of "
109  "aero_drag_param_ptr was not initialized. "
110  "Please initialize both of these values.");
111  }
112 
113  /* --
114  dimensionless parameter equal to the square root of the ratio of the
115  vehicle velocity to the mean free-molecular thermal speed (not
116  velocity, not relative speed) */
117  double s = rel_vel_mag / sqrt(2.0 * aero_drag_param_ptr->gas_const * aero_drag_param_ptr->temp_free_stream);
118 
119  double s_2; // s^2 (see definition of s)
120  double exp_ssa2; // e^(-s_sa2)
121  double temp_ratio; // ratio of the temperature of the reflected molecules to that of the incident molecules
122  switch(coef_method)
123  {
124  case AeroDragEnum::Calc_coef:
125  {
126  /* Calculate variables used in coef calcs */
127  // This is an implementation of the coefficient calculations found
128  // in the documentation
129  double one_p_epsilon = 1 + epsilon; /* epsilon is the fraction of incident molecules that "bounce"; this
130  is 1+epsilon */
131  double one_m_epsilon = 1 - epsilon; /* 1-epsilon (see one_p_epsilon) */
132  s_2 = s * s;
133  double s_sinalpha = s * sin_alpha;
134  double s_sa2 = s_sinalpha * s_sinalpha;
135  exp_ssa2 = exp(-s_sa2);
136 
137  double erf_ssa = erf(s_sinalpha);
138 
139  // temperature of the reflected molecules
140  double local_temp_reflect = (one_m_epsilon)*temperature +
141  epsilon * aero_drag_param_ptr->temp_free_stream;
142  // ratio of the temperature of the reflected molecules to that of the incident molecules
143  temp_ratio = local_temp_reflect / aero_drag_param_ptr->temp_free_stream;
144 
145  drag_coef_norm = ((M_2_SQRTPI * one_p_epsilon) * s_sinalpha * exp_ssa2 +
146  one_m_epsilon * sqrt(temp_ratio) * (2.0 / M_2_SQRTPI) * s_sinalpha +
147  one_p_epsilon * (1.0 + 2.0 * s_sa2) * erf_ssa) /
148  (s_2);
149 
150  // if plate is full-on (alpha = 90 degrees), there is no tangential drag
151  if(!Numerical::compare_exact(sin_alpha, 1))
152  {
153  double cos_alpha = sqrt(1 - (sin_alpha * sin_alpha));
154  Vector3::scale(normal, sin_alpha, tangent);
155  Vector3::diff(rel_vel_struct_hat, tangent, tangent);
156  double cos_a_inv = 1 / cos_alpha;
157  Vector3::scale(cos_a_inv, tangent);
158 
159  drag_coef_tang = fabs((2.0 * one_m_epsilon / (2.0 / M_2_SQRTPI)) * s * cos_alpha *
160  (exp(-s_sinalpha * s_sinalpha) +
161  (2.0 / M_2_SQRTPI) * s_sinalpha * erf(s_sinalpha))) /
162  (s * s);
163  // Note that fabs was added to the equation because direction is
164  // accounted for later
165  }
166  break;
167  }
168  case AeroDragEnum::Specular:
169  {
170  // assumes all molecules "bounce" from the plate
171 
172  // Calculate the drag coef
173  // assumes angle of attack = 90 degrees for unknown reason, therefore
174  // s_sinalpha = s
175  // assumes epsilon = 1; so 1+epsilon =2; 1-epsilon = 0; 2*(1+epsilon)=4
176  s_2 = s * s;
177  exp_ssa2 = exp(-s_2);
178  drag_coef_spec = ((2.0 * M_2_SQRTPI) * s * exp_ssa2 + (2.0 + 4.0 * s_2)) / (s_2);
179  break;
180  }
181  case AeroDragEnum::Diffuse:
182  {
183  // Calculate the drag coef
184  // assumes angle of attack = 90 degrees for unknown reason, therefore
185  // s_sinalpha = s
186  // assumes epsilon = 0; so 1+epsilon = 1-epsilon = 1; 2*(1+epsilon) = 4
187  s_2 = s * s;
188  exp_ssa2 = exp(-s_2);
189 
190  /* Assume the temperature of reflected molecules */
191  temp_ratio = temperature / aero_drag_param_ptr->temp_free_stream;
192 
193  drag_coef_diff = ((M_2_SQRTPI)*s * exp_ssa2 + sqrt(temp_ratio) * (2.0 / M_2_SQRTPI) * s +
194  (1.0 + 2.0 * s_2)) /
195  (s * s);
196  break;
197  }
198  case AeroDragEnum::Mixed:
199  {
200  s_2 = s * s;
201  exp_ssa2 = exp(-s_2);
202  drag_coef_spec = ((2.0 * M_2_SQRTPI) * s * exp_ssa2 + (2.0 + 4.0 * s_2)) / (s_2);
203  temp_ratio = temperature / aero_drag_param_ptr->temp_free_stream;
204  drag_coef_diff = ((M_2_SQRTPI)*s * exp_ssa2 + sqrt(temp_ratio) * (2.0 / M_2_SQRTPI) * s +
205  (1.0 + 2.0 * s_2)) /
206  (s * s);
207  break;
208  }
209  default:
210  MessageHandler::fail(__FILE__,
211  __LINE__,
212  AerodynamicsMessages::runtime_error,
213  "The choice for calculating the coefficients of drag in the "
214  "flat plate model for aerodynamics was invalid. Please supply "
215  "a valid choice.\n");
216  }
217  }
218 
219  // Have calculated drag coeffs if desired, or using pre-defined values
220  // otherwise. Now calculate the forces.
221  // force_base is negative because the force is opposite in direction to all
222  // the vectors we know.
223  force_base = -aero_drag_param_ptr->dynamic_pressure * area;
224 
225  switch(coef_method)
226  {
227  case AeroDragEnum::Calc_coef:
228  force_n = force_base * drag_coef_norm;
229 
230  // if plate is full-on (alpha = 90 degrees)
231  if(Numerical::compare_exact(sin_alpha, 1))
232  {
233  Vector3::scale(normal, force_n, force);
234  force_t = 0.0;
235  }
236  else
237  {
238  force_t = force_base * drag_coef_tang;
239  Vector3::scale(tangent, force_t, vec_force_t);
240  Vector3::scale(normal, force_n, vec_force_n);
241  Vector3::sum(vec_force_t, vec_force_n, force);
242  }
243  break;
244 
245  case AeroDragEnum::Specular:
246  // For the Specular case the atmospheric particles "bounce" from the
247  // the plates after a collision creating a force that is normal to the
248  // area of impact. The specular drag force is proportional to the product
249  // of the plate are and the square of the relative velocity normal to
250  // the plate area.
251  // Velocity normal = V*sin(alpha), therefore,
252  // Vnormal^2 = V^2*sin^2(alpha) */
253 
254  force_n = force_base * drag_coef_spec * sin_alpha * sin_alpha;
255  // The force is applied in the normal direction
256  Vector3::scale(normal, force_n, force);
257  break;
258 
259  case AeroDragEnum::Diffuse:
260  // For the Diffuce case the atmospheric particles "stick" to the plates
261  // after a collision, creating a force that is parallel to the relative
262  // velocity vector of a body. The diffuse drag force is proportional to
263  // the product of the plate area projected in a direction normal to the
264  // relative velocity and the square of the relative velocity.
265  // Plate area projected in the direction normal to the relative velocity =
266  // area * sin(alpha)
267 
268  // force_t is not the force normal to the plate in this case; it
269  // is the force parallel to the velocity.
270  force_t = force_base * drag_coef_diff * sin_alpha;
271 
272  // The force is applied in the direction of the velocity
273  Vector3::scale(rel_vel_struct_hat, force_t, force);
274 
275  break;
276 
277  case AeroDragEnum::Mixed:
278 
279  // The calculations for the Mixed case have been updated from previous
280  // releases. The formulation is more intuitive, easier to read, and
281  // produces the exact same output as before.
282 
283  // Drag is a mixture of specular (along plate normal) and diffuse (along
284  // relative velocity vector) forces. These are summed to produce the
285  // total drag force.
286 
287  force_n = epsilon * force_base * drag_coef_spec * sin_alpha * sin_alpha;
288  force_t = (1.0 - epsilon) * force_base * drag_coef_diff * sin_alpha;
289 
290  // calculate the force on the plate
291  Vector3::scale(normal, force_n, vec_force_n);
292  Vector3::scale(rel_vel_struct_hat, force_t, vec_force_t);
293  Vector3::sum(vec_force_t, vec_force_n, force);
294 
295  break;
296 
297  default:
298  MessageHandler::fail(__FILE__,
299  __LINE__,
300  AerodynamicsMessages::runtime_error,
301  "The choice for calculating the aerodynamic drag in the "
302  "flat plate model for aerodynamics was invalid. Please supply "
303  "a valid choice.\n");
304  }
305 
306  double cpres[3];
307 
308  Vector3::diff(center_pressure, center_grav, cpres);
309  Vector3::initialize(torque);
310  Vector3::cross_incr(cpres, force, torque);
311  return;
312  // empty for now
313 }
