59  * PrecessionMars specific implementation of update_rotation,
60  * to calculate precession.
61  */
62 void PrecessionMars::update_rotation()
63 {
64  // Set up the relevant precession equation from pg. 39 of Konopliv 2006:
65  // Psi(t) = Psi_o + Psi_dot*time + Psi_nut
66 
67  // Access nutation of longitude term for use in equation
68  if(nutation == nullptr)
69  {
70  MessageHandler::fail(__FILE__,
71  __LINE__,
72  RNPMessages::setup_error,
73  "PrecessionMars is not currently pointing "
74  "to a nutation object.");
75  return;
76  }
77  double psi_nut = 0.0;
78  psi_nut = nutation->nutation_in_longitude;
79 
80  // Calculate the current precession angle
81  psi_precess = psi_at_j2000 + psi_dot * current_time + psi_nut;
82 
83  // Take trigs of the precession angle to build the precession matrix
84  double cos_psi = cos(psi_precess);
85  double sin_psi = sin(psi_precess);
86 
87  double precession_matrix[3][3];
88  // Now populate the precession matrix, as follows:
89 
90  precession_matrix[0][0] = cos_psi;
91  precession_matrix[0][1] = -sin_psi;
92  precession_matrix[0][2] = 0.0;
93 
94  precession_matrix[1][0] = sin_psi;
95  precession_matrix[1][1] = cos_psi;
96  precession_matrix[1][2] = 0.0;
97 
98  precession_matrix[2][0] = 0.0;
99  precession_matrix[2][1] = 0.0;
100  precession_matrix[2][2] = 1.0;
101 
102  // celestial frame = NJ_matrix * rot_z(-Psi) * mars frame
103  Matrix3x3::product(NJ_matrix, precession_matrix, rotation);
104 }

109 void PrecessionMars::compute_fixed_matrices()
110 {
111  // Take trigs of the angles to build their rotation matrices
112  double cos_N = cos(N);
113  double sin_N = sin(N);
114  double cos_J = cos(J);
115  double sin_J = sin(J);
116 
117  // Populate the N rotation matrix as follows:
118  // celestial frame = rot_z(-N) * mars frame
119  double N_matrix[3][3];
120  Matrix3x3::initialize(N_matrix);
121  N_matrix[0][0] = cos_N;
122  N_matrix[0][1] = -sin_N;
123  N_matrix[0][2] = 0.0;
124 
125  N_matrix[1][0] = sin_N;
126  N_matrix[1][1] = cos_N;
127  N_matrix[1][2] = 0.0;
128 
129  N_matrix[2][0] = 0.0;
130  N_matrix[2][1] = 0.0;
131  N_matrix[2][2] = 1.0;
132 
133  // Populate the J rotation matrix as follows:
134  // celestial frame = rot_x(-J) * mars frame
135  double J_matrix[3][3];
136  Matrix3x3::initialize(J_matrix);
137  J_matrix[0][0] = 1.0;
138  J_matrix[0][1] = 0.0;
139  J_matrix[0][2] = 0.0;
140 
141  J_matrix[1][0] = 0.0;
142  J_matrix[1][1] = cos_J;
143  J_matrix[1][2] = -sin_J;
144 
145  J_matrix[2][0] = 0.0;
146  J_matrix[2][1] = sin_J;
147  J_matrix[2][2] = cos_J;
148 
149  // Combine the matrices and store in the appropriate member data variable
150  Matrix3x3::product(N_matrix, J_matrix, NJ_matrix);
151 }
