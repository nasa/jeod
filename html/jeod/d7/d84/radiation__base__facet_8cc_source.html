48  * initializes the base surface
49  */
50 void RadiationBaseFacet::initialize()
51 {
52  // albedo and albedo_vis are synonymous at initialization. During runtime,
53  // albedo can take the value albedo_vis or albedo_IR, but legacy code had
54  // only one albedo (repesenting albedo_vis), called albedo.
55  bool setequ = false;
56  if(albedo < 0.0)
57  {
58  albedo = albedo_vis;
59  setequ = true;
60  }
61  if(albedo_vis < 0.0)
62  {
63  albedo_vis = albedo;
64  setequ = true;
65  }
66  // If albedo_IR not set, assume equal to albedo.
67  if(albedo_IR < 0.0)
68  {
69  albedo_IR = albedo;
70  }
71 
72  if(!setequ)
73  { // So both albedo and albedo_vis were set to legal values.
74  double epsilon = 1.0E-12;
75  if((albedo - albedo_vis) > epsilon || (albedo - albedo_vis) < -epsilon)
76  {
77  MessageHandler::fail(__FILE__,
78  __LINE__,
79  RadiationMessages::invalid_setup_error,
80  "\n"
81  "The values albedo and albedo_vis are synonymous at initialization.\n"
82  "However, they have been set to different values (%f and %f"
83  "respectively)\n. Unsure how to interpret these data.\n",
84  albedo,
85  albedo_vis);
86  }
87  }
88 }
