72 /**
73  * initializes the base surface
74  */
75 void
76 RadiationBaseFacet::initialize (
77  void)
78 {
79  // albedo and albedo_vis are synonymous at initialization. During runtime,
80  // albedo can take the value albedo_vis or albedo_IR, but legacy code had
81  // only one albedo (repesenting albedo_vis), called albedo.
82  bool setequ = false;
83  if (albedo < 0.0) {
84  albedo = albedo_vis;
85  setequ = true;
86  }
87  if (albedo_vis < 0.0) {
88  albedo_vis = albedo;
89  setequ = true;
90  }
91  // If albedo_IR not set, assume equal to albedo.
92  if (albedo_IR < 0.0) {
93  albedo_IR = albedo;
94  }
95 
96  if (!setequ) { // So both albedo and albedo_vis were set to legal values.
97  double epsilon = 1.0E-12;
98  if ((albedo - albedo_vis) > epsilon ||
99  (albedo - albedo_vis) < -epsilon) {
100  MessageHandler::fail (
101  __FILE__, __LINE__, RadiationMessages::invalid_setup_error, "\n"
102  "The values albedo and albedo_vis are synonymous at initialization.\n"
103  "However, they have been set to different values (%f and %f"
104  "respectively)\n. Unsure how to interpret these data.\n",
105  albedo, albedo_vis);
106  }
107  }
108  return;
109 }
