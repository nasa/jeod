182 /**
183  A SectionedInputStream is a std::istream that reads from a section in a
184  checkpoint file. This class will indicate EOF when the input pointer in the
185  checkpoint file file buffer goes beyond the end of the section.
186 
187  This is a barebones implementation. It does not provide buffering, it does
188  not support seek and tell, and it does not support putback or unget.
189 
190  @par Usage
191  A SectionedInputStream object is used in a preload_checkpoint or restart
192  job to read and then act on contents stored in a checkpoint file.
193  @code
194  return_type function_name (
195  SomeStructureType & stuff_to_restore)
196  {
197  std::string section_name;
198  double number;
199  char c_style_line[256];
200  std::string cpp_line;
201  char character;
202  int char_as_int;
203 
204  std::string section_name;
205  // Set to name of the checkpoint section
206 
207  // Construct a checkpoint input stream.
208  // Notes:
209  // - This object must go out of scope by the end of the job.
210  // - DO NOT make a copy of this object.
211  // - DO NOT save a pointer to this object in a permanent structure.
212  // - The code below assumes that function_name is called as a
213  // preload_checkpoint or a restart job.
214  SectionedInputStream reader (
215  JeodSimulationInterface::get_checkpoint_reader(section_name));
216 
217  // Activate the reader.
218  // Fail to do so and you'll get EOF on the first read.
219  reader.activate();
220 
221  // You can use the C++ operator >> to read various kinds of data ...
222  reader >> number;
223 
224  // ... even data structures if the structure has a deserializer.
225  reader >> stuff_to_restore;
226 
227  // Lines can be read with the getline member or std::getline global.
228  reader.getline (c_style_line, 255);
229  std::getline (reader, cpp_line);
230 
231  // Individual characters can be read in a variety of ways.
232  reader >> std::noskipws >> character;
233  reader.get (character);
234  char_as_int = reader.rdbuf()->sbumpc();
235 
236  // A bunch of numbers can be read using operator >>:
237  while (!! (reader >> number)) {
238  stuff_to_restore.add_number (number);
239  }
240 
241  // An alternative is to implicitly use operator void*:
242  while (reader >> number) {
243  stuff_to_restore.add_number (number);
244  }
245 
246  // The file can be scanned via getline, here using the bang-bang trick:
247  while (!! std::getline (reader, cpp_string)) {
248  process_line (cpp_string);
249  }
250 
251  // Same as the above, but implicitly using operator void*:
252  while (std::getline (reader, cpp_string)) {
253  process_line (cpp_string);
254  }
255 
256  // The file can be processed a character at a time.
257  // Once again, either the bang-bang trick or operator void* can be
258  // used to check for EOF.
259  while (!! std::get (reader, character)) {
260  stuff_to_restore.add_char (character);
261  }
262 
263  // Yet another alternative is to test for EOF using sbumpc:
264  while ((char_as_int = rdbuf->sbumpc()) != EOF) {
265  stuff_to_restore.add_char ((char)char_as_int);
266  }
267 
268  // Or use sgetc/sbumpc if the above grates too much:
269  while (reader.rdbuf->sgetc()) != EOF) {
270  stuff_to_restore.add_char ((char)reader.rdbuf()->sbumpc());
271  }
272  }
273  @endcode
274 
275  @par Diagnosing problems
276  - Nothing is being read.
277  This can be caused by several problems, described below.
278  - Is the JEOD checkpoint file open for input?\n
279  Checkpoint file sections can only be read from a JEOD checkpoint file
280  that is open for input. In a Trick context, the checkpoint file is only
281  open for preload_checkpoint and restart jobs. Reading from a checkpoint
282  file in other contexts won't work.
283  - Are multiple threads trying to read from the same checkpoint file?\n
284  Don't do that. This package is not thread-safe.
285  - Have you cached some another active checkpoint reader somewhere?\n
286  Don't do that, either. Only one reader can be active at a time.
287  - Is the checkpoint file section in the checkpoint file?\n
288  You will get a diagnostic message if the section doesn't exist.
289  - Is the checkpoint reader viable?\n
290  The above problems will result in a non-viable checkpoint reader.
291  The method is_activatable() can be called prior to calling activate()
292  to check whether the stream is viable.
293  - Did you call reader.activate()?\n
294  Whether compilers make two different objects in the construction of the
295  SectionedInputStream or just one object depends on the compiler and on the
296  optimization level. Making the package robustly handle the complexities of
297  RVO (return value optimization) was too much for the author of the package.
298  The call to reader.activate() is essential.
299  - Did the call to reader.activate() work?\n
300  The method activate() returns true or false to indicate success or failure.
301  While the above code did not check status, doing so is a good idea.
302  - Did you call reader.deactivate()?\n
303  Don't do that until you are done reading. The call to deactivate() is
304  irreversible.
305  - Did you mix scanned input with line reading?\n
306  As with any other stream, operator >> will mark the stream as failed
307  if the operator fails to parse.
308  */
