168 /**
169  A SectionedInputStream is a std::istream that reads from a section in a
170  checkpoint file. This class will indicate EOF when the input pointer in the
171  checkpoint file file buffer goes beyond the end of the section.
172 
173  This is a barebones implementation. It does not provide buffering, it does
174  not support seek and tell, and it does not support putback or unget.
175 
176  @par Usage
177  A SectionedInputStream object is used in a preload_checkpoint or restart
178  job to read and then act on contents stored in a checkpoint file.
179  @code
180  return_type function_name (
181  SomeStructureType & stuff_to_restore)
182  {
183  std::string section_name;
184  double number;
185  char c_style_line[256];
186  std::string cpp_line;
187  char character;
188  int char_as_int;
189 
190  std::string section_name;
191  // Set to name of the checkpoint section
192 
193  // Construct a checkpoint input stream.
194  // Notes:
195  // - This object must go out of scope by the end of the job.
196  // - DO NOT make a copy of this object.
197  // - DO NOT save a pointer to this object in a permanent structure.
198  // - The code below assumes that function_name is called as a
199  // preload_checkpoint or a restart job.
200  SectionedInputStream reader (
201  JeodSimulationInterface::get_checkpoint_reader(section_name));
202 
203  // Activate the reader.
204  // Fail to do so and you'll get EOF on the first read.
205  reader.activate();
206 
207  // You can use the C++ operator >> to read various kinds of data ...
208  reader >> number;
209 
210  // ... even data structures if the structure has a deserializer.
211  reader >> stuff_to_restore;
212 
213  // Lines can be read with the getline member or std::getline global.
214  reader.getline (c_style_line, 255);
215  std::getline (reader, cpp_line);
216 
217  // Individual characters can be read in a variety of ways.
218  reader >> std::noskipws >> character;
219  reader.get (character);
220  char_as_int = reader.rdbuf()->sbumpc();
221 
222  // A bunch of numbers can be read using operator >>:
223  while (!! (reader >> number)) {
224  stuff_to_restore.add_number (number);
225  }
226 
227  // An alternative is to implicitly use operator void*:
228  while (reader >> number) {
229  stuff_to_restore.add_number (number);
230  }
231 
232  // The file can be scanned via getline, here using the bang-bang trick:
233  while (!! std::getline (reader, cpp_string)) {
234  process_line (cpp_string);
235  }
236 
237  // Same as the above, but implicitly using operator void*:
238  while (std::getline (reader, cpp_string)) {
239  process_line (cpp_string);
240  }
241 
242  // The file can be processed a character at a time.
243  // Once again, either the bang-bang trick or operator void* can be
244  // used to check for EOF.
245  while (!! std::get (reader, character)) {
246  stuff_to_restore.add_char (character);
247  }
248 
249  // Yet another alternative is to test for EOF using sbumpc:
250  while ((char_as_int = rdbuf->sbumpc()) != EOF) {
251  stuff_to_restore.add_char ((char)char_as_int);
252  }
253 
254  // Or use sgetc/sbumpc if the above grates too much:
255  while (reader.rdbuf->sgetc()) != EOF) {
256  stuff_to_restore.add_char ((char)reader.rdbuf()->sbumpc());
257  }
258  }
259  @endcode
260 
261  @par Diagnosing problems
262  - Nothing is being read.
263  This can be caused by several problems, described below.
264  - Is the JEOD checkpoint file open for input?\n
265  Checkpoint file sections can only be read from a JEOD checkpoint file
266  that is open for input. In a Trick context, the checkpoint file is only
267  open for preload_checkpoint and restart jobs. Reading from a checkpoint
268  file in other contexts won't work.
269  - Are multiple threads trying to read from the same checkpoint file?\n
270  Don't do that. This package is not thread-safe.
271  - Have you cached some another active checkpoint reader somewhere?\n
272  Don't do that, either. Only one reader can be active at a time.
273  - Is the checkpoint file section in the checkpoint file?\n
274  You will get a diagnostic message if the section doesn't exist.
275  - Is the checkpoint reader viable?\n
276  The above problems will result in a non-viable checkpoint reader.
277  The method is_activatable() can be called prior to calling activate()
278  to check whether the stream is viable.
279  - Did you call reader.activate()?\n
280  Whether compilers make two different objects in the construction of the
281  SectionedInputStream or just one object depends on the compiler and on the
282  optimization level. Making the package robustly handle the complexities of
283  RVO (return value optimization) was too much for the author of the package.
284  The call to reader.activate() is essential.
285  - Did the call to reader.activate() work?\n
286  The method activate() returns true or false to indicate success or failure.
287  While the above code did not check status, doing so is a good idea.
288  - Did you call reader.deactivate()?\n
289  Don't do that until you are done reading. The call to deactivate() is
290  irreversible.
291  - Did you mix scanned input with line reading?\n
292  As with any other stream, operator >> will mark the stream as failed
293  if the operator fails to parse.
294  */
