38 /**
39  * Calculate the aggregate cm for a composite MassBody.
40  *
41  * \par Assumptions and Limitations
42  * - Rigid bodies
43  * - Must calculate from bottom to top of tree for meaningful results
44  */
45 void MassBody::calc_composite_cm()
46 {
47  double mass; // kg Total mass
48  double mass_times_com[3]; // M*kg Total mass * CoM
49 
50  // For composite bodies, the mass is the sum of the component masses and
51  // the center of mass is given by M*CM = sum (m*cm), summed over the
52  // component masses. The component masses comprise the core mass of this body
53  // alone plus the composite masses of each attached child body.
54 
55  // Initialize the mass and mass*center of mass to the core values.
56  mass = core_properties.mass;
57  Vector3::scale(core_properties.position, core_properties.mass, mass_times_com);
58 
59  // Add each child body's contribution to the accumulators.
60  for(auto * child_link : TreeLinksChildrenRange<MassBodyLinks>(links))
61  {
62  const MassBody * child = &(child_link->container());
63 
64  // Accumulate the mass and mass * center of mass for this child body.
65  mass += child->composite_properties.mass;
66  Vector3::scale_incr(child->composite_wrt_pstr.position, child->composite_properties.mass, mass_times_com);
67  }
68 
69  // Compute the composite center of mass.
70  if(mass > 0.0)
71  {
72  composite_properties.inverse_mass = 1.0 / mass;
73  Vector3::scale(mass_times_com, composite_properties.inverse_mass, composite_properties.position);
74  }
75  else
76  {
77  composite_properties.inverse_mass = 0.0;
78  Vector3::initialize(composite_properties.position);
79  }
80 
81  // Store the total mass back into composite properties
82  composite_properties.mass = mass;
83 }
