260 void LsodeControlDataInterface::allocate_arrays()
261 {
262  unsigned int abs_vec_size = abs_tolerance_error_control_vec.size();
263  unsigned int rel_vec_size = rel_tolerance_error_control_vec.size();
264 
265  if(num_odes < rel_vec_size)
266  {
267  rel_vec_size = num_odes;
268  er7_utils::MessageHandler::error(__FILE__,
269  __LINE__,
270  er7_utils::IntegrationMessages::invalid_request,
271  "The sim has been configured with %d equations, but the relative \n"
272  "error-control vector has been populated with %d entries.\n"
273  "Only the first %d of those values will be copied into the usable \n"
274  "error-control array.\n",
275  num_odes,
276  rel_vec_size,
277  num_odes);
278  }
279  if(num_odes < abs_vec_size)
280  {
281  abs_vec_size = num_odes;
282  er7_utils::MessageHandler::error(__FILE__,
283  __LINE__,
284  er7_utils::IntegrationMessages::invalid_request,
285  "The sim has been configured with %d equations, but the absolute \n"
286  "error-control vector has been populated with %d entries.\n"
287  "Only the first %d of those values will be copied into the usable \n"
288  "error-control array.\n",
289  num_odes,
290  abs_vec_size,
291  num_odes);
292  }
293  abs_tolerance_error_control = er7_utils::alloc::allocate_array<double>(num_odes);
294  rel_tolerance_error_control = er7_utils::alloc::allocate_array<double>(num_odes);
295 
296  for(unsigned int ii = 0; ii < abs_vec_size; ii++)
297  {
298  abs_tolerance_error_control[ii] = abs_tolerance_error_control_vec[ii];
299  }
300  for(unsigned int ii = 0; ii < rel_vec_size; ii++)
301  {
302  rel_tolerance_error_control[ii] = rel_tolerance_error_control_vec[ii];
303  }
304  error_control_vector_copied_over = true;
305  num_odes_at_alloc = num_odes; // Counts for the data arrays also since
306  // they are done at the same time.
307 }

325 void LsodeControlDataInterface::set_rel_tol(int index, double value)
326 {
327  if(index < 0)
328  {
329  er7_utils::MessageHandler::error(__FILE__,
330  __LINE__,
331  er7_utils::IntegrationMessages::invalid_request,
332  "Error in configuration of Lsode integrator:\n"
333  "Attempt to set rel_tolerance_error_control element with negative index.\n"
334  "Index = %d. Attempt failed. \n",
335  index);
336  return;
337  }
338 
339  auto uint_index = static_cast<unsigned int>(index);
340 
341  if(error_control_vector_copied_over)
342  {
343  if(uint_index < num_odes_at_alloc)
344  {
345  rel_tolerance_error_control[uint_index] = value;
346  }
347  else
348  {
349  er7_utils::MessageHandler::error(__FILE__,
350  __LINE__,
351  er7_utils::IntegrationMessages::invalid_request,
352  "Error in configuration of Lsode integrator:\n"
353  "Attempt to set rel_tolerance_error_control with index that exceeds\n"
354  "the number of ODEs at allocation. Index = %u. num_odes = %u.\n"
355  "Attempt failed.\n",
356  index,
357  num_odes_at_alloc);
358  }
359  }
360  else
361  {
362  unsigned int size = rel_tolerance_error_control_vec.size();
363  if(uint_index < size)
364  {
365  rel_tolerance_error_control_vec[uint_index] = value;
366  }
367  else if(uint_index == size)
368  {
369  rel_tolerance_error_control_vec.push_back(value);
370  }
371  else
372  {
373  for(unsigned int ii = size; ii < uint_index; ii++)
374  {
375  rel_tolerance_error_control_vec.push_back(-1.0);
376  }
377  rel_tolerance_error_control_vec.push_back(value);
378  }
379  }
