92 template<typename CollectType, typename value_type> void collect_insert(CollectType & collect_in, value_type & elem)
93 {
94  bool entry_found = false;
95  for(auto it = collect_in.begin(); it != collect_in.end(); ++it)
96  {
97  value_type collect_ptr = *it;
98  if(*collect_ptr == *elem)
99  {
100  entry_found = true;
101  break;
102  }
103  }
104  if(!entry_found)
105  {
106  collect_in.insert(collect_in.end(), elem);
107  }
108  else
109  {
110  if(JEOD_IS_ALLOCATED(elem))
111  {
112  jeod::JeodMemoryManager::destroy_memory(jeod::jeod_alloc_get_allocated_pointer(elem),
113  false,
114  __FILE__,
115  __LINE__);
116  MessageHandler::inform(__FILE__,
117  __LINE__,
118  "collect.insert",
119  "pointer to vector already exists in STL. Deallocating Collect container");
120  }
121  else
122  {
123  MessageHandler::inform(__FILE__, __LINE__, "collect.insert", "pointer to vector already exists in STL");
124  }
125  }
126 }
