186  /**

189  * needed just after a reset. The behavior should revert to nominal after

191  */

195  trick_integrator.first_step_deriv = true;

197

201  * the most recent call to reset_first_step_derivs_flag.
202  */
203  void restore_first_step_derivs_flag() override
204  {
205  trick_integrator.first_step_deriv = default_first_step_deriv;
206  }
