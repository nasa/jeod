40 /**
41  * Re-attach a child MassBody to a parent MassBody.
42  *
43  * \par Assumptions and Limitations
44  * - Rigid Bodies
45  * - Tree attachment structure
46  * - Child is known to be a root or atomic body
47  * - Re-establishing an attachment that previously existed
48  * \param[in] offset Desired offset from parent struct frame to attached child's struct frame, expressed in parent
49  * struct frame\n Units: M \param[in] T_pstr_cstr Desired transformation matrix from parent struct frame to attached
50  * child's struct frame
51  */
52 bool MassBody::reattach(double offset[3], double T_pstr_cstr[3][3])
53 {
54  MassBody * parent = links.parent();
55 
56  /* Check to see if the body is attached to anything. */
57  if(links.is_root())
58  {
59  MessageHandler::fail(__FILE__,
60  __LINE__,
61  MassBodyMessages::invalid_node,
62  "Body '%s' is a root body and cannot be reattached.",
63  name.c_str());
64 
65  // Not reached
66  return false;
67  }
68 
69  // Update the attachment attributes.
70  structure_point.update_orientation(T_pstr_cstr);
71  structure_point.update_point(offset);
72 
73  // Construct the transformation from the parent body's structural frame
74  // to child body's composite body frame:
75  // T_pstr_cbdy = T_cstr_cbdy * T_pstr_cstr
76  composite_properties.Q_parent_this.multiply(structure_point.Q_parent_this, composite_wrt_pstr.Q_parent_this);
77  composite_wrt_pstr.compute_transformation();
78 
79  // Compute the displace from the parent body's structural origin to the
80  // child body's composite CoM in the parent body's structural frame:
81  // x_pstr_cbdy = x_pstr_cstr + (T_cstr_pstr^T)*x_cstr_cbdy
82  Vector3::transform_transpose(structure_point.T_parent_this,
83  composite_properties.position,
84  composite_wrt_pstr.position);
85  Vector3::incr(structure_point.position, composite_wrt_pstr.position);
86 
87  // Construct the transformation from the parent body's body frame
88  // to child body's composite body frame.
89  composite_wrt_pstr.Q_parent_this.multiply_conjugate(parent->composite_properties.Q_parent_this,
90  composite_wrt_pbdy.Q_parent_this);
91  composite_wrt_pbdy.compute_transformation();
92 
93  // Update the parent's mass properties.
94  // Note that this sets composite_wrt_pbdy.position.
95  parent->set_update_flag();
96  parent->get_root_body_internal()->update_mass_properties();
97 
98  return true;
99 }
