52 // suffix: Find the part of the name that follows the prefix.
53 std::string NamedItem::suffix(const std::string & prefix, const std::string & name)
54 {
55  std::size_t prefix_len = prefix.length();
56 
57  // Return the full name if
58  // - the name is NULL (sanity check) or
59  // - the name does not start with "prefix."
60  if(name.empty() || (name.compare(0, prefix_len, prefix) != 0) || (name[prefix_len] != '.'))
61  {
62  return name;
63  }
64 
65  // Name starts with "prefix.": return the suffix.
66  else
67  {
68  return name.substr(prefix_len + 1);
69  }
