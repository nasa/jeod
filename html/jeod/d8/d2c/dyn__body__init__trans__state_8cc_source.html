75 /**
76  * Indicate whether this initializer is ready to be applied.
77  * The full state of the reference reference frame must be known
78  * to compute the position and velocity of the subject reference frame.
79  * @return Is initializer ready?
80  */
81 bool DynBodyInitTransState::is_ready()
82 {
83  RefFrameItems required_items;
84  BodyRefFrame * body_frame;
85 
86  body_frame = dynamic_cast<BodyRefFrame *>(reference_ref_frame);
87 
88  // The initializer is not ready if internal dependencies are not satisfied.
89  if(!DynBodyInit::is_ready())
90  {
91  return false;
92  }
93 
94  // Initializing from a non-body, non-ephemeris frame is potentially
95  // problematic. The initialization will proceed, but ...
96  if((body_frame == nullptr) && (dynamic_cast<EphemerisRefFrame *>(reference_ref_frame) == nullptr))
97  {
98  MessageHandler::error(__FILE__,
99  __LINE__,
100  BodyActionMessages::invalid_object,
101  "%s is suspect:\n"
102  "Reference frame '%s' is of a unknown derived type.",
103  action_identifier.c_str(),
104  reference_ref_frame->get_name().c_str());
105  }
106 
107  // Determine the items needed from the reference reference frame.
108  // Position: x_A:C = x_A:B + T_B:A*x_B:C
109  // Velocity: v_A:C = v_A:B + T_B:A*(v_B:C + w_A:B x X_B:C)
110  // Position needs reference (A->B, B->A) position, attitude
111  // Velocity needs reference velocity, attitude, rate
112  required_items.set(initializes_what());
113  required_items.add(RefFrameItems::Att);
114  if(required_items.contains(RefFrameItems::Vel))
115  {
116  required_items.add(RefFrameItems::Rate);
117  }
118 
119  // The initializer is ready if the either the reference reference frame
120  // is not a body frame or if reference body frame contains the
121  // requisite required_items.
122  return (body_frame == nullptr) || body_frame->initialized_items.contains(required_items.get());
123 }
