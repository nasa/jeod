69 /**
70  * Provides a view into an object that can conceptually be indexed by a single
71  * integer (currently unsigned). For example, part of a one dimensional array
72  * can be encapsulated as a view, as can part of a specific row or column
73  * of a two dimensional array.
74  *
75  * This class template is not instantiable. An instantiable derived class must
76  * at a minimum implement the index operator.
77  *
78  * @warning Destructing the object that underlies a view invalidates the view.
79  * Changing the shape or size of the underlying object may invalidate the view.
80  * Referencing into an invalidated view is undefined behavior.
81  *
82  * @todo Add python-like referencing, where some_view[-1] references the last
83  * value in the view, some_view[-2] references the penultimate value, etc.
84  *
85  * @todo Add an unsafe copy constructor in the case that the source does
86  * not provide a size() member.
87  *
88  * @todo Consider replacing use of unsigned with std::size_t.
89  *
90  * @tparam ElemType The type of the elements held by the underlying object.
91  * This is assumed to be a primitive type.
92  */
93 template<typename ElemType, typename RefType> class VectorView
94 {
95 public:
96  /**
97  * Non-default constructor.
98  * @param n_elems_in The number of elements in the view.
99  */
100  VectorView(unsigned n_elems_in)
101  : n_elems(n_elems_in)
102  {
103  }
104 
105  /**
106  * Destructor. There's nothing to destruct here.
107  */
108  virtual ~VectorView() = default;
109 
110  /**
111  * Non-const index operator.
112  * @param i_elem Element to be indexed.
113  * @return Reference to the indexed value.
114  */
115  virtual RefType operator[](unsigned i_elem) = 0;
116 
117  /**
118  * Copy assignment from an object that implements size().
119  * The default implementation does an item-for-item copy,
120  * which might be suboptimal (e.g., memcpy might be faster).
121  * @param src The view whose data are to be copied into this.
122  * @throw std::length_error if this view and the source view are
123  * incompatibly sized.
124  */
125  template<typename SourceType, typename T = VectorView>
126  typename std::enable_if<!std::is_reference<RefType>::value, T &>::type operator=(const SourceType & src)
127  {
128  if(src.size() != n_elems)
129  {
130  JEOD_THROW(std::length_error("Incompatible sizes"));
131  }
132 
133  for(unsigned ii = 0; ii < n_elems; ++ii)
134  {
135  (*this)[ii] = src[ii];
136  }
137 
138  return *this;
139  }
140 
141  /**
142  * Copy assignment from a primitive array.
143  * The default implementation does an item-for-item copy,
144  * which might be suboptimal (e.g., memcpy might be faster).
145  * @param src The view whose data are to be copied into this.
146  * @throw std::length_error if this view and the source view are
147  * incompatibly sized.
148  */
149  template<std::size_t N, typename T = VectorView>
150  typename std::enable_if<!std::is_reference<RefType>::value, T &>::type operator=(const ElemType src[N])
151  {
152  if(N != n_elems)
153  {
154  JEOD_THROW(std::length_error("Incompatible sizes"));
155  }
156 
157  for(unsigned ii = 0; ii < n_elems; ++ii)
158  {
159  (*this)[ii] = src[ii];
160  }
161 
162  return *this;
163  }
164 
165  /**
166  * Const index operator.
167  * @param i_elem Element to be indexed.
168  * @return The indexed value.
169  */
170  virtual ElemType operator[](unsigned i_elem) const
171  {
172  return const_cast<VectorView &>(*this)[i_elem];
173  }
174 
175  /**
176  * Safe (bounds-checked) index operator.
177  * @param i_elem Element to be indexed.
178  * @return Reference to the indexed value.
179  * @throw std::out_of_range if index is outside the range of the view.
180  */
181  virtual RefType at(unsigned i_elem)
182  {
183  if(i_elem >= n_elems)
184  {
185  JEOD_THROW(std::out_of_range("Illegal element access"));
186  }
187 
188  return (*this)[i_elem];
189  }
190 
191  /**
192  * Const safe (bounds-checked) index operator.
193  * @param i_elem Element to be indexed.
194  * @return The indexed value.
195  * @throw std::out_of_range if index is outside the range of the view.
196  */
197  virtual ElemType at(unsigned i_elem) const
198  {
199  return const_cast<VectorView &>(*this).at(i_elem);
200  }
201 
202  /**
203  * Get the number of data elements encapsulated by this view.
204  * @return View size.
205  */
206  std::size_t size() const
207  {
208  return n_elems;
209  }

211  /**
212  * Create a view into this view.
213  * @param range The begin and end elements in the to-be-created view.

217  */
218  SubVectorView<VectorView<ElemType, RefType>, ElemType, RefType, RefType> make_view(

222  * Create a view into this view.
