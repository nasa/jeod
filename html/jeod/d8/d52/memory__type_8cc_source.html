93 /**
94  * Get the descriptor for the base (non-pointer) of some pointer type.
95  * @note
96  * Assumes GNU c++ name mangling, where 'const' is always preceded by a space.
97  */
98 const JeodMemoryTypeDescriptor * JeodMemoryTypeDescriptor::base_type( // Return: -- Base type name
99  const std::string & demangled_name) // In: -- Type name
100 {
101  std::string base_name(demangled_name);
102  std::size_t len = base_name.length();
103  std::size_t start_pos;
104  std::size_t pos;
105 
106  // Skip past the template instantiation part of the type name.
107  start_pos = base_name.find_last_of('>');
108  if(start_pos == std::string::npos)
109  {
110  start_pos = 0;
111  }
112  else
113  {
114  ++start_pos;
115  }
116 
117  // Get rid of consts in the type name.
118  for(pos = base_name.find(" const", start_pos); pos < len; pos = base_name.find(" const", pos))
119  {
120  // Delete the " const" if the next character is
121  // the null character, a space, an ampersand, or an asterisk.
122  // Otherwise, false alarm (e.g. constable); skip past it.
123  switch(base_name[pos + 6])
124  {
125  case '\0':
126  case ' ':
127  case '&':
128  case '*':
129  base_name.erase(pos, 6);
130  len -= 6;
131  break;
132 
133  default:
134  pos += 6;
135  break;
136  }
137  }
138 
139  // Get rid of asterisks in the type name.
140  pos = base_name.find('*', start_pos);
141  if(pos != std::string::npos)
142  {
143  base_name.erase(pos);
144  len = base_name.length();
145  }
146 
147  // Get rid of trailing blanks, if present.
148  if(base_name[len - 1] == ' ')
149  {
150  pos = base_name.find_last_not_of(' ');
151  base_name.erase(pos + 1);
152  }
153 
154  // The base type for void* is void*.
155  if(base_name.compare("void") == 0)
156  {
157  base_name += "*";
158  }
159 
160  // Get the descriptor for this mangled-up demangled name.
161  return JeodMemoryManager::get_type_descriptor(JeodMemoryManager::Demangled_type_name, base_name);
162 }
