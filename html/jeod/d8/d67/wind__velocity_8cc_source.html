53 void WindVelocity::update_wind(double inertial_pos[3], double altitude, double wind_inertial[3])
54 {
55  if(!active)
56  {
57  return;
58  }
59 
60  if(omega_scale_table == nullptr || inertial_pos == nullptr || wind_inertial == nullptr)
61  {
62  MessageHandler::error(__FILE__,
63  __LINE__,
64  AtmosphereMessages::framework_error,
65  "One of the required pointers is NULL.\n"
66  "Wind cannot be computed.\n"
67  "Deactivating model to prevent this message repeating.\n");
68  active = false;
69  return;
70  }
71 
72  /* Value by which omega is scaled depending on alt. */
73  double omega_scale = 0;
74 
75  if(num_layers == 1)
76  {
77  omega_scale = omega_scale_table[0].scale_factor;
78  }
79  else
80  {
81  // Check if increasing, decreasing, or invalid the first time this is called
82  if(first_pass)
83  {
84  bool hasIncreased = false;
85  bool hasDecreased = false;
86  for(unsigned int i = 0; i < num_layers - 1; i++)
87  {
88  if(omega_scale_table[i + 1].altitude > omega_scale_table[i].altitude)
89  {
90  hasIncreased = true;
91  if(hasDecreased)
92  {
93  break;
94  }
95  }
96  else
97  {
98  hasDecreased = true;
99  if(hasIncreased)
100  {
101  break;
102  }
103  }
104  }
105 
106  if(hasIncreased && hasDecreased)
107  {
108  MessageHandler::error(
109  __FILE__,
110  __LINE__,
111  AtmosphereMessages::framework_error,
112  "The provided table has neither strictly increasing nor strictly decreasing altitude.\n"
113  "Wind cannot be computed.\n"
114  "Deactivating model to prevent this message repeating.\n");
115  active = false;
116  }
117  else if(hasIncreased)
118  {
119  increasing_altitude = true;
120  }
121  else if(hasDecreased)
122  {
123  increasing_altitude = false;
124  }
125  else
126  {
127  MessageHandler::error(__FILE__,
128  __LINE__,
129  AtmosphereMessages::framework_error,
130  "The provided table somehow has no increases or decreases in altitude.\n"
131  "Wind cannot be computed.\n"
132  "Deactivating model to prevent this message repeating.\n");
133  active = false;
134  }
135 
136  first_pass = false;
137  }
138 
139  // Check if current altitude is outside the bounds of the table and assign the top/bottom value as necessary
140  if((increasing_altitude && altitude < omega_scale_table[0].altitude) ||
141  (!increasing_altitude && altitude > omega_scale_table[0].altitude))
142  {
143  omega_scale = omega_scale_table[0].scale_factor;
144  array_index = 0;
145  }
146  else if((increasing_altitude && altitude > omega_scale_table[num_layers - 1].altitude) ||
147  (!increasing_altitude && altitude < omega_scale_table[num_layers - 1].altitude))
148  {
149  omega_scale = omega_scale_table[num_layers - 1].scale_factor;
150  array_index = num_layers - 2; // Note that the array_index is the lower of the two indices used for
151  // interpolation. It can never be greater than num_layers-2.
152  }
153  // If the current altitude lies within the bounds of the table, determine the correct indices to interpolate
154  // between Start with most recent index to save time. It's reasonable to assume it's close to where it was
155  // before.
156  else
157  {
158  if(increasing_altitude)
159  {
160  if(altitude < omega_scale_table[array_index].altitude)
161  {
162  while(altitude < omega_scale_table[array_index].altitude)
163  {
164  array_index--;
165  }
166  }
167  else
168  {
169  while(altitude > omega_scale_table[array_index + 1].altitude)
170  {
171  array_index++;
172  }
173  }
174  }
175  else
176  {
177  if(altitude < omega_scale_table[array_index + 1].altitude)
178  {
179  while(altitude < omega_scale_table[array_index + 1].altitude)
180  {
181  array_index++;
182  }
183  }
184  else
185  {
186  while(altitude > omega_scale_table[array_index].altitude)
187  {
188  array_index--;
189  }
190  }
191  }
192 
193  // Interpolate between the two indices
194  double cell_fraction = (altitude - omega_scale_table[array_index].altitude) /
195  (omega_scale_table[array_index + 1].altitude -
196  omega_scale_table[array_index].altitude);
197  omega_scale = omega_scale_table[array_index].scale_factor +
198  cell_fraction * (omega_scale_table[array_index + 1].scale_factor -
199  omega_scale_table[array_index].scale_factor);
200  }
201  }
202 
203  // Scale omega by the proper scaling factor.
204  double omega_scaled = omega * omega_scale;
205 
206  // Cross product of the omega vector and the target position vector
207  wind_inertial[0] = -omega_scaled * inertial_pos[1];
208  wind_inertial[1] = omega_scaled * inertial_pos[0];
209  wind_inertial[2] = 0.0;
210 }
