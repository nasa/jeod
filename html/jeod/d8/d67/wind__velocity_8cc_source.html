75 void
76 WindVelocity::update_wind (
77  double inertial_pos[3],
78  double altitude,
79  double wind_inertial[3])
80 {
81  if (!active) {
82  return;
83  }
84 
85  if (omega_scale_table == nullptr ||
86  inertial_pos == nullptr ||
87  wind_inertial == nullptr) {
88  MessageHandler::error(
89  __FILE__,__LINE__, AtmosphereMessages::framework_error,
90  "One of the required pointers is NULL.\n"
91  "Wind cannot be computed.\n"
92  "Deactivating model to prevent this message repeating.\n");
93  active = false;
94  return;
95  }
96 
97  /* Value by which omega is scaled depending on alt. */
98  double omega_scale = 0;
99 
100  if (num_layers == 1) {
101  omega_scale = omega_scale_table[0].scale_factor;
102  }
103  else {
104  //Check if increasing, decreasing, or invalid the first time this is called
105  if (first_pass) {
106  bool hasIncreased = false;
107  bool hasDecreased = false;
108  for (unsigned int i=0; i<num_layers-1; i++) {
109  if (omega_scale_table[i+1].altitude > omega_scale_table[i].altitude) {
110  hasIncreased = true;
111  if(hasDecreased) break;
112  }
113  else {
114  hasDecreased = true;
115  if (hasIncreased) break;
116  }
117  }
118 
119  if (hasIncreased && hasDecreased) {
120  MessageHandler::error(
121  __FILE__,__LINE__, AtmosphereMessages::framework_error,
122  "The provided table has neither strictly increasing nor strictly decreasing altitude.\n"
123  "Wind cannot be computed.\n"
124  "Deactivating model to prevent this message repeating.\n");
125  active = false;
126  }
127  else if (hasIncreased) {
128  increasing_altitude = true;
129  }
130  else if (hasDecreased) {
131  increasing_altitude = false;
132  }
133  else {
134  MessageHandler::error(
135  __FILE__,__LINE__, AtmosphereMessages::framework_error,
136  "The provided table somehow has no increases or decreases in altitude.\n"
137  "Wind cannot be computed.\n"
138  "Deactivating model to prevent this message repeating.\n");
139  active = false;
140  }
141 
142  first_pass = false;
143  }
144 
145  //Check if current altitude is outside the bounds of the table and assign the top/bottom value as necessary
146  if ((increasing_altitude && altitude < omega_scale_table[0].altitude) || (!increasing_altitude && altitude > omega_scale_table[0].altitude)) {
147  omega_scale = omega_scale_table[0].scale_factor;
148  array_index = 0;
149  }
150  else if ((increasing_altitude && altitude > omega_scale_table[num_layers - 1].altitude) || (!increasing_altitude && altitude < omega_scale_table[num_layers - 1].altitude)) {
151  omega_scale = omega_scale_table[num_layers - 1].scale_factor;
152  array_index = num_layers - 2; //Note that the array_index is the lower of the two indices used for interpolation. It can never be greater than num_layers-2.
153  }
154  //If the current altitude lies within the bounds of the table, determine the correct indices to interpolate between
155  //Start with most recent index to save time. It's reasonable to assume it's close to where it was before.
156  else {
157  if (increasing_altitude) {
158  if (altitude < omega_scale_table[array_index].altitude) {
159  while (altitude < omega_scale_table[array_index].altitude) {
160  array_index--;
161  }
162  }
163  else {
164  while (altitude > omega_scale_table[array_index + 1].altitude) {
165  array_index++;
166  }
167  }
168  }
169  else {
170  if (altitude < omega_scale_table[array_index + 1].altitude) {
171  while (altitude < omega_scale_table[array_index + 1].altitude) {
172  array_index++;
173  }
174  }
175  else {
176  while (altitude > omega_scale_table[array_index].altitude) {
177  array_index--;
178  }
179  }
180  }
181 
182  //Interpolate between the two indices
183  double cell_fraction = (altitude - omega_scale_table[array_index].altitude) / (omega_scale_table[array_index + 1].altitude - omega_scale_table[array_index].altitude);
184  omega_scale = omega_scale_table[array_index].scale_factor + cell_fraction * (omega_scale_table[array_index + 1].scale_factor - omega_scale_table[array_index].scale_factor);
185  }
186  }
187 
188  // Scale omega by the proper scaling factor.
189  double omega_scaled = omega * omega_scale;
190 
191  // Cross product of the omega vector and the target position vector
192  wind_inertial[0] = -omega_scaled * inertial_pos[1];
193  wind_inertial[1] = omega_scaled * inertial_pos[0];
194  wind_inertial[2] = 0.0;
195 }
