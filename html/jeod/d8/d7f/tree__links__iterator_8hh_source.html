70 /**
71  * @file
72  * The JEOD 4.0 version of the tree links iterators is motivated by the c++11
73  * range-based for, which requires a range expression from which a begin
74  * iterator and an end sentinel can be formed. One way the compiler can form the
75  * begin iterator and end sentinel is to have the range expression be an object
76  * that implements begin() and end() member functions. The loops that use
77  * the JEOD 4.0 tree links iterators are of the form
78  *
79  * for (auto element : TreeLinksSomeRange<LinksType>(arglist)) {
80  * body;
81  * }
82  */
83 
84 /**
85  * Class template that defines member types ForwardIterator and ReverseIterator
86  * for walking over a std::vector of pointers to Links objects.
87  * This primary template definition is for a non-const Links type.
88  * @tparam Links Link object type.
89  */
90 template<class Links> struct JeodLinksIterators
91 {
92  using ForwardIterator = typename std::vector<Links *>::iterator;
93  using ReverseIterator = typename std::vector<Links *>::reverse_iterator;
94 };
95 
96 /**
97  * Partial specialization of JeodLinksIterators for const Links types.
98  * Like the primary definition, this specialization defines member types
99  * ForwardIterator and ReverseIterator, but this are now const iterators.
100  * @tparam Links Link object type.
101  */
102 template<class Links> struct JeodLinksIterators<const Links>
103 {
104  using ForwardIterator = typename std::vector<Links *>::const_iterator;
105  using ReverseIterator = typename std::vector<Links *>::const_reverse_iterator;
106 };

108 /**
109  * Base class template for all tree links range types.
110  * @tparam Iterator The type of iterator stored as the begin_ and end_ data
111  * members and returned by the begin and end member functions.
112  */
113 template<class Iterator> class TreeLinksRange
114 {
115 public:
116  /**
117  * Constructor.
118  * @tparam T The type of argument begin_in.
119  * @tparam U The type of argument end_in.
120  * @param begin_in Value used to construct the begin_ data member.
121  * @param end_in Value used to construct the end_ data member.
122  */
123  template<typename T, typename U>
124  TreeLinksRange(T begin_in, U end_in)
125  : begin_(begin_in),
126  end_(end_in)
127  {
128  }
129 
130  /**
131  * Mutable accessor to the begin_ data member.
132  */
133  Iterator & begin()
134  {
135  return begin_;
136  }
137 
138  /**
139  * Mutable accessor to the end_ data member.
140  */
141  Iterator & end()
142  {
143  return end_;
144  }
145 
146 private:
147  /**
148  * Object returned (by reference) by the begin member function.
149  */
150  Iterator begin_; //!< trick_units(--)
151 
152  /**
153  * Object returned (by reference) by the end member function.
154  */
155  Iterator end_; //!< trick_units(--)
156 };

158 /**
159  * A TreeLinksAscendRange walks up a Links object's path_to_node_ data
160  * member, starting at the start node and ending just before the end node.
161  */
162 template<class Links>
163 class TreeLinksAscendRange : public TreeLinksRange<typename JeodLinksIterators<Links>::ReverseIterator>
164 {
165 public:
166  using ReverseIterator = typename JeodLinksIterators<Links>::ReverseIterator;
167 
168  /**
169  * Non-default constructor.
170  * Create a TreeLinksAscendRange that walks over the entire path_to_node_
171  * from the bottom to the top.
172  */
173  explicit TreeLinksAscendRange(Links & links)
174  : TreeLinksRange<ReverseIterator>(ReverseIterator(links.path_to_node_.end()),
175  ReverseIterator(links.path_to_node_.begin()))
176  {
177  }
178 
179  /**
180  * Non-default constructor.
181  * Create a TreeLinksAscendRange given the start and end indices in the
182  * input Links object's path_to_node_ vector.
183  * Behavior is undefined if start_index > path_to_node_.size() or if
184  * end_index >= start_index.
185  * @param links Object whose path_to_node_ vector is to be traversed,
186  * in reverse.
187  * @param start_index Index of the element in the path_to_node_ vector
188  * that immediately follows the initial element to be visited in a
189  * range-based for loop. For example, using path_to_node_.size()
190  * starts at the final element of the vector.
191  * @param end_index Index of the element in the path_to_node_ vector
192  * that is the last element to be visited in a range-based for loop.
193  * For example, using zero stops the iteration at the initial element
194  * in the vector.
195  */
196  TreeLinksAscendRange(Links & links, unsigned int start_index, unsigned int end_index = 0)
197  : TreeLinksRange<ReverseIterator>(ReverseIterator(links.path_to_node_.begin() + start_index),
198  ReverseIterator(links.path_to_node_.begin() + end_index))
199  {
200  }
201 };
