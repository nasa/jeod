108 void ContactSurface::allocate_interaction_facet(Facet * facet,
109  InteractionFacetFactory * factory,
110  FacetParams * params,
111  unsigned int index)
112 {
113  if(facets_size <= index)
114  {
115  MessageHandler::fail(__FILE__,
116  __LINE__,
117  ContactMessages::initialization_error,
118  "ContactSurface::allocate_interaction_facet was asked to allocate "
119  "from the Facet at array index %d. This is out of bounds "
120  "of the array of facets, which is %d long",
121  index,
122  facets_size);
123 
124  return;
125  }
126 
127  /* need to temporarily save off the InteractionFacet returned before
128  dynamic casting it. If the dynamic cast fails, we want to destroy
129  the InteractionFacet so we don't get a memory leak */
130 
131  InteractionFacet * temp_facet = nullptr;
132 
133  // attempt to create the facet
134  temp_facet = factory->create_facet(facet, params);
135 
136  // if the facet is NULL, then we have a problem
137  if(temp_facet == nullptr)
138  {
139  MessageHandler::fail(__FILE__,
140  __LINE__,
141  ContactMessages::initialization_error,
142  "There was an error when attempting to create an interaction "
143  "facet for the facet found at array index %d",
144  index);
145  }
146 
147  // Facet is currently an interaction facet. Try to make it an ContactFacet
148 
149  auto * temp_contact_facet = dynamic_cast<ContactFacet *>(temp_facet);
150 
151  // If that fails, it doesn't belong in this surface so there is a problem
152  if(temp_contact_facet == nullptr)
153  {
154  // temp_facet can NOT be NULL, since it was already checked for above
155  JEOD_DELETE_OBJECT(temp_facet);
156 
157  MessageHandler::fail(__FILE__,
158  __LINE__,
159  ContactMessages::initialization_error,
160  "The InteractionFacet created from the Facet found at array "
161  "index %d was not of a type that inherits from ContactFacet. "
162  "This is required",
163  index);
164 
165  return;
166  }
167 
168  // Store the contact_facet into the contact_facets array
169  contact_facets[index] = temp_contact_facet;
170 }

177 void ContactSurface::collect_forces_torques()
178 {

181 
182  for(unsigned int i = 0; i < facets_size; i++)
183  {
184  Vector3::sum(contact_facets[i]->force, contact_force, contact_force);
185 
186  Vector3::sum(contact_facets[i]->torque, contact_torque, contact_torque);
187 
188  Vector3::initialize(contact_facets[i]->force);
189  Vector3::initialize(contact_facets[i]->torque);
190  }
191 
192  Vector3::zero_small(1e-10, contact_force);
193  Vector3::zero_small(1e-10, contact_torque);
194 }
