68 /**
69  * Represents a view into a column of an array.
70  * The index operator provides unprotected access to an element
71  * of the view. The member function at() ensures that the indexed
72  * element is valid.
73  * @tparam ArrayType The type of the underlying object.
74  * This must, at a minimum, provide an function call operator that
75  * accesses array elements and a size method that returns a pair.
76  * @tparam ElemType The type of the elements held by the underlying object.
77  * This is assumed to be a primitive type.
78  */
79 template <class ArrayType, typename ElemType, typename RefType>
80 class ColumnView : public VectorView<ElemType, RefType>
81 {
82 public:
83 
84  /**
85  * Non-default constructor.
86  * @note This can create a zero-sized view.
87  * @param array_in The 2D array that underlies this view.
88  * @param row_range A begin, end pair that denote the extent of the
89  * rows in this view.
90  * The begin value corresponds to this view's zeroth element.
91  * The end value is one past the last element indexed by this view.
92  * To be valid, the end value must not be less than the begin value
93  * and must not be greater than the underlying vector's size.
94  * @param col_in The column of the array to be represented by this view.
95  * This must reference a valid column of the underlying array.
96  * @throw std::out_of_range if the range is invalid.
97  */
98  ColumnView (
99  ArrayType& array_in,
100  const SolverTypes::IndexPairT& row_range,
101  unsigned col_in)
102  :
103  VectorView<ElemType>(row_range.second-row_range.first),
104  array(array_in),
105  the_col(col_in),
106  beg_row(row_range.first)
107  {
108  const auto& array_size = array.size();
109  if ((the_col >= array_size.second) ||
110  (row_range.second < row_range.first) ||
111  (row_range.second > array_size.first))
112  {
113  throw std::out_of_range(
114  "Attempt to create an invalid column view into an array");
115  }
116  }
117 
118  /**
119  * Non-const index operator.
120  * @param ielem View element to be indexed.
121  * @return Reference to the indexed value of the view, which is the
122  * beg_elem+ielem element of the underlying std::vector.
123  */
124  RefType operator[] (unsigned ielem) override
125  {
126  return array(beg_row+ielem, the_col);
127  }
128 
129 protected:
130 
131  /**
132  * The underlying array.
133  */
134  ArrayType& array; //!< trick_units(--)

144  */
145  const unsigned beg_row; //!< trick_units(--)
146 };
