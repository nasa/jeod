67 /**
68  * Represents a view into a column of an array.
69  * The index operator provides unprotected access to an element
70  * of the view. The member function at() ensures that the indexed
71  * element is valid.
72  * @tparam ArrayType The type of the underlying object.
73  * This must, at a minimum, provide an function call operator that
74  * accesses array elements and a size method that returns a pair.
75  * @tparam ElemType The type of the elements held by the underlying object.
76  * This is assumed to be a primitive type.
77  */
78 template<class ArrayType, typename ElemType, typename RefType> class ColumnView : public VectorView<ElemType, RefType>
79 {
80 public:
81  /**
82  * Non-default constructor.
83  * @note This can create a zero-sized view.
84  * @param array_in The 2D array that underlies this view.
85  * @param row_range A begin, end pair that denote the extent of the
86  * rows in this view.
87  * The begin value corresponds to this view's zeroth element.
88  * The end value is one past the last element indexed by this view.
89  * To be valid, the end value must not be less than the begin value
90  * and must not be greater than the underlying vector's size.
91  * @param col_in The column of the array to be represented by this view.
92  * This must reference a valid column of the underlying array.
93  * @throw std::out_of_range if the range is invalid.
94  */
95  ColumnView(ArrayType & array_in, const SolverTypes::IndexPairT & row_range, unsigned col_in)
96  : VectorView<ElemType>(row_range.second - row_range.first),
97  array(array_in),
98  the_col(col_in),
99  beg_row(row_range.first)
100  {
101  const auto & array_size = array.size();
102  if((the_col >= array_size.second) || (row_range.second < row_range.first) ||
103  (row_range.second > array_size.first))
104  {
105  JEOD_THROW(std::out_of_range("Attempt to create an invalid column view into an array"));
106  }
107  }
108 
109  /**
110  * Non-const index operator.
111  * @param ielem View element to be indexed.
112  * @return Reference to the indexed value of the view, which is the
113  * beg_elem+ielem element of the underlying std::vector.
114  */
115  RefType operator[](unsigned ielem) override
116  {
117  return array(beg_row + ielem, the_col);
118  }
119 
120 protected:
121  /**
122  * The underlying array.
123  */
124  ArrayType & array; //!< trick_units(--)
125 
126  /**
127  * The array column encapsulated by this view.
128  */
129  const unsigned the_col; //!< trick_units(--)
130 
131  /**
132  * The array row that corresponds to the zeroth view element.
133  */
134  const unsigned beg_row; //!< trick_units(--)
135 };
