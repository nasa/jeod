212  // Compute derived constants.
213  header.be_em_dist_ratio = 1.0 / (1.0 + header.em_mass_ratio);
214  header.bm_em_dist_ratio = header.be_em_dist_ratio *

270 /**
271  * Calculate the location of the L1 point as a ratio.
272  * @return Ratio of body1 to L1-point distance to body1 to body2 distance
273  * \param[in] b1b2_mass_ratio Body1 to body2 mass ratio
274  */
275 static double
276 l1_point (
277  double b1b2_mass_ratio)
278 {
279  double k, kp1, k3p2, k3p1;
280  double z, dz;
281 
282  // The ratio of the body1-L1 distance to the body1-body2 distance 'z'
283  // is the real solution to the fifth-order equation
284  // (k+1)*z^5 - (3k+2)*z^4 + (3k+1)*z^3 - z^2 + 2z - 1 = 0
285  // where 'k' is the body2-to-body1 mass ratio.
286  // Solve for z via Newton-Raphson.
287  k = 1.0 / b1b2_mass_ratio;
288  kp1 = k + 1.0;
289  k3p1 = 3.0*k + 1.0;
290  k3p2 = 3.0*k + 2.0;
291  z = 0.8;
292  dz = 1.0;
293  while ((dz > 1e-15) || (dz < -1e-15)) {
294  double f = -1.0 + z*(2.0 - z*(1.0 - z*(k3p1 - z*(k3p2 - z*kp1))));
295  double df = 2.0 - z*(2.0 - z*(3.0*k3p1 - z*(4.0*k3p2 - z*5.0*kp1)));
296  dz = -f / df;
297  z += dz;
298  }
299 
300  return z;
301 }
