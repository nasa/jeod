180  double f;

182  // Perform items common to the initial open / and restart re-open.
183  pre_initialize();
184 
185  // Allocate buffers for reading and interpreting ephemeris file records.
186  coef.chebypoly = JEOD_ALLOC_PRIM_ARRAY(io.max_terms, double);
187  coef.chebyderiv = JEOD_ALLOC_PRIM_ARRAY(io.max_terms, double);
188 
189  // Grab important constants from the symbol file
190  header.au = io.metaData->de_constants[De4xxBase::De4xx_Const_AU];
191  header.em_mass_ratio = io.metaData->de_constants[De4xxBase::De4xx_Const_EMRAT];
192  header.vlight = io.metaData->de_constants[De4xxBase::De4xx_Const_CLIGHT];
193 
194  header.gmbody[De4xxBase::De4xx_Ephem_Mercury] = io.metaData->de_constants[De4xxBase::De4xx_Const_GM1];
195  header.gmbody[De4xxBase::De4xx_Ephem_Venus] = io.metaData->de_constants[De4xxBase::De4xx_Const_GM2];
196  header.gmbody[De4xxBase::De4xx_Ephem_EMbary] = io.metaData->de_constants[De4xxBase::De4xx_Const_GMB];
197  header.gmbody[De4xxBase::De4xx_Ephem_Mars] = io.metaData->de_constants[De4xxBase::De4xx_Const_GM4];
198  header.gmbody[De4xxBase::De4xx_Ephem_Jupiter] = io.metaData->de_constants[De4xxBase::De4xx_Const_GM5];
199  header.gmbody[De4xxBase::De4xx_Ephem_Saturn] = io.metaData->de_constants[De4xxBase::De4xx_Const_GM6];
200  header.gmbody[De4xxBase::De4xx_Ephem_Uranus] = io.metaData->de_constants[De4xxBase::De4xx_Const_GM7];
201  header.gmbody[De4xxBase::De4xx_Ephem_Neptune] = io.metaData->de_constants[De4xxBase::De4xx_Const_GM8];
202  header.gmbody[De4xxBase::De4xx_Ephem_Pluto] = io.metaData->de_constants[De4xxBase::De4xx_Const_GM9];
203  header.gmbody[De4xxBase::De4xx_Ephem_Sun] = io.metaData->de_constants[De4xxBase::De4xx_Const_GMS];
204 
205  // Compute derived constants.
206  header.be_em_dist_ratio = 1.0 / (1.0 + header.em_mass_ratio);
207  header.bm_em_dist_ratio = header.be_em_dist_ratio * header.em_mass_ratio;
208  header.e1_em_dist_ratio = l1_point(header.em_mass_ratio);
209  header.b1_em_dist_ratio = header.e1_em_dist_ratio - header.be_em_dist_ratio;
210 
211  // Compute the Moon and Earth gravitational constants.
212  header.gmbody[De4xxBase::De4xx_Ephem_Moon] = header.gmbody[De4xxBase::De4xx_Ephem_EMbary] * header.be_em_dist_ratio;
213  header.gmbody[De4xxBase::De4xx_Ephem_Earth] = header.gmbody[De4xxBase::De4xx_Ephem_Moon] * header.em_mass_ratio;
214 
215  // Convert distance and velocity constants from km, km/sec to MKS units.
216  header.au *= 1e3;
217  header.vlight *= 1e3;
218 
219  // Convert the gravitational constants from AU^3/day^2 to MKS units.
220  f = (header.au * header.au * header.au) / (86400.0 * 86400.0);
221  for(uint32_t ii = 0; ii < De4xxBase::number_grav_models(file_spec.denum); ii++)
222  {
223  header.gmbody[ii] *= f;
224  }
225 
226  // Save the reference time.
227  del_day += (time_offset + init_time) / 86400.0;
228  ref_time.epoch_date = std::floor(epoch_time + del_day) + 0.5;
229  ref_time.fdate = (epoch_time - ref_time.epoch_date) + del_day;
230  day_offset = std::floor(ref_time.fdate);
231  ref_time.epoch_date += day_offset;
232  ref_time.fdate -= day_offset;
233  ref_time.init_time = init_time;
234 
235  // Compute the reference block.
236  day_offset = (ref_time.epoch_date - io.segmentData[0].start_epoch) + ref_time.fdate;
237  ref_time.block_no = day_offset / io.metaData->delta_epoch;
238 
239  // Sanity check: Ensure the timepoint of interest is in the database.

246  }
247

249  // Force an update by setting the time to an invalid value.
250  update_time = init_time - 1.0;

253 /**
254  * Calculate the location of the L1 point as a ratio.
255  * @return Ratio of body1 to L1-point distance to body1 to body2 distance
256  * \param[in] b1b2_mass_ratio Body1 to body2 mass ratio
257  */
258 static double l1_point(double b1b2_mass_ratio)
259 {
260  double k, kp1, k3p2, k3p1;
261  double z, dz;
262 
263  // The ratio of the body1-L1 distance to the body1-body2 distance 'z'
264  // is the real solution to the fifth-order equation
265  // (k+1)*z^5 - (3k+2)*z^4 + (3k+1)*z^3 - z^2 + 2z - 1 = 0
266  // where 'k' is the body2-to-body1 mass ratio.
267  // Solve for z via Newton-Raphson.
268  k = 1.0 / b1b2_mass_ratio;
269  kp1 = k + 1.0;
270  k3p1 = 3.0 * k + 1.0;
271  k3p2 = 3.0 * k + 2.0;
272  z = 0.8;
273  dz = 1.0;
274  while((dz > 1e-15) || (dz < -1e-15))
275  {
276  double f = -1.0 + z * (2.0 - z * (1.0 - z * (k3p1 - z * (k3p2 - z * kp1))));
277  double df = 2.0 - z * (2.0 - z * (3.0 * k3p1 - z * (4.0 * k3p2 - z * 5.0 * kp1)));
278  dz = -f / df;
279  z += dz;
280  }
281 
282  return z;
283 }
