153 /**
154  * Apply the initializer.
155  * \param[in,out] dyn_manager Dynamics manager
156  */
157 void DynBodyInitOrbit::apply(DynManager & dyn_manager)
158 {
159  OrbitalElements elem;
160 
161  enum
162  {
163  ShapeAltitudes,
164  ShapeSemiMajorAxis,
165  ShapeSemiLatusRectum
166  } shape;
167 
168  enum
169  {
170  LocationTimePeri,
171  LocationMeanAnom,
172  LocationTrueAnom
173  } location;
174 
175  // Convert the input elements to inclination, ascending node, arg. peri PLUS
176  // - altitudes at periapsis, apoapsis, OR
177  // semi-major axis and eccentricity OR
178  // semi-latus rectum and eccentricity
179  // - time since periapsis, mean anomaly, or true anomaly.
180  switch(set)
181  {
182  // SmaEccIncAscnodeArgperTimeperi:
183  // Have a,e,i,Omega,omega,T.
184  case SmaEccIncAscnodeArgperTimeperi:
185  shape = ShapeSemiMajorAxis;
186  location = LocationTimePeri;
187  break;
188 
189  // SmaEccIncAscnodeArgperManom:
190  // Have a,e,i,Omega,omega,M.
191  case SmaEccIncAscnodeArgperManom:
192  shape = ShapeSemiMajorAxis;
193  location = LocationMeanAnom;
194  break;
195 
196  // SlrEccIncAscnodeArgperTanom:
197  // Have alpha,e,i,Omega,omega,nu.
198  case SlrEccIncAscnodeArgperTanom:
199  shape = ShapeSemiLatusRectum;
200  location = LocationTrueAnom;
201  break;
202 
203  // IncAscnodeAltperAltapoArgperTanom
204  // Have i,Omega,perialt,apoalt,omega,nu.
205  case IncAscnodeAltperAltapoArgperTanom:
206  case CaseEleven:
207  shape = ShapeAltitudes;
208  location = LocationTrueAnom;
209  break;
210 
211  // IncAscnodeAltperAltapoArgperTimeperi:
212  // Have i,Omega,perialt,apoalt,omega,T.
213  case IncAscnodeAltperAltapoArgperTimeperi:
214  shape = ShapeAltitudes;
215  location = LocationTimePeri;
216  break;
217 
218  // SmaIncAscnodeArglatRadRadvel:
219  // Have a,i,Omega,arglat,r,rdot.
220  // Compute omega,e,nu from the above.
221  case SmaIncAscnodeArglatRadRadvel:
222  {
223  double ecosE, esinE, ecc_sq;
224 
225  // Compute the eccentricity times the
226  // sine and cosine of the eccentric anomaly.
227  ecosE = (semi_major_axis - orb_radius) / semi_major_axis;
228  esinE = (radial_vel * orb_radius) / std::sqrt(planet->grav_source->mu * semi_major_axis);
229 
230  // Compute the eccentricity.
231  ecc_sq = ecosE * ecosE + esinE * esinE;
232  eccentricity = std::sqrt(ecc_sq);
233 
234  // Non-circular orbit: Compute the true anomaly.
235  if(eccentricity >= 1.0e-14)
236  {
237  double kcost = ecosE - ecc_sq;
238  double ksint = std::sqrt(1.0 - ecc_sq) * esinE;
239  true_anomaly = std::atan2(ksint, kcost);
240  }
241 
242  // Circular orbit: Set the true anomaly to zero.
243  else
244  {
245  true_anomaly = 0.0;
246  }
247 
248  // Compute the argument of periapsis.
249  arg_periapsis = arg_latitude - true_anomaly;
250 
251  shape = ShapeSemiMajorAxis;
252  location = LocationTrueAnom;
253  break;
254  }
255 
256  // SmaEccIncAscnodeArgperTanom:
257  // Have a,e,i,Omega,omega,nu.
258  case SmaEccIncAscnodeArgperTanom:
259  shape = ShapeSemiMajorAxis;
260  location = LocationTrueAnom;
261  break;
262 
263  // All others are illegal.
264  case InvalidSet:
265  default:
266  MessageHandler::fail(__FILE__,
267  __LINE__,
268  BodyActionMessages::illegal_value,
269  "%s error:\n"
270  "Invalid orbital set",
271  action_identifier.c_str());
272 
273  // Not reached
274  return;
275  }
276 
277  // Compute the semi-major axis and eccentricity from altitudes.
278  if(shape == ShapeAltitudes)
279  {
280  semi_major_axis = planet->r_eq + 0.5 * (alt_apoapsis + alt_periapsis);
281  eccentricity = (alt_apoapsis - alt_periapsis) / (2.0 * semi_major_axis);
282  shape = ShapeSemiMajorAxis;
283  }
284 
285  // Compute the semi-latus rectum from the semi-major axis and eccentricity.
286  if(shape == ShapeSemiMajorAxis)
287  {
288  semi_latus_rectum = semi_major_axis * (1.0 - eccentricity * eccentricity);
289  // shape = ShapeSemiLatusRectum;
290  }
291 
292  // Compute the mean anomaly from the time since periapsis passage.
293  if(location == LocationTimePeri)
294  {
295  mean_anomaly = time_periapsis * std::sqrt(planet->grav_source->mu / semi_major_axis) / semi_major_axis;
296  location = LocationMeanAnom;
297  }
298 
299  // At this point, have alpha,e,i,Omega,omega and either nu or M.
300  // All that remains to be resolved is mean versus true anomaly.
301  // Populate the orbital elements structure.
302  elem.semiparam = semi_latus_rectum;
303  elem.e_mag = eccentricity;
304  elem.inclination = inclination;
305  elem.arg_periapsis = arg_periapsis;
306  elem.long_asc_node = ascending_node;
307 
308  // Compute true anomaly if needed.
309  if(location == LocationMeanAnom)
310  {
311  elem.mean_anom = mean_anomaly;
312  elem.mean_anom_to_nu();
313  }
314  else
315  {
316  elem.true_anom = true_anomaly;
317  elem.nu_to_anomalies();
318  }
319 
320  // Compute cartesian coordinates.
321  elem.to_cartesian(planet->grav_source->mu, position, velocity);
322 
323  // Transform to planet-centered inertial.
324  if(orbit_frame != &(planet->inertial))
325  {
326  RefFrameState rel_state;
327  orbit_frame->compute_relative_state(planet->inertial, rel_state);
328  // Sanity check? Should have null position, velocity from the
329  // planet inertial to the orbit frame.
330 
331  Vector3::transform_transpose(rel_state.rot.T_parent_this, position);
332  Vector3::transform_transpose(rel_state.rot.T_parent_this, velocity);
333  }
334 
335  // Pass the message up the chain.
336  DynBodyInitTransState::apply(dyn_manager);
337 }
