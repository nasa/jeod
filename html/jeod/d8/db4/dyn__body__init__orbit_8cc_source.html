187 /**
188  * Apply the initializer.
189  * \param[in,out] dyn_manager Dynamics manager
190  */
191 void
192 DynBodyInitOrbit::apply (
193  DynManager & dyn_manager)
194 {
195  OrbitalElements elem;
196  enum {
197  ShapeAltitudes,
198  ShapeSemiMajorAxis,
199  ShapeSemiLatusRectum
200  } shape;
201  enum {
202  LocationTimePeri,
203  LocationMeanAnom,
204  LocationTrueAnom
205  } location;
206 
207  // Convert the input elements to inclination, ascending node, arg. peri PLUS
208  // - altitudes at periapsis, apoapsis, OR
209  // semi-major axis and eccentricity OR
210  // semi-latus rectum and eccentricity
211  // - time since periapsis, mean anomaly, or true anomaly.
212  switch (set) {
213 
214  // SmaEccIncAscnodeArgperTimeperi:
215  // Have a,e,i,Omega,omega,T.
216  case SmaEccIncAscnodeArgperTimeperi:
217  shape = ShapeSemiMajorAxis;
218  location = LocationTimePeri;
219  break;
220 
221  // SmaEccIncAscnodeArgperManom:
222  // Have a,e,i,Omega,omega,M.
223  case SmaEccIncAscnodeArgperManom:
224  shape = ShapeSemiMajorAxis;
225  location = LocationMeanAnom;
226  break;
227 
228  // SlrEccIncAscnodeArgperTanom:
229  // Have alpha,e,i,Omega,omega,nu.
230  case SlrEccIncAscnodeArgperTanom:
231  shape = ShapeSemiLatusRectum;
232  location = LocationTrueAnom;
233  break;
234 
235  // IncAscnodeAltperAltapoArgperTanom
236  // Have i,Omega,perialt,apoalt,omega,nu.
237  case IncAscnodeAltperAltapoArgperTanom:
238  case CaseEleven:
239  shape = ShapeAltitudes;
240  location = LocationTrueAnom;
241  break;
242 
243  // IncAscnodeAltperAltapoArgperTimeperi:
244  // Have i,Omega,perialt,apoalt,omega,T.
245  case IncAscnodeAltperAltapoArgperTimeperi:
246  shape = ShapeAltitudes;
247  location = LocationTimePeri;
248  break;
249 
250  // SmaIncAscnodeArglatRadRadvel:
251  // Have a,i,Omega,arglat,r,rdot.
252  // Compute omega,e,nu from the above.
253  case SmaIncAscnodeArglatRadRadvel:
254  {
255  double ecosE, esinE, ecc_sq;
256 
257  // Compute the eccentricity times the
258  // sine and cosine of the eccentric anomaly.
259  ecosE = (semi_major_axis - orb_radius) / semi_major_axis;
260  esinE = (radial_vel * orb_radius) /
261  std::sqrt (planet->grav_source->mu * semi_major_axis);
262 
263  // Compute the eccentricity.
264  ecc_sq = ecosE * ecosE + esinE * esinE ;
265  eccentricity = std::sqrt (ecc_sq);
266 
267  // Non-circular orbit: Compute the true anomaly.
268  if (eccentricity >= 1.0e-14) {
269  double kcost = ecosE - ecc_sq;
270  double ksint = std::sqrt (1.0 - ecc_sq) * esinE ;
271  true_anomaly = std::atan2 (ksint, kcost);
272  }
273 
274  // Circular orbit: Set the true anomaly to zero.
275  else {
276  true_anomaly = 0.0 ;
277  }
278 
279  // Compute the argument of periapsis.
280  arg_periapsis = arg_latitude - true_anomaly;
281 
282  shape = ShapeSemiMajorAxis;
283  location = LocationTrueAnom;
284  break;
285  }
286 
287  // SmaEccIncAscnodeArgperTanom:
288  // Have a,e,i,Omega,omega,nu.
289  case SmaEccIncAscnodeArgperTanom:
290  shape = ShapeSemiMajorAxis;
291  location = LocationTrueAnom;
292  break;
293 
294  // All others are illegal.
295  case InvalidSet:
296  default:
297  MessageHandler::fail (
298  __FILE__, __LINE__, BodyActionMessages::illegal_value,
299  "%s error:\n"
300  "Invalid orbital set",
301  action_identifier.c_str());
302 
303  // Not reached
304  return;
305  }
306 
307 
308  // Compute the semi-major axis and eccentricity from altitudes.
309  if (shape == ShapeAltitudes) {
310  semi_major_axis = planet->r_eq +
311  0.5 * (alt_apoapsis + alt_periapsis);
312  eccentricity = (alt_apoapsis - alt_periapsis) /
313  (2.0 * semi_major_axis);
314  shape = ShapeSemiMajorAxis;
315  }
316 
317  // Compute the semi-latus rectum from the semi-major axis and eccentricity.
318  if (shape == ShapeSemiMajorAxis) {
319  semi_latus_rectum = semi_major_axis *
320  ( 1.0 - eccentricity * eccentricity);
321  //shape = ShapeSemiLatusRectum;
322  }
323 
324  // Compute the mean anomaly from the time since periapsis passage.
325  if (location == LocationTimePeri) {
326  mean_anomaly = time_periapsis *
327  std::sqrt (planet->grav_source->mu / semi_major_axis) / semi_major_axis;
328  location = LocationMeanAnom;
329  }
330 
331  // At this point, have alpha,e,i,Omega,omega and either nu or M.
332  // All that remains to be resolved is mean versus true anomaly.
333  // Populate the orbital elements structure.
334  elem.semiparam = semi_latus_rectum;
335  elem.e_mag = eccentricity;
336  elem.inclination = inclination;
337  elem.arg_periapsis = arg_periapsis;
338  elem.long_asc_node = ascending_node;
339 
340  // Compute true anomaly if needed.
341  if (location == LocationMeanAnom) {
342  elem.mean_anom = mean_anomaly;
343  elem.mean_anom_to_nu();
344  }
345  else {
346  elem.true_anom = true_anomaly;
347  elem.nu_to_anomalies();
348  }
349 
350  // Compute cartesian coordinates.
351  elem.to_cartesian (planet->grav_source->mu, position, velocity);
352 
353  // Transform to planet-centered inertial.
354  if (orbit_frame != &(planet->inertial)) {
355  RefFrameState rel_state;
356  orbit_frame->compute_relative_state (planet->inertial, rel_state);
357  // Sanity check? Should have null position, velocity from the
358  // planet inertial to the orbit frame.
359 
360  Vector3::transform_transpose (rel_state.rot.T_parent_this, position);
361  Vector3::transform_transpose (rel_state.rot.T_parent_this, velocity);
362  }
363 
364  // Pass the message up the chain.
365  DynBodyInitTransState::apply (dyn_manager);
366 
367  return;
368 }
