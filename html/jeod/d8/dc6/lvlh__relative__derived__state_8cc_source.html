111 /**
112  * Convert from rectilinear LVLH coordinates to circular curvilinear.
113  * \param[in] rect_rel_state Source state
114  */
115 void LvlhRelativeDerivedState::convert_rect_to_circ(const RefFrameState & rect_rel_state)
116 {
117  // Locally store values to reduce lookups
118  double rect_position[3];
119  double rect_velocity[3];
120  Vector3::copy(rect_rel_state.trans.position, rect_position);
121  Vector3::copy(rect_rel_state.trans.velocity, rect_velocity);
122 
123  // Calculate the radial distance of the target LVLH frame from the origin
124  // of the planet used to define that LVLH frame. Note that <planet>.inertial
125  // is the direct parent of the target LVLH frame by assignation.
126  double reference_radius = Vector3::vmag(target_frame->state.trans.position);
127 
128  // Protect for invalid or zero value for reference radius
129  if(reference_radius <= 1e-9)
130  {
131  MessageHandler::fail(__FILE__,
132  __LINE__,
133  DerivedStateMessages::illegal_value,
134  "Derived state %s: Reference radius for curvilinear LVLH calculation"
135  " is negative or nearly zero.",
136  name.c_str());
137 
138  // Not reached
139  return;
140  }
141 
142  // Calculate the orbital phase angle between the target and subject.
143  double theta = std::atan2(rect_position[0], reference_radius - rect_position[2]);
144 
145  // Calculate the CLVLH x-position using the phase angle
146  rel_state.trans.position[0] = reference_radius * theta;
147 
148  // For this implementation, the y-axis for the RLVLH and CLVLH frames
149  // are co-aligned, i.e., the CLVLH y-axis does not curve
150  rel_state.trans.position[1] = rect_position[1];
151 
152  // The CLVLH z-position is the difference between the radial
153  // distances of the two vehicles.
154  double subject_radius = std::sqrt((reference_radius - rect_position[2]) * (reference_radius - rect_position[2]) +
155  rect_position[0] * rect_position[0]);
156  rel_state.trans.position[2] = reference_radius - subject_radius;
157 
158  // Construct the rotation matrix from RLVLH to CLVLH using phase angle
159  double xform_rect2curvi[3][3];
160  Matrix3x3::initialize(xform_rect2curvi);
161  double cos_theta = std::cos(theta);
162  double sin_theta = std::sin(theta);
163 
164  xform_rect2curvi[0][0] = cos_theta;
165  xform_rect2curvi[0][2] = sin_theta;
166  xform_rect2curvi[1][1] = 1.0;
167  xform_rect2curvi[2][0] = -sin_theta;
168  xform_rect2curvi[2][2] = cos_theta;
169 
170  // Transform subject_vehicle's RLVLH velocity by the phase angle to
171  // obtain the CLVLH velocity
172  Vector3::transform(xform_rect2curvi, rect_velocity, rel_state.trans.velocity);
173 
174  // Adjust subject_vehicle's RLVLH attitude by
175  // the phase angle as well to get the corresponding CLVLH values
176  Matrix3x3::product_right_transpose(rect_rel_state.rot.T_parent_this, xform_rect2curvi, rel_state.rot.T_parent_this);
177 
178  // Account for the rotation of the rectilinear LVLH frame
179  Vector3::copy(rect_rel_state.rot.ang_vel_this, rel_state.rot.ang_vel_this);
180  // Correct the angular velocity to account for theta dot
181  do_theta_dot_correction(rel_state.rot.ang_vel_this, rect_rel_state, reference_radius, false);
182  rel_state.rot.compute_quaternion();
183  rel_state.rot.compute_ang_vel_products();
184 }

186 /**
187  * Convert from circular curvilinear LVLH coordinates to rectilinear.
188  * \param[in] curvi_rel_state Source state
189  */
190 void LvlhRelativeDerivedState::convert_circ_to_rect(const RefFrameState & curvi_rel_state)
191 {
192  // Locally store values to reduce lookups
193  double curvi_position[3];
194  double curvi_velocity[3];
195  Vector3::copy(curvi_rel_state.trans.position, curvi_position);
196  Vector3::copy(curvi_rel_state.trans.velocity, curvi_velocity);
197 
198  // Calculate the radial distance of the target LVLH frame from the origin
199  // of the planet used to define that LVLH frame. Note that <planet>.inertial
200  // is the direct parent of the target LVLH frame by assignation.
201  double reference_radius = Vector3::vmag(target_frame->state.trans.position);
202 
203  // Protect for invalid or zero value for reference radius
204  if(reference_radius <= 1e-9)
205  {
206  MessageHandler::fail(__FILE__,
207  __LINE__,
208  DerivedStateMessages::illegal_value,
209  "Derived state %s: Reference radius for curvilinear LVLH calculation"
210  " is negative or nearly zero.",
211  name.c_str());
212 
213  // Not reached
214  return;
215  }
216 
217  // Calculate the radius of subject vehicle; note that CLVLH z-pos is
218  // the difference between the radial distances of the two vehicles.
219  double subject_radius = reference_radius - curvi_position[2];
220 
221  // Calculate the orbital phase angle between the target and subject.
222  // Note curvilinear x-position = reference_radius * theta.
223  double theta = curvi_position[0] / reference_radius;
224 
225  double cos_theta = std::cos(theta);
226  double sin_theta = std::sin(theta);
227 
228  // Project CLVLH z-position into RLVLH coordinates using phase angle.
229  rel_state.trans.position[2] = reference_radius - (subject_radius * cos_theta);
230 
231  // For this implementation, the y-axis for the RLVLH and CLVLH frames
232  // are co-aligned, i.e., the CLVLH y-axis does not curve.
233  rel_state.trans.position[1] = curvi_position[1];
234 
235  // Calculate the RLVLH x-position using the phase angle.
236  rel_state.trans.position[0] = subject_radius * sin_theta;
237 
238  // Construct the rotation matrix from CLVLH to RLVLH using phase angle
239  double xform_curvi2rect[3][3];
240  Matrix3x3::initialize(xform_curvi2rect);
241 
242  xform_curvi2rect[0][0] = cos_theta;
243  xform_curvi2rect[0][2] = -sin_theta;
244  xform_curvi2rect[1][1] = 1.0;
245  xform_curvi2rect[2][0] = sin_theta;
246  xform_curvi2rect[2][2] = cos_theta;
247 
248  // Transform subject_vehicle's CLVLH velocity by the phase angle to
249  // obtain the RLVLH velocity
250  Vector3::transform(xform_curvi2rect, curvi_velocity, rel_state.trans.velocity);
251 
252  // Adjust subject_vehicle's CLVLH attitude by
253  // the phase angle as well to get the corresponding RLVLH value
254  Matrix3x3::product_right_transpose(curvi_rel_state.rot.T_parent_this,
255  xform_curvi2rect,
256  rel_state.rot.T_parent_this);
257  Vector3::copy(curvi_rel_state.rot.ang_vel_this, rel_state.rot.ang_vel_this);
258  do_theta_dot_correction(rel_state.rot.ang_vel_this, rel_state, reference_radius, true);
259  rel_state.rot.compute_quaternion();
260  rel_state.rot.compute_ang_vel_products();
261 }

263 /**
264  * Compute thetadot correction to omega.
265  */
266 void LvlhRelativeDerivedState::do_theta_dot_correction(double omega[3],
267  const RefFrameState & s,
268  const double r,
269  bool c2r)
270 {
271  if(!use_theta_dot_correction)
272  {
273  return;
274  }
275 
276  double x = r - s.trans.position[2];
277  double y = s.trans.position[0];
278  double rdot = Vector3::dot(target_frame->state.trans.position, target_frame->state.trans.velocity) / r;
279  double xdot = rdot - s.trans.velocity[2];
280  double ydot = s.trans.velocity[0];
281  double delta_omega[3] = {0, (x * ydot - y * xdot) / (x * x + y * y), 0};
282 
283  Vector3::transform(s.rot.T_parent_this, delta_omega);
284  if(c2r)
285  {
286  Vector3::decr(delta_omega, omega);
287  }
288  else
289  {
290  Vector3::incr(delta_omega, omega);
291  }
292 }
