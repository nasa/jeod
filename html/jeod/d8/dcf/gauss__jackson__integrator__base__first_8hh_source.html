251 template<>
252 inline void GaussJacksonIntegratorBase<GaussJacksonOneState, er7_utils::FirstOrderODEIntegrator>::correct(
253  double dt, const double * ER7_UTILS_RESTRICT acc, GaussJacksonOneState & state)
254 {
255  double vfact = velocity_corrector;
256  double * ER7_UTILS_RESTRICT first_csum = corrector_sum.first;
257  double * ER7_UTILS_RESTRICT first_dinv = delinv.first;
258  double * ER7_UTILS_RESTRICT first_state = state.first;
259 
260  for(unsigned int ii = 0; ii < size; ++ii)
261  {
262  double temp = first_csum[ii] + vfact * acc[ii];
263  first_state[ii] = dt * (first_dinv[ii] + temp);

270 template<>
271 inline bool GaussJacksonIntegratorBase<GaussJacksonOneState, er7_utils::FirstOrderODEIntegrator>::test_for_convergence(
272  const GaussJacksonOneState & state, double * ER7_UTILS_RESTRICT hist_data)
273 {
274  const double * ER7_UTILS_RESTRICT new_data = state.first;
275 
276  bool passed = true;
277  for(unsigned int ii = 0; ii < size; ++ii)
278  {
279  double error = std::abs(new_data[ii] - hist_data[ii]);
280  if((error > absolute_tolerance) && (error > relative_tolerance * std::abs(new_data[ii])))
281  {
282  passed = false;
283  }
284  hist_data[ii] = new_data[ii];
285  }
