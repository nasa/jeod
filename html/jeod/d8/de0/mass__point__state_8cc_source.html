60 /*******************************************************************************
61 
62  Nomenclature of mathematical formulations.
63  x,Q,T = position, quaternion, transform, respectively.
64  A,B,C = various reference frames.
65 
66 
67  Long-hand notations.
68  x_A->B:C = Position of frame B origin with respect to frame A origin,
69  expressed in frame A coordinates.
70  T_A->B = Column vector transformation matrix from frame A to frame B.
71 
72 
73  Short-hand notations.
74  x_A:B = x_A->B:A (Location of frame B origin in frame A coordinates.)
75  T_A:B = T_A->B
76  S_A:B = {x_A:B, T_A:B}
77  In other words, the state of frame A with respect to frame B.
78 
79 
80  Operators.
81  Conceptually, there are three primitive operators on reference frame states:
82  assignment (=), unary negation (-), and addition (+). Note that addition
83  is nonabelian. Subtraction is in terms of addition and negation.
84 
85  Negation: S_B:A = - S_A:B
86  T_B:A = T_A:B^T
87  x_B:A = - (T_A:B * x_A:B)
88 
89  Addition: S_A:C = S_A:B + S_B:C
90  T_A:C = T_B:C * T_A:B
91  x_A:C = x_A:B + T_A:B^T * x_B:C
92 
93 
94  Methods.
95  Method This(input) Operand Result
96  negate N/A S_A:B this <- -S_A:B = S_B:A
97  incr_left S_B:C S_A:B this <- arg + this = S_A:C
98  incr_right S_A:B S_B:C this <- this + arg = S_A:C
99  decr_left S_A:C S_A:B this <- (-arg) + this = S_B:C
100  decr_right S_A:C S_B:C this <- this + (-arg) = S_A:B
101 
102 
103  Shortcut:
104  There is no need to transform a vector if the transformation is an identity.
105  The scalar part of the left transformation quaternion is exactly one in the
106  case of an identity transformation.
107 
108 *******************************************************************************/

120 void MassPointState::negate(const MassPointState & source)
121 {
122  // Relevant equations:
123  // T_B:A = T_A:B^T
124  // x_B:A = - (T_A:B * x_A:B)
125 
126  // Transform the matrix and translational state to the alternate frame.
127  // This only needs to be performed for nontrivial transformation matrices.
128  if(!Numerical::compare_exact(source.Q_parent_this.scalar, 1.0))
129  {
130  source.Q_parent_this.conjugate(Q_parent_this);
131  Matrix3x3::transpose(source.T_parent_this, T_parent_this);
132  Vector3::transform_transpose(T_parent_this, source.position, position);
133  }
134 
135  // Identity transform: Just copy position.
136  else
137  {
138  Matrix3x3::identity(T_parent_this);
139  Q_parent_this.make_identity();
140  Vector3::copy(source.position, position);
141  }
142 
143  // The above made the position have the wrong sign. Fix that.
144  Vector3::negate(position);
145 }

147 /**
148  * Compute S_A:C = S_A:B + S_B:C,
149  * with this initially containing S_B:C,
150  * the supplied argument containing S_A:B, and
151  * the resultant composition of states stored in this.
152  * \param[in] s_ab Left addend
153  */
154 void MassPointState::incr_left(const MassPointState & s_ab)
155 {
156  // Relevant equations:
157  // T_A:C = T_B:C * T_A:B
158  // x_A:C = x_A:B + T_A:B^T * x_B:C
159 
160  // Order dependency:
161  // Quantity Uses S_B:C items Uses S_A:C items Order
162  // T_A:C T None 1
163  // x_A:C x None 2
164 
165  // Compute T_A:C and x_A:C via
166  // T_A:C = T_B:C * T_A:B
167  // x_A:C = x_A:B + T_A:B^T * x_B:C
168  // Shortcut: only do the transformations if T_A:B is not the identity matrix.
169  if(!Numerical::compare_exact(s_ab.Q_parent_this.scalar, 1.0))
170  {
171  // Compute Q_A:C = Q_B:C * Q_A:B (and then normalize result)
172  Q_parent_this.multiply(s_ab.Q_parent_this);
173  Q_parent_this.normalize();
174 
175  // Compute the corresponding transformation matrix.
176  compute_transformation();
177 
178  // Transform x_B:C to frame A.
179  Vector3::transform_transpose(s_ab.T_parent_this, position);
180 
181  // No else. x_B:C is already in frame in as T_A:B is identity.
182  }
183 
184  // Add X_A:B to the state.
185  Vector3::incr(s_ab.position, position);
186 }

188 /**
189  * Compute S_A:C = S_A:B + S_B:C,
190  * with this initially containing S_A:B,
191  * the supplied argument containing S_B:C, and
192  * the resultant composition of states stored in this.
193  * \param[in] s_bc Right addend
194  */
195 void MassPointState::incr_right(const MassPointState & s_bc)
196 {
197  // Relevant equations:
198  // T_A:C = T_B:C * T_A:B
199  // x_A:C = x_A:B + T_A:B^T * x_B:C
200 
201  // Order dependency:
202  // Quantity Uses S_A:B items Uses S_A:C items Order
203  // T_A:C T None 2
204  // x_A:C x, T None 1
205 
206  // Compute x_A:C, v_A:C, and T_A:C via
207  // T_A:C = T_B:C * T_A:B
208  // x_A:C = x_A:B + T_A:B^T * x_B:C
209  // Shortcut: only do the transformations if T_A:B is not the identity matrix.
210  if(!Numerical::compare_exact(Q_parent_this.scalar, 1.0))
211  {
212  // Compute x_A:C.
213  // Note: This must be computed before computing T_A:C as it uses T_A:B.
214  Vector3::transform_transpose_incr(T_parent_this, s_bc.position, position);
215 
216  // Compute Q_A:C = Q_B:C * Q_A:B (and then normalize result)
217  Q_parent_this.multiply_left(s_bc.Q_parent_this);
218  Q_parent_this.normalize();
219 
220  // Compute the corresponding transformation matrix.
221  compute_transformation();
222  }
223 
224  // Shortcuts for the case T_A:B is identity.
225  else
226  {
227  Q_parent_this = s_bc.Q_parent_this;
228  Matrix3x3::copy(s_bc.T_parent_this, T_parent_this);
229  Vector3::incr(s_bc.position, position);
230  }
231 }

233 /**
234  * Compute S_B:C = (-S_A:B) + S_A:C,
235  * with this initially containing S_A:C,
236  * the supplied argument containing S_A:B, and
237  * the resultant composition of states stored in this.
238  * \param[in] s_ab Left addend
239  */
240 void MassPointState::decr_left(const MassPointState & s_ab)
241 {
242  // Relevant equations:
243  // T_B:C = T_A:C * T_A:B^T
244  // x_B:C = T_A:B * (x_A:C - x_A:B)
245 
246  // Order dependency:
247  // Quantity Uses S_A:C items Uses S_B:C items Order
248  // T_B:C T None 1
249  // x_B:C x None 2
250 
251  // Compute T_B:C and x_B:C via
252  // T_B:C = T_A:C * T_A:B^T
253  // x_B:C = T_A:B * (x_A:C - x_A:B)
254  // Shortcut: only do the transformations if T_A:B is not the identity matrix.
255  if(!Numerical::compare_exact(s_ab.Q_parent_this.scalar, 1.0))
256  {
257  // Compute Q_B:C = Q_A:C * Q_A:B^Q
258  Q_parent_this.multiply_conjugate(s_ab.Q_parent_this);
259  Q_parent_this.normalize();
260 
261  // Compute the corresponding transformation matrix.
262  compute_transformation();
263 
264  // Compute x_B:C.
265  Vector3::decr(s_ab.position, position);
266  Vector3::transform(s_ab.T_parent_this, position);
267  }
268 
269  // Shortcuts for the case T_A:B is identity.
270  else
271  {
272  Vector3::decr(s_ab.position, position);
273  }
274 }

283 void MassPointState::decr_right(const MassPointState & s_bc)
284 {
285  // Relevant equations:
286  // T_A:B = T_B:C^T * T_A:C
287  // x_A:B = x_A:C - T_A:B^T * x_B:C
288 
289  // Order dependency:
290  // Quantity Uses S_A:C items Uses S_A:B items Order
291  // T_A:B T None 1
292  // x_A:B x T 2
293 
294  // Compute T_A:B via
295  // T_A:B = T_B:C^T * T_A:C
296  // Shortcut: only do the transformations if T_B:C is not the identity matrix.
297  if(!Numerical::compare_exact(s_bc.Q_parent_this.scalar, 1.0))
298  {
299  Q_parent_this.multiply_left_conjugate(s_bc.Q_parent_this);
300  Q_parent_this.normalize();
301  compute_transformation();
302 
303  // No else. T_A:B = T_A:C as T_B:C is identity.
304  }
305 
306  // Compute x_A:B via
307  // x_A:B = x_A:C - T_A:B^T * x_B:C
308  if(!Numerical::compare_exact(Q_parent_this.scalar, 1.0))
309  {
310  Vector3::transform_transpose_decr(T_parent_this, s_bc.position, position);
311  }
312  // Shortcuts for the case T_A:B is identity.
313  else
314  {
315  Vector3::decr(s_bc.position, position);
