146 /**
147  * Name an ephemeris item.
148  * \param[in] new_name New name
149  */
150 void
151 EphemerisItem::set_name (
152  const char * new_name)
153 {
154 
155  // Validate the name.
156  validate_name (__FILE__, __LINE__, new_name, name, "new_name");
157 
158  // Ephemeris reference frames have a name of the form "xxx.yyy".
159  // Lacking such a form, assume the caller meant the ".yyy" to be the
160  // default suffix for this item class.
161  if (std::strchr (new_name, '.') == nullptr) {
162  MessageHandler::inform (
163  __FILE__, __LINE__, EphemeridesMessages::invalid_name,
164  "Ephemeris item names must be of the form \"object.frame_name\".\n"
165  "Input value \"%s\" is invalid. Using \"%s.%s\".",
166  new_name, new_name, default_suffix());
167  set_name (new_name, default_suffix());
168  }
169 
170  // Name is of the form "xxx.yyy":
171  // Set the name per the argument, but do so via a string copy.
172  else {
173  set_name_internal (JEOD_STRDUP (new_name));
174  }
175 }

197 /**
198  * Set the target frame.
199  * All ephemeris items that share a common name must point
200  * to the same target frame.
201  * \param[in] frame Target frame
202  */
203 void
204 EphemerisItem::set_target_frame (
205  EphemerisRefFrame & frame)
206 {
207  // The head item needs to handle this message.
208  if (this != head) {
209  head->set_target_frame (frame);
210  return;
211  }
212 
213  // Nothing to do if the target frame has already been set on head and all
214  // connected EphemItems. If it has been set on the head item, set it on any
215  // follow-on items that have not already been set.
216  if (target_frame == &frame) {
217  for (EphemerisItem * ephem_item = head;
218  ephem_item != nullptr;
219  ephem_item = ephem_item->next) {
220  if (ephem_item->target_frame == nullptr) {
221  ephem_item->target_frame = &frame;
222  }
223  }
224  return;
225  }
226 
227  // Items that update translational state must have integration frames as
228  // their target, while items that update rotational state should target
229  // non-integration frames.
230  if ((updates_what() == EphemerisItem::Translation) !=
231  manager->is_integ_frame (frame)) {
232  MessageHandler::fail (
233  __FILE__, __LINE__, EphemeridesMessages::invalid_item,
234  "Incompatible item type and target frame for ephemeris item '%s'.",
235  name);
236  return;
237  }
238 
239  // Changing the target frame is verboten.
240  if (target_frame != nullptr) {
241  MessageHandler::fail (
242  __FILE__, __LINE__, EphemeridesMessages::invalid_item,
243  "Illegal attempt to change the target frame for ephemeris item '%s'.",
244  name);
245  return;
246  }
247 
248  // Set the target frame for all ephemeris items in the list.
249  for (EphemerisItem * ephem_item = head;
250  ephem_item != nullptr;
251  ephem_item = ephem_item->next) {
252  ephem_item->target_frame = &frame;
253  }
254 
255  //Sanity check target frame should not be NULL at this point.
256  if (target_frame == NULL) {
257  MessageHandler::fail (
258  __FILE__, __LINE__, EphemeridesMessages::invalid_item,
259  "Target frame not set for ephemeris item '%s'.",
260  name);
261  return;
262  }
263 
264  // Make the enabled item, if there is one, be the 'owner' of this frame.
265  EphemerisItem * enabled_item = get_enabled_item ();
266  if (enabled_item != nullptr) {
267  target_frame->set_owner (enabled_item);
268  if (target_frame->is_active()) {
269  enabled_item->activate();
270  }
271  }
272 }

276  * Enable an EphemerisItem object.

278 void
279 EphemerisItem::enable (
280  void)
281 {
282  // Nothing to do if the object is already enabled.
283  if (enabled) {
284  return;
285  }
286 
287  EphemerisItem * enabled_item = get_enabled_item ();
288 
289  // If there is a currently enabled item,
290  // Set the activity of this object to that of currently-enabled item and
291  // disable the enabled object.
292  if (enabled_item != nullptr) {
293  active = enabled_item->active;
294  enabled_item->disable ();
295  }
296 
297  // Transfer ownership of the target frame to this object.
298  if (target_frame != nullptr) {
299  target_frame->set_owner (this);
300  }
301 
302  // Let the manager know that things have changed.
303  if (manager != nullptr) {
304  manager->ephem_note_tree_status_change ();
305  }
306 
307  // Finally, mark the object as enabled.
308  enabled = true;
309 }

360 /**
361  * Is the item activatable?
362  * @return True if item can be activated.
363  */
364 bool
365 EphemerisItem::is_activatable (
366  void)
367 const
368 {
369  // An item must be enabled before it can be activated.
370  if (! enabled) {
371  return false;
372  }
373 
374  // An item that is already activated is (trivially) activatable.
375  if (active) {
376  return true;
377  }
378 
379  // Only one item with the same name can be active.
380  for (EphemerisItem * ephem_item = head;
381  ephem_item != nullptr;
382  ephem_item = ephem_item->next) {
383  if (ephem_item->active) {
384  return false;
385  }
386  }
387 
388  return true;
389 }
