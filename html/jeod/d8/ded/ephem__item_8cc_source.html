107 void EphemerisItem::set_name(const std::string & new_name)
108 {
109  // Validate the name.
110  validate_name(__FILE__, __LINE__, new_name, name, "new_name");
111 
112  // Ephemeris reference frames have a name of the form "xxx.yyy".

145 /**
146  * Set the target frame.
147  * All ephemeris items that share a common name must point
148  * to the same target frame.
149  * \param[in] frame Target frame
150  */
151 void EphemerisItem::set_target_frame(EphemerisRefFrame & frame)
152 {
153  // The head item needs to handle this message.
154  if(this != head)
155  {
156  head->set_target_frame(frame);
157  return;
158  }
159 
160  // Nothing to do if the target frame has already been set on head and all
161  // connected EphemItems. If it has been set on the head item, set it on any
162  // follow-on items that have not already been set.
163  if(target_frame == &frame)
164  {
165  for(EphemerisItem * ephem_item = head; ephem_item != nullptr; ephem_item = ephem_item->next)
166  {
167  if(ephem_item->target_frame == nullptr)
168  {
169  ephem_item->target_frame = &frame;
170  }
171  }
172  return;
173  }
174 
175  // Items that update translational state must have integration frames as
176  // their target, while items that update rotational state should target
177  // non-integration frames.
178  if((updates_what() == EphemerisItem::Translation) != manager->is_integ_frame(frame))
179  {
180  MessageHandler::fail(__FILE__,
181  __LINE__,
182  EphemeridesMessages::invalid_item,
183  "Incompatible item type and target frame for ephemeris item '%s'.",
184  name.c_str());
185  return;
186  }
187 
188  // Changing the target frame is verboten.
189  if(target_frame != nullptr)
190  {
191  MessageHandler::fail(__FILE__,
192  __LINE__,
193  EphemeridesMessages::invalid_item,
194  "Illegal attempt to change the target frame for ephemeris item '%s'.",
195  name.c_str());
196  return;
197  }
198 
199  // Set the target frame for all ephemeris items in the list.
200  for(EphemerisItem * ephem_item = head; ephem_item != nullptr; ephem_item = ephem_item->next)
201  {
202  ephem_item->target_frame = &frame;
203  }
204 
205  // Sanity check target frame should not be NULL at this point.
206  if(target_frame == nullptr)
207  {
208  MessageHandler::fail(__FILE__,
209  __LINE__,
210  EphemeridesMessages::invalid_item,
211  "Target frame not set for ephemeris item '%s'.",
212  name.c_str());
213  return;
214  }
215 
216  // Make the enabled item, if there is one, be the 'owner' of this frame.
217  EphemerisItem * enabled_item = get_enabled_item();
218  if(enabled_item != nullptr)
219  {
220  target_frame->set_owner(enabled_item);
221  if(target_frame->is_active())
222  {
223  enabled_item->activate();
224  }
225  }
226 }

228 /**
229  * Enable an EphemerisItem object.
230  */
231 void EphemerisItem::enable()
232 {
233  // Nothing to do if the object is already enabled.
234  if(enabled)
235  {
236  return;
237  }
238 
239  EphemerisItem * enabled_item = get_enabled_item();
240 
241  // If there is a currently enabled item,
242  // Set the activity of this object to that of currently-enabled item and
243  // disable the enabled object.
244  if(enabled_item != nullptr)
245  {
246  active = enabled_item->active;
247  enabled_item->disable();
248  }
249 
250  // Transfer ownership of the target frame to this object.
251  if(target_frame != nullptr)
252  {
253  target_frame->set_owner(this);
254  }
255 
256  // Let the manager know that things have changed.
257  if(manager != nullptr)
258  {
259  manager->ephem_note_tree_status_change();
260  }
261 
262  // Finally, mark the object as enabled.
263  enabled = true;
264 }

313 /**
314  * Is the item activatable?
315  * @return True if item can be activated.
316  */
317 bool EphemerisItem::is_activatable() const
318 {
319  // An item must be enabled before it can be activated.
320  if(!enabled)
321  {
322  return false;
323  }
324 
325  // An item that is already activated is (trivially) activatable.
326  if(active)
327  {
328  return true;
329  }
330 
331  // Only one item with the same name can be active.
332  for(EphemerisItem * ephem_item = head; ephem_item != nullptr; ephem_item = ephem_item->next)
333  {
334  if(ephem_item->active)
335  {
336  return false;
337  }
338  }
339 
340  return true;
341 }
