76 void TimeConverter_TAI_UT1::initialize(JeodBaseTime * parent_ptr, JeodBaseTime * child_ptr, const int int_dir)
77 {
78  verify_setup(parent_ptr, child_ptr, int_dir);
79 
80  /* The time in the lookup table is TAI. Initializing from TAI is preferred,
81  but possible from UT1. */
82  if(int_dir == 1)
83  {
84  tai_ptr = dynamic_cast<TimeTAI *>(parent_ptr);
85  ut1_ptr = dynamic_cast<TimeUT1 *>(child_ptr);

94  initialize_tai_to_ut1();

97  initialization. */
98  tai_ptr->trunc_julian_time = ut1_ptr->trunc_julian_time - a_to_b_offset;
99  if((tai_ptr->trunc_julian_time > next_when) || (tai_ptr->trunc_julian_time < prev_when))
100  {
101  initialize_tai_to_ut1();
102  }
103  }
104  else
105  {
106  MessageHandler::fail(__FILE__,
107  __LINE__,

115 /**
116  * The conversion from Atomic Time (TAI) to Universal Time (UT1)
117  * involves the addition of value that is a continuous function of TAI.
118  * That value is tabulated at regular points of TAI. This function
119  * initializes that table and sets the preliminary values.
120  *
121  * \par Assumptions and Limitations
122  * - The table does not go into the future.
123  */
124 void TimeConverter_TAI_UT1::initialize_tai_to_ut1()
125 {
126  /* if the user inputs their own value for a_to_b_offset, make sure that it
127  does not get overridden. */
128  if(override_data_table)
129  {
130  ut1_ptr->true_ut1 = false;
131  a_to_b_offset = tai_to_ut1_override_val / 86400;
132  return;
133  }
134 
135  if((when_vec == nullptr) || (val_vec == nullptr))
136  {
137  MessageHandler::fail(__FILE__,
138  __LINE__,
139  TimeMessages::invalid_data_error,
140  "\n"
141  "The TAI to UT1 conversion function requires a"
142  " data lookup table, but none has been provided.\n");
143  return;
144  }
145 
146  double tai_time = tai_ptr->trunc_julian_time; /* day local value of TAI time */
147  off_table_end = false;
148 
149  if(tai_time < when_vec[0])
150  {
151  // Julian date is less than initial date in table
152  if(tai_ptr->time_manager->dyn_time.scale_factor >= 0)
153  {
154  MessageHandler::inform(__FILE__,
155  __LINE__,
156  TimeMessages::invalid_data_error,
157  "\n"
158  "The current time precedes the first entry in the"
159  " look-up table for TAI to UT1 conversions. \n"
160  "The conversion value is being set to the first value in the table"
161  " (%f s) .\n"
162  "Simulation will continue to run and will continue to monitor\n"
163  "time for future crossing into the tabulated range.\n",
164  val_vec[0]);
165  index = -1; // because update will look at index+1, i.e. 0
166  prev_when = tai_time - 1;
167  next_when = when_vec[0];
168  next_value = val_vec[0];
169  }
170  else
171  {
172  MessageHandler::inform(__FILE__,
173  __LINE__,
174  TimeMessages::invalid_data_error,
175  "\n"
176  "The current time precedes the first entry in the"
177  " look-up table for TAI to UT1 conversions. \n"
178  "The conversion value is being set to the first value in the table"
179  " (%f s) .\n"
180  "Simulation will continue to run, and this value will be\n"
181  "maintained unless the time-direction changes.\n",
182  val_vec[0]);
183  off_table_end = true;
184  index = 0;
185  }
186  a_to_b_offset = val_vec[0] / 86400.0;
187  }
188  else if(tai_time >= when_vec[last_index])
189  {
190  // Julian date is beyond the table
191  index = last_index;
192  if(tai_ptr->time_manager->dyn_time.scale_factor >= 0)
193  {
194  MessageHandler::inform(__FILE__,
195  __LINE__,
196  TimeMessages::invalid_data_error,
197  "\n"
198  "The current time follows the last entry in the"
199  " look-up table for TAI to UT1 conversions. \n"
200  "The conversion value is being set to the last value in the table"
201  " (%f s) .\n"
202  "Simulation will continue to run, and this value will be\n"
203  "maintained unless the time-direction changes.\n",
204  val_vec[last_index]);
205  off_table_end = true;
206  }
207  else
208  {
209  MessageHandler::inform(__FILE__,
210  __LINE__,
211  TimeMessages::invalid_data_error,
212  "\n"
213  "The current time follows the last entry in the"
214  " look-up table for TAI to UT1 conversions. \n"
215  "The conversion value is being set to the last value in the table"
216  " (%f s) .\n"
217  "Simulation will continue to run and will continue to monitor\n"
218  "time for future crossing into the tabulated range.\n",
219  val_vec[last_index]);
220  prev_when = when_vec[last_index];
221  prev_value = val_vec[last_index];
222  next_when = tai_time + 1;
223  }
224  a_to_b_offset = val_vec[last_index] / 86400.0;
225  }
226  else
227  {
228  /* Find the last entry that comes before the current time (trunc_julian_time)
229  Highest value that can be reached is index = last_index - 1
230  because tai_time < when_vec[last_index] by earlier test. */
231  for(index = 0; tai_time >= when_vec[index + 1]; ++index)
232  {
233  ;
234  }
235  prev_when = when_vec[index];
236  prev_value = val_vec[index];
237  next_when = when_vec[index + 1];
238  next_value = val_vec[index + 1];
239  gradient = (next_value - prev_value) / (next_when - prev_when);
240  a_to_b_offset = (prev_value + (tai_time - prev_when) * gradient) / 86400.0;
241  }
242 }

244 /**
245  * Convert from TimeTAI to TimeUT1.
246  *
247  * \par Assumptions and Limitations
248  * - Time class MET is based on time class TAI, and counts the elapsed
249  * TAI time only
250  */
251 void TimeConverter_TAI_UT1::convert_a_to_b()
252 {
253  /* "when" is in days
254  a_to_b_offset is in seconds
255  gradient is in sec / day. */
256 
257  double tai_time = tai_ptr->trunc_julian_time; // -- local value of TAI time
258 
259  if(off_table_end || !ut1_ptr->true_ut1)
260  { // a_to_b_offset does not change
261  ut1_ptr->set_time_by_trunc_julian(tai_ptr->trunc_julian_time + a_to_b_offset);
262  // ut1_ptr->trunc_julian_time = tai_time + a_to_b_offset;
263  // ut1_ptr->days = ut1_ptr->trunc_julian_time - ut1_ptr->tjt_at_epoch;
264  // ut1_ptr->seconds = ut1_ptr->days * 86400;
265  return;
266  } // else
267  // for conventional (forward-time) simulations:
268 
269  /* "while" is used because updates need not be done regularly, may have to
270  catch up with accumulation of points. */
271  while(tai_time > next_when)
272  {
273  index++;
274  /* Doing a full assignment at each scan because in most cases, this loop is
275  skipped; a partial assignment (index and next_when
276  only) would require the assignment of the other values after the
277  while-loop, regardless of whether the while-loop was entered. */
278  prev_value = next_value;
279  prev_when = next_when;
280 
281  if(index < last_index)
282  {
283  next_value = val_vec[index + 1];
284  next_when = when_vec[index + 1];
285  gradient = (next_value - prev_value) / (next_when - prev_when);
286  }
287  else
288  {
289  /* Julian date is beyond the table, set to last value */
290  MessageHandler::inform(__FILE__,
291  __LINE__,
292  TimeMessages::invalid_data_error,
293  "\n"
294  "Current time is beyond the range covered by the TAI-UT1 table.\n"
295  "The difference between TAI and UT1 is being set to the last value \n"
296  "in the table. The simulation will continue to run with this value."
297  "\n");
298  off_table_end = true;
299  gradient = 0.0;
300  break;
301  }
302  }
303  /* do same for unconventional (reverse-time) sims.
304  prev still references the table entry that precedes current real-time,
305  even though it will be reached later in the sim (due to reverse
306  direction). */
307  while(tai_time < prev_when)
308  {
309  index--;
310  next_value = prev_value;
311  next_when = prev_when;
312 
313  if(index >= 0)
314  {
315  prev_value = val_vec[index];
316  prev_when = when_vec[index];
317  gradient = (next_value - prev_value) / (next_when - prev_when);
318  }
319  else
320  {
321  /* Julian date is less than initial date in table. */
322  MessageHandler::inform(__FILE__,
323  __LINE__,
324  TimeMessages::invalid_data_error,
325  "\n"
326  "Current time is beyond the range covered by the TAI-UT1 table.\n"
327  "The difference between TAI and UT1 is being set to the first value\n"
328  "in the table. The simulation will continue to run with this value."
329  "\n");
330  off_table_end = true;
331  gradient = 0.0;
332  break;
333  }
334  }
335  a_to_b_offset = (prev_value + (tai_time - prev_when) * gradient) / 86400.0;
336 
337  ut1_ptr->set_time_by_trunc_julian(tai_ptr->trunc_julian_time + a_to_b_offset);
338  // ut1_ptr->trunc_julian_time = tai_time + a_to_b_offset;
339  // ut1_ptr->days = ut1_ptr->trunc_julian_time - ut1_ptr->tjt_at_epoch;
340  // ut1_ptr->seconds = ut1_ptr->days * 86400;
341 }

440 /**
441  * Used when time reverses direction.
442  * Checks whether the table lookup function is using input values that
443  * are outside the scope of the table, and sets the flags appropriately
444  *
445  * \par Assumptions and Limitations
446  * - None
447  */
448 void TimeConverter_TAI_UT1::verify_table_lookup_ends()
449 {
450  if(!ut1_ptr->true_ut1)
451  { // table is not monitored
452  return;
453  }
454 
455  // if now going forward in time
456  if(tai_ptr->time_manager->dyn_time.scale_factor >= 0)
457  {
458  // and past the end of the table, past the table
459  if(index >= last_index)
460  {
461  off_table_end = true;
462  }
463  /* or if before the beginning of the table, no longer past the table. */
464  /* Change "prev_when" to be before "now", or the converter will
465  automatically revert to being off the table */
466  else if(index < 0)
467  {
468  prev_when = tai_ptr->trunc_julian_time - 1;
469  next_when = when_vec[0];
470  index = -1;
471  off_table_end = false;
472  }
473  }
474  else
475  { // reverse
476  if(index >= last_index)
477  {
478  next_when = tai_ptr->trunc_julian_time + 1;
479  prev_when = when_vec[last_index];
480  index = last_index;
481  off_table_end = false;
482  }
483  else if(index < 0)
484  {
485  off_table_end = true;
486  }
487  }
488 }
