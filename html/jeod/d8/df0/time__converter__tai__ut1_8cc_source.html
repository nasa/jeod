95 TimeConverter_TAI_UT1::initialize (

104  if (int_dir == 1) {
105  tai_ptr = dynamic_cast<TimeTAI *> (parent_ptr);

135 /**
136  * The conversion from Atomic Time (TAI) to Universal Time (UT1)
137  * involves the addition of value that is a continuous function of TAI.
138  * That value is tabulated at regular points of TAI. This function
139  * initializes that table and sets the preliminary values.
140  *
141  * \par Assumptions and Limitations
142  * - The table does not go into the future.
143  */
144 void
145 TimeConverter_TAI_UT1::initialize_tai_to_ut1 (
146  void)
147 {
148  /* if the user inputs their own value for a_to_b_offset, make sure that it
149  does not get overridden. */
150  if (override_data_table) {
151  ut1_ptr->true_ut1 = false;
152  a_to_b_offset = tai_to_ut1_override_val / 86400;
153  return;
154  }
155 
156  if ((when_vec == nullptr) || (val_vec == nullptr)) {
157  MessageHandler::fail (
158  __FILE__, __LINE__, TimeMessages::invalid_data_error, "\n"
159  "The TAI to UT1 conversion function requires a"
160  " data lookup table, but none has been provided.\n");
161  return;
162  }
163 
164  double tai_time = tai_ptr->trunc_julian_time; /* day local value of TAI time */
165  off_table_end = false;
166 
167  if (tai_time < when_vec[0]) {
168  // Julian date is less than initial date in table
169  if (tai_ptr->time_manager->dyn_time.scale_factor >= 0) {
170  MessageHandler::inform (
171  __FILE__, __LINE__, TimeMessages::invalid_data_error, "\n"
172  "The current time precedes the first entry in the"
173  " look-up table for TAI to UT1 conversions. \n"
174  "The conversion value is being set to the first value in the table"
175  " (%f s) .\n"
176  "Simulation will continue to run and will continue to monitor\n"
177  "time for future crossing into the tabulated range.\n",
178  val_vec[0]);
179  index = -1; // because update will look at index+1, i.e. 0
180  prev_when = tai_time - 1;
181  next_when = when_vec[0];
182  next_value = val_vec[0];
183  }
184  else {
185  MessageHandler::inform (
186  __FILE__, __LINE__, TimeMessages::invalid_data_error, "\n"
187  "The current time precedes the first entry in the"
188  " look-up table for TAI to UT1 conversions. \n"
189  "The conversion value is being set to the first value in the table"
190  " (%f s) .\n"
191  "Simulation will continue to run, and this value will be\n"
192  "maintained unless the time-direction changes.\n",
193  val_vec[0]);
194  off_table_end = true;
195  index = 0;
196  }
197  a_to_b_offset = val_vec[0] / 86400.0;
198  }
199  else if (tai_time >= when_vec[last_index]) {
200  // Julian date is beyond the table
201  index = last_index;
202  if (tai_ptr->time_manager->dyn_time.scale_factor >= 0) {
203  MessageHandler::inform (
204  __FILE__, __LINE__, TimeMessages::invalid_data_error, "\n"
205  "The current time follows the last entry in the"
206  " look-up table for TAI to UT1 conversions. \n"
207  "The conversion value is being set to the last value in the table"
208  " (%f s) .\n"
209  "Simulation will continue to run, and this value will be\n"
210  "maintained unless the time-direction changes.\n",
211  val_vec[last_index]);
212  off_table_end = true;
213  }
214  else {
215  MessageHandler::inform (
216  __FILE__, __LINE__, TimeMessages::invalid_data_error, "\n"
217  "The current time follows the last entry in the"
218  " look-up table for TAI to UT1 conversions. \n"
219  "The conversion value is being set to the last value in the table"
220  " (%f s) .\n"
221  "Simulation will continue to run and will continue to monitor\n"
222  "time for future crossing into the tabulated range.\n",
223  val_vec[last_index]);
224  prev_when = when_vec[last_index];
225  prev_value = val_vec[last_index];
226  next_when = tai_time + 1;
227  }
228  a_to_b_offset = val_vec[last_index] / 86400.0;
229  }
230  else {
231 
232  /* Find the last entry that comes before the current time (trunc_julian_time)
233  Highest value that can be reached is index = last_index - 1
234  because tai_time < when_vec[last_index] by earlier test. */
235  for (index = 0; tai_time >= when_vec[index + 1]; ++index) {
236  ;
237  }
238  prev_when = when_vec[index];
239  prev_value = val_vec[index];
240  next_when = when_vec[index + 1];
241  next_value = val_vec[index + 1];
242  gradient = (next_value - prev_value) / (next_when - prev_when);
243  a_to_b_offset =
244  (prev_value + (tai_time - prev_when) * gradient) / 86400.0;
245  }
246 
247  return;
248 }

251 /**
252  * Convert from TimeTAI to TimeUT1.
253  *
254  * \par Assumptions and Limitations
255  * - Time class MET is based on time class TAI, and counts the elapsed
256  * TAI time only
257  */
258 void
259 TimeConverter_TAI_UT1::convert_a_to_b (
260  void)
261 {
262  /* "when" is in days
263  a_to_b_offset is in seconds
264  gradient is in sec / day. */
265 
266  double tai_time = tai_ptr->trunc_julian_time; // -- local value of TAI time
267 
268  if (off_table_end || !ut1_ptr->true_ut1) { // a_to_b_offset does not change
269  ut1_ptr->set_time_by_trunc_julian (tai_ptr->trunc_julian_time +
270  a_to_b_offset);
271  // ut1_ptr->trunc_julian_time = tai_time + a_to_b_offset;
272  // ut1_ptr->days = ut1_ptr->trunc_julian_time - ut1_ptr->tjt_at_epoch;
273  // ut1_ptr->seconds = ut1_ptr->days * 86400;
274  return;
275  } // else
276  // for conventional (forward-time) simulations:
277 
278  /* "while" is used because updates need not be done regularly, may have to
279  catch up with accumulation of points. */
280  while (tai_time > next_when) {
281  index++;
282  /* Doing a full assignment at each scan because in most cases, this loop is
283  skipped; a partial assignment (index and next_when
284  only) would require the assignment of the other values after the
285  while-loop, regardless of whether the while-loop was entered. */
286  prev_value = next_value;
287  prev_when = next_when;
288 
289  if (index < last_index) {
290  next_value = val_vec[index + 1];
291  next_when = when_vec[index + 1];
292  gradient = (next_value - prev_value) / (next_when - prev_when);
293  }
294  else {
295  /* Julian date is beyond the table, set to last value */
296  MessageHandler::inform (
297  __FILE__, __LINE__, TimeMessages::invalid_data_error, "\n"
298  "Current time is beyond the range covered by the TAI-UT1 table.\n"
299  "The difference between TAI and UT1 is being set to the last value \n"
300  "in the table. The simulation will continue to run with this value."
301  "\n");
302  off_table_end = true;
303  gradient = 0.0;
304  break;
305  }
306  }
307  /* do same for unconventional (reverse-time) sims.
308  prev still references the table entry that precedes current real-time,
309  even though it will be reached later in the sim (due to reverse
310  direction). */
311  while (tai_time < prev_when) {
312  index--;
313  next_value = prev_value;
314  next_when = prev_when;
315 
316  if (index >= 0) {
317  prev_value = val_vec[index];
318  prev_when = when_vec[index];
319  gradient = (next_value - prev_value) / (next_when - prev_when);
320  }
321  else {
322  /* Julian date is less than initial date in table. */
323  MessageHandler::inform (
324  __FILE__, __LINE__, TimeMessages::invalid_data_error, "\n"
325  "Current time is beyond the range covered by the TAI-UT1 table.\n"
326  "The difference between TAI and UT1 is being set to the first value\n"
327  "in the table. The simulation will continue to run with this value."
328  "\n");
329  off_table_end = true;
330  gradient = 0.0;
331  break;
332  }
333  }
334  a_to_b_offset = (prev_value + (tai_time - prev_when) * gradient) / 86400.0;
335 
336  ut1_ptr->set_time_by_trunc_julian (tai_ptr->trunc_julian_time +
337  a_to_b_offset);
338  // ut1_ptr->trunc_julian_time = tai_time + a_to_b_offset;
339  // ut1_ptr->days = ut1_ptr->trunc_julian_time - ut1_ptr->tjt_at_epoch;
340  // ut1_ptr->seconds = ut1_ptr->days * 86400;
341 
342  return;
343 }

428  (86400.0 + gradient);

431  a_to_b_offset);

441 /**
442  * Used when time reverses direction.
443  * Checks whether the table lookup function is using input values that
444  * are outside the scope of the table, and sets the flags appropriately
445  *
446  * \par Assumptions and Limitations
447  * - None
448  */
449 void
450 TimeConverter_TAI_UT1::verify_table_lookup_ends (
451  void)
452 {
453  if (!ut1_ptr->true_ut1) { // table is not monitored
454  return;
455  }
456 
457  // if now going forward in time
458  if (tai_ptr->time_manager->dyn_time.scale_factor >= 0) {
459  // and past the end of the table, past the table
460  if (index >= last_index) {
461  off_table_end = true;
462  }
463  /* or if before the beginning of the table, no longer past the table. */
464  /* Change "prev_when" to be before "now", or the converter will
465  automatically revert to being off the table */
466  else if (index < 0) {
467  prev_when = tai_ptr->trunc_julian_time - 1;
468  next_when = when_vec[0];
469  index = -1;
470  off_table_end = false;
471  }
472  }
473  else { // reverse
474  if (index >= last_index) {
475  next_when = tai_ptr->trunc_julian_time + 1;
476  prev_when = when_vec[ last_index ];
477  index = last_index;
478  off_table_end = false;
479  }
480  else if (index < 0) {
481  off_table_end = true;
482  }
483  }
484  return;
485 }
