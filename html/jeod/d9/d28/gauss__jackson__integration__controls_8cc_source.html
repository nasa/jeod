178 unsigned int
179 GaussJacksonIntegrationControls::integrate (
180  double start_time,
181  double sim_dt,
182  er7_utils::TimeInterface & time_interface,
183  er7_utils::IntegratorInterface & integ_interface,
184  er7_utils::BaseIntegrationGroup & integ_group)
185 {
186  // Starting a new integration tour needs special processing.
187  if (step_number == 0) {
188 
189  // Reset the integrators, time if the meaning of time has changed.
190  if (reset_needed || (!Numerical::compare_exact(integ_simdt,sim_dt))) {
191 
192  // Reset integrators.
193  integ_group.reset_body_integrators ();
194  reset_integrator ();
195 
196  // Update timing information.
197  reset_time = start_time;
198  time_scale_factor = time_interface.get_time_scale_factor();
199  integ_simdt = sim_dt;
200  integ_dyndt = sim_dt * time_scale_factor;
201 
202  cycle_simdt = sim_dt * state_machine.get_cycle_scale();
203  cycle_dyndt = cycle_simdt * time_scale_factor;
204 
205  // Mark the reset as handled.
206  reset_needed = false;
207  integ_interface.restore_first_step_derivs_flag ();
208  }
209 
210  // Cache the integration tour start time.
211  integ_starttime = start_time;
212  }
213 
214  // Short-cut for operational mode.
215  if (fsm_state == GaussJacksonStateMachine::Operational) {
216 
217  // The state machine is no longer needed.
218  // All that remains is to set the cycle start time.
219  if (cycle_stage == 0) {
220  cycle_starttime = start_time;
221  }
222  // Integrate using the Gauss-Jackson predictor/corrector.
223  integrate_gj (time_interface, integ_group);
224  }
225 
226  else {
227  // Advance the state machine at start of cycle.
228  if (cycle_stage == 0) {
229  start_cycle (sim_dt);
230  }
231 
232  // Integrate per the state machine state.
233  switch (fsm_state) {
234 
235  // Priming: Let the priming controls take control.
236  case GaussJacksonStateMachine::Priming:
237  priming_controls->integrate (
238  cycle_starttime, cycle_simdt,
239  time_interface, integ_interface, integ_group);
240  cycle_stage = priming_controls->get_step_number();
241  break;
242 
243  // Edit mode: Edit one point in the collected history.
244  case GaussJacksonStateMachine::BootstrapEdit:
245  integrate_edit (time_interface, integ_group);
246  break;
247 
248  // Step and operational modes: Integrate using the predictor/corrector.
249  case GaussJacksonStateMachine::BootstrapStep:
250  case GaussJacksonStateMachine::Operational:
251  integrate_gj (time_interface, integ_group);
252  break;
253  case GaussJacksonStateMachine::Reset:
254  default:
255  assert ("Internal error" == nullptr);
256  break;
257  }
258  }
259 
260  // Set the step number, the value to be returned to the caller.
261  // Clear if the integration completed a cycle and we're at the end of an
262  // integration tour, otherwise increment.
263  if ((cycle_stage == 0) && at_end_of_tour) {
264  step_number = 0;
265  }
266  else {
267  ++step_number;
268  }
269 
270  return step_number;
271 }

274 void
275 GaussJacksonIntegrationControls::integrate_edit (
276  er7_utils::TimeInterface & time_interface,
277  er7_utils::BaseIntegrationGroup & integ_group)
278 {
279  // Integrate the bodies. Note: Edit mode is always at target stage = 1.
280  const er7_utils::IntegratorResult & integ_status =
281  integ_group.integrate_bodies (cycle_dyndt, 1);
282 
283  // The integrator should have edited the element corresponding to the
284  // state machine's history length.
285  unsigned int edit_target = state_machine.get_history_length();
286 
287  assert ((integ_status.get_merge_count() == 0) ||
288  (Numerical::compare_exact(integ_status.get_time_scale(),edit_target)));
289 
290  // Update timing info.
291  integ_simtime = cycle_starttime;
292  time_interface.update_time (integ_simtime);
293 
294  // Check whether another sequence of corrections is needed.
295  if ((edit_count < max_correction_iterations) &&
296  ! integ_status.get_passed()) {
297  state_machine.set_bootstrap_edit_redo_needed();
298  }
299 
300  // Bump the edit count if this the end of the edit sequence.
301  if (edit_target == order) {
302  ++edit_count;
303  }
304 
305  // Edit mode is always at cycle stage 0 (target stage 1).
306  cycle_stage = 0;
307 }

310 void
311 GaussJacksonIntegrationControls::integrate_gj (
312  er7_utils::TimeInterface & time_interface,
313  er7_utils::BaseIntegrationGroup & integ_group)
314 {
315  unsigned int target_stage = cycle_stage + 1;
316  const er7_utils::IntegratorResult & integ_status =
317  integ_group.integrate_bodies (cycle_dyndt, target_stage);
318 
319  assert (Numerical::compare_exact(integ_status.get_time_scale(),1.0));
320 
321  if (target_stage == 1) {
322  assert (integ_status.get_passed());
323  integ_simtime = cycle_starttime + cycle_simdt;
324  time_interface.update_time (integ_simtime);
325  if (max_correction_iterations == 0) {
326  cycle_stage = 0;
327  }
328  else {
329  cycle_stage = 1;
330  edit_count = 0;
331  }
332  }
333  else {
334  ++edit_count;
335  if (integ_status.get_passed() ||
336  (edit_count == max_correction_iterations)) {
337  cycle_stage = 0;
338  }
339  }
340 }

343 void
344 GaussJacksonIntegrationControls::start_cycle (
345  double sim_dt)
346 {
347  // Apply the state machine and get the new state.
348  state_machine.perform_step ();
349  fsm_state = state_machine.get_fsm_state();
350 
351  // Compute the start time of this integration cycle.
352  cycle_starttime = reset_time +
353  sim_dt * state_machine.get_cycle_start_time();
354  at_end_of_tour = state_machine.get_at_end_of_tour();
355 
356  if (state_machine.get_at_downsample()) {
357  cycle_simdt = sim_dt * state_machine.get_cycle_scale();
358  cycle_dyndt = cycle_simdt * time_scale_factor;
359  edit_count = 0;
360  }
361 
362  if (state_machine.get_at_order_change()) {
363  order = state_machine.get_current_order();
364  coeff.compute_coeffs (order);
365  edit_count = 0;
366  }
367 
368  if (state_machine.get_at_reinitialize()) {
369  cycle_stage = 0;
370  if (fsm_state == GaussJacksonStateMachine::BootstrapEdit) {
371  ++edit_count;
372  }
373  }
374 }
