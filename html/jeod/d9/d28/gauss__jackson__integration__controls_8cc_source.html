126 unsigned int GaussJacksonIntegrationControls::integrate(double start_time,
127  double sim_dt,
128  er7_utils::TimeInterface & time_interface,
129  er7_utils::IntegratorInterface & integ_interface,
130  er7_utils::BaseIntegrationGroup & integ_group)
131 {
132  // Starting a new integration tour needs special processing.
133  if(step_number == 0)
134  {
135  // Reset the integrators, time if the meaning of time has changed.
136  if(reset_needed || (!Numerical::compare_exact(integ_simdt, sim_dt)))
137  {
138  // Reset integrators.
139  integ_group.reset_body_integrators();
140  reset_integrator();
141 
142  // Update timing information.
143  reset_time = start_time;
144  time_scale_factor = time_interface.get_time_scale_factor();
145  integ_simdt = sim_dt;
146  integ_dyndt = sim_dt * time_scale_factor;
147 
148  cycle_simdt = sim_dt * state_machine.get_cycle_scale();
149  cycle_dyndt = cycle_simdt * time_scale_factor;
150 
151  // Mark the reset as handled.
152  reset_needed = false;
153  integ_interface.restore_first_step_derivs_flag();
154  }
155 
156  // Cache the integration tour start time.
157  integ_starttime = start_time;
158  }
159 
160  // Short-cut for operational mode.
161  if(fsm_state == GaussJacksonStateMachine::Operational)
162  {
163  // The state machine is no longer needed.
164  // All that remains is to set the cycle start time.
165  if(cycle_stage == 0)
166  {
167  cycle_starttime = start_time;
168  }
169  // Integrate using the Gauss-Jackson predictor/corrector.
170  integrate_gj(time_interface, integ_group);
171  }
172 
173  else
174  {
175  // Advance the state machine at start of cycle.
176  if(cycle_stage == 0)
177  {
178  start_cycle(sim_dt);
179  }
180 
181  // Integrate per the state machine state.
182  switch(fsm_state)
183  {
184  // Priming: Let the priming controls take control.
185  case GaussJacksonStateMachine::Priming:
186  priming_controls->integrate(cycle_starttime, cycle_simdt, time_interface, integ_interface, integ_group);
187  cycle_stage = priming_controls->get_step_number();
188  break;
189 
190  // Edit mode: Edit one point in the collected history.
191  case GaussJacksonStateMachine::BootstrapEdit:
192  integrate_edit(time_interface, integ_group);
193  break;
194 
195  // Step and operational modes: Integrate using the predictor/corrector.
196  case GaussJacksonStateMachine::BootstrapStep:
197  case GaussJacksonStateMachine::Operational:
198  integrate_gj(time_interface, integ_group);
199  break;
200  case GaussJacksonStateMachine::Reset:
201  default:
202  assert("Internal error" == nullptr);
203  break;
204  }
205  }
206 
207  // Set the step number, the value to be returned to the caller.
208  // Clear if the integration completed a cycle and we're at the end of an
209  // integration tour, otherwise increment.
210  if((cycle_stage == 0) && at_end_of_tour)
211  {
212  step_number = 0;
213  }
214  else
215  {
216  ++step_number;
217  }
218 
219  return step_number;
220 }

222 void GaussJacksonIntegrationControls::integrate_edit(er7_utils::TimeInterface & time_interface,
223  er7_utils::BaseIntegrationGroup & integ_group)
224 {
225  // Integrate the bodies. Note: Edit mode is always at target stage = 1.
226  const er7_utils::IntegratorResult & integ_status = integ_group.integrate_bodies(cycle_dyndt, 1);
227 
228  // The integrator should have edited the element corresponding to the
229  // state machine's history length.
230  unsigned int edit_target = state_machine.get_history_length();
231 
232  assert((integ_status.get_merge_count() == 0) ||
233  (Numerical::compare_exact(integ_status.get_time_scale(), edit_target)));
234 
235  // Update timing info.
236  integ_simtime = cycle_starttime;
237  time_interface.update_time(integ_simtime);
238 
239  // Check whether another sequence of corrections is needed.
240  if((edit_count < max_correction_iterations) && !integ_status.get_passed())
241  {
242  state_machine.set_bootstrap_edit_redo_needed();
243  }
244 
245  // Bump the edit count if this the end of the edit sequence.
246  if(edit_target == order)
247  {
248  ++edit_count;
249  }
250 
251  // Edit mode is always at cycle stage 0 (target stage 1).
252  cycle_stage = 0;
253 }

255 void GaussJacksonIntegrationControls::integrate_gj(er7_utils::TimeInterface & time_interface,
256  er7_utils::BaseIntegrationGroup & integ_group)
257 {
258  unsigned int target_stage = cycle_stage + 1;
259  const er7_utils::IntegratorResult & integ_status = integ_group.integrate_bodies(cycle_dyndt, target_stage);
260 
261  assert(Numerical::compare_exact(integ_status.get_time_scale(), 1.0));
262 
263  if(target_stage == 1)
264  {
265  assert(integ_status.get_passed());
266  integ_simtime = cycle_starttime + cycle_simdt;
267  time_interface.update_time(integ_simtime);
268  if(max_correction_iterations == 0)
269  {
270  cycle_stage = 0;
271  }
272  else
273  {
274  cycle_stage = 1;
275  edit_count = 0;
276  }
277  }
278  else
279  {
280  ++edit_count;
281  if(integ_status.get_passed() || (edit_count == max_correction_iterations))
282  {
283  cycle_stage = 0;
284  }
285  }

288 void GaussJacksonIntegrationControls::start_cycle(double sim_dt)
289 {
290  // Apply the state machine and get the new state.
291  state_machine.perform_step();
292  fsm_state = state_machine.get_fsm_state();
293 
294  // Compute the start time of this integration cycle.
295  cycle_starttime = reset_time + sim_dt * state_machine.get_cycle_start_time();
296  at_end_of_tour = state_machine.get_at_end_of_tour();
297 
298  if(state_machine.get_at_downsample())
299  {
300  cycle_simdt = sim_dt * state_machine.get_cycle_scale();
301  cycle_dyndt = cycle_simdt * time_scale_factor;
302  edit_count = 0;
303  }
304 
305  if(state_machine.get_at_order_change())
306  {
307  order = state_machine.get_current_order();
308  coeff.compute_coeffs(order);
309  edit_count = 0;
310  }
311 
312  if(state_machine.get_at_reinitialize())
313  {
314  cycle_stage = 0;
315  if(fsm_state == GaussJacksonStateMachine::BootstrapEdit)
316  {
317  ++edit_count;
318  }
319  }
320 }
