121 /**
122  * A Subscription object provides two approaches of marking something as
123  * being active or inactive: The activate and deactivate methods versus the
124  * subscribe and unsubscribe methods. The class also provides a mean for
125  * selecting only one of these two approaches as valid.
126  *
127  * This class uses the non-virtual interface design pattern. Derived classes
128  * should not override the non-virtual public interfaces. They should instead
129  * override the private set_active_state method.
130  */
131 class Subscription
132 {
133  JEOD_MAKE_SIM_INTERFACES(jeod, Subscription)
134 
135  // Data types
136 public:
137  /**
138  * The Subscription::Mode enum specifies the mode in which a Subscription
139  * object is operating.
140  */
141  enum Mode
142  {
143  Detect = 0, ///< First scheme used wins.
144  Subscribe = 1, ///< Activation is via subscribe/unsubscribe only.
145  Activate = 2, ///< Activation is via activate/deactivate only.
146  Freeform = 3 ///< Users can use either scheme; conflicts may arise.
147  };
148 
149  // Member functions
150 public:
151  Subscription() = default;
152  explicit Subscription(Mode);
153  virtual ~Subscription() = default;
154 
155  // Note well: The public interfaces are not virtual methods.
156  // These should not be be overridden.
157 
158  // Getters
159 
160  // is_active: Is the object active?
161  bool is_active() const;
162 
163  // subscriptions: Return the subscription count
164  unsigned int subscriptions() const;
165 
166  // get_subscription_mode: Return the object's mode.
167  Mode get_subscription_mode() const;
168 
169  // Setters
170 
171  // activate: Mark the object as active
172  void activate();
173 
174  // deactivate: Mark the object as inactive
175  void deactivate();
176 
177  // subscribe: Increment the subscription count and activate
178  void subscribe();
179 
180  // unsubscribe: Decrement the subscription count and deactivate when zero
181  void unsubscribe();
182 
183 protected:
184  // set_subscription_mode: Set the subscription mode.
185  virtual void set_subscription_mode(Mode value);
186 
187  // set_active_status: This is the low-level method that derived classes can
188  // override. The default implementation simply sets the value of the
189  // active flag. Compliant derived classes must forward a call to this
190  // method to either the parent class or to this class.
191  virtual void set_active_status(bool value);
192 
193  // Member data
194  // Note well: These data are protected; they should but cannot be private.
195  // Derived classes should use the getters to examine these data members.
196  // Simulation users should not set these members in normal simulation input
197  // files.
