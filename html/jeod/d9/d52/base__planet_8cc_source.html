88 /**
89  * Use the celestial and ecliptic poles to set the conventional fixed
90  * transformation from J2000 to alt_inertial
91  *
92  * \par Assumptions and Limitations
93  * - Method only works once
94  * - Celestial and ecliptic poles are not the same
95  * \param[in] cp celestial pole unit vector
96  * \param[in] ep Ecliptic pole unit vector
97  */
98 void
99 BasePlanet::set_alt_inertial(
100  const double cp[3],
101  const double ep[3])
102 {
103  double m[3][3];
104 
105  Vector3::normalize(cp, m[2]);
106  Vector3::cross(cp, ep, m[0]);
107  Vector3::normalize(m[0]);
108  Vector3::cross(m[2], m[0], m[1]);
109  Vector3::normalize(m[1]);
110  set_alt_inertial(m);
