72  * Use the celestial and ecliptic poles to set the conventional fixed
73  * transformation from J2000 to alt_inertial
74  *
75  * \par Assumptions and Limitations
76  * - Method only works once
77  * - Celestial and ecliptic poles are not the same
78  * \param[in] cp celestial pole unit vector

80  */
81 void BasePlanet::set_alt_inertial(const double cp[3], const double ep[3])
82 {
83  double m[3][3];
84 
85  Vector3::normalize(cp, m[2]);
86  Vector3::cross(cp, ep, m[0]);
87  Vector3::normalize(m[0]);
88  Vector3::cross(m[2], m[0], m[1]);
89  Vector3::normalize(m[1]);
