59 {

61  // pointers to this frame. Sever those links by making each child a root.

64  MassPointLinks * child_links = links.child_tail();
65  child_links->detach();
66  }
67 
68  // Sever the links from the parent and sibling nodes as well.
69  links.detach();

81 /**
82  * Compute the state of this point with respect to some reference
83  * point, which may be any point in the same tree as this point.
84  * \param[in] ref_point The point with respect to which the state is to be expressed
85  * \param[out] rel_state The relative state
86  */
87 void MassPoint::compute_relative_state(const MassPoint & ref_point, MassPointState & rel_state) const
88 {
89  int common_node_index; /* Index of last node in common
90  between this point and the
91  ref_point */
92  const MassPoint * common_node_point; /* Last common point */
93 
94  // Determine the point of intersection in the paths to the root node
95  // between this point and the reference point.
96  common_node_index = find_last_common_index(ref_point);
97  common_node_point = links.nth_from_root(common_node_index);
98 
99  // A negative number indicates a *serious* problem.
100  if(common_node_index < 0)
101  {
102  MessageHandler::fail(__FILE__,
103  __LINE__,
104  MassBodyMessages::invalid_node,
105  "Mass points are not in the same tree");
106 
107  // Not reached
108  return;
109  }
110 
111  // There are four cases to cover:
112  // 1. The trivial case: The ref_point is this point.
113  // 2. The ref_point is a direct predecessor of this point.
114  // 3. This point is a direct predecessor of the ref_point.
115  // 4. General case.
116 
117  // Trivial case.
118  if(&ref_point == this)
119  {
120  rel_state.initialize_mass_point();
121  }
122 
123  // Case 2: The ref_point is a direct predecessor of this point.
124  else if(common_node_point == &ref_point)
125  {
126  compute_state_wrt_pred(common_node_index, rel_state);
127  }
128 
129  // Case 3: This point is a direct predecessor of the ref_point.
130  else if(common_node_point == this)
131  {
132  ref_point.compute_pred_rel_state(*this, rel_state);
133  }
134 
135  // General case: Nodes descend from some common node.
136  else
137  {
138  // Compute this point's state wrt the common point.
139  compute_state_wrt_pred(common_node_index, rel_state);
140 
141  // Move the reference down to the ref_point
142  for(auto * link : TreeLinksDescentRange<const MassPointLinks>(ref_point.links, common_node_index + 1))
143  {
144  rel_state.decr_left(link->container());
145  }
146  }
147 }

175 /**
176  * Compute the state of this point with respect to some reference
177  * point, which must be at or above this point via the parent links.
178  * \param[in] ref_point_index Reference point index
179  * \param[out] rel_state Point state wrt ref. pt.
180  */
181 void MassPoint::compute_state_wrt_pred(unsigned int ref_point_index, MassPointState & rel_state) const
182 {
183  // The state of a point with respect to itself is the trivial state.
184  if(ref_point_index == links.path_length() - 1)
185  {
186  rel_state.initialize_mass_point();
187  }
188 
189  // Child body: Walk up the point tree, accumulating offsets and transforms.
190  else
191  {
192  double r_temp[3]; // M Transformed point location
193 
194  // Copy the current state to the relative state.
195  // r_B->C:B = vector from frame B origin to frame C origin
196  // T_B:C = transform from frame B to frame C
197  // Frame B = parent, frame C = point frame
198  Vector3::copy(position, rel_state.position);
199  Matrix3x3::copy(T_parent_this, rel_state.T_parent_this);
200  rel_state.Q_parent_this = Q_parent_this;
201 
202  // Walk up the tree, accumulating offsets and transforms.
203  for(auto * link :
204  TreeLinksAscendRange<const MassPointLinks>(links, links.path_length() - 1, ref_point_index + 1))
205  {
206  const MassPoint & node = link->container();
207 
208  // r_A->C:A = r_A->B:A + T_A:B^T * r_B:C:B
209  Vector3::transform_transpose(node.T_parent_this, rel_state.position, r_temp);
210  Vector3::sum(node.position, r_temp, rel_state.position);
211 
212  // T_A:C = T_B:C * T_A:B
213  // Q_A:C = Q_B:C * Q_A:B
214  rel_state.Q_parent_this.multiply(node.Q_parent_this);
215  rel_state.Q_parent_this.normalize();
216  rel_state.Q_parent_this.left_quat_to_transformation(rel_state.T_parent_this);
217  }
218  }
219 }

251 /**
252  * Compute the state of some reference point with respect to this
253  * point. The reference point must be at or above this point via the
254  * parent links.
255  * \param[in] ref_point_index Reference point index
256  * \param[out] rel_state Relative state
257  */
258 void MassPoint::compute_pred_rel_state(unsigned int ref_point_index, MassPointState & rel_state) const
259 {
260  // The state of a root point wrt itself is the trivial state.
261  if(ref_point_index == links.path_length() - 1)
262  {
263  rel_state.initialize_mass_point();
264  }
265 
266  // Child body: Walk up the point tree, accumulating offsets and transforms.
267  else
268  {
269  // Reverse the current state to form the parent state wrt the point.
270  // T_C:B = T_B:C^T
271  // r_C->B:C = - r_B->C:C = - T_B:C * r_B->C:B
272  Vector3::transform(T_parent_this, position, rel_state.position);
273  Vector3::negate(rel_state.position);
274  Matrix3x3::transpose(T_parent_this, rel_state.T_parent_this);
275  Q_parent_this.conjugate(rel_state.Q_parent_this);
276 
277  // Walk up the tree, accumulating offsets and transforms.
278  for(auto * link :
279  TreeLinksAscendRange<const MassPointLinks>(links, links.path_length() - 1, ref_point_index + 1))
280  {
281  const MassPoint & node = link->container();
282 
283  // T_C:A = T_B:A * T_C:B = T_A:B^T * T_C:B
284  // Q_C:A = Q_B:A * Q_C:B = Q_A:B^T * Q_C:B
285  rel_state.Q_parent_this.multiply_left_conjugate(node.Q_parent_this);
286  rel_state.Q_parent_this.normalize();
287  rel_state.Q_parent_this.left_quat_to_transformation(rel_state.T_parent_this);
288 
289  // r_C->A:C = r_C->B:C + r_B->A:C
290  // = r_C->B:C - r_A->B:C
291  // = r_C->B:C - T_A:C r_A->B:A
292  Vector3::transform_transpose_decr(rel_state.T_parent_this, node.position, rel_state.position);
293  }
294  }
295 }
