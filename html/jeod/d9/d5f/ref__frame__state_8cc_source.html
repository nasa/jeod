41 /*******************************************************************************
42 
43  Nomenclature of mathematical formulations.
44  x,v,Q,T,w = position, velocity, quaternion, transform, rate, respectively.
45  A,B,C = various reference frames.
46 
47 
48  Long-hand notations.
49  x_A->B:C = Position of frame B origin with respect to frame A origin,
50  expressed in frame A coordinates.
51  v_A->B:C = Velocity of frame B origin with respect to frame A origin,
52  expressed in frame C (observer is fixed wrt frame C).
53  w_A->B:C = Angular velocity of frame B axes wrt frame A axes,
54  expressed in frame C (observer is fixed wrt frame C).
55  T_A->B = Column vector transformation matrix from frame A to frame B.
56 
57 
58  Short-hand notations.
59  x_A:B = x_A->B:A (Location of frame B origin in frame A coordinates.)
60  v_A:B = v_A->B:A
61  T_A:B = T_A->B
62  w_A:B = w_A->B:B (Note that w_A:B is expressed in frame B coordinates.)
63  S_A:B = {x_A:B, V_A:B, T_A:B, w_A:B}
64  In other words, the state of frame with respect to frame B.
65 
66 
67  Operators.
68  Conceptually, there are three primitive operators on reference frame states:
69  assignment (=), unary negation (-), and addition (+). Note that addition
70  is nonabelian. Subtraction is in terms of addition and negation.
71 
72  Negation: S_B:A = - S_A:B
73  T_B:A = T_A:B^T
74  w_B:A = - (T_B:A * w_A:B)
75  x_B:A = - (T_A:B * x_A:B)
76  v_B:A = -(T_A:B * v_A:B + w_A:B X x_B:A)
77 
78  Addition: S_A:C = S_A:B + S_B:C
79  T_A:C = T_B:C * T_A:B
80  w_A:C = T_B:C * w_A:B + w_B:C
81  x_A:C = x_A:B + T_A:B^T * x_B:C
82  v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
83 
84 
85  Methods.
86  Method This(input) Operand Result
87  negate N/A S_A:B this <- -S_A:B = S_B:A
88  incr_left S_B:C S_A:B this <- arg + this = S_A:C
89  incr_right S_A:B S_B:C this <- this + arg = S_A:C
90  decr_left S_A:C S_A:B this <- (-arg) + this = S_B:C
91  decr_right S_A:C S_B:C this <- this + (-arg) = S_A:B
92 
93 
94  Shortcuts.
95  The methods defined below two shortcuts to bypass various calculations.
96 
97  - Identity transformation.
98  There is no need to transform a vector if the transformation is an identity.
99  The scalar part of the left transformation quaternion is exactly one in the
100  case of an identity transformation.
101 
102  - Null rotation.
103  This is no need to calculate wXr and the like if the angular velocity is
104  identically zero. The magnitude of the angular velocity is used to detect
105  whether this is the case.

174 /**
175  * Copy a reference frame state, negated.
176  * \param[in] source Source state
177  */
178 void RefFrameState::negate(const RefFrameState & source)
179 {
180  // Relevant equations:
181  // T_B:A = T_A:B^T
182  // w_B:A = - (T_B:A * w_A:B)
183  // x_B:A = - (T_A:B * x_A:B)
184  // v_B:A = -(T_A:B * v_A:B + w_A:B X x_B:A)
185 
186  // Transform the matrix and translational state to the alternate frame.
187  // This only needs to be performed for nontrivial transformation matrices.
188  if(!Numerical::compare_exact(source.rot.Q_parent_this.scalar, 1.0))
189  {
190  source.rot.Q_parent_this.conjugate(rot.Q_parent_this);
191  Matrix3x3::transpose(source.rot.T_parent_this, rot.T_parent_this);
192  Vector3::transform_transpose(rot.T_parent_this, source.trans.position, trans.position);
193  Vector3::transform_transpose(rot.T_parent_this, source.trans.velocity, trans.velocity);
194  }
195 
196  // Identity transform: Just copy.
197  else
198  {
199  rot.initialize();
200  Vector3::copy(source.trans.position, trans.position);
201  Vector3::copy(source.trans.velocity, trans.velocity);
202  }
203 
204  // Compute the wXr velocity component and transform the angular velocity.
205  // This only needs to be performed for non-zero angular velocities.
206  if(std::fpclassify(source.rot.ang_vel_mag) != FP_ZERO)
207  {
208  Vector3::cross_decr(source.rot.ang_vel_this, trans.position, trans.velocity);
209 
210  Vector3::transform(rot.T_parent_this, source.rot.ang_vel_this, rot.ang_vel_this);
211  }
212 
213  // Zero rate: Set the new rate to zero (the unit vector is ill-defined).
214  else
215  {
216  rot.ang_vel_mag = 0.0;
217  Vector3::initialize(rot.ang_vel_unit);
218  Vector3::initialize(rot.ang_vel_this);
219  }
220 
221  // The above made the vectors have the wrong sign. Fix that.
222  Vector3::negate(trans.position);
223  Vector3::negate(trans.velocity);
224  Vector3::negate(rot.ang_vel_this);
225 }

227 /**
228  * Compute S_A:C = S_A:B + S_B:C,
229  * with this initially containing S_B:C,
230  * the supplied argument containing S_A:B, and
231  * the resultant composition of states stored in this.
232  * \param[in] s_ab Left addend
233  */
234 void RefFrameState::incr_left(const RefFrameState & s_ab)
235 {
236  // Relevant equations:
237  // T_A:C = T_B:C * T_A:B
238  // w_A:C = T_B:C * w_A:B + w_B:C
239  // x_A:C = x_A:B + T_A:B^T * x_B:C
240  // v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
241 
242  // Order dependency:
243  // Quantity Uses S_B:C items Uses S_A:C items Order
244  // T_A:C T None 2
245  // w_A:C w, T None 1
246  // x_A:C x None 4
247  // v_A:C v, x None 3
248 
249  // Compute the angular velocity of frame C wrt frame A in frame C:
250  // w_A:C = T_B:C * w_A:B + w_B:C
251  // Shortcut: Don't bother if s_ab is not rotating wrt its parent.
252  if(std::fpclassify(s_ab.rot.ang_vel_mag) != FP_ZERO)
253  {
254  // Compute as this <- this + T_B:C * w_A:B
255  Vector3::transform_incr(rot.T_parent_this, s_ab.rot.ang_vel_this, rot.ang_vel_this);
256 
257  // Compute the angular velocity magnitude and unit vector.
258  rot.compute_ang_vel_products();
259  }
260 
261  // Compute T_A:C, v_A:C, and x_A:C via
262  // T_A:C = T_B:C * T_A:B
263  // x_A:C = x_A:B + T_A:B^T * x_B:C
264  // v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
265  // Shortcut: only do the transformations if T_A:B is not the identity matrix.
266  if(!Numerical::compare_exact(s_ab.rot.Q_parent_this.scalar, 1.0))
267  {
268  // Compute Q_A:C = Q_B:C * Q_A:B (and then normalize result)
269  rot.Q_parent_this.multiply(s_ab.rot.Q_parent_this);
270  rot.Q_parent_this.normalize();
271 
272  // Compute the corresponding transformation matrix.
273  rot.compute_transformation();
274 
275  // Compute v_B->C:A = T_A:B^T * (v_B:C + w_a:B X x_B:C)
276  // Note: This must be performed prior to computing the position
277  // because it uses x_B:C = trans.position.
278  if(std::fpclassify(s_ab.rot.ang_vel_mag) != FP_ZERO)
279  {
280  Vector3::cross_incr(s_ab.rot.ang_vel_this, trans.position, trans.velocity);
281  }
282 
283  // Transform x_B:C to frame A.
284  Vector3::transform_transpose(s_ab.rot.T_parent_this, trans.position);
285 
286  Vector3::transform_transpose(s_ab.rot.T_parent_this, trans.velocity);
287  }
288 
289  // T_A:B is identity. The only quantity needed is w_A:B X x_B:C.
290  else
291  {
292  if(std::fpclassify(s_ab.rot.ang_vel_mag) != FP_ZERO)
293  {
294  Vector3::cross_incr(s_ab.rot.ang_vel_this, trans.position, trans.velocity);
295  }
296  }
297 
298  // Add the X_A:B, v_A:B to the state.
299  Vector3::incr(s_ab.trans.velocity, trans.velocity);
300  Vector3::incr(s_ab.trans.position, trans.position);
301 }

303 /**
304  * Compute S_A:C = S_A:B + S_B:C,
305  * with this initially containing S_A:B,
306  * the supplied argument containing S_B:C, and
307  * the resultant composition of states stored in this.
308  * Note that this function is untested, as it is not
309  * used in the Reference Frame Model at any point, and
310  * is only given here as a utility function.
311  * \param[in] s_bc Right addend
312  */
313 void RefFrameState::incr_right(const RefFrameState & s_bc)
314 {
315  // Relevant equations:
316  // T_A:C = T_B:C * T_A:B
317  // w_A:C = T_B:C * w_A:B + w_B:C
318  // x_A:C = x_A:B + T_A:B^T * x_B:C
319  // v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
320 
321  // Order dependency:
322  // Quantity Uses S_A:B items Uses S_A:C items Order
323  // T_A:C T None 5
324  // w_A:C w None 2
325  // x_A:C x, T None 3
326  // v_A:C v, T, w None 1,4
327 
328  double v_bc_in_b_obs_a[3]; // M/s Vel. of C wrt B in B, observed from A
329 
330  // The velocity of frame C origin wrt frame B as observed in frame A
331  // includes the velocity itself plus and omega cross r term.
332  // The velocity term does not depend on the frame rate: copy it.
333  Vector3::copy(s_bc.trans.velocity, v_bc_in_b_obs_a);
334 
335  // Compute
336  // 1. The velocity of frame C wrt frame B observed from frame A
337  // v_B->C:B(A) = v_B:C + w_A:B X x_B:C
338  // 2. The angular velocity of frame C wrt frame A in frame C:
339  // w_A:C = T_B:C * w_A:B + w_B:C
340  // Shortcut: Don't bother if frame B is not rotating wrt frame A.
341  if(std::fpclassify(rot.ang_vel_mag) != FP_ZERO)
342  {
343  // Add the omega cross r term to the observed velocity.
344  Vector3::cross_incr(rot.ang_vel_this, s_bc.trans.position, v_bc_in_b_obs_a);
345 
346  // Compute w_A:C as this = T_B:C * this + w_B:C
347  Vector3::transform(s_bc.rot.T_parent_this, rot.ang_vel_this);
348  Vector3::incr(s_bc.rot.ang_vel_this, rot.ang_vel_this);
349 
350  // Compute the angular velocity magnitude and unit vector.
351  rot.compute_ang_vel_products();
352  }
353 
354  // Compute x_A:C, v_A:C, and T_A:C via
355  // T_A:C = T_B:C * T_A:B
356  // x_A:C = x_A:B + T_A:B^T * x_B:C
357  // v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
358  // Shortcut: only do the transformations if T_A:B is not the identity matrix.
359  if(!Numerical::compare_exact(rot.Q_parent_this.scalar, 1.0))
360  {
361  // Compute x_A:C and v_A:C.
362  // Note: These must be computed before computing T_A:C as they use T_A:B.
363  Vector3::transform_transpose_incr(rot.T_parent_this, s_bc.trans.position, trans.position);
364  Vector3::transform_transpose_incr(rot.T_parent_this, v_bc_in_b_obs_a, trans.velocity);
365 
366  // Compute Q_A:C = Q_B:C * Q_A:B (and then normalize result)
367  rot.Q_parent_this.multiply_left(s_bc.rot.Q_parent_this);
368  rot.Q_parent_this.normalize();
369 
370  // Compute the corresponding transformation matrix.
371  rot.compute_transformation();
372  }
373 
374  // Shortcuts for the case T_A:B is identity.
375  else
376  {
377  Vector3::incr(s_bc.trans.position, trans.position);
378  Vector3::incr(s_bc.trans.velocity, trans.velocity);
379  rot.copy(s_bc.rot);
380  }
381 }

383 /**
384  * Compute S_B:C = (-S_A:B) + S_A:C,
385  * with this initially containing S_A:C,
386  * the supplied argument containing S_A:B, and
387  * the resultant composition of states stored in this.
388  * \param[in] s_ab Left addend
389  */
390 void RefFrameState::decr_left(const RefFrameState & s_ab)
391 {
392  // Relevant equations:
393  // T_B:C = T_A:C * T_A:B^T
394  // w_B:C = w_A:C - T_B:C * w_A:B
395  // x_B:C = T_A:B * (x_A:C - x_A:B)
396  // v_B:C = T_A:B * (v_A:C - v_A:B) - w_A:B X x_B:C
397 
398  // Order dependency:
399  // Quantity Uses S_A:C items Uses S_B:C items Order
400  // T_B:C T None 1
401  // w_B:C w T 2
402  // x_B:C x None 3
403  // v_B:C v x 4
404 
405  // Compute T_B:C, v_B:C, and x_B:C via
406  // T_B:C = T_A:C * T_A:B^T
407  // x_B:C = T_A:B * (x_A:C - x_A:B)
408  // v_B:C = T_A:B * (v_A:C - v_A:B) - w_A:B X x_B:C
409  // Shortcut: only do the transformations if T_A:B is not the identity matrix.
410  if(!Numerical::compare_exact(s_ab.rot.Q_parent_this.scalar, 1.0))
411  {
412  // Compute Q_B:C = Q_A:C * Q_A:B^Q
413  rot.Q_parent_this.multiply_conjugate(s_ab.rot.Q_parent_this);
414  rot.Q_parent_this.normalize();
415 
416  // Compute the corresponding transformation matrix.
417  rot.compute_transformation();
418 
419  // Compute x_B:C and v_B:C.
420  // Note that x_B:C must be computed before v_B:C as V_B:C depends on
421  // the former.
422  Vector3::decr(s_ab.trans.position, trans.position);
423  Vector3::transform(s_ab.rot.T_parent_this, trans.position);
424 
425  Vector3::decr(s_ab.trans.velocity, trans.velocity);
426  Vector3::transform(s_ab.rot.T_parent_this, trans.velocity);
427  Vector3::cross_decr(s_ab.rot.ang_vel_this, trans.position, trans.velocity);
428  }
429 
430  // Shortcuts for the case T_A:B is identity.
431  else
432  {
433  Vector3::decr(s_ab.trans.position, trans.position);
434  Vector3::decr(s_ab.trans.velocity, trans.velocity);
435  Vector3::cross_decr(s_ab.rot.ang_vel_this, trans.position, trans.velocity);
436  }
437 
438  // Compute w_B:C = w_A:C - T_B:C * w_A:B and related products.
439  if(std::fpclassify(s_ab.rot.ang_vel_mag) != FP_ZERO)
440  {
441  Vector3::transform_decr(rot.T_parent_this, s_ab.rot.ang_vel_this, rot.ang_vel_this);
442  rot.compute_ang_vel_products();
443  }
444 }

446 /**
447  * Compute S_A:B = S_A:C + (-S_B:C)
448  * with this initially containing S_A:C,
449  * the supplied argument containing S_B:C, and
450  * the resultant composition of states stored in this.
451  * \param[in] s_bc Left addend
452  */
453 void RefFrameState::decr_right(const RefFrameState & s_bc)
454 {
455  // Relevant equations:
456  // T_A:B = T_B:C^T * T_A:C
457  // w_A:B = T_B:C^T * (w_A:C - w_B:C)
458  // x_A:B = x_A:C - T_A:B^T * x_B:C
459  // v_A:B = v_A:C - T_A:B^T * (v_B:C + w_A:B X x_B:C)
460 
461  // Order dependency:
462  // Quantity Uses S_A:C items Uses S_A:B items Order
463  // T_A:B T None 1
464  // w_A:B w None 2
465  // x_A:B x T 3
466  // v_A:B v T, w 4
467 
468  double v_bc_in_b_obs_a[3]; // M/s Vel. of C wrt B in B, observed from A
469 
470  // Compute T_A:B, w_A:B, via
471  // T_A:B = T_B:C^T * T_A:C
472  // w_A:B = T_B:C^T * (w_A:C - w_B:C)
473  // Shortcut: only do the transformations if T_B:C is not the identity matrix.
474  if(!Numerical::compare_exact(s_bc.rot.Q_parent_this.scalar, 1.0))
475  {
476  rot.Q_parent_this.multiply_left_conjugate(s_bc.rot.Q_parent_this);
477  rot.Q_parent_this.normalize();
478  rot.compute_transformation();
479 
480  Vector3::decr(s_bc.rot.ang_vel_this, rot.ang_vel_this);
481  Vector3::transform_transpose(s_bc.rot.T_parent_this, rot.ang_vel_this);
482  rot.compute_ang_vel_products();
483  }
484 
485  // Shortcuts for the case T_B:C is identity.
486  else if(std::fpclassify(s_bc.rot.ang_vel_mag) != FP_ZERO)
487  {
488  Vector3::decr(s_bc.rot.ang_vel_this, rot.ang_vel_this);
489  rot.compute_ang_vel_products();
490  }
491 
492  // Compute the velocity of frame C wrt frame B observed from frame A
493  // v_B->C:B(A) = v_B:C + w_A:B X x_B:C
494  Vector3::copy(s_bc.trans.velocity, v_bc_in_b_obs_a);
495  if(std::fpclassify(rot.ang_vel_mag) != FP_ZERO)
496  {
497  Vector3::cross_incr(rot.ang_vel_this, s_bc.trans.position, v_bc_in_b_obs_a);
498  }
499 
500  // Compute x_A:B, V_A:B via
501  // x_A:B = x_A:C - T_A:B^T * x_B:C
502  // v_A:B = v_A:C - T_A:B^T * (v_B:C + w_A:B X x_B:C)
503  if(!Numerical::compare_exact(rot.Q_parent_this.scalar, 1.0))
504  {
505  Vector3::transform_transpose_decr(rot.T_parent_this, s_bc.trans.position, trans.position);
506  Vector3::transform_transpose_decr(rot.T_parent_this, v_bc_in_b_obs_a, trans.velocity);
507  }
508  // Shortcuts for the case T_A:B is identity.
509  else
510  {
511  Vector3::decr(s_bc.trans.position, trans.position);
512  Vector3::decr(v_bc_in_b_obs_a, trans.velocity);
513  }
514 }
