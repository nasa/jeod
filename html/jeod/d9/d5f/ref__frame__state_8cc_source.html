42 /*******************************************************************************
43 
44  Nomenclature of mathematical formulations.
45  x,v,Q,T,w = position, velocity, quaternion, transform, rate, respectively.
46  A,B,C = various reference frames.
47 
48 
49  Long-hand notations.
50  x_A->B:C = Position of frame B origin with respect to frame A origin,
51  expressed in frame A coordinates.
52  v_A->B:C = Velocity of frame B origin with respect to frame A origin,
53  expressed in frame C (observer is fixed wrt frame C).
54  w_A->B:C = Angular velocity of frame B axes wrt frame A axes,
55  expressed in frame C (observer is fixed wrt frame C).
56  T_A->B = Column vector transformation matrix from frame A to frame B.
57 
58 
59  Short-hand notations.
60  x_A:B = x_A->B:A (Location of frame B origin in frame A coordinates.)
61  v_A:B = v_A->B:A
62  T_A:B = T_A->B
63  w_A:B = w_A->B:B (Note that w_A:B is expressed in frame B coordinates.)
64  S_A:B = {x_A:B, V_A:B, T_A:B, w_A:B}
65  In other words, the state of frame with respect to frame B.
66 
67 
68  Operators.
69  Conceptually, there are three primitive operators on reference frame states:
70  assignment (=), unary negation (-), and addition (+). Note that addition
71  is nonabelian. Subtraction is in terms of addition and negation.
72 
73  Negation: S_B:A = - S_A:B
74  T_B:A = T_A:B^T
75  w_B:A = - (T_B:A * w_A:B)
76  x_B:A = - (T_A:B * x_A:B)
77  v_B:A = -(T_A:B * v_A:B + w_A:B X x_B:A)
78 
79  Addition: S_A:C = S_A:B + S_B:C
80  T_A:C = T_B:C * T_A:B
81  w_A:C = T_B:C * w_A:B + w_B:C
82  x_A:C = x_A:B + T_A:B^T * x_B:C
83  v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
84 
85 
86  Methods.
87  Method This(input) Operand Result
88  negate N/A S_A:B this <- -S_A:B = S_B:A
89  incr_left S_B:C S_A:B this <- arg + this = S_A:C
90  incr_right S_A:B S_B:C this <- this + arg = S_A:C
91  decr_left S_A:C S_A:B this <- (-arg) + this = S_B:C
92  decr_right S_A:C S_B:C this <- this + (-arg) = S_A:B
93 
94 
95  Shortcuts.
96  The methods defined below two shortcuts to bypass various calculations.
97 
98  - Identity transformation.
99  There is no need to transform a vector if the transformation is an identity.
100  The scalar part of the left transformation quaternion is exactly one in the
101  case of an identity transformation.
102 
103  - Null rotation.
104  This is no need to calculate wXr and the like if the angular velocity is
105  identically zero. The magnitude of the angular velocity is used to detect
106  whether this is the case.

190 /**
191  * Copy a reference frame state, negated.
192  * \param[in] source Source state
193  */
194 void
195 RefFrameState::negate (
196  const RefFrameState & source)
197 {
198  // Relevant equations:
199  // T_B:A = T_A:B^T
200  // w_B:A = - (T_B:A * w_A:B)
201  // x_B:A = - (T_A:B * x_A:B)
202  // v_B:A = -(T_A:B * v_A:B + w_A:B X x_B:A)
203 
204  // Transform the matrix and translational state to the alternate frame.
205  // This only needs to be performed for nontrivial transformation matrices.
206  if (!Numerical::compare_exact(source.rot.Q_parent_this.scalar,1.0)) {
207  source.rot.Q_parent_this.conjugate (rot.Q_parent_this);
208  Matrix3x3::transpose (source.rot.T_parent_this, rot.T_parent_this);
209  Vector3::transform_transpose (rot.T_parent_this, source.trans.position,
210  trans.position);
211  Vector3::transform_transpose (rot.T_parent_this, source.trans.velocity,
212  trans.velocity);
213  }
214 
215  // Identity transform: Just copy.
216  else {
217  rot.initialize();
218  Vector3::copy (source.trans.position, trans.position);
219  Vector3::copy (source.trans.velocity, trans.velocity);
220  }
221 
222  // Compute the wXr velocity component and transform the angular velocity.
223  // This only needs to be performed for non-zero angular velocities.
224  if (std::fpclassify(source.rot.ang_vel_mag) != FP_ZERO) {
225 
226  Vector3::cross_decr (source.rot.ang_vel_this, trans.position,
227  trans.velocity);
228 
229  Vector3::transform (rot.T_parent_this, source.rot.ang_vel_this,
230  rot.ang_vel_this);
231  }
232 
233  // Zero rate: Set the new rate to zero (the unit vector is ill-defined).
234  else {
235  rot.ang_vel_mag = 0.0;
236  Vector3::initialize (rot.ang_vel_unit);
237  Vector3::initialize (rot.ang_vel_this);
238  }
239 
240  // The above made the vectors have the wrong sign. Fix that.
241  Vector3::negate (trans.position);
242  Vector3::negate (trans.velocity);
243  Vector3::negate (rot.ang_vel_this);
244 
245  return;
246 }

249 /**
250  * Compute S_A:C = S_A:B + S_B:C,
251  * with this initially containing S_B:C,
252  * the supplied argument containing S_A:B, and
253  * the resultant composition of states stored in this.
254  * \param[in] s_ab Left addend
255  */
256 void
257 RefFrameState::incr_left (
258  const RefFrameState & s_ab)
259 {
260  // Relevant equations:
261  // T_A:C = T_B:C * T_A:B
262  // w_A:C = T_B:C * w_A:B + w_B:C
263  // x_A:C = x_A:B + T_A:B^T * x_B:C
264  // v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
265 
266  // Order dependency:
267  // Quantity Uses S_B:C items Uses S_A:C items Order
268  // T_A:C T None 2
269  // w_A:C w, T None 1
270  // x_A:C x None 4
271  // v_A:C v, x None 3
272 
273 
274  // Compute the angular velocity of frame C wrt frame A in frame C:
275  // w_A:C = T_B:C * w_A:B + w_B:C
276  // Shortcut: Don't bother if s_ab is not rotating wrt its parent.
277  if (std::fpclassify(s_ab.rot.ang_vel_mag) != FP_ZERO) {
278 
279  // Compute as this <- this + T_B:C * w_A:B
280  Vector3::transform_incr (rot.T_parent_this, s_ab.rot.ang_vel_this,
281  rot.ang_vel_this);
282 
283  // Compute the angular velocity magnitude and unit vector.
284  rot.compute_ang_vel_products ();
285  }
286 
287  // Compute T_A:C, v_A:C, and x_A:C via
288  // T_A:C = T_B:C * T_A:B
289  // x_A:C = x_A:B + T_A:B^T * x_B:C
290  // v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
291  // Shortcut: only do the transformations if T_A:B is not the identity matrix.
292  if (!Numerical::compare_exact(s_ab.rot.Q_parent_this.scalar,1.0)) {
293 
294  // Compute Q_A:C = Q_B:C * Q_A:B (and then normalize result)
295  rot.Q_parent_this.multiply (s_ab.rot.Q_parent_this);
296  rot.Q_parent_this.normalize ();
297 
298  // Compute the corresponding transformation matrix.
299  rot.compute_transformation ();
300 
301  // Compute v_B->C:A = T_A:B^T * (v_B:C + w_a:B X x_B:C)
302  // Note: This must be performed prior to computing the position
303  // because it uses x_B:C = trans.position.
304  if (std::fpclassify(s_ab.rot.ang_vel_mag) != FP_ZERO) {
305  Vector3::cross_incr (s_ab.rot.ang_vel_this, trans.position,
306  trans.velocity);
307  }
308 
309  // Transform x_B:C to frame A.
310  Vector3::transform_transpose (s_ab.rot.T_parent_this,
311  trans.position);
312 
313  Vector3::transform_transpose (s_ab.rot.T_parent_this,
314  trans.velocity);
315  }
316 
317  // T_A:B is identity. The only quantity needed is w_A:B X x_B:C.
318  else {
319  if (std::fpclassify(s_ab.rot.ang_vel_mag) != FP_ZERO) {
320  Vector3::cross_incr (s_ab.rot.ang_vel_this, trans.position,
321  trans.velocity);
322  }
323  }
324 
325  // Add the X_A:B, v_A:B to the state.
326  Vector3::incr (s_ab.trans.velocity, trans.velocity);
327  Vector3::incr (s_ab.trans.position, trans.position);
328 
329  return;
330 }

333 /**
334  * Compute S_A:C = S_A:B + S_B:C,
335  * with this initially containing S_A:B,
336  * the supplied argument containing S_B:C, and
337  * the resultant composition of states stored in this.
338  * Note that this function is untested, as it is not
339  * used in the Reference Frame Model at any point, and
340  * is only given here as a utility function.
341  * \param[in] s_bc Right addend
342  */
343 void
344 RefFrameState::incr_right (
345  const RefFrameState & s_bc)
346 {
347  // Relevant equations:
348  // T_A:C = T_B:C * T_A:B
349  // w_A:C = T_B:C * w_A:B + w_B:C
350  // x_A:C = x_A:B + T_A:B^T * x_B:C
351  // v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
352 
353  // Order dependency:
354  // Quantity Uses S_A:B items Uses S_A:C items Order
355  // T_A:C T None 5
356  // w_A:C w None 2
357  // x_A:C x, T None 3
358  // v_A:C v, T, w None 1,4
359 
360 
361  double v_bc_in_b_obs_a[3]; // M/s Vel. of C wrt B in B, observed from A
362 
363 
364  // The velocity of frame C origin wrt frame B as observed in frame A
365  // includes the velocity itself plus and omega cross r term.
366  // The velocity term does not depend on the frame rate: copy it.
367  Vector3::copy (s_bc.trans.velocity, v_bc_in_b_obs_a);
368 
369  // Compute
370  // 1. The velocity of frame C wrt frame B observed from frame A
371  // v_B->C:B(A) = v_B:C + w_A:B X x_B:C
372  // 2. The angular velocity of frame C wrt frame A in frame C:
373  // w_A:C = T_B:C * w_A:B + w_B:C
374  // Shortcut: Don't bother if frame B is not rotating wrt frame A.
375  if (std::fpclassify(rot.ang_vel_mag) != FP_ZERO) {
376 
377  // Add the omega cross r term to the observed velocity.
378  Vector3::cross_incr (rot.ang_vel_this, s_bc.trans.position,
379  v_bc_in_b_obs_a);
380 
381  // Compute w_A:C as this = T_B:C * this + w_B:C
382  Vector3::transform (s_bc.rot.T_parent_this, rot.ang_vel_this);
383  Vector3::incr (s_bc.rot.ang_vel_this, rot.ang_vel_this);
384 
385  // Compute the angular velocity magnitude and unit vector.
386  rot.compute_ang_vel_products ();
387  }
388 
389  // Compute x_A:C, v_A:C, and T_A:C via
390  // T_A:C = T_B:C * T_A:B
391  // x_A:C = x_A:B + T_A:B^T * x_B:C
392  // v_A:C = v_A:B + T_A:B^T * (v_B:C + w_A:B X x_B:C)
393  // Shortcut: only do the transformations if T_A:B is not the identity matrix.
394  if (!Numerical::compare_exact(rot.Q_parent_this.scalar,1.0)) {
395 
396  // Compute x_A:C and v_A:C.
397  // Note: These must be computed before computing T_A:C as they use T_A:B.
398  Vector3::transform_transpose_incr (rot.T_parent_this, s_bc.trans.position,
399  trans.position);
400  Vector3::transform_transpose_incr (rot.T_parent_this, v_bc_in_b_obs_a,
401  trans.velocity);
402 
403  // Compute Q_A:C = Q_B:C * Q_A:B (and then normalize result)
404  rot.Q_parent_this.multiply_left (s_bc.rot.Q_parent_this);
405  rot.Q_parent_this.normalize ();
406 
407  // Compute the corresponding transformation matrix.
408  rot.compute_transformation ();
409  }
410 
411  // Shortcuts for the case T_A:B is identity.
412  else {
413  Vector3::incr (s_bc.trans.position, trans.position);
414  Vector3::incr (s_bc.trans.velocity, trans.velocity);
415  rot.copy (s_bc.rot);
416  }
417 
418  return;
419 }

422 /**
423  * Compute S_B:C = (-S_A:B) + S_A:C,
424  * with this initially containing S_A:C,
425  * the supplied argument containing S_A:B, and
426  * the resultant composition of states stored in this.
427  * \param[in] s_ab Left addend
428  */
429 void
430 RefFrameState::decr_left (
431  const RefFrameState & s_ab)
432 {
433  // Relevant equations:
434  // T_B:C = T_A:C * T_A:B^T
435  // w_B:C = w_A:C - T_B:C * w_A:B
436  // x_B:C = T_A:B * (x_A:C - x_A:B)
437  // v_B:C = T_A:B * (v_A:C - v_A:B) - w_A:B X x_B:C
438 
439  // Order dependency:
440  // Quantity Uses S_A:C items Uses S_B:C items Order
441  // T_B:C T None 1
442  // w_B:C w T 2
443  // x_B:C x None 3
444  // v_B:C v x 4
445 
446 
447  // Compute T_B:C, v_B:C, and x_B:C via
448  // T_B:C = T_A:C * T_A:B^T
449  // x_B:C = T_A:B * (x_A:C - x_A:B)
450  // v_B:C = T_A:B * (v_A:C - v_A:B) - w_A:B X x_B:C
451  // Shortcut: only do the transformations if T_A:B is not the identity matrix.
452  if (!Numerical::compare_exact(s_ab.rot.Q_parent_this.scalar,1.0)) {
453 
454  // Compute Q_B:C = Q_A:C * Q_A:B^Q
455  rot.Q_parent_this.multiply_conjugate (s_ab.rot.Q_parent_this);
456  rot.Q_parent_this.normalize ();
457 
458  // Compute the corresponding transformation matrix.
459  rot.compute_transformation ();
460 
461  // Compute x_B:C and v_B:C.
462  // Note that x_B:C must be computed before v_B:C as V_B:C depends on
463  // the former.
464  Vector3::decr (s_ab.trans.position, trans.position);
465  Vector3::transform (s_ab.rot.T_parent_this, trans.position);
466 
467  Vector3::decr (s_ab.trans.velocity, trans.velocity);
468  Vector3::transform (s_ab.rot.T_parent_this, trans.velocity);
469  Vector3::cross_decr (s_ab.rot.ang_vel_this, trans.position,
470  trans.velocity);
471  }
472 
473  // Shortcuts for the case T_A:B is identity.
474  else {
475  Vector3::decr (s_ab.trans.position, trans.position);
476  Vector3::decr (s_ab.trans.velocity, trans.velocity);
477  Vector3::cross_decr (s_ab.rot.ang_vel_this, trans.position,
478  trans.velocity);
479  }
480 
481  // Compute w_B:C = w_A:C - T_B:C * w_A:B and related products.
482  if (std::fpclassify(s_ab.rot.ang_vel_mag) != FP_ZERO) {
483  Vector3::transform_decr (rot.T_parent_this, s_ab.rot.ang_vel_this,
484  rot.ang_vel_this);
485  rot.compute_ang_vel_products ();
486  }
487 
488  return;
489 }

492 /**
493  * Compute S_A:B = S_A:C + (-S_B:C)
494  * with this initially containing S_A:C,
495  * the supplied argument containing S_B:C, and
496  * the resultant composition of states stored in this.
497  * \param[in] s_bc Left addend
498  */
499 void
500 RefFrameState::decr_right (
501  const RefFrameState & s_bc)
502 {
503  // Relevant equations:
504  // T_A:B = T_B:C^T * T_A:C
505  // w_A:B = T_B:C^T * (w_A:C - w_B:C)
506  // x_A:B = x_A:C - T_A:B^T * x_B:C
507  // v_A:B = v_A:C - T_A:B^T * (v_B:C + w_A:B X x_B:C)
508 
509  // Order dependency:
510  // Quantity Uses S_A:C items Uses S_A:B items Order
511  // T_A:B T None 1
512  // w_A:B w None 2
513  // x_A:B x T 3
514  // v_A:B v T, w 4
515 
516 
517  double v_bc_in_b_obs_a[3]; // M/s Vel. of C wrt B in B, observed from A
518 
519 
520  // Compute T_A:B, w_A:B, via
521  // T_A:B = T_B:C^T * T_A:C
522  // w_A:B = T_B:C^T * (w_A:C - w_B:C)
523  // Shortcut: only do the transformations if T_B:C is not the identity matrix.
524  if (!Numerical::compare_exact(s_bc.rot.Q_parent_this.scalar,1.0)) {
525  rot.Q_parent_this.multiply_left_conjugate (s_bc.rot.Q_parent_this);
526  rot.Q_parent_this.normalize ();
527  rot.compute_transformation ();
528 
529  Vector3::decr (s_bc.rot.ang_vel_this, rot.ang_vel_this);
530  Vector3::transform_transpose (s_bc.rot.T_parent_this, rot.ang_vel_this);
531  rot.compute_ang_vel_products ();
532  }
533 
534  // Shortcuts for the case T_B:C is identity.
535  else if (std::fpclassify(s_bc.rot.ang_vel_mag) != FP_ZERO) {
536  Vector3::decr (s_bc.rot.ang_vel_this, rot.ang_vel_this);
537  rot.compute_ang_vel_products ();
538  }
539 
540 
541  // Compute the velocity of frame C wrt frame B observed from frame A
542  // v_B->C:B(A) = v_B:C + w_A:B X x_B:C
543  Vector3::copy (s_bc.trans.velocity, v_bc_in_b_obs_a);
544  if (std::fpclassify(rot.ang_vel_mag) != FP_ZERO) {
545  Vector3::cross_incr (rot.ang_vel_this, s_bc.trans.position,
546  v_bc_in_b_obs_a);
547  }
548 
549  // Compute x_A:B, V_A:B via
550  // x_A:B = x_A:C - T_A:B^T * x_B:C
551  // v_A:B = v_A:C - T_A:B^T * (v_B:C + w_A:B X x_B:C)
552  if (!Numerical::compare_exact(rot.Q_parent_this.scalar,1.0)) {
553  Vector3::transform_transpose_decr (rot.T_parent_this, s_bc.trans.position,
554  trans.position);
555  Vector3::transform_transpose_decr (rot.T_parent_this, v_bc_in_b_obs_a,
556  trans.velocity);
557  }
558  // Shortcuts for the case T_A:B is identity.
559  else {
560  Vector3::decr (s_bc.trans.position, trans.position);
561  Vector3::decr (v_bc_in_b_obs_a, trans.velocity);
562  }
563 
564  return;
565 }
