61 {
62  // Make a copy of the orientation to preserve constness.
63  Orientation orientation(pt_orientation);
64 
65  // Compute the transformation matrix and quaternion from the user-specified
66  // values, in the user-specified frame.
67  orientation.compute_transformation_and_quaternion();
68 
69  // Convert to the parent-to-child form required for a mass point.
70  switch(pt_frame_spec)
71  {
72  case StructToPoint:
73  Matrix3x3::copy(orientation.trans, mass_point.T_parent_this);
74  mass_point.Q_parent_this = orientation.quat;
