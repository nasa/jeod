309  /**
310  * Create the integrator to be managed.
311  * @param[in] technique_in Integration technique; generalized step vs deriv.
312  * @param[in] generator Integrator constructor used to create the integrator.
313  * @param[in,out] controls Integration controls to be passed to the generator.
314  */
315  void create_integrator(GeneralizedSecondOrderODETechnique::TechniqueType technique_in,
316  const er7_utils::IntegratorConstructor & generator,
317  er7_utils::IntegrationControls & controls)
318  {
319  // Save the integration technique for checkpoint/restart.
320  technique = technique_in;
321 
322  // Generate the integration generator.
323  if(technique == GeneralizedSecondOrderODETechnique::Cartesian)
324  {
325  generalized_deriv_integrator_manager.set_integrator_reference(integrator);
326  generalized_deriv_integrator_manager.create_integrator(generator, controls);
327  }
328  else if(technique == GeneralizedSecondOrderODETechnique::LieGroup)
329  {
330  generalized_step_integrator_manager.set_integrator_reference(integrator);
331  generalized_step_integrator_manager.create_integrator(generator, controls);
332  }
333  else
334  {
335  MessageHandler::fail(__FILE__,
336  __LINE__,
337  IntegrationMessages::invalid_item,
338  "The integration option is unspecified.");
339  }
340  }
341 
342  /**
343  * Destroy the integrator.
344  */
345  void destroy_integrator()
346  {
347  generalized_deriv_integrator_manager.destroy_integrator();
348  generalized_step_integrator_manager.destroy_integrator();
349 
350  technique = GeneralizedSecondOrderODETechnique::Unspecified;
351  }

365  * @param[in,out] position Generalized position vector.

389  /**
390  * Restore the integrator on restart.
391  */
392  void simple_restore() override
393  {
394  generalized_step_integrator_manager.clear_integrator_reference();
395  generalized_deriv_integrator_manager.clear_integrator_reference();
396 
397  // Restore the integrator from the checkpoint-restored value in the
398  // external pointer and perform technique-specific actions on it.
399  if(technique == GeneralizedSecondOrderODETechnique::Cartesian)
400  {
401  generalized_deriv_integrator_manager.set_integrator_reference(integrator);
402  generalized_deriv_integrator_manager.simple_restore();
403  }
404  else if(technique == GeneralizedSecondOrderODETechnique::LieGroup)
405  {
406  generalized_step_integrator_manager.set_integrator_reference(integrator);
407  generalized_step_integrator_manager.simple_restore();
408  }
409  }

433  */
434  RestartableGeneralizedStepSecondOrderODEIntegrator<4, 3, er7_utils::LeftQuaternionGeneralizedPositionFunctions>
