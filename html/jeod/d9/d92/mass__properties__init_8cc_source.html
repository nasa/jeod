65 void MassPropertiesInit::initialize_mass_properties(MassProperties & properties) const
66 {
67  double offset_inertia[3][3];
68 
69  // Initialize the center of mass and structure-to-body transform
70  initialize_mass_point(properties);
71 
72  // Copy the mass to the properties.
73  properties.mass = mass;
74 
75  // Initialize the inertia tensor per the inertia_spec.
76  // Inertia_spec == NoSpec means the inertia tensor in the target MassProperties
77  // object has (presumably) been set by some other means (e.g., direct user
78  // input); the inertia tensor in this MassPropertiesInit object is not
79  // used.
80  // For all other cases, the inertia tensor in the target MassProperties is
81  // to be based on this MassPropertiesInit object, with inertia_spec
82  // specifying in the frame and origin in which / about which this
83  // object's inertia tensor is specified.
84  switch(inertia_spec)
85  {
86  // NoSpec: Ignore the inertia tensor in this MassPropertiesInit object.
87  case NoSpec:
88  break;
89 
90  // Body: This inertia tensor is in the target body frame and about the
91  // target body CoM; exactly what is needed.
92  // Load this inertia directly into the mass properties.
93  case Body:
94 
95  Matrix3x3::copy(inertia, properties.inertia);
96  break;
97 
98  // StructCG: This inertia tensor is represented in target structural
99  // coordinates and is about the target body CoM.
100  // Transform this inertia tensor from structural frame to body.
101  case StructCG:
102 
103  Matrix3x3::transform_matrix(properties.T_parent_this, inertia, properties.inertia);
104  break;
105 
106  // Struct: This inertia tensor is represented in target structural
107  // coordinates and is about the target structural origin.
108  // Transform and translate this inertia tensor from structure to body.
109  case Struct:
110 
111  // Compute the inertia tensor for a point mass located at the CoM
112  // in structural coordinates.
113  MassBody::compute_point_mass_inertia(properties.mass, properties.position, offset_inertia);
114 
115  // Translate the input inertia to the CoM.
116  Matrix3x3::subtract(inertia, offset_inertia, offset_inertia);
117 
118  // Transform to body.
119  Matrix3x3::transform_matrix(properties.T_parent_this, offset_inertia, properties.inertia);
120  break;
121 
122  // SpecCG: This inertia tensor is represented in a user-specified frame
123  // and is about the target body CoM.
124  // Transform this inertia tensor from the inertia frame to body.
125  case SpecCG:
126  {
127  // Make a copy of the orientation to preserve constness.
128  Orientation orientation(inertia_orientation);
129 
130  // Compute the transformation matrix from user inputs.
131  orientation.compute_transform();
132 
133  // Transform the inertia tensor.
134  Matrix3x3::transform_matrix(orientation.trans, inertia, properties.inertia);
135  break;
136  }
137 
138  // Spec: This inertia tensor is represented in a user-specified frame
139  // and is about a user-specified origin.
140  // Transform and translate this inertia tensor from the user frame to body.
141  case Spec:
142  {
143  // Make a copy of the orientation to preserve constness.
144  Orientation orientation(inertia_orientation);
145 
146  // Compute the transformation matrix from user inputs.
147  orientation.compute_transform();
148 
149  // Compute the inertia tensor for a point mass located at the CoM
150  // in user frame coordinates.
151  MassBody::compute_point_mass_inertia(properties.mass, inertia_offset, offset_inertia);
152 
153  // Translate the input inertia to the CoM.
154  Matrix3x3::subtract(inertia, offset_inertia, offset_inertia);
155 
156  // Transform to body.
157  Matrix3x3::transform_matrix(orientation.trans, offset_inertia, properties.inertia);
158  break;
159  }
160 
161  default:
162  MessageHandler::fail(__FILE__,
163  __LINE__,
164  MassBodyMessages::invalid_enum,
165  "Illegal enum value (%d) for MassPropertiesInit.inertia_spec",
166  static_cast<int>(inertia_spec));
167 
168  // Not reached
169  return;
170  }
