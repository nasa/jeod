162 /**
163  * Add a DynBody to the set of bodies whose states are integrated by this group.
164  * @param dyn_body DynBody to be added to the group.
165  */
166 void DynamicsIntegrationGroup::add_dyn_body(DynBody & dyn_body)
167 {
168  bool found = false;
169 
170  // Check if the body is already in our list of bodies.
171  for(std::vector<DynBody *>::const_iterator it = dyn_bodies.begin(); it != dyn_bodies.end(); ++it)
172  {
173  DynBody * body = *it;
174  if(body == &dyn_body)
175  {
176  found = true;
177  break;
178  }
179  }
180 
181  // Don't add the body if it is already here.
182  if(found)
183  {
184  MessageHandler::error(__FILE__,
185  __LINE__,
186  DynManagerMessages::duplicate_entry,
187  "DynBody '%s' is already registered with this integration group.",
188  dyn_body.name.c_str());
189  return;
190  }
191 
192  // Not a duplicate. Add the body to the list.
193  dyn_bodies.push_back(&dyn_body);
194 
195  // Let the body know it was assigned to this group.
196  dyn_body.set_integration_group(*this);
197 
198  // Create the integrators for this body,
199  // but only if the bodies are to be integrated via DynBody::integrate().
200  if((integ_controls != nullptr) && bodies_integrated_separately)
201  {
202  // Create the integrators for this body.
203  dyn_body.create_body_integrators(*integ_constructor, *integ_controls, *jeod_time_manager);
204 
205  // Add any integrable objects associated with the dyn_body.
206  JeodPointerVector<er7_utils::IntegrableObject>::type associated_objects = dyn_body.get_integrable_objects();
207 
208  for(std::vector<er7_utils::IntegrableObject *>::const_iterator it = associated_objects.begin();
209  it != associated_objects.end();
210  ++it)
211  {
212  add_integrable_object(**it);
213  }
214  }
215 }

222 void DynamicsIntegrationGroup::delete_dyn_body(DynBody & dyn_body)
223 {
224  std::vector<DynBody *>::iterator it;
225 
226  // Find the body in our list of bodies.
227  for(it = dyn_bodies.begin(); it != dyn_bodies.end(); ++it)
228  {
229  DynBody * body = *it;
230  if(body == &dyn_body)
231  {
232  break;
233  }
234  }
235 
236  // Don't delete the body if it isn't ours.
237  if(it == dyn_bodies.end())
238  {
239  MessageHandler::error(__FILE__,

246

248  dyn_bodies.erase(it);
249 
250  // Let the body know it has been removed from this group.
251  dyn_body.clear_integration_group();

266 /**
267  * Compute the gravitational acceleration of each root dynamic body.
268  * @param dyn_manager Dynamics manager.
269  * @param gravity_manager Gravity Manager.
270  */
271 void DynamicsIntegrationGroup::gravitation(DynManager & dyn_manager, GravityManager & gravity_manager)
272 {
273  // Update ephemerides if this is to be done at the derivative rate
274  // or if the reference frame tree is out of whack.
275  if(deriv_ephem_update || dyn_manager.ref_frame_tree_needs_rebuild())
276  {
277  dyn_manager.update_ephemerides();
278  }
279 
280  // Compute gravitational effects on each root body.
281  for(std::vector<DynBody *>::const_iterator it = dyn_bodies.begin(); it != dyn_bodies.end(); ++it)
282  {
283  DynBody * body = *it;
284 
285  // Only process root bodies.
286  // The gravitational acceleration is not needed for child bodies.
287  if(body->is_root_body())
288  {
289  // Ask the Gravity Manager to compute the acceleration.
290  gravity_manager.gravitation(body->composite_body, body->grav_interaction);
291  }
292  }
293 }

332 /**
333  * Integrate the states of the DynBody objects that comprise the group.
334  * @param[in] cycle_dyndt Dynamic time step, in dynamic time seconds.
335  * @param[in] target_stage The stage of the integration process
336  * that the integrator should try to attain.
337  * @return The status (time advance, pass/fail status) of the integration.
338  */
339 er7_utils::IntegratorResult DynamicsIntegrationGroup::integrate_bodies(double cycle_dyndt, unsigned int target_stage)
340 {
341  er7_utils::IntegratorResult status(false);
342 
343  // This method requires that bodies_integrated_separately be set.
344  if(!bodies_integrated_separately)
345  {
346  // We should *never* get here. If we do, we've got trouble, big time.
347  MessageHandler::fail(__FILE__,
348  __LINE__,
349  DynManagerMessages::inconsistent_setup,
350  "Bodies must be integrated separately in this context.\n");
351  }
352 
353  // Integrate non-DynBody objects that are to be integrated as a part
354  // of this integration group.
355  if(!integrable_objects.empty())
356  {
357  status = integrate_container(cycle_dyndt, target_stage, integrable_objects);
358  }
359 
360  // Propagate state of each body to the end of the intermediate step.
361  for(std::vector<DynBody *>::const_iterator it = dyn_bodies.begin(); it != dyn_bodies.end(); ++it)
362  {
363  DynBody * body = *it;
364 
365  // Only process root bodies.
366  // The state of a non-root body is updated by virtue of the
367  // propagate_state() method of that body's root body.
368  if(body->is_root_body())
369  {
370  // Integrate the body's state and merge the integration status.
371  integ_merger.merge_integrator_result(body->integrate(cycle_dyndt, target_stage), status);
372  }
373  }
374 
375  return status;
376 }
