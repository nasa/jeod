186 /**
187  * Add a DynBody to the set of bodies whose states are integrated by this group.
188  * @param dyn_body DynBody to be added to the group.
189  */
190 void
191 DynamicsIntegrationGroup::add_dyn_body (
192  DynBody & dyn_body)
193 {
194  bool found = false;
195 
196  // Check if the body is already in our list of bodies.
197  for (std::vector<DynBody *>::const_iterator it = dyn_bodies.begin();
198  it != dyn_bodies.end();
199  ++it) {
200  DynBody * body = *it;
201  if (body == &dyn_body) {
202  found = true;
203  break;
204  }
205  }
206 
207  // Don't add the body if it is already here.
208  if (found) {
209  MessageHandler::error (
210  __FILE__, __LINE__, DynManagerMessages::duplicate_entry,
211  "DynBody '%s' is already registered with this integration group.",
212  dyn_body.name.c_str());
213  return;
214  }
215 
216  // Not a duplicate. Add the body to the list.
217  dyn_bodies.push_back (&dyn_body);
218 
219  // Let the body know it was assigned to this group.
220  dyn_body.set_integration_group (*this);
221 
222  // Create the integrators for this body,
223  // but only if the bodies are to be integrated via DynBody::integrate().
224  if ((integ_controls != nullptr) && bodies_integrated_separately) {
225 
226  // Create the integrators for this body.
227  dyn_body.create_body_integrators (
228  *integ_constructor, *integ_controls, *jeod_time_manager);
229 
230  // Add any integrable objects associated with the dyn_body.
231  JeodPointerVector<er7_utils::IntegrableObject>::type associated_objects =
232  dyn_body.get_integrable_objects();
233 
234  for (std::vector<er7_utils::IntegrableObject *>::const_iterator it =
235  associated_objects.begin();
236  it != associated_objects.end();
237  ++it) {
238  add_integrable_object(**it);
239  }
240  }
241 }

249 void
250 DynamicsIntegrationGroup::delete_dyn_body (
251  DynBody & dyn_body)
252 {
253  std::vector<DynBody *>::iterator it;
254 
255  // Find the body in our list of bodies.
256  for (it = dyn_bodies.begin(); it != dyn_bodies.end(); ++it) {
257  DynBody * body = *it;
258  if (body == &dyn_body) {
259  break;
260  }
261  }
262 
263  // Don't delete the body if it isn't ours.
264  if (it == dyn_bodies.end()) {
265  MessageHandler::error (
266  __FILE__, __LINE__, DynManagerMessages::inconsistent_setup,
267  "DynBody '%s' is not registered with this integration group.",
268  dyn_body.name.c_str());
269  return;
270  }
271 
272  // It's one of ours. Delete the body from the list.
273  dyn_bodies.erase (it);
274 
275  // Let the body know it has been removed from this group.
276  dyn_body.clear_integration_group ();
277 }

297  * @param dyn_manager Dynamics manager.

300 void
301 DynamicsIntegrationGroup::gravitation (
302  DynManager & dyn_manager,
303  GravityManager & gravity_manager)
304 {
305  // Update ephemerides if this is to be done at the derivative rate
306  // or if the reference frame tree is out of whack.
307  if (deriv_ephem_update || dyn_manager.ref_frame_tree_needs_rebuild()) {
308  dyn_manager.update_ephemerides ();
309  }
310 
311  // Compute gravitational effects on each root body.
312  for (std::vector<DynBody *>::const_iterator it = dyn_bodies.begin();
313  it != dyn_bodies.end();
314  ++it) {
315  DynBody * body = *it;
316 
317  // Only process root bodies.
318  // The gravitational acceleration is not needed for child bodies.
319  if (body->is_root_body()) {
320 
321  // Ask the Gravity Manager to compute the acceleration.
322  gravity_manager.gravitation (
323  body->composite_body,
324  body->grav_interaction);
325  }
326  }
327 }

373 /**
374  * Integrate the states of the DynBody objects that comprise the group.
375  * @param[in] cycle_dyndt Dynamic time step, in dynamic time seconds.
376  * @param[in] target_stage The stage of the integration process
377  * that the integrator should try to attain.
378  * @return The status (time advance, pass/fail status) of the integration.
379  */
380 er7_utils::IntegratorResult
381 DynamicsIntegrationGroup::integrate_bodies (
382  double cycle_dyndt,
383  unsigned int target_stage)
384 {
385  er7_utils::IntegratorResult status (false);
386 
387  // This method requires that bodies_integrated_separately be set.
388  if (! bodies_integrated_separately) {
389 
390  // We should *never* get here. If we do, we've got trouble, big time.
391  MessageHandler::fail (
392  __FILE__, __LINE__, DynManagerMessages::inconsistent_setup,
393  "Bodies must be integrated separately in this context.\n");
394  }
395 
396  // Integrate non-DynBody objects that are to be integrated as a part
397  // of this integration group.
398  if (! integrable_objects.empty()) {
399  status = integrate_container (
400  cycle_dyndt, target_stage, integrable_objects);
401 
402  }
403 
404  // Propagate state of each body to the end of the intermediate step.
405  for (std::vector<DynBody *>::const_iterator it = dyn_bodies.begin();
406  it != dyn_bodies.end();
407  ++it) {
408  DynBody * body = *it;
409 
410  // Only process root bodies.
411  // The state of a non-root body is updated by virtue of the
412  // propagate_state() method of that body's root body.
413  if (body->is_root_body()) {
414 
415  // Integrate the body's state and merge the integration status.
416  integ_merger.merge_integrator_result (
417  body->integrate (cycle_dyndt, target_stage),
418  status);
419  }
420  }
421 
422  return status;
423 }
