54 const JEOD_ATTRIBUTES_POINTER_TYPE
55 JeodTrickMemoryInterface::find_attributes (
56  const std::string & type_name)
57 const
58 {
59  std::string attr_name = "attr" + type_name;
60  for (size_t idx = attr_name.find(':');
61  idx < std::string::npos;
62  idx = attr_name.find(':', idx)) {
63  attr_name[idx] = '_';
64  }
65 
66  void * symbol = dlsym (dlhandle, attr_name.c_str());
67 
68  if (symbol == nullptr) {
69  MessageHandler::inform (
70  __FILE__, __LINE__, SimInterfaceMessages::interface_error,
71  "Unable to find Trick attributes for type '%s'.\n"
72  "Logging and checkpointing such objects will result in failure.",
73  type_name.c_str());
74  }
75 
76  return reinterpret_cast<ATTRIBUTES*> (symbol);
77 }

200 /**
201  * Create an attributes structure that represents a pointer type.
202  * @return Constructed pointer attributes.
203  * \param[in] target_attr Pointed-to type attributes.
204  */
205 JEOD_ATTRIBUTES_TYPE
206 JeodTrickMemoryInterface::pointer_attributes (
207  const JEOD_ATTRIBUTES_TYPE & target_attr)
208 const
209 {
210 #pragma GCC diagnostic ignored "-Wmissing-field-initializers"
211  ATTRIBUTES attr = {};
212  int pointer_index = target_attr.num_index;
213 
214  // Check if the parent is an array.
215  // A non-zero size for some index indicates that that element is an array.
216  for (int ii = 0; ii < target_attr.num_index; ++ii) {
217  if (target_attr.index[ii].size == 0) {
218  pointer_index = ii;
219  break;
220  }
221  }
222 
223  // Construct the output attributes.
224  attr.language = Language_CPP;
225  attr.io = 3;
226  attr.size = sizeof(void *);
227  attr.num_index = target_attr.num_index + 1;
228  attr.type = target_attr.type;
229 
230  // Set the attribute's attributes for a structured type.
231  if (target_attr.type == TRICK_STRUCTURED) {
232  attr.attr = target_attr.attr;
233  }
234 
235  // For non-structured types, attr.attr is the null pointer.
236  else {
237  attr.attr = nullptr;
238  }
239 
240  // Copy the parent array dimensions, if any.
241  for (int ii = 0; ii < pointer_index; ++ii) {
242  attr.index[ii] = target_attr.index[ii];
243  }
244 
245  return attr;
246 }
