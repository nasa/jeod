53 const JEOD_ATTRIBUTES_POINTER_TYPE JeodTrickMemoryInterface::find_attributes(const std::string & type_name) const
54 {
55  std::string attr_name = "attr" + type_name;
56  for(size_t idx = attr_name.find(':'); idx < std::string::npos; idx = attr_name.find(':', idx))
57  {
58  attr_name[idx] = '_';
59  }
60 
61  void * symbol = dlsym(dlhandle, attr_name.c_str());
62 
63  if(symbol == nullptr)
64  {
65  MessageHandler::inform(__FILE__,
66  __LINE__,
67  SimInterfaceMessages::interface_error,
68  "Unable to find Trick attributes for type '%s'.\n"
69  "Logging and checkpointing such objects will result in failure.",
70  type_name.c_str());
71  }
72 
73  return reinterpret_cast<ATTRIBUTES *>(symbol);
74 }

207 /**
208  * Create an attributes structure that represents a pointer type.
209  * @return Constructed pointer attributes.
210  * \param[in] target_attr Pointed-to type attributes.
211  */
212 JEOD_ATTRIBUTES_TYPE
213 JeodTrickMemoryInterface::pointer_attributes(const JEOD_ATTRIBUTES_TYPE & target_attr) const
214 {
215 #pragma GCC diagnostic ignored "-Wmissing-field-initializers"
216  ATTRIBUTES attr{};
217  int pointer_index = target_attr.num_index;
218 
219  // Check if the parent is an array.
220  // A non-zero size for some index indicates that that element is an array.
221  for(int ii = 0; ii < target_attr.num_index; ++ii)
222  {
223  if(target_attr.index[ii].size == 0)
224  {
225  pointer_index = ii;
226  break;
227  }
228  }
229 
230  // Construct the output attributes.
231  attr.language = Language_CPP;
232  attr.io = 3;
233  attr.size = sizeof(void *);
234  attr.num_index = target_attr.num_index + 1;
235  attr.type = target_attr.type;
236 
237  // Set the attribute's attributes for a structured type.
238  if(target_attr.type == TRICK_STRUCTURED)
239  {
240  attr.attr = target_attr.attr;
241  }
242 
243  // For non-structured types, attr.attr is the null pointer.
244  else
245  {
246  attr.attr = nullptr;
247  }
248 
249  // Copy the parent array dimensions, if any.
250  for(int ii = 0; ii < pointer_index; ++ii)
251  {
252  attr.index[ii] = target_attr.index[ii];
253  }
254 
255  return attr;
256 }
