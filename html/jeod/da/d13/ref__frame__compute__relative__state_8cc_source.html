46 /**
47  * Compute the complete state of the invoking reference frame (*this)
48  * with respect to the supplied wrt_frame reference frame.
49  * The state will include:
50  * - The position and velocity of the invoking frame with respect to the
51  * supplied wrt_frame, expressed in the coordinates of the wrt_frame.
52  * - The angular velocity of the invoking frame with respect to the
53  * supplied wrt_frame, expressed in the coordinates of invoking frame.
54  * - The transformation (as a matrix and a quaternion) from the supplied
55  * wrt_frame to the invoking frame.
56  *
57  * \par Assumptions and Limitations
58  * - The two frames are in the same tree.
59  * \param[in] wrt_frame The frame with respect to which the state is to be expressed
60  * \param[out] rel_state The relative state
61  */

69  this frame and the wrt_frame */

85  common_node_index = find_last_common_index (wrt_frame);

87  // A negative number indicates a *serious* problem.
88  if (common_node_index < 0) {
89  MessageHandler::fail (__FILE__, __LINE__,
90  RefFrameMessages::invalid_node,

99  common_node_frame = links.nth_from_root (common_node_index);

102  // There are four cases to cover:
103  // 1. The trivial case: The wrt_frame is this frame.
104  // 2. The wrt_frame is a direct predecessor of this frame.
105  // 3. This frame is a direct predecessor of the wrt_frame.
106  // 4. General case.
107 
108  // Trivial case.
109  if (&wrt_frame == this) {
110  rel_state.initialize ();
111 
112  }
113

115  else if (common_node_frame == &wrt_frame) {
116  compute_state_wrt_pred (common_node_index, rel_state);

120  else if (common_node_frame == this) {
121  wrt_frame.compute_pred_rel_state (common_node_index, rel_state);
122  }
123

125  else {
126

128  compute_state_wrt_pred (common_node_index, rel_state);

139 /**
140  * Compute the complete state of the invoking reference frame (*this)
141  * with respect to the supplied wrt_frame reference frame.
142  * If reverse_sense is false, the results are those from
143  * the simpler two argument form of RefFrame::compute_relative_state.
144  * If reverse_sense is true, the results from the two
145  * argument form are transformed as follows:
146  * - The position and velocity are those the invoking frame with respect to the
147  * supplied wrt_frame, but expressed in invoking frame coordinates.
148  * - The angular velocity of the invoking frame with respect to the
149  * supplied wrt_frame, expressed in the coordinates of supplied wrt_frame.
150  * - The transformation (as a matrix and a quaternion) from the invoking
151  * frame to the supplied wrt_frame.
152  *
153  * \par Assumptions and Limitations
154  * - The two frames are in the same tree.
155  * \param[in] wrt_frame The frame with respect to which the state is to be expressed
156  * \param[in] reverse_sense Express position and velocity in this frame,
157  * angular velocity in the wrt_frame, and the transformations from this
158  * frame to the wrt_frame.
159  * \param[out] rel_state The relative state
160  */
161 void
162 RefFrame::compute_relative_state (
163  const RefFrame & wrt_frame,
164  bool reverse_sense,
165  RefFrameState & rel_state)
166 const
167 {
168  // Compute the relative state as if express_in_wrt_frame is true.
169  compute_relative_state (wrt_frame, rel_state);
170 
171  // Reverse the sense of the computed state if so indicated.
172  if (reverse_sense) {
173  // Transform relative pos and vel from the wrt_frame to this frame.
174  Vector3::transform (
175  rel_state.rot.T_parent_this, rel_state.trans.position);
176  Vector3::transform (
177  rel_state.rot.T_parent_this, rel_state.trans.velocity);
178 
179  // Transform relative angular vel. from this frame to the wrt_frame.
180  Vector3::transform_transpose (
181  rel_state.rot.T_parent_this, rel_state.rot.ang_vel_this);
182  Vector3::transform_transpose (
183  rel_state.rot.T_parent_this, rel_state.rot.ang_vel_unit);
184 
185  // Transpose the tranformations so they are from this to the wrt frame.
186  Matrix3x3::transpose (rel_state.rot.T_parent_this);
187  rel_state.rot.Q_parent_this.conjugate ();
188  }
189 }

226 /**
227  * Compute the complete state of the invoking reference
228  * frame with respect to the supplied reference frame, which *must*
229  * be a predecessor of the invoking frame.
230  *
231  * \par Assumptions and Limitations
232  * - The predecessor frame is a predecessor.
233  * \param[in] pred_frame_index The frame with respect to which the state is to be expressed
234  * \param[out] rel_state The relative state
235  */
236 void
237 RefFrame::compute_state_wrt_pred (
238  unsigned int pred_frame_index,
239  RefFrameState & rel_state)
240 const
241 {
242  // Initialize the relative state as that of the state of the invoking frame
243  // wrt its immediate parent.
244  rel_state.copy (state);
245 
246  // Compute the state of this frame with respect to the predecessor node by
247  // successively computing the state of this frame with respect to frames in
248  // this frame's link path from the bottom up.
249  // The loop below starts just above the last entry in the link path (note
250  // that rel_state was initialized above to be the state with respect to
251  // this frame's immediate parent) and ends one shy of the common node.
252  // For each step, the state is re-represented with respect to the link
253  // frame's parent frame.
254  // Note: This loop will not be executed if the common node is this frame
255  // (the relative state is a null transformation) or if the common node
256  // is the immediate parent (relative state already has desired contents).
257  for (auto* link : TreeLinksAscendRange<const RefFrameLinks>(
258  links, links.path_length()-1, pred_frame_index+1)) {
259 
260  // Update the relative state by adding (on the left) the state of the
261  // current node with respect to it's parent. Conceptually,
262  // rel_state = link_frame->state + rel_state;
263  rel_state.incr_left (link->container().state);
264  }
265 }

302 /**
303  * Compute the complete state of the supplied reference frame
304  * wrt the invoking reference frame. The supplied reference frame
305  * must be a predecessor of the invoking frame.
306  *
307  * \par Assumptions and Limitations
308  * - The predecessor frame is a predecessor.
309  * \param[in] pred_frame_index The frame with respect to which the state is to be expressed
310  * \param[out] rel_state The relative state
311  */
312 void
313 RefFrame::compute_pred_rel_state (
314  unsigned int pred_frame_index,
315  RefFrameState & rel_state)
316 const
317 {
318  // Initialize the relative state as the negative of state of the invoking
319  // frame wrt its immediate parent.
320  rel_state.negate (state);
321 
322  // Compute the state of this frame with respect to the common node by
323  // successive computing the state of this frame with respect to frames in
324  // this frame's link path from the bottom up.
325  // The loop below starts just above the last entry in the link path (note
326  // that rel_state was initialized above to be the state with respect to
327  // this frame's immediate parent) and ends one shy of the common node.
328  // For each step, the state is re-represented with respect to the link
329  // frame's parent frame.
330  // Note: This loop will not be executed if the common node is this frame
331  // (the relative state is a null transformation) or if the common node
332  // is the immediate parent (relative state already has desired contents).
333  for (auto* link : TreeLinksAscendRange<const RefFrameLinks>(
334  links, links.path_length()-1, pred_frame_index+1)) {
335 
336  // Update the relative state by adding (on the right) the state of the
337  // current node's parent with respect to the current node. This is
338  // the equivalent of subtracting (on the left) the state of the current
339  // node wrt its parent. Conceptually,
340  // rel_state = rel_state - link_frame->state;
341  rel_state.decr_right (link->container().state);
342  }
343 }

346 /**
347  * Compute the relative position vector from the origin of the
348  * supplied reference frame to the origin of this reference frame,
349  * expressed in the coordinates of the supplied frame.
350  * \param[in] in_frame Relative position vector origin
351  * \param[out] rel_pos Relative position vector\n Units: M
352  */
353 void
354 RefFrame::compute_position_from (
355  const RefFrame& in_frame,
356  double rel_pos[3])
357 const
358 {
359  int common_node_index; /* Index of last node in common between
360  this frame and the wrt_frame */
361  const RefFrame * link_frame; /* Frame in path to node */
362 
363 
364  // Find the index of the node below which the path to the two frames diverge.
365  common_node_index = find_last_common_index (in_frame);
366 
367  // A negative number indicates a *serious* problem.
368  if (common_node_index < 0) {
369  MessageHandler::fail (__FILE__, __LINE__,
370  RefFrameMessages::invalid_node,
371  "Frames '%s' and '%s' are not in the same tree",
372  name.c_str(), in_frame.name.c_str());
373 
374  // Not reached
375  return;
376  }
377 
378 
379  Vector3::initialize (rel_pos);
380 
381  // First we step the current position back to the common node
382  // The end result will be the position of the invoking frame in the
383  // common node frame.
384  for (auto* link : TreeLinksAscendRange<const RefFrameLinks>(
385  links, links.path_length(), common_node_index+1)) {
386  link_frame = &link->container();
387 
388  // Transform the current position to the link frame's parent
389  if (!Numerical::compare_exact(link_frame->state.rot.Q_parent_this.scalar,1.0)) {
390  Vector3::transform_transpose (link_frame->state.rot.T_parent_this,
391  rel_pos);
392  }
393 
394  // Add in the position to the link frame's parent.
395  Vector3::incr (link_frame->state.trans.position, rel_pos);
396  }
397 
398  // Next step the current position down to the reference node.
399  for (auto* link : TreeLinksDescentRange<const RefFrameLinks>(
400  in_frame.links, common_node_index+1)) {
401  link_frame = &link->container();
402 
403  // Subtract the position to the link frame's parent.
404  Vector3::decr (link_frame->state.trans.position, rel_pos);
405 
406  // Transform the current position to the link frame
407  if (!Numerical::compare_exact(link_frame->state.rot.Q_parent_this.scalar,1.0)) {
408  Vector3::transform (link_frame->state.rot.T_parent_this, rel_pos);
409  }
410  }
411 }
