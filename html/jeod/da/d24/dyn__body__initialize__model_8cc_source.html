43 void DynBody::initialize_model(BaseDynManager & dyn_manager_in)
44 {
45  // Sanity check: protect against an empty name for this body
46  if(name.size() == 0)
47  {
48  MessageHandler::fail(__FILE__,
49  __LINE__,
50  DynBodyMessages::invalid_name,
51  "All DynBody instances must have names");
52 
53  // Not reached
54  return;
55  }
56 
57  // Setting the integration frame from the input file is not proper.
58  if(integ_frame != nullptr)
59  {
60  MessageHandler::error(__FILE__,
61  __LINE__,
62  DynBodyMessages::invalid_frame,
63  "DynBody '%s' integ_frame member was set in the input file.\n"
64  "This is not the recognized interface. "
65  "Set the integration frame name (integ_frame_name) instead.",
66  name.c_str());

71  if(integ_frame_name.empty())
72  {
73  MessageHandler::fail(__FILE__,
74  __LINE__,
75  DynBodyMessages::invalid_name,
76  "DynBody '%s' does not have an integration frame.",
77  name.c_str());
78 
79  // Not reached
80  return;
81  }
82 
83  // Cache the dynamics manager for later use.
84  dyn_manager = &dyn_manager_in;
85 
86  // Tell the dynamics manager to manage this DynBody.
87  dyn_manager->add_dyn_body(*this);
88 
89  // For each reference frame associated with this body,
90  // - Denote that the frame is owned by this body.
91  // - Set the name of the frame based on the body name.
92  // - Tell the dynamics manager that the frame exists.
93  // - Subscribe to the frame (which will mark the frame as active).
94 
95  // Core body frame
96  core_body.set_owner(this);
97  core_body.set_name(name.get_name(), "core_body");
98  dyn_manager->add_ref_frame(core_body);
99  core_body.subscribe();
100 
101  // Composite body frame
102  composite_body.set_owner(this);
103  composite_body.set_name(name.get_name(), "composite_body");
104  dyn_manager->add_ref_frame(composite_body);
105  composite_body.subscribe();
106 
107  // Structural frame
108  structure.set_owner(this);
109  structure.set_name(name.get_name(), "structure");
110  dyn_manager->add_ref_frame(structure);
111  structure.subscribe();
112 
113  // Mark the body as unitialized.
114  mass.needs_update = true;
115  initialized_states.set(RefFrameItems::No_Items);
116 
117  // Set the integration frame.
118  // Note that this links each of the three primary frames as children of
119  // the integration frame.
120  set_integ_frame(integ_frame_name);
