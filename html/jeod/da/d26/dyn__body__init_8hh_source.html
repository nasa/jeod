118  /**

128  /**

135  */

137 
138  /**
139  * Relative orientation between the subject and reference reference frame
140  * axes. The normal sense (reverse_sense is not set) is the transformation
141  * from reference to subject. The reverse meaning (reverse_sense set) is the
142  * transformation from subject to reference.
143  */
144  Orientation orientation; //!< trick_units(--)
145 
146  /**
147  * Relative angular velocity between the subject and reference axes. The
148  * flags reverse_sense and rate_in_parent give four interpretations:
149  * - Default (both reverse_sense and rate_in_parent are false):\n
150  * Angular velocity of the subject frame with respect to the reference
151  * frame, expressed in subject frame coordinates.
152  * - reverse_sense is clear, rate_in_parent is set:\n
153  * Angular velocity of the subject frame with respect to the reference
154  * frame, expressed in reference frame coordinates.
155  * - reverse_sense is set, rate_in_parent is clear:\n
156  * Angular velocity of the reference frame with respect to the subject
157  * frame, expressed in reference frame coordinates.
158  * - Both reverse_sense and rate_in_parent are set:\n
159  * Angular velocity of the reference frame with respect to the subject
160  * frame, expressed in subject frame coordinates.
161  */
162  double ang_velocity[3]; //!< trick_units(rad/s)
163 
164  /**
165  * Indicates how the user input state items are to be interpreted.
166  * If clear (default setting), indicates that the user input position,
167  * velocity, orientation, and angular velocity are to be interpreted in
168  * the standard JEOD parent to child sense. The meaning is reversed when
169  * this flag is set.
170  * See the descriptions of the individual state items for details.
171  */
172  bool reverse_sense; //!< trick_units(--)
173 
174  /**
175  * Indicates how the user input angular velocity is to be interpreted.
176  * This item works in conjunction with reverse_sense.
177  * See ang_velocity for a complete description.
178  */
179  bool rate_in_parent; //!< trick_units(--)
