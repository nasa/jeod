125  /**

128  * the subject origin with respect to the reference origin, expressed in and

132  */
133  double velocity[3]{}; //!< trick_units(m/s)
134 
135  /**
136  * Relative orientation between the subject and reference reference frame
137  * axes. The normal sense (reverse_sense is not set) is the transformation
138  * from reference to subject. The reverse meaning (reverse_sense set) is the
139  * transformation from subject to reference.
140  */
141  Orientation orientation; //!< trick_units(--)
142 
143  /**
144  * Relative angular velocity between the subject and reference axes. The
145  * flags reverse_sense and rate_in_parent give four interpretations:
146  * - Default (both reverse_sense and rate_in_parent are false):\n
147  * Angular velocity of the subject frame with respect to the reference
148  * frame, expressed in subject frame coordinates.
149  * - reverse_sense is clear, rate_in_parent is set:\n
150  * Angular velocity of the subject frame with respect to the reference
151  * frame, expressed in reference frame coordinates.
152  * - reverse_sense is set, rate_in_parent is clear:\n
153  * Angular velocity of the reference frame with respect to the subject
154  * frame, expressed in reference frame coordinates.
155  * - Both reverse_sense and rate_in_parent are set:\n
156  * Angular velocity of the reference frame with respect to the subject
157  * frame, expressed in subject frame coordinates.
158  */
159  double ang_velocity[3]{}; //!< trick_units(rad/s)
160 
161  /**
162  * Indicates how the user input state items are to be interpreted.
163  * If clear (default setting), indicates that the user input position,
164  * velocity, orientation, and angular velocity are to be interpreted in
165  * the standard JEOD parent to child sense. The meaning is reversed when
166  * this flag is set.
167  * See the descriptions of the individual state items for details.
168  */
169  bool reverse_sense{}; //!< trick_units(--)
170 
171  /**
172  * Indicates how the user input angular velocity is to be interpreted.

174  * See ang_velocity for a complete description.
175  */
176  bool rate_in_parent{}; //!< trick_units(--)
