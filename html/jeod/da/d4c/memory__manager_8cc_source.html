398 /**
399  * Allocate memory if that was not already done by the caller and
400  * register the memory with JEOD and with an external agent.
401  *
402  * \par Assumptions and Limitations
403  * - This method will be invoked via the JEOD memory allocation macros.
404  * Use in any other context is caveat emptor.
405  * - The corresponding delete macro will be used to delete the memory.
406  * Using the C free function or the C++ delete operator can cause *big*
407  * problems.
408  * - The delete macro will be expanded with the same placement new option
409  * as was used in the allocation macro that resulted in this call.
410  * - The memory is not constructed.
411  * That is the job of the expansion of the JEOD_ALLOC macro.
412  * \param[in] addr Memory to be registered
413  * \param[in] unique_id Unique id
414  * \param[in] placement_new Was memory allocated by this model?
415  * \param[in] is_array Was memory allocated as an array?
416  * \param[in] nelems Array size
417  * \param[in] tentry Type entry
418  * \param[in] file Source file containing JEOD_ALLOC
419  * \param[in] line Line number containing JEOD_ALLOC
420  */
421 void JeodMemoryManager::register_memory_internal(const void * addr,
422  uint32_t unique_id,
423  bool placement_new,
424  bool is_array,
425  unsigned int nelems,
426  const TypeEntry & tentry,
427  const char * file,
428  unsigned int line)
429 {
430  unsigned int alloc_idx = 0;
431  uint32_t tidx = tentry.index;
432  const JeodMemoryTypeDescriptor & tdesc = *(tentry.tdesc);
433 
434  // Sanity check: nelems and elem_size must be positive to continue.
435  // (Note that c++ allows double foo = new double[0]. JEOD does not.)
436  // Complain & return NULL; this will most likely raise havoc elsewhere.
437  if(nelems == 0)
438  {
439  MessageHandler::error(__FILE__,
440  __LINE__,
441  MemoryMessages::invalid_size,
442  "User allocation failed at %s:%d\n",
443  "The number of elements must be positive.",
444  file,
445  line);
446  return;
447  }
448 
449  // Construct and save a string describing the place the memory was
450  // allocated, but only if debugging levels are high enough.
451  if((file == nullptr) || (debug_level <= 1))
452  {

454  }
455  else
456  {
457  std::ostringstream id;
458  id << file << ":" << line;
459 
460  alloc_idx = add_string_atomic(id.str());
461  }

463  // Create the memory item that describes the allocated memory.
464  JeodMemoryItem item(placement_new, is_array, true, tdesc.is_structured(), nelems, tidx, alloc_idx);

493  add_allocation_atomic(addr, item, tdesc, file, line);
494 
495  // Report the allocation.

550 {
551  void * found_addr = nullptr;
552  JeodMemoryItem found_item;
553  const JeodMemoryTypeDescriptor * found_type = nullptr;
554 
555  // Nothing to do if address is null.
556  if(addr == nullptr)
557  {
558  MessageHandler::warn(__FILE__,
559  __LINE__,
560  MemoryMessages::null_pointer,
561  "Attempt to free a null pointer at %s:%d",
562  file,
563  line);
564  return;
565  }
566 
567  // Find and delete the matching allocation table entry for this address.
568  find_alloc_entry_atomic(addr, true, file, line, found_addr, found_item, found_type);
569 
570  // Item not found:
571  // The most likely cause is trying to delete something not allocated by JEOD,
572  // e.g., allocated by new, by the simulation engine, or from the stack.
573  // The response here is to do nothing other than complain.
574  if(found_addr == nullptr)
575  {
576  MessageHandler::error(__FILE__,
577  __LINE__,
578  MemoryMessages::suspect_pointer,
579  "JEOD memory error: Memory at %p was not created by JEOD.\n"

581  addr,
582  file,
583  line);
584  return;
585  }
586

588  // Complain and continue.
589  // destroy_memory() should be able to address the mismatch.
590  if(delete_array != found_item.get_is_array())
591  {

593  __LINE__,

595  "JEOD memory warning: Deletion of %p is inconsistent with "

598  addr,
599  file,
600  line);
601  }
602

606  if(found_item.get_is_registered())
607  {
608  sim_interface.deregister_allocation(found_addr, found_item, *found_type, file, line);
609  }
610 
611  // Destruct and delete the item consistent with its creation.
612  found_type->destroy_memory(found_item.get_placement_new(),
613  found_item.get_is_array(),
614  found_item.get_nelems(),

616

624  __LINE__,
625  MemoryMessages::debug,
626  "JEOD memory debug: Deleted %p at %s:%d\n"
627  " Type=%s\n",
628  found_addr,
629  file,
630  line,
631  found_type->type_spec(found_item).c_str());
632  }
633  else

637  MemoryMessages::debug,
638  "JEOD memory debug: Deleted %p at %s:%d\n"
639  " Type=%s\n"
640  " Allocated at %s",
641  found_addr,
642  file,
643  line,

646  }

721  // Compute the allocation buffer size, including guards.
722  buf_len = length + start_offset + end_offset;
723 
724  // Allocate the requisite amount of memory.
725  buf = new char[buf_len];
726 
727  // Failure to allocate memory means either the program has consumed all
728  // memory or the user asked for a huge amount.
729  // *Try* to print a message.
730  // Note to the user: If the program drops core in the message handler call
731  // it means your sim has consumed all memory.
732  if(buf == nullptr)
733  {
734  MessageHandler::fail(__FILE__, __LINE__, MemoryMessages::out_of_memory, "Out of memory");
735  return nullptr;
736  }
737 
738  // Insert guards.
739  if(guard)
740  {
741  uint32_t * guard_lo; // The four guard words at the start of the buffer
742  uint32_t * guard_hi; // The two guard words at the end of the buffer
743 
744  // Compute the address of the guards.
745  guard_lo = reinterpret_cast<uint32_t *>(buf);
746  guard_hi = reinterpret_cast<uint32_t *>(buf + buf_len - end_offset);
747 
748  // Set the guard words.
749  guard_lo[0] = guard_lo[2] = guard_hi[0] = MAGIC0;
750  guard_lo[1] = guard_lo[3] = guard_hi[1] = MAGIC1;
751  }
752 
753  // Determine the address to be returned and fill with the fill pattern.
754  addr = reinterpret_cast<void *>(buf + start_offset);
755  std::memset(addr, fill, length);
756 
757  return addr;
758 }
759 
760 /**
761  * Release memory.
762  *
763  * \par Assumptions and Limitations
764  * - This is a low-level de-allocation function.
765  * It does not
766  * - De-register the memory with JEOD or with an external agent.
767  * - Destruct the memory.
768  * \param[in,out] addr Memory to be freed
769  * \param[in] length Buffer size
770  * \param[in] guard Memory was guarded if set
771  * \param[in] alloc_idx Allocation index
772  * \param[in] file Source file containing delete
773  * \param[in] line Line number containing delete
774  */
775 void JeodMemoryManager::free_memory(
776  void * addr, std::size_t length, bool guard, unsigned int alloc_idx, const char * file, unsigned int line) const
777 {
778  char * buf; // The address of the allocated buffer
779  std::size_t buf_len; // The length of the allocated buffer
780 
781  // Memory is guarded:
782  // Free the allocated memory, which is offset from the input address.
783  if(guard)
784  {
785  uint32_t * guard_lo; // The four guard words at the start of the buffer
786  uint32_t * guard_hi; // The two guard words at the end of the buffer
787 
788  // Compute the addresses of the allocated memory and the guards
789  buf = reinterpret_cast<char *>(addr) - 16;
790  guard_lo = reinterpret_cast<uint32_t *>(buf);
791  guard_hi = reinterpret_cast<uint32_t *>(buf + length + 16);
792  buf_len = length + 24;
793 
794  // Check whether the guard words have been overwritten.
795  // Issue a complaint if this is the case.
796  if((guard_lo[0] != MAGIC0) || (guard_lo[2] != MAGIC0) || (guard_hi[0] != MAGIC0) || (guard_lo[1] != MAGIC1) ||
797  (guard_lo[3] != MAGIC1) || (guard_hi[1] != MAGIC1))
798  {
799  if(alloc_idx == 0)
800  {
801  MessageHandler::warn(__FILE__,
802  __LINE__,
803  MemoryMessages::corrupted_memory,
804  "JEOD memory warning: Memory at %p is corrupted!\n"
805  " Freed at %s:%d",
806  addr,
807  file,
808  line);
809  }
810  else
811  {
812  MessageHandler::warn(__FILE__,
813  __LINE__,
814  MemoryMessages::corrupted_memory,
815  "JEOD memory warning: Memory at %p is corrupted!\n"
816  " Allocated at %s\n"
817  " Freed at %s:%d",
818  addr,
819  get_string_atomic(alloc_idx).c_str(),
820  file,
821  line);
822  }
823  }
824  }
825 
826  // Memory is not guarded:
827  // The input address is the address of allocated memory.
828  else
829  {
830  buf = reinterpret_cast<char *>(addr);
831  buf_len = length;
832  }
833 
834  // Fill the buffer with garbage and free it.
835  std::memset(buf, 0xa5, buf_len);
836  delete[] buf;
837 }
