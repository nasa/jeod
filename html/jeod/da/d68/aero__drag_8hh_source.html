94  /**
95  * Dictates how the coefficients of drag will be
96  * calculated when using a flat plate model.
97  */
98  enum CoefCalcMethod {
99  Specular = 0 , /*
100  Molecules "bounce" producing forces in the normal-direction only */
101  Diffuse , /*
102  Molecules "stick" to the plate, producing forces in the direction of
103  the relative velocity */
104  Mixed , /*
105  A method that combines the specular and diffuse methods */
106  Calc_coef /*
107  Drag coefficients are calculated for direct pressure (normal to the
108  surface) and slip-friction (tangential to the surface) based on
109  angle of attack, material diffuse/specular properties, wall temperature,
110  altitude and gas constant */
111  };
