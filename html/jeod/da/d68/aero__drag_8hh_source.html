85  * Contains enumerations associated with aerodynamic drag
86  */
87 class AeroDragEnum
88 {
89 public:
90  /**
91  * Dictates how the coefficients of drag will be
92  * calculated when using a flat plate model.
93  */
94  enum CoefCalcMethod
95  {
96  Specular = 0, /*
97  Molecules "bounce" producing forces in the normal-direction only */
98  Diffuse, /*
99  Molecules "stick" to the plate, producing forces in the direction of
100  the relative velocity */
101  Mixed, /*
102  A method that combines the specular and diffuse methods */
103  Calc_coef /*
104  Drag coefficients are calculated for direct pressure (normal to the
105  surface) and slip-friction (tangential to the surface) based on
106  angle of attack, material diffuse/specular properties, wall temperature,
107  altitude and gas constant */
108  };
109 };
