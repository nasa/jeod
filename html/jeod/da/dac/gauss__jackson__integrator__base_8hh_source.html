278  /**
279  * Propagate state to the specified target_stage.
280  * @param[in] dyn_dt Dynamic time step, in dynamic time seconds.
281  * @param[in] target_stage The stage of the integration process
282  * that the integrator should try to attain.
283  * @param[in] deriv Acceleration vector.
284  * @param[in,out] state State vector(s).
285  *
286  * @return The status (time advance, pass/fail status) of the integration.
287  */
288  er7_utils::IntegratorResult base_integrate(double dyn_dt,
289  unsigned int target_stage,
290  const double * ER7_UTILS_RESTRICT deriv,
291  State state)
292  {
293  er7_utils::IntegratorResult result;
294  bool passed = true;
295 
296  // Optimization for operational mode.
297  // The integrator will spend most of its time in this mode.
298  // Note that this only applies if the state is already operational.
299  // It does not cover the very first step in operational mode;
300  // that transition is covered in the else below.
301  if(fsm_state == GaussJacksonStateMachine::Operational)
302  {
303  // Perform start of cycle operations at the start of a cycle:
304  // Rotate the acceleration history and save the incoming accels.
305  // This is what start_cycle() would do, but that function would do so
306  // with a meaningless function call and a bunch of meaningless tests.
307  if(target_stage == 1)
308  {
309  rotate_acc_hist();
310  er7_utils::integ_utils::copy_array(deriv, size, acc_hist[order]);
311  }
312 
313  // Integrate state via the Gauss-Jackson integrator.
314  passed = integrate_gj(dyn_dt, // Dynamic time step
315  target_stage, // 1=predict, 2=correct
316  order, // acc_hist index to be used for advance
317  order, // pos_hist index for computed state
318  deriv, // Input derivatives
319  acc_hist, // Acceleration history (no offset)
320  state); // Output state
321  }
322 
323  // Things are a bit hairier until we reach operational mode.
324  else
325  {
326  // Perform start of cycle operations at the start of a cycle.
327  // Note that each edit step is at start of cycle.
328  // Also note that this can transition to operational mode.
329  if(target_stage == 1)
330  {
331  start_cycle(dyn_dt, deriv, state);
332  }
333 
334  // Integrate per the state machine's state.
335  switch(fsm_state)
336  {
337  // Priming: Integrate via the primer, saving the integrated state
338  // for use during BootstrapEdit.
339  case GaussJacksonStateMachine::Priming:
340  result = integrate_primer(dyn_dt, target_stage, deriv, state);
341  save_comparison_data(state, pos_hist[history_length]);
342  break;
343 
344  // BootstrapEdit: Edit the point specified by history_length.
345  case GaussJacksonStateMachine::BootstrapEdit:
346  result.set_time_scale(history_length);
347  passed = edit_point(dyn_dt, deriv, state);
348  break;
349 
350  // BootstrapStep: Integrate via Gauss-Jackson proper,
351  // using the order+1 most recent history points.
352  // Note: acc_hist+history_length-(order+1) points to the start of the
353  // subarray that contains the order+1 most recent history points.
354  case GaussJacksonStateMachine::BootstrapStep:
355  passed = integrate_gj(dyn_dt, // Dynamic time step
356  target_stage, // 1=predict, 2=correct
357  history_length - 1, // Index to be used for advance
358  history_length, // pos_hist index for computed state
359  deriv, // Input derivatives
360  acc_hist + // Acceleration history
361  history_length - (order + 1), // offset
362  state); // Output state
363  break;
364 
365  // Operational: Integrate via Gauss-Jackson proper.
366  // History always starts at element 0 in operational mode.
367  // Note that this case will be hit once. After that the code will
368  // use the operational mode optimization above.
369  case GaussJacksonStateMachine::Operational:
370  passed = integrate_gj(dyn_dt, // Dynamic time step
371  target_stage, // 1=predict, 2=correct
372  order, // acc_hist index to be used for advance
373  order, // pos_hist index for computed state
374  deriv, // Input derivatives
375  acc_hist, // Acceleration history (no offset)
376  state); // Output state
377  break;
378  case GaussJacksonStateMachine::Reset:
379  default:
380  assert("Internal error" == nullptr);
381  break;
382  }
383  }
384 
385  // Mark the result as failed if the integration didn't converge.
386  if(!passed)
387  {
388  result.set_failed();
389  }
390  return result;
391  }

450  /**
451  * Start an integration cycle.
452  * @param[in] dt Dynamic time step, in dynamic time seconds.
453  * @param[in] acc Acceleration vector.
454  * @param[in] state State vector(s).
455  */
456  void start_cycle(double dt, const double * ER7_UTILS_RESTRICT acc, const State & state)
457  {
458  // Reset: Save the epoch point and the initial acceleration.
459  if(fsm_state == GaussJacksonStateMachine::Reset)
460  {
461  save_epoch_data(acc, state);
462  history_length = 1;
463  }
464 
465  // Non-reset start of cycle: Save incoming acceleration.
466  else
467  {
468  er7_utils::integ_utils::copy_array(acc, size, acc_hist[history_length]);
469  ++history_length;
470  }
471 
472  // Set the state per the state machine.
473  fsm_state = state_machine->get_fsm_state();
474 
475  // Downsample if so indicated.
476  if(state_machine->get_at_downsample())
477  {
478  downsample_hist();
479  }
480 
481  // Change order if so indicated.
482  if(state_machine->get_at_order_change())
483  {
484  order = state_machine->get_current_order();
485  velocity_corrector = 1.0 + coeff->corrector[order].sa_coefs[order];
486  position_corrector = coeff->corrector[order].gj_coefs[order];
487  }
488 
489  // Reinitialize the integration constants if so indicated.
490  if(state_machine->get_at_reinitialize())
491  {
492  if(fsm_state == GaussJacksonStateMachine::BootstrapEdit)
493  {
494  initialize_edit_integration_constants(dt);
495  history_length = 1;
496  }
497  else
498  {
499  initialize_predictor_integration_constants(dt);
500  }
501  }
502  }
503 
504  /**
505  * Edit the specified point using the mid-corrector that pertains
506  * to the point being edited, which is that at history_length.
507  * @param[in] dt Dynamic time step, in dynamic time seconds.
508  * @param[in] acc Acceleration vector.
509  * @param[out] state State vector(s).
510  */
511  bool edit_point(double dt, const double * ER7_UTILS_RESTRICT acc ER7_UTILS_UNUSED, State & state)
512  {
513  assert(history_length <= order);
514  advance_edit_integration_constants(history_length);
515  mid_correct(history_length, dt, state);
516  return test_for_convergence(state, pos_hist[history_length]);
517  }

519  /**
520  * Integrate using the Gauss-Jackson predictor and corrector.
521  * @param[in] dt Dynamic time step, in dynamic time seconds.
522  * @param[in] target_stage The stage of the integration process
523  * that the integrator should try to attain.
524  * @param[in] advance_index Acceleration history index.
525  * @param[in] target_index Position history index.
526  * @param[in] acc Acceleration vector.
527  * @param[in] ahist Acceleration vector history.
528  * @param[out] state State vector(s).
529  *
530  * @return True if step was successful, false otherwise.
531  */
532  bool integrate_gj(double dt,
533  unsigned int target_stage,
534  int advance_index,
535  int target_index,
536  const double * ER7_UTILS_RESTRICT acc,
537  const double * const * ahist,
538  State & state)
539  {
540  if(target_stage == 1)
541  {
542  advance_predictor_integration_constants(advance_index);
543  predict(dt, ahist, state);
544  save_comparison_data(state, pos_hist[target_index]);
545 
546  coeff->corrector[order].apply(size, order, ahist + 1, corrector_sum);
547  return true;
548  }
549  else
550  {
551  correct(dt, acc, state);
552  return test_for_convergence(state, pos_hist[target_index]);
553  }
554  }
