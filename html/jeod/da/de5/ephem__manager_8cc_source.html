244 /**
245  * Add an ephemeris item to the set of ephemeris items known to the ephemerides
246  * manager. Each ephemeris model calls this method for each ephemeris item
247  * represented by that ephemeris model. The provided item is added to
248  * - The ephem_items list if the provided item's name is unique or
249  * - The tail of the list of items with the same name if an item with the
250  * provided item's name is already in the ephem_items list.
251  *
252  * Multiple ephemerides models might represent the same conceptual item. The
253  * ephem_items list contains only items with unique names. These head items link
254  * to other items with the same name via the items' next data members, which
255  * form a singly-linked list of commonly-named items.
256  *
257  * ## Assumptions and limitations:
258  * - The item must have a valid name.
259  * - Only one item with a given name can be enabled.
260  *
261  * @param ephem_item Ephemeris item to be added to the registry.
262  */
263 void EphemeridesManager::add_ephem_item(EphemerisItem & ephem_item)
264 {
265  // The ephemeris item must have a minimally valid name.
266  if(!validate_name(__FILE__, __LINE__, ephem_item.get_name(), "Argument", "ephem_item.name"))
267  {
268  return;
269  }
270 
271  // Reject items that represents points when in single ephemeris mode.
272  // (Items that represent rotations are allowed in single ephemeris mode.)
273  if(single_ephem_mode && (ephem_item.updates_what() == EphemerisItem::Translation))
274  {
275  MessageHandler::inform(__FILE__,
276  __LINE__,
277  EphemeridesMessages::single_ephem_mode,
278  "Ephemeris item '%s' is being disabled and ignored.",
279  ephem_item.get_name().c_str());
280  ephem_item.disable();
281  return;
282  }
283 
284  // Set the item's ephemeris manager to this object.
285  ephem_item.set_manager(this);
286 
287  // Find the head of the list of items with the same name.
288  EphemerisItem * head_item = find_ephem_item(ephem_item.get_name());
289 
290  // No item yet:
291  // Add the provided item to the ephem_items list.
292  if(head_item == nullptr)
293  {
294  ephem_item.set_head(&ephem_item);
295  ephem_item.set_next(nullptr);
296  ephem_items.push_back(&ephem_item);
297  }
298 
299  // Existing item: validate and add to head item's list.
300  else
301  {
302  EphemerisItem * last_item = head_item;
303 
304  // Find the last item and verify that the item has not been registered.
305  for(EphemerisItem * item = head_item; item != nullptr; item = item->get_next())
306  {
307  last_item = item;
308  if(item == &ephem_item)
309  {
310  MessageHandler::error(__FILE__,
311  __LINE__,
312  EphemeridesMessages::duplicate_entry,
313  "Ephemeris item '%s' is already registered.",
314  ephem_item.get_name().c_str());
315  return;
316  }
317  }
318 
319  // At most one item can be enabled.
320  if((head_item->get_enabled_item() != nullptr) && ephem_item.is_enabled())
321  {
322  MessageHandler::warn(__FILE__,
323  __LINE__,
324  EphemeridesMessages::inconsistent_setup,
325  "Multiple ephemeris items named '%s' are enabled.\n"
326  "Disabling item from %s model.",
327  ephem_item.get_name().c_str(),
328  ephem_item.get_owner()->get_name().c_str());
329  ephem_item.disable();
330  }
331 
332  // Link the item into the list of items with the same name.
333  ephem_item.set_head(head_item);
334  last_item->set_next(&ephem_item);
335  ephem_item.set_next(nullptr);
336  }
337 
338  RefFrame * frame = find_ref_frame(ephem_item.get_name());
339  auto * ephem_frame = dynamic_cast<EphemerisRefFrame *>(frame);
340 
341  if(ephem_frame != nullptr)
342  {
343  ephem_item.set_target_frame(*ephem_frame);
344  }
345 
346  else if(frame != nullptr)
347  {
348  MessageHandler::fail(__FILE__,
349  __LINE__,
350  EphemeridesMessages::internal_error,
351  "Frame '%s' is not an ephemeris reference frame.",
352  frame->get_name().c_str());
353  }
354 }

546 /**
547  * Set the target-frame reference for the ref-frame and all known EphemItems
548  * with similar target-frame names.
549  * @param ref_frame Reference frame to be used as the target-frame.
550  */
551 void EphemeridesManager::set_target_frame(RefFrame & ref_frame)
552 {
553  // An ephemeris reference frame needs special treatment.
554  // There should be an ephemeris item with the same name that needs to
555  // to be tied to the frame, and the frame needs to be tied to the
556  // ephemerides manager (this object).
557  auto * ephem_frame = dynamic_cast<EphemerisRefFrame *>(&ref_frame);
558  EphemerisItem * head_item = find_ephem_item(ref_frame.get_name());
559  if(ephem_frame != nullptr)
560  {
561  // A change in the active status of an ephemeris reference frames means
562  // that the reference frame tree needs to be rebuilt.
563  // To accomodate this, ephemeris reference frames contain a link to the
564  // ephemerides manager (this object).
565  ephem_frame->set_ephem_manager(this);
566 
567  // The target frame of an ephemeris item is the ephemeris reference frame
568  // with the same name as the ephemeris item.
569  // Note that this sets the target frame for the head item and all linked
570  // items that follow it through the "next" link.
571  if(head_item != nullptr)
572  {
573  head_item->set_target_frame(*ephem_frame);
574  }
575 
576  // No else; the frame will be tied to the ephemeris item when the
577  // appropriately-named item is added.
578  }
579 
580  // Non-ephemeris reference frames that map to an ephemeris item are
581  // highly problematic. The simulation was build incorrectly.
582  else if(head_item != nullptr)
583  {
584  MessageHandler::fail(__FILE__,
585  __LINE__,
586  EphemeridesMessages::inconsistent_setup,
587  "Frame '%s' is not an ephemeris reference frame.",
588  ref_frame.get_name().c_str());
589  }
590 }

631 /**
632  * Activate ephemeris items based on frame subscription status,
633  * activate ephemeris models, and build the reference frame tree.
634  */
635 void EphemeridesManager::activate_ephemerides()
636 {
637  // There's nothing to do if the tree doesn't need to be built/rebuilt.
638  if(!regenerate_ref_frame_tree)
639  {
640  return;
641  }
642 
643  // Disconnect the registered ephemeris items from the tree.
644  for(std::vector<EphemerisItem *>::const_iterator it = ephem_items.begin(); it != ephem_items.end(); ++it)
645  {
646  EphemerisItem * ephem_item = *it;
647  ephem_item->disconnect_from_tree();
648  }
649 
650  // Forget that the reference frame tree has a root.
651  reset_tree_root_node();
652 
653  // Activate the ephemeris models in reverse order.
654  // Activating an ephemeris model that depend on some other ephemeris model
655  // may activate additional ephemeris items in that higher-level model.
656  // This dependency-based activation necessitates the reverse order.
657  for(auto it = ephemerides.rbegin(); it != ephemerides.rend(); ++it)
658  {
659  EphemerisInterface * ephem = *it;
660  ephem->ephem_activate(*this);
661  }
662 
663  // Build the tree starting from the first ephemeris model.
664  // Here we start from the base (dependency-free) ephemeris models and
665  // proceed downward to ensure that the base of the tree is built before
666  // adding to it.
667  for(std::vector<EphemerisInterface *>::const_iterator it = ephemerides.begin(); it != ephemerides.end(); ++it)
668  {
669  EphemerisInterface * ephem = *it;
670  ephem->ephem_build_tree(*this);
671  }
672 
673  // Finally, mark the tree as no longer needed to be regenerated.
674  // Note that this assumes a single pass is all that is needed.
675  // This will be the case if the base (dependency-free) ephemeris models
676  // are registered first and if the dependencies form an acyclic graph.
677  regenerate_ref_frame_tree = false;
678 }
