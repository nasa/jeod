74 /**
75  * The SimpleCheckpointable class provides a simple checkpoint/restart interface
76  * by which an object can complete the restart process. Typical use of the class
77  * is to restore inherently uncheckpointable data such as file streams and
78  * function pointers.
79  *
80  * The SimpleCheckpointable is an incomplete class. Derived classes must define
81  * a simple_restore() method to make the derived class complete. This method
82  * will be called as a part of the container restart process. Those derived
83  * classes should not override the overrides provided by this class. Derived
84  * classes can override the pre_ and post_ checkpoint and restart methods.
85  */
86 class SimpleCheckpointable : public JeodCheckpointable {

145  /**
146  * In general, respond to the actions recorded in the checkpoint file.
147  * For this class, the only recorded action is "restore", and the response
148  * is to invoke the (undefined) simple_restore method.
149  * @param action_name The name of the action; here just "restore".
150  * @param action_value The value of the action; here ignored.
151  * @return Success (zero) / failure (non-zero).
152  */
153  int perform_restore_action (
154  const std::string & action_name,
155  const std::string & action_value JEOD_UNUSED) override
156  {
157  int return_code;
158 
159  // "restore" action is to call the (undefined) simple_restore method.
160  if (action_name.compare("restore") == 0) {
161  simple_restore ();
162  return_code = 0;
163  }
164 
165  // All other actions are illegal and should never happen.
166  else {
167  return_code = -1;
168  }
169 
170  return return_code;
171  }
