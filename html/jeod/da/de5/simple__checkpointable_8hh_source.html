73 /**
74  * The SimpleCheckpointable class provides a simple checkpoint/restart interface
75  * by which an object can complete the restart process. Typical use of the class
76  * is to restore inherently uncheckpointable data such as file streams and
77  * function pointers.
78  *
79  * The SimpleCheckpointable is an incomplete class. Derived classes must define
80  * a simple_restore() method to make the derived class complete. This method
81  * will be called as a part of the container restart process. Those derived
82  * classes should not override the overrides provided by this class. Derived
83  * classes can override the pre_ and post_ checkpoint and restart methods.
84  */
85 class SimpleCheckpointable : public JeodCheckpointable
86 {
87  JEOD_MAKE_SIM_INTERFACES(jeod, SimpleCheckpointable)
88

100  /**
101  * Return the name of the initial restart action, in this case "restore".
102  * A derived class can of course override this.
103  */
104  const std::string get_init_name() override

106  return "restore";
107  }

140  void advance_checkpoint() override {}
141 
142  /**
143  * In general, indicate when checkpointing is complete.
144  * For this class, always return true.
145  */
146  bool is_checkpoint_finished() override
147  {
148  return true;
149  }
150 
151  /**
152  * In general, respond to the actions recorded in the checkpoint file.
153  * For this class, the only recorded action is "restore", and the response
154  * is to invoke the (undefined) simple_restore method.
155  * @param action_name The name of the action; here just "restore".
156  * @param action_value The value of the action; here ignored.
157  * @return Success (zero) / failure (non-zero).
158  */
159  int perform_restore_action(const std::string & action_name, const std::string & action_value JEOD_UNUSED) override
160  {
161  int return_code;
162 
163  // "restore" action is to call the (undefined) simple_restore method.
164  if(action_name.compare("restore") == 0)
165  {
166  simple_restore();
167  return_code = 0;
168  }
169 
170  // All other actions are illegal and should never happen.
171  else
172  {
173  return_code = -1;
174  }
175 
176  return return_code;
177  }
178 
179  SimpleCheckpointable(const SimpleCheckpointable &) = delete;
180  SimpleCheckpointable & operator=(const SimpleCheckpointable &) = delete;
181 
182 protected:
183  /**
184  * Perform the sole restore action.
185  */
186  virtual void simple_restore() = 0;
187 };
