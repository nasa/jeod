317 /**
318  * Uses a string comparison to find where in the TimeManager record a
319  * time type of a particular name is located. Returns the integer
320  * corresponding to the time type's index in the TimeManager.
321  *
322  * \par Assumptions and Limitations
323  * - Rarely used. If the time type address is known, it is easier to
324  * access its index "time_type.index" which returns the same result.
325  * @return index value of time-type
326  * \param[in] name name of time-type
327  */
328 int TimeManager::time_lookup(const std::string & name) const
329 {
330  int index = -1; // -1 corresponds to unfound
331 
332  // Sanity check:
333  // Test for a NULL or empty input name.
334  if(name.empty())
335  {
336  index = -2; // -2 corresponds to search name error
337  }
338 
339  // Input name is of a valid form.
340  // Find the matching item, if any.
341  else
342  {
343  for(unsigned ii = 0; ii < time_vector.size(); ++ii)
344  {
345  // See if we have a match to time_type_ptrs[ii].
346  // First, sanity check that time_types_ptrs[ii]->name is valid
347  // If so, test whether the name of that item matches the input.
348  if(name == time_vector[ii]->name)
349  {
350  // Have a match.
351  // See if we already had a match.
352  // This should never happen (it means there are 2
353  // time types with the same name), but just in case ...
354  if(index > -1)
355  {
356  MessageHandler::fail(__FILE__,
357  __LINE__,
358  TimeMessages::invalid_setup_error,
359  "\n"
360  "Looked up time type by name (%s) but found two"
361  " occurrences. Unable to distinguish between them.\n",
362  name.c_str());
363  return -3;
364  }
365  else
366  {
367  index = ii;
368  }
369  }
370  }
371  }
372 
373  // returns -1 if no match found, -2 if found an "undefined".
374  // Check these at each calling function for appropriate action.
375 
376  return (index);
377 }

379 /**
380  * This function manages the time update process.
381  * It first updates each of the representations of time, calling the
382  * update functions for each time representation in dependency order.
383  * After updating the representations of time, the function then
384  * updates the dynamic time scale factor. Time change subscribers are
385  * notified if the scale factor has changed.
386  *
387  * Note that by updating first and then checking for a change in the
388  * rate/direction of time means that these changes in rate/direction
389  * will first take affect on the next call to update_time or update.
390  *
391  * \par Assumptions and Limitations
392  * - Derived times must have a parent; this should be defined by the
393  * user, or if not, already determined when the update_tree was built
394  * \param[in] current_simtime input time from simulation engine; it always runs forwards and allows for determination of
395  * what has and has not already been done.\n Units: s
396  */
397 void TimeManager::update(double current_simtime)
398 {
399  if(!Numerical::compare_exact(current_simtime, simtime))
400  {
401  simtime = current_simtime;
402 
403  // update all times that are to be updated with the manager
404  // These are ordered in some update hierarchy, such that if x updates
405  // from y, y appears in the ordered_update_list array first.
406  for(int ii = 0; ii < num_types; ++ii)
407  {
408  time_vector[ii]->update();
409  }
410  }
411 
412  // check for run-time changes to scaling factor or time direction.
413  time_change_flag = dyn_time.update_offset();
414 
415  // notify subscribers if time has changed meaning.
416  if(time_change_flag)
417  {
418  notify_time_change_subscribers();
419  }
420 }
