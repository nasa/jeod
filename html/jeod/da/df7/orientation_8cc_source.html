160 /*******************************************************************************
161  * compute_<single_representation> and compute_<multiple_representation> methods
162  *
163  * Each compute_<single_representation> method computes the single
164  * representation of the orientation if possible to do so. (The only way this is
165  * not possible is when the orientation object is void of data.)
166  * - No computation is needed if the requested item is already available.
167  * - A direct conversion from the data source is used if possible.
168  * - An indirect path is used in those cases where a direct conversion is not
169  * available.
170  *
171  * Direct conversions exist for the follow representation schemes:
172  * To From Method
173  * Matrix Quaternion Quaternion::left_quat_to_transformation
174  * Eigen rot Orientation::compute_matrix_from_eigen_rotation
175  * Euler seq Orientation::compute_matrix_from_euler_angles
176  * Quaternion Matrix Quaternion::left_quat_from_transformation
177  * Eigen rot Quaternion::left_quat_from_eigen_rotation
178  * Eigen rot Quaternion Quaternion::left_quat_to_eigen_rotation
179  * Matrix Orientation::compute_eigen_rotation_from_matrix
180  * Euler seq Matrix Orientation::compute_euler_angles_from_matrix
181  *
182  * Missing from the above are methods to transform a quaternion or a single
183  * axis rotation to an Euler rotation sequence and methods to transform an
184  * Euler rotation sequence to a quaternion or a single axis rotation.
185  *
186  * The compute_<multiple_representation> methods are convenience methods that
187  * compute more than one representation of the orientation from the source.
188  ******************************************************************************/

198 void Orientation::compute_transform()
199 {
200  // Matrix is already available: We're done.
201  if(have_transformation_)
202  {
203  return;
204  }
205 
206  // Ensure the availability flag for the user input is set.
207  mark_input_as_available();
208 
209  // A direct path exists from each data source to a transformation matrix.
210  // Use the data source as the definitive source.
211  switch(data_source)
212  {
213  // Source is a matrix: Nothing to do.
214  case InputMatrix:
215  break;
216 
217  // Source is a quaternion: Extract the matrix from the quaternion.
218  case InputQuaternion:
219  quat.left_quat_to_transformation(trans);
220  break;
221 
222  // Source is a single axis rotation: Extract the matrix from that.
223  case InputEigenRotation:
224  compute_matrix_from_eigen_rotation();
225  break;
226 
227  // Source is an Euler rotation: Extract the matrix from that.
228  case InputEulerRotation:
229  compute_matrix_from_euler_angles();
230  break;
231 
232  // Invalid marker: Just return.
233  // A complaint was just issued by the call to mark_input_as_available.
234  case InputNone:
235  default:
236  return;
237  }
238 
239  have_transformation_ = true;
240 }
