118 /**
119  * Base method used by the other ColorText methods to wrap a chunk of text
120  * between an ASCII start color and ASCII restore pair of markers.
121  * \return A std::string whose c_str() value is the input text sandwiched
122  * between an ASCII start color escape sequence and an ASCII restore
123  * color escape sequence
124  * \param[in] text The text to be processed
125  * \param[in] text_color The color to be applied
126  */
127 inline std::string ColorText::color(const std::string & text, TextColor text_color)
128 {
129  unsigned int color_idx = static_cast<unsigned int>(text_color) - 30;
130  std::string result = ascii_color[color_idx] + text + ascii_color[0];
131  return result;

218 /**
219  * Generate a string whose c_str() value is the number as formatted per the
220  * format string but sandwiched betweeen an ASCII color text/restore monitor
221  * pair. The text is colored green if the passed flag is true, red otherwise.
222  * \return An ASCII-colored, c-formatted string
223  * \param[in] passed Pass/fail flags
224  * \param[in] format C-language sprintf format (e.g. "%g" or "%.7f")
225  * \param[in] nsp1st Number of spaces before first item
226  * \param[in] nspmid Number of spaces between items
227  * \param[in] number The vector to be formatted and colored
228  */
229 inline std::string ColorText::pass_fail(
230  const bool passed[3], const char * format, unsigned int nsp1st, unsigned int nspmid, double number[3])
231 {
232  std::snprintf(buf, sizeof(buf), "%*s", nsp1st, "");
233  std::string result(buf);
234 
235  std::snprintf(buf, sizeof(buf), "%*s", nspmid, "");
236  std::string sep(buf);
237 
238  for(unsigned int ii = 0; ii < 3; ++ii)
239  {
240  std::snprintf(buf, sizeof(buf), format, number[ii]);
241  if(passed[ii])
242  {
243  result += color(buf, Green);
244  }
245  else
246  {
247  result += color(buf, Red);
248  }
249  if(ii < 2)
250  {
251  result += sep;
252  }
253  }
254 
255  return result;
256 }

262  * \return An ASCII-colored, c-formatted string

267  * \param[in] number The matrix to be formatted and colored
