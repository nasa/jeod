70  * Invokes the calculation for all rotation models contained in the
71  * RNP, used on the last time set in each model through
72  * PlanetRotation::set_time. Then multiplies out and updates the
73  * attitude of the planet found during initialization.

75 void PlanetRNP::update_rnp()
76 {
77  // Update the nutation and precession, checking both that they are
78  // required and that they exist
79  if(rnp_type == FullRNP)
80  {
81  if(nutation == nullptr)
82  {
83  MessageHandler::fail(__FILE__,
84  __LINE__,
85  RNPMessages::fidelity_error,
86  "The PlanetRotation* nutation pointer"
87  " was not set in the PlanetRNP object, which is"
88  " required with a full fidelity RNP\n");
89  }
90  else
91  {
92  nutation->update_rotation();
93  }
94  if(precession == nullptr)
95  {
96  MessageHandler::fail(__FILE__,
97  __LINE__,
98  RNPMessages::fidelity_error,
99  "The PlanetRotation* precession pointer"
100  " was not set in the PlanetRNP object, which is"
101  " required with a full fidelity RNP\n");
102  }
103  else
104  {
105  precession->update_rotation();
106  // If it is a full fidelity RNP, then we need to update the
107  // internal NP matrix
108  Matrix3x3::product_transpose_transpose(nutation->rotation, precession->rotation, NP_matrix);
109  }
110  }
111 
112  // Update the rotation, first checking if it has been set to something
113  if(rotation == nullptr)
114  {
115  MessageHandler::fail(__FILE__,
116  __LINE__,
117  RNPMessages::setup_error,
118  "The PlanetRotation* rotation pointer was"
119  " not set in the PlanetRNP object, causing an error\n");
120  }
121  else
122  {
123  rotation->update_rotation();
124  }
125 
126  // Update the polar motion, checking if it has been set to something
127  // Only do this if polar motion is enabled
128  if(enable_polar == true)
129  {
130  if(polar_motion == nullptr)
131  {
132  MessageHandler::fail(__FILE__,
133  __LINE__,
134  RNPMessages::fidelity_error,
135  "The PlanetRotation* polar_motion pointer"
136  " was not set in the PlanetRNP object, which is"
137  " required with polar motion enabled\n");
138  }
139  else
140  {
141  polar_motion->update_rotation();
142  }
143  }
144 
145  propagate_rnp();

170 /**
171  * Multiples out the (up to) four planet rotation models (nutation,
172  * precession, polar motion and rotation), leaving out models whose
173  * pointers are NULL, and feeds it to the planet attitude found
174  * in the dyn manager given in initialize
175  */
176 void PlanetRNP::propagate_rnp()
177 {
178  // rotation should ALWAYS going to be present, so check if it is present
179 
180  if(rotation == nullptr)
181  {
182  MessageHandler::fail(__FILE__,
183  __LINE__,
184  RNPMessages::setup_error,
185  "The PlanetRotation* rotation pointer was"
186  " not set in the PlanetRNP object, causing an error\n");
187  }
188 
189  // order is:
190  // polar_motion^T * rotation^T * nutation^T precession^T
191 
192  // If rnp_type == Identity_RNP, only rotation exists, so just copy
193  // that into the final number and return the function
194  if(rnp_type == RotationOnly)
195  {
196  Matrix3x3::transpose(rotation->rotation, planet_rot_state->T_parent_this);
197  // set the rotational velocity components, and calculate the quaternion
198  // from the already set transformation matrix
199  planet_rot_state->ang_vel_mag = planet_omega;
200  planet_rot_state->ang_vel_this[2] = planet_omega;
201  planet_rot_state->compute_quaternion();
202  return;
203  }
204  // If it's a full fidelity RNP, calculate NP
205 
206  // Rotation is ALWAYS involved. If polar motion is involved, copy
207  // polar_motion^T * rotation^T into the temp matrix. Otherwise,
208  // copy only rotation^T into the temp matrix.
209  if(enable_polar == true)
210  {
211  if(polar_motion == nullptr)
212  {
213  MessageHandler::fail(__FILE__,
214  __LINE__,
215  RNPMessages::fidelity_error,
216  "The PlanetRotation* polar_motion pointer"
217  " was not set in the PlanetRNP object, which is"
218  " required with polar motion enabled\n");
219  }
220  else
221  {
222  Matrix3x3::product_transpose_transpose(polar_motion->rotation, rotation->rotation, scratch_matrix);
223  }
224  }
225  else
226  {
227  Matrix3x3::transpose(rotation->rotation, scratch_matrix);
228  }
229 
230  // If rnp_type is NOT Identity_RNP, multiply rotation^T * NP
231  // scratch_matrix currently contains either
232  // polar_motion^T * rotation^T, or rotation^T, depending on if
233  // polar motion is enabled.
234  // Note that if NP needs to be calculated, it was already calculated
235  // in "update_rnp"
236  if((rnp_type == FullRNP) || (rnp_type == ConstantNP))
237  {
238  Matrix3x3::product(scratch_matrix, NP_matrix, planet_rot_state->T_parent_this);
239  }
240 
241  // set the rotational velocity components, and calculate the quaternion
242  // from the already set transformation matrix
243  planet_rot_state->ang_vel_mag = planet_omega;
244  planet_rot_state->ang_vel_this[2] = planet_omega;
245  planet_rot_state->compute_quaternion();
246 }
