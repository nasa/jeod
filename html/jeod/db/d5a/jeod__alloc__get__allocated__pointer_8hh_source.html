78 /**
79  * Class template that provides a static function @a cast that casts a pointer
80  * to an object of type @a T to a void* pointer.
81  * @tparam T Type
82  * @tparam is_poly True if the type T is a polymorphic class.
83  *
84  * This class template is intended for used by jeod_alloc_get_allocated_pointer.
85  * Use in any other context is caveat emptor.
86  *
87  * This template provides a default implementation for non-polymorphic classes
88  * (<tt>is_poly == false</tt>) that uses implicit cast.
89  * The partial template instantiation that immediately follows provides n
90  * an implementation that uses dynamic_cast when @a is_poly is true.
91  */
92 template<typename T, bool is_poly> class JeodAllocHelperAllocatedPointer
93 {
94 public:
95  /**
96  * Cast a pointer to a non-polymorphic class via an implicit cast.
97  * @return Input pointer cast to void* via implicit cast.
98  * @param pointer Pointer
99  */
100  static void * cast(T * pointer)
101  {
102  return pointer;
103  }
104 };
105 
106 /**
107  * Partial instantiation of JeodAllocHelperAllocatedPointer for polymorphic
108  * classes. This class uses the fact that <tt>dynamic_cast<void*>(ptr)</tt>
109  * results in a pointer to the most derived object pointed to by <tt>ptr</tt>.
110  * See ISO/IEC 14882:2003 section 5.2.7.7 for details.
111  */
112 template<typename T> class JeodAllocHelperAllocatedPointer<T, true>
113 {
114 public:
115  /**
116  * Cast a pointer to a non-class object via dynamic_cast.
117  * @return Input pointer cast to void* via dynamic_cast.
118  * @param pointer Pointer
119  */
120  static void * cast(T * pointer)
121  {
122  return dynamic_cast<void *>(pointer);
123  }
124 };

133 /**
134  * Cast a pointer to some object to a pointer to void* such that a pointer to a
135  * polymorphic object, downcast to a base class pointer, becomes a pointer to
136  * the original object, but also such that a pointer to an instance of a
137  * non-polymorphic class or a pointer to a non-class type is handled correctly.
138  * @tparam T Pointed-to type.
139  * @param pointer Pointer to be cast to void*.
140  *
141  * <b>Usage:</b>@n
142  * @verbatim jeod_alloc_get_allocated_pointer(pointer) @endverbatim@n
143  * Note that the template parameter does not need to be specified.
144  * The compiler automagically determines the type.
145  *
146  * <b>Assumptions and limitations:</b>@n
147  * - The function argument @a pointer is a pointer.
148  * - The pointer is not @a cv qualified (i.e., neither const nor volatile).
149  * - Inheritance is public for polymorphic classes.
150  * - @verbatim jeod_alloc_get_allocated_pointer(&array[1]) @endverbatim
151  * does not return a pointer to @verbatim &array[0] @endverbatim.
152  */
153 template<typename T> inline void * jeod_alloc_get_allocated_pointer(T * pointer)
154 {
155  const bool is_poly = std::is_polymorphic<T>::value;
156  return JeodAllocHelperAllocatedPointer<T, is_poly>::cast(pointer);
157 }
