62 /**
63  * Compute the inverse of a 3x3 matrix.
64  *
65  * \par Assumptions and Limitations
66  * - Input and output matrices are distinct.
67  * - Input matrix is well-conditioned.
68  * @return 0=success, non-zero=singular
69  * \param[in] matrix Matrix to invert
70  * \param[out] inverse Inverse
71  */
72 int Matrix3x3::invert(const double matrix[3][3], double inverse[3][3])
73 {
74  double determinant; // The determinant of the input matrix
75  double det_inv; // The inverse of the above
76 
77  // Calculate the determinant
78  // (Save the intermediate calculations in 'inverse' for later use)
79  inverse[0][0] = matrix[1][1] * matrix[2][2] - matrix[2][1] * matrix[1][2];
80  inverse[0][1] = matrix[2][1] * matrix[0][2] - matrix[0][1] * matrix[2][2];
81  inverse[0][2] = matrix[0][1] * matrix[1][2] - matrix[1][1] * matrix[0][2];
82 
83  determinant = matrix[0][0] * inverse[0][0] + matrix[1][0] * inverse[0][1] + matrix[2][0] * inverse[0][2];
84 
85  // Test for an ill-conditioned matrix
86  if(std::fpclassify(determinant) == FP_ZERO)
87  {
88  MessageHandler::error(__FILE__,
89  __LINE__,
90  MathMessages::ill_conditioned,
91  "Matrix is ill-conditioned in Matrix3x3::invert.");
92  return 1;
93  }
94 
95  // Calculate the inverse matrix
96  det_inv = 1.0 / determinant;
97 
98  inverse[0][0] *= det_inv;
99  inverse[0][1] *= det_inv;
100  inverse[0][2] *= det_inv;
101  inverse[1][0] = (matrix[2][0] * matrix[1][2] - matrix[1][0] * matrix[2][2]) * det_inv;
102  inverse[1][1] = (matrix[0][0] * matrix[2][2] - matrix[2][0] * matrix[0][2]) * det_inv;
103  inverse[1][2] = (matrix[1][0] * matrix[0][2] - matrix[0][0] * matrix[1][2]) * det_inv;
104  inverse[2][0] = (matrix[1][0] * matrix[2][1] - matrix[2][0] * matrix[1][1]) * det_inv;
105  inverse[2][1] = (matrix[2][0] * matrix[0][1] - matrix[0][0] * matrix[2][1]) * det_inv;
106  inverse[2][2] = (matrix[0][0] * matrix[1][1] - matrix[1][0] * matrix[0][1]) * det_inv;
107 
108  // Return success code (0).
109  return 0;
110 }
