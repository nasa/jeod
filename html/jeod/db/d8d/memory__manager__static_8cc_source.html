322 void JeodMemoryManager::register_container(const void * container,
323  const std::type_info & container_type,
324  const std::string & elem_name,
325  JeodCheckpointable & checkpointable)
326 {
327  // Throw a fatal error if the singleton memory manager is not available.
328  if(check_master(true, __LINE__))
329  {
330  const JeodMemoryTypeDescriptor * tdesc(Master->get_type_descriptor_atomic(container_type));
331 
332  // Protect against A null type descriptor (otherwise get core dump)
333  if(tdesc == nullptr)
334  {
335  std::string type_name(NamedItem::demangle(container_type));
336  MessageHandler::fail(__FILE__,
337  __LINE__,
338  MemoryMessages::null_pointer,
339  "Illegal attempt to register container %s:%s contained in %p\n"
340  "Class %s has not been registered with the memory manager.",
341  type_name.c_str(),
342  elem_name.c_str(),
343  container,
344  type_name.c_str());
345  return;
346  }
347 
348  // Register the checkpointable object with the sim interface.
349  Master->sim_interface.register_container(container, *tdesc, elem_name, checkpointable);
350 
351  // Tell the checkpointable object to register / store type info.
352  checkpointable.initialize_checkpointable(container, container_type, elem_name);
353  }
