102 /**
103  * Given a value of seconds, propagate to other reps.
104  *
105  * \par Assumptions and Limitations
106  * - 86400 seconds = 1 day
107  * \param[in] new_seconds new value for seconds\n Units: s
108  */
109 void TimeGPS::set_time_by_seconds(const double new_seconds)
110 {
111  TimeStandard::set_time_by_seconds(new_seconds);
112  // Split time into day number and seconds of day.
113  auto gps_time_int = static_cast<int>(days);
114  seconds_of_day = (days - gps_time_int) * 86400.0;
115 
116  // GPS time rolls over every 1024 weeks, or every 7168 days.
117  rollover_count = gps_time_int / 7168;
118  // In newer systems, GPS time rolls over every 8192 weeks, or 57344 days.
119  rollover_count_13_bit = gps_time_int / 57344;
120  week_13_bit = (gps_time_int - 57344 * rollover_count_13_bit) / 7;
121  gps_time_int -= rollover_count * 7168;
122  week = gps_time_int / 7; // weeks since rollover
123 
124  day_of_week = gps_time_int - week * 7; /* number of full days in this
125  (partial) week */
126  seconds_of_week = day_of_week * 86400 + seconds_of_day; // convert to seconds
127 }
