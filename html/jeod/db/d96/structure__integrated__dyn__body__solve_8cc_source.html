140 // Solve for and apply constraints acting on the vehicle.
141 void
142 StructureIntegratedDynBody::solve_constraints ()
143 {
144 
145  // Both translational and rotational dynamics need to be on, and
146  // this must be a root body. (The constraint solver collects the
147  // constraints on attached bodies.)
148  if (! (translational_dynamics && rotational_dynamics
149  && (dyn_parent == nullptr)))
150  {
151  return;
152  }
153 
154  if (constraints_solver == nullptr)
155  {
156  MessageHandler::fail (
157  __FILE__, __LINE__, DynBodyMessages::invalid_body,
158  "solve_constraints called prior to setting the solver.");
159  return;
160  }
161 
162  // Populate the non-gravitational state structure.
163  Vector3::copy (
164  composite_body.state.rot.ang_vel_this, non_grav_state.omega_body);
165  Vector3::copy (
166  structure.state.rot.ang_vel_this, non_grav_state.omega_struct);
167  Vector3::copy (
168  derivs.rot_accel, non_grav_state.omega_dot_body);
169  Vector3::copy (
170  struct_derivs.rot_accel, non_grav_state.omega_dot_struct);
171  Vector3::scale (
172  collect.extern_forc_struct, mass.composite_properties.inverse_mass, non_grav_state.accel_struct);
173  Vector3::transform_transpose (
174  mass.composite_properties.T_parent_this,
175  collect.inertial_torq,
176  non_grav_state.inertial_torque_struct);
177 
178  // Solve the constraints, yielding an updated non-gravitational structure.
179  constraints_solver->solve (vehicle_properties, non_grav_state);
180 
181  // Update the rotational state from the solver's solution.
182  Vector3::copy (non_grav_state.omega_dot_struct, struct_derivs.rot_accel);
183  Vector3::copy (non_grav_state.omega_dot_body, derivs.rot_accel);
184 
185  // ... and the translational state.
186  Vector3::transform_transpose (
187  structure.state.rot.T_parent_this,
188  non_grav_state.accel_struct,
189  derivs.non_grav_accel);
190  complete_translational_acceleration();
191 }
