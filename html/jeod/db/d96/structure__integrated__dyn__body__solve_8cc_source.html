126 // Solve for and apply constraints acting on the vehicle.
127 void StructureIntegratedDynBody::solve_constraints()
128 {
129  // Both translational and rotational dynamics need to be on, and
130  // this must be a root body. (The constraint solver collects the
131  // constraints on attached bodies.)
132  if(!(translational_dynamics && rotational_dynamics && (dyn_parent == nullptr)))
133  {
134  return;
135  }
136 
137  if(constraints_solver == nullptr)
138  {
139  MessageHandler::fail(__FILE__,
140  __LINE__,
141  DynBodyMessages::invalid_body,
142  "solve_constraints called prior to setting the solver.");
143  return;
144  }
145 
146  // Populate the non-gravitational state structure.
147  Vector3::copy(composite_body.state.rot.ang_vel_this, non_grav_state.omega_body);
148  Vector3::copy(structure.state.rot.ang_vel_this, non_grav_state.omega_struct);
149  Vector3::copy(derivs.rot_accel, non_grav_state.omega_dot_body);
150  Vector3::copy(struct_derivs.rot_accel, non_grav_state.omega_dot_struct);
151  Vector3::scale(collect.extern_forc_struct, mass.composite_properties.inverse_mass, non_grav_state.accel_struct);
152  Vector3::transform_transpose(mass.composite_properties.T_parent_this,
153  collect.inertial_torq,
154  non_grav_state.inertial_torque_struct);
155 
156  // Solve the constraints, yielding an updated non-gravitational structure.
157  constraints_solver->solve(vehicle_properties, non_grav_state);
158 
159  // Update the rotational state from the solver's solution.
160  Vector3::copy(non_grav_state.omega_dot_struct, struct_derivs.rot_accel);
161  Vector3::copy(non_grav_state.omega_dot_body, derivs.rot_accel);
162 
163  // ... and the translational state.
164  Vector3::transform_transpose(structure.state.rot.T_parent_this, non_grav_state.accel_struct, derivs.non_grav_accel);
165  complete_translational_acceleration();
166 }
