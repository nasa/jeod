156 void
157 RNPJ2000::update_rnp (
158  const TimeTT& time_tt,
159  TimeGMST& time_gmst,
160  const TimeUT1& time_ut1)
161 {
162 
163  // check if active and get out if not
164  if(!active) {
165  return;
166  }
167 
168  // If the DynTime pointer has not been filled out yet, then we need
169  // to go and get that.
170 
171  if(time_dyn_ptr == nullptr) {
172  get_dyn_time_ptr(time_gmst);
173  }
174 
175  // If the TimeGMST pointer hasn't been cached off yet, do that.
176  if(gmst_ptr == nullptr) {
177  gmst_ptr = &time_gmst;
178  }
179 
180  // check if it even needs to be updated. If it has been previously
181  // updated, and if the time has not been changed, then we don't
182  // need to update.
183  if(!never_updated_full &&
184  Numerical::compare_exact(last_updated_time_full,time_dyn_ptr->seconds)) {
185  return;
186  }
187 
188  // If we are this far, then we are updating! and can set our update flags
189 
190  // This is the full RNP update, so it will ALSO get the rotational update.
191  // so set all flags to be correct.
192  never_updated_full = false;
193  never_updated_rotational = false;
194  last_updated_time_full = last_updated_time_rotational = time_dyn_ptr->seconds;
195 
196  double time = 0.0;
197 
198  // rotation needs seconds since J2000
199  time = time_gmst.seconds;
200 
201  if (enable_polar == true) {
202 
203  // polar motion needs UT1 based modified julian date
204  polar_motion->update_time (
205  time_ut1.trunc_julian_time + (2440000.5 - 2400000.5));
206  }
207 
208  if (rotation != nullptr) {
209  if (rnp_type == FullRNP) {
210  rotation->update_time (time);
211  }
212  else {
213  time = time_gmst.time_manager->dyn_time.seconds;
214  rotation->update_time (time);
215  }
216  }
217  /* compute the fraction of a julian century from the standard
218  epoch J2000 to julian_date.
219  To compute the precession and nutation matrices. Ported from
220  Jeod 1.52 */
221  // time = ( time_tt.trunc_julian_time - 2451545.0 + 2440000.5 ) / 36525.0 ;
222  time = (time_tt.trunc_julian_time + (2440000.5 - 2451545.0)) / 36525.0;
223  if (rnp_type == FullRNP) {
224  nutation->update_time (time);
225  precession->update_time (time);
226  }
227 
228  // update the timestamp of the controlled reference frame.
229  planet->pfix.set_timestamp(time_dyn_ptr->seconds);
230 
231  PlanetRNP::update_rnp();
232 
233  return;
234 }
