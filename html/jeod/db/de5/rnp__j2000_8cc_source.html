133 void RNPJ2000::update_rnp(const TimeTT & time_tt, TimeGMST & time_gmst, const TimeUT1 & time_ut1)
134 {
135  // check if active and get out if not
136  if(!active)
137  {
138  return;
139  }
140 
141  // If the DynTime pointer has not been filled out yet, then we need
142  // to go and get that.
143 
144  if(time_dyn_ptr == nullptr)
145  {
146  get_dyn_time_ptr(time_gmst);
147  }
148 
149  // If the TimeGMST pointer hasn't been cached off yet, do that.
150  if(gmst_ptr == nullptr)
151  {
152  gmst_ptr = &time_gmst;
153  }
154 
155  // check if it even needs to be updated. If it has been previously
156  // updated, and if the time has not been changed, then we don't
157  // need to update.
158  if(!never_updated_full && Numerical::compare_exact(last_updated_time_full, time_dyn_ptr->seconds))
159  {
160  return;
161  }
162 
163  // If we are this far, then we are updating! and can set our update flags
164 
165  // This is the full RNP update, so it will ALSO get the rotational update.
166  // so set all flags to be correct.
167  never_updated_full = false;
168  never_updated_rotational = false;
169  last_updated_time_full = last_updated_time_rotational = time_dyn_ptr->seconds;
170 
171  double time = 0.0;
172 
173  // rotation needs seconds since J2000
174  time = time_gmst.seconds;
175 
176  if(enable_polar == true)
177  {
178  // polar motion needs UT1 based modified julian date
179  polar_motion->update_time(time_ut1.trunc_julian_time + (2440000.5 - 2400000.5));
180  }
181 
182  if(rotation != nullptr)
183  {
184  if(rnp_type == FullRNP)
185  {
186  rotation->update_time(time);
187  }
188  else
189  {
190  time = time_gmst.time_manager->dyn_time.seconds;
191  rotation->update_time(time);
192  }
193  }
194  /* compute the fraction of a julian century from the standard
195  epoch J2000 to julian_date.
196  To compute the precession and nutation matrices. Ported from
197  Jeod 1.52 */
198  // time = ( time_tt.trunc_julian_time - 2451545.0 + 2440000.5 ) / 36525.0 ;
199  time = (time_tt.trunc_julian_time + (2440000.5 - 2451545.0)) / 36525.0;
200  if(rnp_type == FullRNP)
201  {
202  nutation->update_time(time);
203  precession->update_time(time);
204  }
205 
206  // update the timestamp of the controlled reference frame.
207  planet->pfix.set_timestamp(time_dyn_ptr->seconds);
208 
209  PlanetRNP::update_rnp();
210 }
