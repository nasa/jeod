96 // Add an integrable object to the vector of such.
97 void JeodIntegrationGroup::add_integrable_object(er7_utils::IntegrableObject & integrable_object)
98 {
99  // Remove the entry for this object if it's already present.
100  auto iter = std::find(integrable_objects.begin(), integrable_objects.end(), &integrable_object);
101  if(iter != integrable_objects.end())
102  {
103  // Make the object destroy its integrator(s).
104  if(integ_controls != nullptr)
105  {
106  integrable_object.destroy_integrators();
107  }
108 
109  // Delete the object from the list of integrable objects.
110  integrable_objects.erase(iter);
111  }
112 
113  // Make the object create its integrator(s).
114  if(integ_controls != nullptr)
115  {
116  integrable_object.set_integration_group(*this);
117  integrable_object.create_integrators(*integ_constructor, *integ_controls, *time_interface);
118  }
119 
120  // Add the object to the list of integrable objects.
121  integrable_objects.push_back(&integrable_object);
122 }
