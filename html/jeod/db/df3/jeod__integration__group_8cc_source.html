112 // Add an integrable object to the vector of such.
113 void
114 JeodIntegrationGroup::add_integrable_object (
115  er7_utils::IntegrableObject & integrable_object)
116 {
117  // Remove the entry for this object if it's already present.
118  std::vector<er7_utils::IntegrableObject*>::iterator iter =
119  std::find (integrable_objects.begin(), integrable_objects.end(),
120  &integrable_object);
121  if (iter != integrable_objects.end()) {
122  // Make the object destroy its integrator(s).
123  if (integ_controls != nullptr) {
124  integrable_object.destroy_integrators ();
125  }
126 
127  // Delete the object from the list of integrable objects.
128  integrable_objects.erase (iter);
129  }
130 
131  // Make the object create its integrator(s).
132  if (integ_controls != nullptr) {
133  integrable_object.set_integration_group (*this);
134  integrable_object.create_integrators (
135  *integ_constructor, *integ_controls, *time_interface);
136  }
137 
138  // Add the object to the list of integrable objects.
139  integrable_objects.push_back (&integrable_object);
140 }
