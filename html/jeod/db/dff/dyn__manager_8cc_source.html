164 /**
165  * Add a body action to the list of such.
166  * \param[in,out] body_action Body action
167  */
168 void DynManager::add_body_action(BodyAction & body_action)
169 {
170  // Handle errors.
171  // Note well: These are treated as severe rather than fatal errors.
172 
173  // The action must not yet be in the list of actions.
174  auto action_matcher = [&body_action](const BodyAction * bod_act)
175  {
176  return bod_act == &body_action;
177  };
178 
179  if(std::any_of(body_actions.begin(), body_actions.end(), action_matcher))
180  {
181  MessageHandler::error(__FILE__,
182  __LINE__,
183  DynManagerMessages::duplicate_entry,
184  "Duplicate entry passed to add_body_action()\n"
185  "Addition request ignored.");
186  return;
187  }
188 
189  // Initialize the action if the DynManager has already been initialized.
190  // (Enqueued body actions will be initialized en-masse during DynManager
191  // initialization.)
192  if(initialized)
193  {
194  body_action.initialize(*this);
195  }
196 
197  // Add the action to the list of such.
198  body_actions.push_back(&body_action);
199 }

201 /**
202  * Remove a body action to the list of such.
203  * \param[in] action_name_in Name of the action to remove
204  */
205 void DynManager::remove_body_action(const std::string & action_name_in)
206 {
207  if(action_name_in.empty())
208  {
209  return;
210  }
211  for(auto it = body_actions.begin(); it != body_actions.end(); ++it)
212  {
213  BodyAction * action = *it;
214 
215  if(action_name_in == action->action_name)
216  {
217  action->shutdown(); // frees the memory allocated to action_identifier
218  body_actions.erase(it); // remove from list.
219  return;
220  }
221  }
222 }
