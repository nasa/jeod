70 void RadiationPressure::initialize(DynManager & dyn_mgr, RadiationSurface * surf_ptr, double center_grav[3])
71 {
72  // make the model thermally dynamic as default
73  thermal.active = true;
74 
75  // set the Dyn Manager pointer
76  dyn_manager_ptr = &(dyn_mgr);
77 
78  // initialize the 3rd bodies (provides compatibility with Trick7 methodology
79  // of defining 3rd bodies in default data)
80  for(unsigned int ii_body = 0; ii_body < source.num_bodies; ++ii_body)
81  {
82  add_third_body(source.third_body[ii_body]);
83  int old_shadow_geom = source.shadow_geometry;
84  third_bodies.back()->convert_shadow_from_int(old_shadow_geom);
85  }
86 
87  // initialize the primary source.
88  initialize_environment(dyn_manager_ptr);
89 
90  // initialize the surface.
91  surface_ptr = surf_ptr;
92  surface_ptr->initialize(center_grav);
93 }

96  * Used to update the model when the surface comprises facets.
97  */
98 void RadiationPressure::update_facet_surface()
99 {
100  // initialize each of the facets here, because paths diverge after this.
101  surface_ptr->initialize_runtime_values();
102 
103  if(source.flux_mag > 0)
104  {
105  surface_ptr->incident_radiation(source.flux_mag, source.flux_struc_hat, calculate_forces);
106  }
107 
108  for(unsigned int ii_body = 0; ii_body < num_third_bodies; ++ii_body)
109  {
110  if(third_bodies[ii_body]->is_interactive() && third_bodies[ii_body]->active)
111  {
112  surface_ptr->interact_with_third_body(third_bodies[ii_body], calculate_forces);
113  }
114  }
115 
116  // Note: must always do the thermal update, even if it thermal.active =
117  // false. In that case (i.e. facet temperatures are stationary)
118  // there will still be radiative thermal emission corresponding
119  // to the radiation absorption. This generates a force which must be
120  // included in the radiation pressure calculation.
121  thermal.update(surface_ptr);
122 
123  // If the thermal model is active, individual inactive facets have already had
124  // power_absorb set to power_emit in the thermal.integrate function, called
125  // from the previous thermal.update() call. If not,
126  // that must still be carried out for all facets.
127  if(!thermal.active)
128  {
129  surface_ptr->equalize_absorption_emission();
130  }
131 
132  // Forces are not always desired, this model can be used to obtain irradiance
133  // measurements for lighting models, electrical subsystems (solar panels),
134  // thermal subsystems. But typically, this flag is true.
135  if(calculate_forces)
136  {
137  surface_ptr->radiation_pressure();
138  Vector3::copy(surface_ptr->force, force);
139  Vector3::copy(surface_ptr->torque, torque);
140  }
141 }
