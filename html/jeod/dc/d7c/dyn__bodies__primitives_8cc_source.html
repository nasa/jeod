53  * @return Pointer to found DynBody; NULL if not found.

55 DynBody * DynManager::find_dyn_body(const std::string & body_name) const
56 {
57  DynBody * found_body = nullptr;
58 
59  // Ensure the passed name has a minimally valid value.
60  if(!validate_name(__FILE__, __LINE__, body_name, "Argument", "name"))
61  {
62  return nullptr;
63  }
64 
65  // Find the body by name.
66  auto name_compare = [&body_name](const DynBody * body)
67  {
68  return body->name.get_name().compare(body_name) == 0;
69  };
70 
71  const auto iter = std::find_if(dyn_bodies.begin(), dyn_bodies.end(), name_compare);
72 
73  if(iter != dyn_bodies.end())
74  {
75  found_body = *iter;
76  }
77 
78  return found_body;
79 }

91 /**
92  * Add a dynamic body to the dynamic body registry.
93  * @param dyn_body Dynamic body to be added to the registry.
94  */
95 void DynManager::add_dyn_body(DynBody & dyn_body)
96 {
97  MassBody * mass_body;
98 
99  // Handle errors.
100  // 1. The body must have a minimally valid name.
101  if(!validate_name(__FILE__, __LINE__, dyn_body.name.get_name(), "Argument", "dyn_body.name"))
102  {
103  return;
104  }
105 
106  // 2. The body must not have been previously registered.
107  if(is_dyn_body_registered(&dyn_body))
108  {
109  MessageHandler::error(__FILE__,
110  __LINE__,
111  DynManagerMessages::duplicate_entry,
112  "Dynamic body '%s' is already registered.",
113  dyn_body.name.c_str());
114  return;
115  }
116 
117  // 3. The body must have a unique name.
118  if(find_dyn_body(dyn_body.name.get_name()) != nullptr)
119  {
120  MessageHandler::error(__FILE__,
121  __LINE__,
122  DynManagerMessages::duplicate_entry,
123  "Dynamic body with name '%s' was previously registered.",
124  dyn_body.name.c_str());
125  return;
126  }
127 
128  // 4. The name uniqueness also applies to mass bodies.
129  mass_body = find_mass_body(dyn_body.name.get_name());
130  if(mass_body != nullptr)
131  {
132  if(mass_body != &dyn_body.mass)
133  {
134  MessageHandler::error(__FILE__,
135  __LINE__,
136  DynManagerMessages::invalid_name,
137  "Attempt to add a DynBody with the same name '%s' "
138  "as a previously registered MassBody.",
139  dyn_body.name.c_str());
140  return;
141  }
142  else
143  {
144  MessageHandler::warn(__FILE__,
145  __LINE__,
146  DynManagerMessages::duplicate_entry,
147  "Adding a DynBody whose MassBody member already "
148  "previously registered. Implies that add_mass_body was called when "
149  "add_dyn_body was more appropriate.");
150  }
151  }
152  else
153  {
154  // All tests passed: Add the body to the registries.
155 
156  // Add the mass to the list of mass body registry.
157  add_mass_body(dyn_body.mass);
158  }
159 
160  // Add the body to the list of dynamic body registry.
161  dyn_bodies.push_back(&dyn_body);
162 }
