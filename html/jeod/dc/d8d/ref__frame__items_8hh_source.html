92  /**
93  * The Items enumeration identifies the major items that can be
94  * set in a RefFrameState structure -- position, velocity, attitude, and
95  * attitude rate. The enumeration values are implemented as bit flags.
96  * The four basic items, position, velocity, attitude, and rate, have
97  * values of 1, 2, 4, and 8, respectively. Combinations thereof have values
98  * corresponding to the bitwise or of the basic components.
99  */
100  enum Items {
101

103  No_Items = 0, ///< Nothing set
104  Pos = 1, ///< Position
105  Vel = 2, ///< Velocity
106  Pos_Vel = 3, ///< Position + velocity
107  Att = 4, ///< Attitude
108  Pos_Att = 5, ///< Position + attitude
109  Vel_Att = 6, ///< Velocity + attitude
110  Pos_Vel_Att = 7, ///< Position + velocity + attitude
111  Rate = 8, ///< Attitude rate
112  Pos_Rate = 9, ///< Position + rate
113  Vel_Rate = 10, ///< Velocity + rate
114  Pos_Vel_Rate = 11, ///< Position + velocity + rate
115  Att_Rate = 12, ///< Attitude + Rate
116  Pos_Att_Rate = 13, ///< Position + attitude + Rate
117  Vel_Att_Rate = 14, ///< Velocity + attitude + Rate
118  Pos_Vel_Att_Rate = 15 ///< Position + velocity + attitude + Rate
119  };
