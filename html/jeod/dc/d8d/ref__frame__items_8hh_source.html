78 /**
79  * Identify which aspects of a reference frame's state have been set.
80  * The aspects that are managed are the position, velocity, attitude,
81  * and attitude rate.
82  */
83 class RefFrameItems
84 {
85  JEOD_MAKE_SIM_INTERFACES(jeod, RefFrameItems)
86 
87  // Enumerations
88 public:
89  /**
90  * The Items enumeration identifies the major items that can be
91  * set in a RefFrameState structure -- position, velocity, attitude, and
92  * attitude rate. The enumeration values are implemented as bit flags.
93  * The four basic items, position, velocity, attitude, and rate, have
94  * values of 1, 2, 4, and 8, respectively. Combinations thereof have values
95  * corresponding to the bitwise or of the basic components.
96  */
97  enum Items
98  {
99  // Values.
100  No_Items = 0, ///< Nothing set
101  Pos = 1, ///< Position
102  Vel = 2, ///< Velocity
103  Pos_Vel = 3, ///< Position + velocity
104  Att = 4, ///< Attitude
105  Pos_Att = 5, ///< Position + attitude
106  Vel_Att = 6, ///< Velocity + attitude
107  Pos_Vel_Att = 7, ///< Position + velocity + attitude
108  Rate = 8, ///< Attitude rate
109  Pos_Rate = 9, ///< Position + rate
110  Vel_Rate = 10, ///< Velocity + rate
111  Pos_Vel_Rate = 11, ///< Position + velocity + rate
112  Att_Rate = 12, ///< Attitude + Rate
113  Pos_Att_Rate = 13, ///< Position + attitude + Rate
114  Vel_Att_Rate = 14, ///< Velocity + attitude + Rate
115  Pos_Vel_Att_Rate = 15 ///< Position + velocity + attitude + Rate
116  };
