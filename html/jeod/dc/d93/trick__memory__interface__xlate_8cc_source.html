87  const JeodMemoryTypeDescriptor * tdesc) const
88 {
89  const ATTRIBUTES & item_attr = tdesc->get_attr();
90  std::string result;
91 
92  // A null pointer is a null pointer. Period.
93  if(addr == nullptr)

98  else

100  // Interpret the address as a pointer to the specified type.
101  result = translate_addr_to_name(addr, &item_attr);

113  result = "";
114  }
115  }
116 
117  return result;
118 }

148  int pointer_index = ptr_attr.num_index;
149 
150  // Find the index in the attributes at which the pointer business starts.

153  {

155  {
156  pointer_index = ii;

159  }
160

175 /**
176 Translate the given address specification string to an address.
177 This is the inverse of translate_addr_to_name.
178 @param spec The address specification to be interpreted.
179 @return Address corresponding to the address specification.
180 */
181 void * JeodTrick10MemoryInterface::translate_name_to_addr(const std::string & spec) const
182 {
183  std::string name(spec);
184  void * result;
185  std::size_t position;
186 
187  // Sometimes translate_addr_to_name returns strings like
188  // NULL /* ERROR Explanation of problem */
189  // Strip the C-style comment off the end of the string.
190  // Note: This assumes
191  // - A single slash indicates start of comment.
192  // - There's nothing of interest after the end of the comment.
193  position = name.find_first_of('/');
194  if(position < name.length())
195  {
196  name.erase(position);
197  position = name.find_last_not_of(' ');
198  if(position < name.length())
199  {
200  name.erase(position + 1);
201  }
202  // No else: We'll deal with an empty spec later.
203  }
204 
205  // "NULL" means NULL, but ref_attributes doesn't grok "NULL".
206  if(name.compare("NULL") == 0)
207  {
208  result = nullptr;
209  }
210 
211  else
212  {
213  std::size_t offset = 0;
214 
215  // Deal with strings of the form "&foo.bar + 42":
216  // Interpret and then get rid of the trailing offset.
217  position = name.find_first_of('+');
218  if(position < name.length())
219  {
220  offset = std::strtoul(name.substr(position + 1).c_str(), nullptr, 0);
221  name.erase(position);
222  }
223 
224  // Get rid of the leading ampersand if it is present.
225  if(name[0] == '&')
226  {
227  name = name.substr(1);
228  }
229 
230  // Get rid of trailing spaces and translate.
231  position = name.find_last_not_of(' ');
232  if(position < name.length())
233  {
234  name.erase(position + 1);
235 
236  // We now have a name spec that ref_attributes might understand.
237  const char * name_string = name.c_str();
238  REF2 * ref = ref_attributes(const_cast<char *>(name_string));
239 
240  // We have a good name spec: form the result.
241  if(ref != nullptr)
242  {
243  result = static_cast<void *>(static_cast<char *>(ref->address) + offset);
244 
245  // Clean up the REFidue before returning the result.
246  std::free(ref->reference);
247  std::free(ref->units);
248  std::free(ref);
249  }
250 
251  // Oops. We don't have a name spec that ref_attributes understands.
252  // For example, name.not.in.name.space.known.to.trick
253  // Complain and return NULL.
254  else
255  {
256  MessageHandler::error(__FILE__,
257  __LINE__,
258  SimInterfaceMessages::interface_error,
259  "Name '%s' cannot be translated to an address.",
260  spec.c_str());
261  result = nullptr;
262  }
263  }
264 
265  // Name is all spaces or is empty: Complain and return NULL.
266  else
267  {
268  MessageHandler::error(__FILE__,
269  __LINE__,
270  SimInterfaceMessages::interface_error,
271  "Name '%s' is illegal.",
272  spec.c_str());
273  result = nullptr;
274  }
275  }
276 
277  return result;
278 }
