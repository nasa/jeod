105 /**
106  * NutationMars specific implementation of update_rotation, used here
107  * to describe nutation effects in longitude and obliquity
108  */
109 void
110 NutationMars::update_rotation (
111  void)
112 {
113 
114  // Cache off time locally for speedup and legibility
115  double time = current_time;
116 
117  // Set up the equations to calculate the nutation corrections, from pg. 40
118  // of Konopliv 2006:
119  // Psi_nut = sum[m=1..9]{Psi_m * sin(alpha_m*time + theta_m)}
120  // I_nut = I_oo + sum[m=1..9]{I_m * cos(alpha_m*time + theta_m)}
121 
122  // Set up nutation parameters for zeroth step of summation
123  nutation_in_longitude = 0.0;
124  nutation_in_obliquity = I_m_orig[0];
125 
126  // Initialize the variables used in the summation
127  double alpha_m;
128  double theta_m;
129 
130  // Perform the summation stackup
131  for (unsigned int ii = 1; ii <= 9; ++ii) {
132 
133  if (ii <= 3) {
134  alpha_m = int_to_double[ii] * mean_motion;
135  theta_m = int_to_double[ii] * mean_anomaly_j2000;
136  }
137  else {
138  alpha_m = (int_to_double[ii] - 3.0) * mean_motion;
139  theta_m = (int_to_double[ii] - 3.0) * mean_anomaly_j2000 +
140  q_angle_j2000;
141  }
142 
143  nutation_in_longitude += psi_m_orig[ii] * sin(alpha_m * time + theta_m);
144  nutation_in_obliquity += I_m_orig[ii] * cos(alpha_m * time + theta_m);
145  }
146 
147 
148  // Next, calculate the current obliquity angle, from pg 39 of Konopliv 2006:
149  // I(t) = I_o + I_dot*time + I_nut
150  obliquity_angle = I_at_j2000 + I_dot * time + nutation_in_obliquity;
151 
152  // Take trigs of the obliquity angle to build its rotation matrix
153  double cos_I = cos (obliquity_angle);
154  double sin_I = sin (obliquity_angle);
155 
156  // Populate the rotation matrix as follows:
157  // celestial frame = rot_x(-I) * mars frame
158  Matrix3x3::initialize (rotation);
159 
160  rotation[0][0] = 1.0;
161  rotation[0][1] = 0.0;
162  rotation[0][2] = 0.0;
163 
164  rotation[1][0] = 0.0;
165  rotation[1][1] = cos_I;
166  rotation[1][2] = -sin_I;
167 
168  rotation[2][0] = 0.0;
169  rotation[2][1] = sin_I;
170  rotation[2][2] = cos_I;
171 
172  return;
173 }
