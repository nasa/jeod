70 /**
71  * NutationMars specific implementation of update_rotation, used here
72  * to describe nutation effects in longitude and obliquity
73  */
74 void NutationMars::update_rotation()
75 {
76  // Cache off time locally for speedup and legibility
77  double time = current_time;
78 
79  // Set up the equations to calculate the nutation corrections, from pg. 40
80  // of Konopliv 2006:
81  // Psi_nut = sum[m=1..9]{Psi_m * sin(alpha_m*time + theta_m)}
82  // I_nut = I_oo + sum[m=1..9]{I_m * cos(alpha_m*time + theta_m)}
83 
84  // Set up nutation parameters for zeroth step of summation
85  nutation_in_longitude = 0.0;
86  nutation_in_obliquity = I_m_orig[0];
87 
88  // Initialize the variables used in the summation
89  double alpha_m;
90  double theta_m;
91 
92  // Perform the summation stackup
93  for(unsigned int ii = 1; ii <= 9; ++ii)
94  {
95  if(ii <= 3)
96  {
97  alpha_m = int_to_double[ii] * mean_motion;
98  theta_m = int_to_double[ii] * mean_anomaly_j2000;
99  }
100  else
101  {
102  alpha_m = (int_to_double[ii] - 3.0) * mean_motion;
103  theta_m = (int_to_double[ii] - 3.0) * mean_anomaly_j2000 + q_angle_j2000;
104  }
105 
106  nutation_in_longitude += psi_m_orig[ii] * sin(alpha_m * time + theta_m);
107  nutation_in_obliquity += I_m_orig[ii] * cos(alpha_m * time + theta_m);
108  }
109 
110  // Next, calculate the current obliquity angle, from pg 39 of Konopliv 2006:
111  // I(t) = I_o + I_dot*time + I_nut
112  obliquity_angle = I_at_j2000 + I_dot * time + nutation_in_obliquity;
113 
114  // Take trigs of the obliquity angle to build its rotation matrix
115  double cos_I = cos(obliquity_angle);
116  double sin_I = sin(obliquity_angle);
117 
118  // Populate the rotation matrix as follows:
119  // celestial frame = rot_x(-I) * mars frame
120  Matrix3x3::initialize(rotation);
121 
122  rotation[0][0] = 1.0;
123  rotation[0][1] = 0.0;
124  rotation[0][2] = 0.0;
125 
126  rotation[1][0] = 0.0;
127  rotation[1][1] = cos_I;
128  rotation[1][2] = -sin_I;
129 
130  rotation[2][0] = 0.0;
131  rotation[2][1] = sin_I;
132  rotation[2][2] = cos_I;
133 }
