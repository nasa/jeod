57  * Complete initialization of a DynBodyInit.
58  * The initialize method for all subclasses of DynBodyInit *nust* pass
59  * the initialize call to their immediate parent class, which in turn
60  * must do the same, eventually invoking this method.
61  * \param[in,out] dyn_manager Dynamics manager
62  */
63 void DynBodyInit::initialize(DynManager & dyn_manager)
64 {
65  // Forward the request up the chain.
66  BodyAction::initialize(dyn_manager);
67 
68  // Sanity check:
69  // The subject MassBody must belong a DynBody.
70  if(dyn_subject == nullptr)
71  {
72  MessageHandler::fail(__FILE__,
73  __LINE__,
74  BodyActionMessages::invalid_object,
75  "%s failed:\n"
76  "Subject body '%s' is a %s.\n"
77  "A subject body of a class derived from DynBody is required.",
78  action_identifier.c_str(),
79  mass_subject->name.c_str(),
80  typeid(*mass_subject).name());
81  return;
82  }
83 
84  // Sanity check: The body must be registered with the dynamics manager.
85  if(!dyn_manager.is_dyn_body_registered(dyn_subject))
86  {
87  MessageHandler::fail(__FILE__,
88  __LINE__,
89  BodyActionMessages::invalid_object,
90  "The subject '%s' of action '%s' has not been initialized.\n",
91  dyn_subject->name.c_str(),
92  action_identifier.c_str());
93  return;
94  }
95 
96  // Find the body frame that will be populated by this initialization.
97  // Note: A child class may have already set this, in which case the
98  // search is not needed.
99  if(body_ref_frame == nullptr)
100  {
101  if(body_frame_id.empty())
102  {
103  MessageHandler::warn(__FILE__,
104  __LINE__,
105  "Incomplete assignment to dyn-body-init\n",
106  "The body_ref_frame and body_frame_id have not been set.\n"
107  "Typically, the body_frame_id would be used to identify \n"
108  "the body_ref_frame.\n"
109  "Setting body_frame_id to the default `composite_body`.\n");
110  body_frame_id = "composite_body";
111  }
112  body_ref_frame = find_body_frame(*dyn_subject, body_frame_id, "body_frame_id");
113  }
114 
115  // Find the reference frame to which the initialization data refers.
116  // Note: A child class may have already set this, in which case the
117  // search is not needed.
118  if(reference_ref_frame == nullptr)
119  {
120  reference_ref_frame = find_ref_frame(dyn_manager, reference_ref_frame_name, "reference_ref_frame_name");
121  }
122 
123  // (Temporarily) subscribe to the frame to force it to be placed in the
124  // reference frame tree. The subscription will be withdrawn by apply().
125  subscribed_frame = reference_ref_frame;
126  dyn_manager.subscribe_to_frame(*subscribed_frame);
127 }

140 /**
141  * In general, determine if the initializer is ready to be applied.
142  * This method determines whether the self-dependencies are satisfied.
143  * Dependencies on the reference reference frame are the responsibility
144  * of derived classes.
145  * @return Can initializer run?
146  */
147 bool DynBodyInit::is_ready()
148 {
149  RefFrameItems set_items(initializes_what());
150  RefFrameItems required_items;
151 
152  // All BodyAction derived classes must query the parent class' readiness.
153  if(!BodyAction::is_ready())
154  {
155  return false;
156  }
157 
158  // A backwards (reverse_sense) specification requires some self knowledge.
159  if(reverse_sense)
160  {
161  // Attitude is always needed in the case of a backards specification.
162  required_items.add(RefFrameItems::Att);
163 
164  // Position and rate are also needed for a reverse-sense velocity spec.
165  if(set_items.contains(RefFrameItems::Vel))
166  {
167  required_items.add(RefFrameItems::Pos_Rate);
168  }
169  }
170 
171  // In a normal sense specification, computation of velocity nominally
172  // requires knowledge of relative position and computation of rate requires
173  // knowledge of relative orientation.
174  // This information is *not* needed if the reference reference frame is
175  // known to be non-rotating. However, this information is not available
176  // without extensive kludginess.
177  else
178  {
179  if(set_items.contains(RefFrameItems::Vel))
180  {
181  required_items.add(RefFrameItems::Pos);
182  }
183  if(set_items.contains(RefFrameItems::Rate))
184  {
185  required_items.add(RefFrameItems::Att);
186  }
187  }
188 
189  // Self-knowledge of the attitude is needed when rate is specified in parent.
190  if(set_items.contains(RefFrameItems::Rate) && rate_in_parent)
191  {
192  required_items.add(RefFrameItems::Att);
193  }
194 
195  // The items provided by this initializer do not need to be previously set.
196  required_items.remove(set_items.get());
197 
198  // The initializer is ready-to-run if the requisite elements of the
199  // to-be-initialized state have been set.
200  return body_ref_frame->initialized_items.contains(required_items.get());
201 }

211 {
212  RefFrameItems::Items set_items = // The newly set state elements

214 
215  // Debug.
216  MessageHandler::debug(__FILE__,
217  __LINE__,
218  BodyActionMessages::trace,

233  dyn_subject->propagate_state();

235

237  dyn_manager.unsubscribe_to_frame(*subscribed_frame);

239  // Forward the action up the chain.

268 /**
269  * Compute the state wrt the reference reference frame,
270  * incorporate the user-input items to this relative state, and
271  * compute the state relative to the target frame's parent.
272  */
273 void DynBodyInit::apply_user_inputs()
274 {
275  RefFrame user_frame; /* --
276  The user-specified orientation and angular velocity are specified with
277  respect to some user-specified reference ref frame. This user frame
278  is attached to this reference ref frame and populated with the user-
279  specified rotational state. */
280  RefFrameItems set_items(initializes_what()); /* --
281  The items to be set from the user inputs. */
282 
283  // (Temporarily) connect the above user_frame under the reference ref frame.
284  reference_ref_frame->add_child(user_frame);
285 
286  // Initialize the user_frame state with the existing state information.
287  body_ref_frame->compute_relative_state(*reference_ref_frame, reverse_sense, user_frame.state);
288 
289  // Overlay any user inputs on top of the state.
290  if(set_items.contains(RefFrameItems::Pos))
291  {
292  Vector3::copy(position, user_frame.state.trans.position);
293  }
294  if(set_items.contains(RefFrameItems::Vel))
295  {
296  Vector3::copy(velocity, user_frame.state.trans.velocity);
297  }
298  if(set_items.contains(RefFrameItems::Att))
299  {
300  orientation.compute_transform();
301  Matrix3x3::copy(orientation.trans, user_frame.state.rot.T_parent_this);
302  user_frame.state.rot.compute_quaternion();
303  }
304  if(set_items.contains(RefFrameItems::Rate))
305  {
306  if(rate_in_parent)
307  {
308  Vector3::transform(user_frame.state.rot.T_parent_this, ang_velocity, user_frame.state.rot.ang_vel_this);
309  }
310  else
311  {
312  Vector3::copy(ang_velocity, user_frame.state.rot.ang_vel_this);
313  }
314  user_frame.state.rot.compute_ang_vel_products();
315  }
316 
317  if(reverse_sense)
318  {
319  RefFrameState user_state_copy(user_frame.state);
320  user_frame.state.negate(user_state_copy);
321  }
322 
323  // Compute the state relative to the vehicle's integration frame.
324  user_frame.compute_relative_state(*(dyn_subject->get_integ_frame()), state);
325 
326  // Remove the temporary connection established above.
327  user_frame.remove_from_parent();
328 }
