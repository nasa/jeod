148 int EarthLighting::circle_intersect(double r_bottom, double r_top, double d_centers, double * area)
149 {
150  double d_c2;
151  double r_t2;
152  double r_b2;
153  double diff_r2;
154  double top_ang;
155  double cos_top_ang;
156  double top_area;
157  double bottom_ang;
158  double cos_bottom_ang;
159  double bottom_area;
160 
161  /***************************************************/
162  /* Quick check to see if circles are intersecting. */
163  /***************************************************/
164  if(d_centers > r_bottom + r_top)
165  {
166  /* Compute the area of intersection. */
167  *area = 0.0;
168 
169  /* Return negative intersection flag. */
170  return (0);
171  }
172 
173  /*************************************************************/
174  /* Check to see if one circle completely contains the other. */
175  /*************************************************************/
176  if(r_bottom > r_top)
177  {
178  /* Check to see if top circle is completely inside bottom circle. */
179  if(d_centers < (r_bottom - r_top) + EPSILON)
180  {
181  /* Area of intersection is entire top circle. */
182  *area = M_PI * r_top * r_top;
183 
184  /* Return positive intersection flag. */
185  return (1);
186  }
187  }
188  else
189  {
190  /* Check to see if top circle completely covers bottom circle. */
191  if(d_centers < (r_top - r_bottom) + EPSILON)
192  {
193  /* Area of intersection is entire bottom circle. */
194  *area = M_PI * r_bottom * r_bottom;
195 
196  /* Return positive intersection flag. */
197  return (1);
198  }
199  }
200 
201  /*********************************************************************/
202  /* Compute the area of intersection of the two overlappping circles. */
203  /*********************************************************************/
204 
205  /* Compute working variables. */
206  d_c2 = d_centers * d_centers;
207  r_t2 = r_top * r_top;
208  r_b2 = r_bottom * r_bottom;
209  diff_r2 = r_b2 - r_t2;
210 
211  /* Compute the bottom intersection angle (law of cosines). */
212  cos_bottom_ang = (d_c2 + diff_r2) / (2.0 * d_centers * r_bottom);
213  bottom_ang = acos(cos_bottom_ang);
214 
215  /* Compute the top intersection angles (law of cosines). */
216  cos_top_ang = (d_c2 - diff_r2) / (2.0 * d_centers * r_top);
217  top_ang = acos(cos_top_ang);
218 
219  /* Compute the area of intersection. */
220  top_area = r_t2 * (top_ang - (sin(top_ang) * cos_top_ang));
221  bottom_area = r_b2 * (bottom_ang - (sin(bottom_ang) * cos_bottom_ang));
222  *area = top_area + bottom_area;
223 
224  /* Return a positive for intersection. */
225  return (1);
226 }

233 void EarthLighting::calc_lighting(const double pos_veh[3])
234 {
235  if(active == false)
236  {
237  return;
238  }
239 
240  int iinc;
241 
242  double eclipse_area;
243 
244  double cross_prod[3];
245  double cross_mag;
246  double cos_obs_ang;
247  double sin_obs_ang;
248 
249  // Get the necessary relative positions
250  sun_frame->compute_position_from(*earth_frame, pos_sun);
251  moon_frame->compute_position_from(*earth_frame, pos_moon);
252 
253  /* Compute the relative positions of the celestial bodies. */
254  for(iinc = 0; iinc < 3; iinc++)
255  {
256  moon_body.position[iinc] = pos_moon[iinc] - pos_veh[iinc];
257  sun_body.position[iinc] = pos_sun[iinc] - pos_veh[iinc];
258  earth_body.position[iinc] = -pos_veh[iinc];
259  }
260 
261  /* Compute the distance to the celestial bodies. */
262  moon_body.distance = Vector3::vmag(moon_body.position);
263  sun_body.distance = Vector3::vmag(sun_body.position);
264  earth_body.distance = Vector3::vmag(earth_body.position);
265 
266  /* Compute the apparent half angles of the celestial bodies. */
267  moon_body.half_angle = asin(moon_body.radius / moon_body.distance);
268  sun_body.half_angle = asin(sun_body.radius / sun_body.distance);
269  if(earth_body.distance >= earth_body.radius)
270  {
271  earth_body.half_angle = asin(earth_body.radius / earth_body.distance);
272  }
273  else
274  {
275  earth_body.half_angle = M_PI_2;
276  }
277 
278  /**********************************************************************/
279  /* Compute the observation angle between the earth and light sources. */
280  /**********************************************************************/
281  /* Moon */
282  // cos_obs_ang = V_DOT( moon_body.position, earth_body.position )
283  // / ( moon_body.distance* earth_body.distance );
284  cos_obs_ang = Vector3::dot(moon_body.position, earth_body.position) / (moon_body.distance * earth_body.distance);
285  // V_CROSS( cross_prod, moon_body.position, earth_body.position );
286  Vector3::cross(moon_body.position, earth_body.position, cross_prod);
287  cross_mag = Vector3::vmag(cross_prod);
288  sin_obs_ang = cross_mag / (moon_body.distance * earth_body.distance);
289  moon_earth.obs_angle = atan2(sin_obs_ang, cos_obs_ang);
290 
291  /* Sun */
292  // cos_obs_ang = V_DOT( sun_body.position, earth_body.position )
293  // / ( sun_body.distance* earth_body.distance );
294  cos_obs_ang = Vector3::dot(sun_body.position, earth_body.position) / (sun_body.distance * earth_body.distance);
295  // V_CROSS( cross_prod, sun_body.position, earth_body.position );
296  Vector3::cross(sun_body.position, earth_body.position, cross_prod);
297  cross_mag = Vector3::vmag(cross_prod);
298  sin_obs_ang = cross_mag / (sun_body.distance * earth_body.distance);
299  sun_earth.obs_angle = atan2(sin_obs_ang, cos_obs_ang);
300 
301  /* Determine if earth occludes the sun. */
302  /* Note: these represent arc-lengths on a unit sphere. */
303  circle_intersect(sun_body.half_angle, earth_body.half_angle, sun_earth.obs_angle, &eclipse_area);
304 
305  /* Compute the sun lighting from the eclipse area. */
306  sun_earth.occlusion = eclipse_area / (sun_body.half_angle * sun_body.half_angle * M_PI);
307  sun_earth.visible = 1.0 - sun_earth.occlusion;
308  sun_earth.lighting = sun_earth.phase * sun_earth.visible;
309 
310  /* Determine if earth occludes the moon. */
311  /* Note: these represent arc-lengths on a unit sphere. */
312  circle_intersect(moon_body.half_angle, earth_body.half_angle, moon_earth.obs_angle, &eclipse_area);
313 
314  /* Compute the moon lighting from the eclipse area. */
315  moon_earth.occlusion = eclipse_area / (moon_body.half_angle * moon_body.half_angle * M_PI);
316  moon_earth.visible = 1.0 - moon_earth.occlusion;
317 
318  /* Apply further scaling by apparent lunar phase. */
319  moon_earth.lighting = moon_earth.phase * moon_earth.visible;
320 
321  /* Crude approximation for Earth albedo. */
322  earth_albedo.lighting = fabs(sun_earth.obs_angle / M_PI);
323  earth_albedo.lighting *= sun_earth.lighting;
324 }
