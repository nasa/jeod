50 {
51  MassBody * found_body = nullptr;
52 
53  // Ensure the passed name has a minimally valid value.
54  if(!validate_name(__FILE__, __LINE__, body_name, "Argument", "name"))
55  {
56  return nullptr;
57  }
58 
59  // Find the body by name.
60  auto name_compare = [&body_name](const MassBody * body)
61  {
62  return body->name.ends_with(0, body_name);
63  };
64 
65  const auto iter = std::find_if(mass_bodies.begin(), mass_bodies.end(), name_compare);

89 void DynManager::add_mass_body(MassBody & mass_body)

91  // Handle errors.

108 
109  // 3. The body must have a unique name.
110  if(find_mass_body(mass_body.name.get_name()) != nullptr)
111  {
112  MessageHandler::error(__FILE__,

114  DynManagerMessages::duplicate_entry,
115  "Mass body with name '%s' was previously registered.",
116  mass_body.name.c_str());
117  return;
118  }
119 
120  // All tests passed: Add the body to the mass body registry.
121  mass_bodies.push_back(&mass_body);
122 }
