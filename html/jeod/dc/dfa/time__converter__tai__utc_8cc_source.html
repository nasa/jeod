73 void TimeConverter_TAI_UTC::initialize(JeodBaseTime * parent_ptr, JeodBaseTime * child_ptr, const int int_dir)
74 {
75  verify_setup(parent_ptr, child_ptr, int_dir);
76 
77  if(int_dir == 1)
78  {
79  tai_ptr = dynamic_cast<TimeTAI *>(parent_ptr);
80  utc_ptr = dynamic_cast<TimeUTC *>(child_ptr);
81  // lookup table is calibrated in UTC,
82  if(utc_ptr->is_initialized())
83  {
84  initialize_leap_second();
85  }
86  else
87  {
88  // start with TAI as first iteration
89  utc_ptr->trunc_julian_time = tai_ptr->trunc_julian_time;
90  initialize_leap_second();
91 
92  // Now adjust TAI as second iteration.

94 
95  /* If the leap-second correction takes UTC back before the time of the
96  last leap second, and if it does so by more than the leap second
97  itself, need to reinitialize */
98  if((index > 0) &&
99  ((when_vec[index] - utc_ptr->trunc_julian_time) > ((val_vec[index] - val_vec[index - 1]) / 86400.0)))
100  {
101  initialize_leap_second();
102  }
103  }
104  }
105  else if(int_dir == -1)
106  {
107  tai_ptr = dynamic_cast<TimeTAI *>(child_ptr);
108  utc_ptr = dynamic_cast<TimeUTC *>(parent_ptr);
109  initialize_leap_second(); // utc must be initialized because it is parent
110  }
111  else
112  {
113  MessageHandler::fail(__FILE__,
114  __LINE__,
115  TimeMessages::initialization_error,
116  "\n"

122 /**
123  * The conversion from Atomic Time (TAI) to Universal Time (UTC)
124  * involves the addition of leap seconds. The number of leap seconds at any
125  * given (historical) time is provided in a table. This function initializes
126  * that table and sets the preliminary values.
127  *
128  * \par Assumptions and Limitations
129  * - The table does not go into the future.
130  */
131 void TimeConverter_TAI_UTC::initialize_leap_second()
132 {
133  /* if the user inputs their own value for a_to_b_offset, make sure that it
134  does not get overridden. */
135  if(override_data_table)
136  {
137  utc_ptr->true_utc = false;
138  a_to_b_offset = -leap_sec_override_val / 86400.0;
139  return;
140  }
141 
142  if((when_vec == nullptr) || (val_vec == nullptr))
143  {
144  MessageHandler::fail(__FILE__,
145  __LINE__,
146  TimeMessages::invalid_setup_error,
147  "\n"
148  "The TAI to UTC conversion function requires a"
149  " data lookup table, or the flag 'override_data_table' to be set"
150  " but none has been provided and the flag remains 'false'.\n");
151  return;
152  }
153 
154  double trunc_julian_time = utc_ptr->trunc_julian_time; /* --
155  local value of UTC time in TJT format */
156  off_table_end = false;
157 
158  if(trunc_julian_time < when_vec[0])
159  {
160  /* Julian date is less than initial date in table, set leap_second to 0 */
161  if(tai_ptr->time_manager->dyn_time.scale_factor >= 0)
162  {
163  MessageHandler::inform(__FILE__,
164  __LINE__,
165  TimeMessages::invalid_data_error,
166  "\n"
167  "\nThe current time precedes the first entry in the"
168  " look-up table for leap-second values. \n"
169  "Leap_second value is being set to the first value in the table"
170  " (%i s) .\n"
171  "Simulation will continue to run and will continue to monitor\n"
172  "time for future crossing into the tabulated range.\n",
173  val_vec[0]);
174  index = -1; // because update will increment to index+1, i.e. 0
175  next_when = when_vec[0];
176  prev_when = trunc_julian_time - (when_vec[0] - trunc_julian_time);
177  // something less than trunc_julian_time
178  }
179  else
180  {
181  MessageHandler::inform(__FILE__,
182  __LINE__,
183  TimeMessages::invalid_data_error,
184  "\n"
185  "\nThe current time precedes the first entry in the"
186  " look-up table for leap-second values. \n"
187  "Leap_second value is being set to the first value in the table"
188  " (%i s) .\n"
189  "Simulation will continue to run, and this value will be\n"
190  "maintained unless the time-direction changes.\n",
191  val_vec[0]);
192  off_table_end = true;
193  index = 0;
194  }
195  a_to_b_offset = -val_vec[0] / 86400.0;
196  }
197  else if(trunc_julian_time >= when_vec[last_index])
198  {
199  /* Julian date is beyond the table, set leap-second to last known value */
200  index = last_index;
201  if(tai_ptr->time_manager->dyn_time.scale_factor >= 0)
202  {
203  MessageHandler::inform(__FILE__,
204  __LINE__,
205  TimeMessages::invalid_data_error,
206  "\n"
207  "\nThe current time follows the last entry in the"
208  " look-up table for leap-second values. \n"
209  "Leap_second value is being set to the last value in the table"
210  " (%i s) .\n"
211  "Simulation will continue to run, and this value will be\n"
212  "maintained unless the time-direction changes.\n",
213  val_vec[last_index]);
214  off_table_end = true;
215  }
216  else
217  {
218  MessageHandler::inform(__FILE__,
219  __LINE__,
220  TimeMessages::invalid_data_error,
221  "\n"
222  "\nThe current time follows the last entry in the"
223  " look-up table for leap-second values. \n"
224  "Leap_second value is being set to the last value in the table"
225  " (%i s) .\n"
226  "Simulation will continue to run and will continue to monitor\n"
227  "time for future crossing into the tabulated range.\n",
228  val_vec[last_index]);
229  prev_when = when_vec[last_index];
230  next_when = trunc_julian_time + (trunc_julian_time - when_vec[last_index]);
231  // some value more than trunc_julian_time
232  }
233  a_to_b_offset = -val_vec[last_index] / 86400.0;
234  }
235  else
236  {
237  /* Find the last entry that comes before the current time (trunc_julian_time)
238  Highest value that can be reached is index = last_index -1
239  because trunc_julian_time < leap_when(last_leap_index) by earlier test. */
240  for(index = 0; trunc_julian_time >= when_vec[index + 1]; ++index)
241  {
242  ;
243  }
244  prev_when = when_vec[index];
245  next_when = when_vec[index + 1];
246  /* and take the number of leap seconds recorded in the previous entry (i.e.
247  the last entry before the current time) */
248  a_to_b_offset = -val_vec[index] / 86400.0;
249  }
250 }

252 /**
253  * Convert from TimeTAI to TimeUTC.
254  *
255  * \par Assumptions and Limitations
256  * - Time class MET is based on time class TAI, and counts the
257  * elapsed TAI time only
258  */
259 void TimeConverter_TAI_UTC::convert_a_to_b()
260 {
261  utc_ptr->trunc_julian_time = tai_ptr->trunc_julian_time + a_to_b_offset;
262 
263  // check for potential updates to a_to_b_offset:
264  if(!off_table_end && utc_ptr->true_utc)
265  {
266  // for conventional (forward-time) simulations.
267  if(tai_ptr->time_manager->dyn_time.scale_factor >= 0)
268  {
269  //"while" is used because updates need not be done regularly, may have
270  // to catch up with accumulation of points. */
271  while(utc_ptr->trunc_julian_time >= next_when)
272  {
273  index++;
274  /* assign these here. It is redundant if the loop runs more than once,
275  but assigning after the loop would be redundant if the loop was not
276  entered at all. The latter case is more likely, so assigning here
277  is more efficient. */
278  a_to_b_offset = -val_vec[index] / 86400.0;
279  utc_ptr->set_time_by_trunc_julian(tai_ptr->trunc_julian_time + a_to_b_offset);
280  // utc_ptr->trunc_julian_time = tai_ptr->trunc_julian_time + a_to_b_offset;
281 
282  prev_when = next_when;
283 
284  if(index < last_index)
285  {
286  next_when = when_vec[index + 1];
287  }
288  else
289  { // run out of table!
290  MessageHandler::inform(__FILE__,
291  __LINE__,
292  TimeMessages::invalid_data_error,
293  "\n"
294  " Current time is beyond the range covered by the leap-second table.\n"
295  " The difference between TAI and UTC is being set to the last value \n"
296  " in the table. The simulation will continue to run with this value. "
297  " \n");
298  off_table_end = true;
299  break;
300  }
301  }
302  }
303  else
304  {
305  // for reverse-time simulations.
306  while(utc_ptr->trunc_julian_time < prev_when)
307  {
308  index--;
309  next_when = prev_when;
310 
311  if(index >= 0)
312  {
313  a_to_b_offset = -val_vec[index] / 86400.0;
314  utc_ptr->set_time_by_trunc_julian(tai_ptr->trunc_julian_time + a_to_b_offset);
315  prev_when = when_vec[index];
316  }
317  else
318  { // run out of table!
319  MessageHandler::inform(__FILE__,
320  __LINE__,
321  TimeMessages::invalid_data_error,
322  "\n"
323  "Current time is beyond the range covered by the leap-second table.\n"
324  "The difference between TAI and UTC is being set to the first value \n"
325  "in the table. The simulation will continue to run with this value."
326  " \n");
327  off_table_end = true;
328  a_to_b_offset = 0;
329  utc_ptr->set_time_by_trunc_julian(tai_ptr->trunc_julian_time + a_to_b_offset);
330  break;
331  }
332  }
333  }
334  }
335  /* if no update to leap seconds was needed, do not need to change
336  trunc_julian_time from that calculated earlier.
337  if an update was needed, it has already been recalculated.
338  Either way, do not need to do it again here. Done. */
339  utc_ptr->set_time_by_trunc_julian(tai_ptr->trunc_julian_time + a_to_b_offset);
340 
341  // utc_ptr->days = utc_ptr->trunc_julian_time - utc_ptr->tjt_at_epoch;
342  // utc_ptr->seconds = utc_ptr->days * 86400;
343 }

417 /**
418  * Used when time reverses direction.
419  * Checks whether the table lookup function is using input values that
420  * are outside the scope of the table, and sets the flags appropriately
421  */
422 void TimeConverter_TAI_UTC::verify_table_lookup_ends()
423 {
424  if(!utc_ptr->true_utc)
425  { // table is not monitored
426  return;
427  }
428 
429  // if now going forward in time
430  if(tai_ptr->time_manager->dyn_time.scale_factor >= 0)
431  {
432  // and past the end of the table, past the table
433  if(index >= last_index)
434  {
435  off_table_end = true;
436  }
437  /* or if before the beginning of the table, no longer past the table. */
438  /* Change "prev_when" to be before "now", or the converter will
439  automatically revert to being off the table */
440  else if(index < 0)
441  {
442  prev_when = utc_ptr->trunc_julian_time - 1;
443  next_when = when_vec[0];
444  index = -1;
445  off_table_end = false;
446  }
447  }
448  else
449  { // reverse
450  if(index >= last_index)
451  {
452  next_when = utc_ptr->trunc_julian_time + 1;
453  prev_when = when_vec[last_index];
454  index = last_index;
455  off_table_end = false;
456  }
457  else if(index < 0)
458  {
459  off_table_end = true;
460  }
461  }
462 }
