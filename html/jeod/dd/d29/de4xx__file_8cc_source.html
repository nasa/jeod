427  *
428  * \par Assumptions and Limitations

430  * - Errors are fatal
431  */
432 void
433 De4xxFile::close (
434  void)
435 {
436  int rc;
437 
438  // Nothing to do if file is already closed.
439  if (io.file == nullptr) {
440  return;
441  }
442 
443  // Close the file.
444  rc = dlclose (io.file);
445 
446  // Check for success.
447  // NOTE: Failure is a warning here as the close may be the result of an
448  // exec_terminate call elsewhere.
449  if (rc != 0) {
450  MessageHandler::warn (
451  __FILE__, __LINE__, EphemeridesMessages::file_error,
452  "Error closing ephemeris file: %s",
453  std::strerror(errno));
454  }
455 
456  // Denote that the file is closed.
457  io.file = nullptr;
458  io.metaData = nullptr;
459 
460  // Free allocated memory.
461  JEOD_DELETE_ARRAY (coef.chebypoly);
462  JEOD_DELETE_ARRAY (coef.chebyderiv);

466 /**
467  * Check whether the specified time is represented in the JPL ephemeris file.
468  *
469  * \par Assumptions and Limitations
470  * - Ephemeris file is open for input
471  * - Ephemeris file is blocked per value set in the ephem_file structure
472  * @return True if time is in file
473  * \param[in] time Time since reference\n Units: s
474  */
475 bool
476 De4xxFile::time_is_in_range (
477  double time)
478 const
479 {
480  double recno;
481 
482  // Compute record number corresponding to this time.
483  // Note that this does not quite do the right thing for negative values.
484  // This is OK since the minimum value record number is two.
485  recno = ref_time.block_no +
486  (time - ref_time.init_time) / (86400.0 * io.metaData->delta_epoch);
487 
488 
489  // Return an indicator as to whether the input time is in range.
490  // Note that the first two data records are out of range.
491  return (recno >= 0) &&
492  (recno < io.total_num_recs);

495 void process_mem_usage(double& vm_usage, double& resident_set)
496 {
497  using std::ios_base;
498  using std::ifstream;
499  using std::string;
500 
501  vm_usage = 0.0;
502  resident_set = 0.0;
503 
504  // 'file' stat seems to give the most reliable results
505  //
506  ifstream stat_stream("/proc/self/stat",ios_base::in);
507 
508  // dummy vars for leading entries in stat that we don't care about
509  //
510  string pid, comm, state, ppid, pgrp, session, tty_nr;
511  string tpgid, flags, minflt, cminflt, majflt, cmajflt;
512  string utime, stime, cutime, cstime, priority, nice;
513  string O, itrealvalue, starttime;
514 
515  // the two fields we want
516  //
517  unsigned long vsize;
518  long rss;
519 
520  stat_stream >> pid >> comm >> state >> ppid >> pgrp >> session >> tty_nr
521  >> tpgid >> flags >> minflt >> cminflt >> majflt >> cmajflt
522  >> utime >> stime >> cutime >> cstime >> priority >> nice
523  >> O >> itrealvalue >> starttime >> vsize >> rss; // don't care about the rest
524 
525  stat_stream.close();
526 
527  long page_size_kb = sysconf(_SC_PAGE_SIZE) / 1024; // in case x86-64 is configured to use 2MB pages
528  vm_usage = vsize / 1024.0;
529  resident_set = rss * page_size_kb;
530 }
