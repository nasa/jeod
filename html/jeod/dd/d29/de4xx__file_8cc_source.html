338 void De4xxFile::close()
339 {
340  int rc;
341 
342  // Nothing to do if file is already closed.
343  if(io.file == nullptr)
344  {
345  return;
346  }
347 
348  // Close the file.
349  rc = dlclose(io.file);
350 
351  // Check for success.
352  // NOTE: Failure is a warning here as the close may be the result of an
353  // exec_terminate call elsewhere.
354  if(rc != 0)
355  {
356  MessageHandler::warn(__FILE__,
357  __LINE__,
358  EphemeridesMessages::file_error,

362 
363  // Denote that the file is closed.
364  io.file = nullptr;
365  io.metaData = nullptr;
366 
367  // Free allocated memory.
368  JEOD_DELETE_ARRAY(coef.chebypoly);
369  JEOD_DELETE_ARRAY(coef.chebyderiv);

372 /**
373  * Check whether the specified time is represented in the JPL ephemeris file.
374  *
375  * \par Assumptions and Limitations
376  * - Ephemeris file is open for input
377  * - Ephemeris file is blocked per value set in the ephem_file structure
378  * @return True if time is in file
379  * \param[in] time Time since reference\n Units: s
380  */
381 bool De4xxFile::time_is_in_range(double time) const
382 {
383  double recno;
384 
385  // Compute record number corresponding to this time.
386  // Note that this does not quite do the right thing for negative values.
387  // This is OK since the minimum value record number is two.
388  recno = ref_time.block_no + (time - ref_time.init_time) / (86400.0 * io.metaData->delta_epoch);
389 
390  // Return an indicator as to whether the input time is in range.
391  // Note that the first two data records are out of range.
392  return (recno >= 0) && (recno < io.total_num_recs);
393 }

395 void process_mem_usage(double & vm_usage, double & resident_set)
396 {
397  using std::ifstream;
398  using std::ios_base;
399  using std::string;
400 
401  vm_usage = 0.0;
402  resident_set = 0.0;
403 
404  // 'file' stat seems to give the most reliable results
405  //
406  ifstream stat_stream("/proc/self/stat", ios_base::in);
407 
408  // dummy vars for leading entries in stat that we don't care about
409  //
410  string pid, comm, state, ppid, pgrp, session, tty_nr;
411  string tpgid, flags, minflt, cminflt, majflt, cmajflt;
412  string utime, stime, cutime, cstime, priority, nice;
413  string O, itrealvalue, starttime;
414 
415  // the two fields we want
416  //
417  unsigned long vsize;
418  long rss;
419 
420  stat_stream >> pid >> comm >> state >> ppid >> pgrp >> session >> tty_nr >> tpgid >> flags >> minflt >> cminflt >>
421  majflt >> cmajflt >> utime >> stime >> cutime >> cstime >> priority >> nice >> O >> itrealvalue >> starttime >>
422  vsize >> rss; // don't care about the rest
423 
424  stat_stream.close();
425 
426  long page_size_kb = sysconf(_SC_PAGE_SIZE) / 1024; // in case x86-64 is configured to use 2MB pages
427  vm_usage = vsize / 1024.0;
428  resident_set = rss * page_size_kb;
429 }
