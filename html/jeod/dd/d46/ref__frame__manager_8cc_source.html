64 /**
65  * Add a reference frame to the reference frame registry.
66  * @param ref_frame Reference frame to be added.
67  */
68 void RefFrameManager::add_ref_frame(RefFrame & ref_frame)
69 {
70  // Handle errors.
71  // 1. The frame must have a valid name.
72  if(!validate_name(__FILE__, __LINE__, ref_frame.get_name(), "Argument", "ref_frame.name"))
73  {
74  return;
75  }
76 
77  // 2. The frame must not have been previously registered.
78  for(std::vector<RefFrame *>::const_iterator it = ref_frames.begin(); it != ref_frames.end(); ++it)
79  {
80  if(*it == &ref_frame)
81  {
82  MessageHandler::error(__FILE__,
83  __LINE__,
84  RefFrameMessages::duplicate_entry,
85  "Reference frame '%s' was previously registered.",
86  ref_frame.get_name().c_str());
87  return;
88  }
89  }
90 
91  // 3. The frame must have a unique name.
92  if(find_ref_frame(ref_frame.get_name()) != nullptr)
93  {
94  MessageHandler::error(__FILE__,
95  __LINE__,
96  RefFrameMessages::duplicate_entry,
97  "Reference frame with name '%s' was previously registered.",
98  ref_frame.get_name().c_str());
99  return;
100  }
101 
102  // Add the reference frame to the reference frame table.
103  ref_frames.push_back(&ref_frame);
104 }

106 /**
107  * Remove a reference frame from the reference frame registry.
108  * @param ref_frame Reference frame to be removed.
109  */
110 void RefFrameManager::remove_ref_frame(RefFrame & ref_frame)
111 {
112  auto it = std::find(ref_frames.begin(), ref_frames.end(), &ref_frame);
113 
114  if(it == ref_frames.end())
115  {
116  MessageHandler::error(__FILE__,
117  __LINE__,
118  RefFrameMessages::removal_failed,
119  "Reference frame %s was not found in the registry",
120  ref_frame.get_name().c_str());
121  return;
122  }
123 
124  ref_frames.erase(it);
125 }

150 /**
151  * Find the reference frame with the dot-conjoined name "${prefix}.${suffix}".
152  * @param prefix Reference frame name prefix
153  * @param suffix Reference frame name suffix
154  * @return Found reference frame, or NULL if not found
155  */
156 RefFrame * RefFrameManager::find_ref_frame(const std::string & prefix, const std::string & suffix) const
157 {
158  RefFrame * found_frame = nullptr;
159 
160  // Search the reference frame list until a match is found.
161  for(auto ref_frame : ref_frames)
162  {
163  std::string ref_frame_name = ref_frame->get_name();
164  if((ref_frame_name.compare(0, prefix.length(), prefix) == 0) && (ref_frame_name[prefix.length()] == '.') &&
165  (ref_frame_name.compare(prefix.length() + 1, std::string::npos, suffix) == 0))
166  {
167  found_frame = ref_frame;
168  break;
169  }
170  }
171 
172  // Return the found frame (if any).
173  return found_frame;
174 }
