73 /**
74  * Add a reference frame to the reference frame registry.
75  * @param ref_frame Reference frame to be added.
76  */
77 void
78 RefFrameManager::add_ref_frame (
79  RefFrame & ref_frame)
80 {
81 
82  // Handle errors.
83  // 1. The frame must have a valid name.
84  if (! validate_name (__FILE__, __LINE__,
85  ref_frame.get_name(), "Argument", "ref_frame.name")) {
86  return;
87  }
88 
89 
90  // 2. The frame must not have been previously registered.
91  for (std::vector<RefFrame *>::const_iterator it = ref_frames.begin();
92  it != ref_frames.end();
93  ++it) {
94  if (*it == &ref_frame) {
95  MessageHandler::error (
96  __FILE__, __LINE__, RefFrameMessages::duplicate_entry,
97  "Reference frame '%s' was previously registered.",
98  ref_frame.get_name());
99  return;
100  }
101  }
102 
103  // 3. The frame must have a unique name.
104  if (find_ref_frame (ref_frame.get_name()) != nullptr) {
105  MessageHandler::error (
106  __FILE__, __LINE__, RefFrameMessages::duplicate_entry,
107  "Reference frame with name '%s' was previously registered.",
108  ref_frame.get_name());
109  return;
110  }
111 
112  // Add the reference frame to the reference frame table.
113  ref_frames.push_back (&ref_frame);
114 }

168 /**
169  * Find the reference frame with the dot-conjoined name "${prefix}.${suffix}".
170  * @param prefix Reference frame name prefix
171  * @param suffix Reference frame name suffix
172  * @return Found reference frame, or NULL if not found
173  */
174 RefFrame *
175 RefFrameManager::find_ref_frame (
176  const char * prefix,
177  const char * suffix)
178 const
179 {
180  std::string pname(prefix);
181  RefFrame * found_frame = nullptr;
182 
183  // Search the reference frame list until a match is found.
184  for (std::vector<RefFrame *>::const_iterator it = ref_frames.begin();
185  it != ref_frames.end();
186  ++it) {
187  RefFrame * ref_frame = *it;
188  const char * ref_frame_name = ref_frame->get_name();
189  if ((std::strncmp (ref_frame_name, prefix, pname.length()) == 0) &&
190  (ref_frame_name[pname.length()] == '.') &&
191  (std::strcmp (&(ref_frame_name[pname.length()+1]), suffix) == 0)) {
192  found_frame = ref_frame;
193  break;
194  }
195  }
196 
197  // Return the found frame (if any).
198  return found_frame;
199 }
