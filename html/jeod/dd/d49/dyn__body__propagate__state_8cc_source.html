47 // Compute the relative state between the integrated frame's mass point
48 // and the source frame's mass point.
49 void DynBody::compute_ref_point_transform(const BodyRefFrame & source_frame,
50  const MassPoint ** const ref_point,
51  MassPointState & rel_state)
52 {
53  const MassPoint * source_point;
54  const MassPoint * target_point;
55  const auto * source_body = dynamic_cast<const DynBody *>(source_frame.get_owner());
56 
57  // Integrated frame is the structural frame:
58  // The source frame's mass point is a part of the tree rooted by the root
59  // body's structural point, which corresponds to the integration frame.
60  if(integrated_frame == &structure)
61  {
62  source_point = source_frame.mass_point;
63 
64  // The relative state only needs to be computed if there was no previous
65  // computation or if the previous computation was for a different point.
66  if(*ref_point != source_point)
67  {
68  *ref_point = source_point;
69  source_point->compute_state_wrt_pred(mass.structure_point, rel_state);
70  }
71  }
72 
73  // Integrated frame is the composite body frame:
74  // The path from the source frame to the root body's composite body frame
75  // is taken in one or two steps, depending on the source frame.
76  // First set the source_point and target_point, where
77  // - The source_point is one of source body's mass points that corresponds
78  // to the source_frame.
79  // - The target_point is the root body's mass point that is the root of the
80  // tree containing the source_point.
81  // Ideally, the target_point will correspond to the root body's composite
82  // body frame. That ideal is not always possible. If not, a second step
83  // will be needed to go from the target_point to the composite body frame.
84  // The selections are
85  // - Source frame is a composite body frame:
86  // source_point is the source body's composite_wrt_pbdy point,
87  // target_point is the root body's composite_wrt_pbdy point.
88  // - Source frame is a core body frame:
89  // source_point is the source body's core_wrt_composite point,
90  // target_point is the root body's composite_wrt_pbdy point.
91  // - Source frame is a structural frame or vehicle point frame:
92  // source_point is the source frame's mass_point,
93  // target_point is the root body's structure_point.
94  else if(integrated_frame == &composite_body)
95  {
96  if(&source_frame == &source_body->composite_body)
97  {
98  target_point = &mass.composite_wrt_pbdy;
99  source_point = &source_body->mass.composite_wrt_pbdy;
100  }
101  else if(&source_frame == &source_body->core_body)
102  {
103  target_point = &mass.composite_wrt_pbdy;
104  source_point = &source_body->mass.core_wrt_composite;
105  }
106  else
107  {
108  target_point = &mass.structure_point;
109  source_point = source_frame.mass_point;
110  }
111 
112  // Only perform the computation if this is a new relative state.
113  if(*ref_point != source_point)
114  {
115  *ref_point = source_point;
116 
117  // Compute state of source point with respect to the target.
118  source_point->compute_state_wrt_pred(*target_point, rel_state);
119 
120  // If the target point is the structure point, the rel_state computed
121  // above is the state of the source point/frame wrt the root body's
122  // structure origin/fram
123  // Incorporate the structure-to-body state into the rel_state.
124  if(target_point == &mass.structure_point)
125  {
126  // Nomenclature:
127  // C = source point
128  // B = structure
129  // A = composite
130  // Want r_A->C:A, T_A->C in rel_state
131  // Have r_B->C:B, T_B->C in rel_state (needs to be updated)
132  // Have r_B->A:B, T_B->A in composite_properties
133  // Relevant math:
134  // r_A->C:A = T_B->A * (r_B->C:B - r_B->A:B)
135  // T_A->C = T_B->C * T_A->B = T_B->C * T_B->A^T
136  Vector3::decr(mass.composite_properties.position, rel_state.position);
137  Vector3::transform(mass.composite_properties.T_parent_this, rel_state.position);
138 
139  rel_state.Q_parent_this.multiply_conjugate(mass.composite_properties.Q_parent_this);
140  rel_state.Q_parent_this.normalize();
141  rel_state.Q_parent_this.left_quat_to_transformation(rel_state.T_parent_this);
142  }
143  }
144  }
145 
146  // The only valid integration frames are structure and composite.
147  else
148  {
149  MessageHandler::fail(__FILE__,
150  __LINE__,
151  DynBodyMessages::invalid_frame,
152  "Body '%s' has an invalid integration frame.",
153  name.c_str());
154  }
155 }

157 // Compute a derived state given the source state and the position/
158 // attitude transformation from the source to the derived state.
159 void DynBody::compute_derived_state_forward(const BodyRefFrame & source_frame,
160  const MassPoint & rel_state,
161  BodyRefFrame & derived_frame) const
162 {
163  double wxr[3];
164 
165  // A = integration frame
166  // B = source frame
167  // C = derived frame
168 
169  // T_A->C = T_B->C * T_A->B
170  rel_state.Q_parent_this.multiply(source_frame.state.rot.Q_parent_this, derived_frame.state.rot.Q_parent_this);
171  derived_frame.state.rot.Q_parent_this.normalize();
172  derived_frame.state.rot.compute_transformation();
173 
174  // w_A->C:C = T_B->C w_A->B:B
175  Vector3::transform(rel_state.T_parent_this,
176  source_frame.state.rot.ang_vel_this,
177  derived_frame.state.rot.ang_vel_this);
178  derived_frame.state.rot.compute_ang_vel_products();
179 
180  // r_A->C:A = r_A->B:A + r_B->C:A
181  // = r_A->B:A + T_A->B^T * r_B->C:B
182  Vector3::copy(source_frame.state.trans.position, derived_frame.state.trans.position);
183  Vector3::transform_transpose_incr(source_frame.state.rot.T_parent_this,
184  rel_state.position,
185  derived_frame.state.trans.position);
186 
187  // v_A->C:A = v_A->B:A + v_B->C:A
188  // = v_A->B:A + T_A->B^T * (w_A->B:B X r_B->C:B)
189  Vector3::copy(source_frame.state.trans.velocity, derived_frame.state.trans.velocity);
190  Vector3::cross(source_frame.state.rot.ang_vel_this, rel_state.position, wxr);
191  Vector3::transform_transpose_incr(source_frame.state.rot.T_parent_this, wxr, derived_frame.state.trans.velocity);
192 
193  // Denote that the state has been fully set.
194  derived_frame.initialized_items.set(RefFrameItems::Pos_Vel_Att_Rate);
195 
196  // Time stamp the derived frame per the source frame's timestamp.
197  derived_frame.set_timestamp(source_frame.timestamp());
198 }

356 // Propagate state from state owners to the integrated state.
357 void DynBody::update_integrated_state()
358 {
359  // Non-root bodies do not have an integrated state.
360  // Pass the buck to the root body.
361  if(dyn_parent != nullptr)
362  {
363  get_root_body_internal()->update_integrated_state();
364  return;
365  }
366 
367  MassPointState rel_state; /* --
368  Postion/attitude state of the integrated state with respect to the source
369  of some component of the state, computed by means of the mass tree rather
370  than the ref frame tree (the latter not being known yet). */
371 
372  const MassPoint * ref_point; /* --
373  The point to which the rel_state refers. */
374 
375  // Nomenclature for below:
376  // T_X->Y Transformation from frame X to frame Y
377  // r_X->Y:Z Vector from frame X to frame Y origin, expressed in frame Z
378  // v_X->Y:Z Time derivative of r_X->Y:Z, as observed in frame Z
379  // w_X->Y:Z Angular velocity of frame X wrt frame Y, expressed in frame Z
380  // Reference frames:
381  // A = integration frame
382  // C = integrated frame
383  // B = source frame
384 
385  ref_point = nullptr;
386 
387  // Attitude.
388  // Note that attitude is initialized if the source is the integrated frame.
389  if(attitude_source == integrated_frame)
390  {
391  initialized_states.add(RefFrameItems::Att);
392  }
393 
394  // Have attitude data, but source is not the integrated frame:
395  // Convert to the integrated frame and note that attitude is initialized.
396  else if(attitude_source != nullptr)
397  {
398  // Compute the position/attitude state of the attitude source frame
399  // with respect to the integrated frame.
400  compute_ref_point_transform(*attitude_source, &ref_point, rel_state);
401 
402  // Compute the orientation of the integrated frame wrt the integration
403  // frame. Have at hand
404  // T_A->B Transformation from integration frame to source frame,
405  // in attitude_source->state.rot.T_parent_this
406  // T_C->B Transformation from integrated frame to source frame,
407  // in rel_state.T_parent_this
408  // Conceptually,
409  // T_A->C = T_C->B^T * T_A->B
410  // Calculation is done with left transformation quaternions, which chain
411  // like transformation matrices.
412  rel_state.Q_parent_this.conjugate_multiply(attitude_source->state.rot.Q_parent_this,
413  integrated_frame->state.rot.Q_parent_this);
414  integrated_frame->state.rot.Q_parent_this.normalize();
415  integrated_frame->state.rot.compute_transformation();
416 
417  // Mark attitude as initialized and transfer attitude ownership to the
418  // integrated frame.
419  initialized_states.add(RefFrameItems::Att);
420  attitude_source = integrated_frame;
421  }
422 
423  // Attitude rate:
424  // Note that rate is initialized if the source is the integrated frame.
425  if(rate_source == integrated_frame)
426  {
427  initialized_states.add(RefFrameItems::Rate);
428  }
429 
430  // Have rate data, but source is not the integrated frame:
431  // Convert to the integrated frame if possible and note that rate is
432  // initialized. Attitude is needed to perform this conversion.
433  else if((rate_source != nullptr) && (attitude_source == integrated_frame))
434  {
435  // Compute the position/attitude state of the rate source frame
436  // with respect to the integrated frame.
437  compute_ref_point_transform(*rate_source, &ref_point, rel_state);
438 
439  // With the rigid body assumption, all attached bodies have exactly the
440  // same angular velocity with respect to inertial, but different frames
441  // will have different representations of this vector.
442  // Transform the rate source's angular velocity to root body coordinates:
443  // w_A->C:C = T_C->B^T * w_A->B:B
444  Vector3::transform_transpose(rel_state.T_parent_this,
445  rate_source->state.rot.ang_vel_this,
446  integrated_frame->state.rot.ang_vel_this);
447  integrated_frame->state.rot.compute_ang_vel_products();
448 
449  // Mark rate as initialized and transfer rate ownership to the
450  // integrated frame.
451  initialized_states.add(RefFrameItems::Rate);
452  rate_source = integrated_frame;
453  }
454 
455  // Position:
456  // Note that position is initialized if the source is the integrated frame.
457  if(position_source == integrated_frame)
458  {
459  initialized_states.add(RefFrameItems::Pos);
460  }
461 
462  // Have position data, but source is not the integrated frame:
463  // Convert to the integrated frame if possible and note that rate is
464  // initialized. Attitude is needed to perform this conversion.
465  else if((position_source != nullptr) && (attitude_source == integrated_frame))
466  {
467  // Compute the position/attitude state of the position source frame
468  // with respect to the integrated frame.
469  compute_ref_point_transform(*position_source, &ref_point, rel_state);
470 
471  // r_A->C:A = r_A->B:A + r_B->C:A
472  // = r_A->B:A - T_A->C^T * r_C->B:C
473  Vector3::copy(position_source->state.trans.position, integrated_frame->state.trans.position);
474  Vector3::transform_transpose_decr(integrated_frame->state.rot.T_parent_this,
475  rel_state.position,
476  integrated_frame->state.trans.position);
477 
478  // Mark position as initialized and transfer position ownership to the
479  // integratet frame.
480  initialized_states.add(RefFrameItems::Pos);
481  position_source = integrated_frame;
482  }
483 
484  // Velocity:
485  // Note that velocity is initialized if the source is the integrated frame.
486  if(velocity_source == integrated_frame)
487  {
488  initialized_states.add(RefFrameItems::Vel);
489  }
490 
491  // Have velocity data, but source is not the integrated frame:
492  // Convert to the integrated frame if possible and note that rate is
493  // initialized. Attitude and rate are needed to perform this conversion.
494  else if((velocity_source != nullptr) && (attitude_source == integrated_frame) && (rate_source == integrated_frame))
495  {
496  double wxr[3];
497 
498  // Compute the position/attitude state of the velocity source frame
499  // with respect to the integrated frame.
500  compute_ref_point_transform(*velocity_source, &ref_point, rel_state);
501 
502  // v_A->C:A = v_A->B:A + v_B->C:A
503  // = v_A->B:A - T_A->C^T * (w_A->C:C * r_C->B:C)
504  Vector3::copy(velocity_source->state.trans.velocity, integrated_frame->state.trans.velocity);
505  Vector3::cross(integrated_frame->state.rot.ang_vel_this, rel_state.position, wxr);
506  Vector3::transform_transpose_decr(integrated_frame->state.rot.T_parent_this,
507  wxr,
508  integrated_frame->state.trans.velocity);
509 
510  // Mark velocity as initialized and transfer velocity ownership to the
511  // integrated frame.
512  initialized_states.add(RefFrameItems::Vel);
513  velocity_source = integrated_frame;
514  }
515 
516  integrated_frame->initialized_items = initialized_states;
517 
518  // Time stamp the integrated frame per the time manager.
519  if(time_manager != nullptr)
520  {
521  integrated_frame->set_timestamp(time_manager->get_timestamp_time());
522  }
523 }

525 // Propagate state to attached bodies.
526 void DynBody::propagate_state()
527 {
528  // Pass the buck to the root body.
529  if(dyn_parent != nullptr)
530  {
531  dyn_parent->propagate_state();
532  return;
533  }
534 
535  // Update the integrated state from the sources.
536  update_integrated_state();
537 
538  // Propagate the state so long as it is not empty.
539  // NOTE: 3DOF simulations still need attitude it's just not propagated.
540  if(!initialized_states.is_empty())
541  {
542  if(integrated_frame == &structure)
543  {
544  propagate_state_from_structure();
545  }
546 
547  else if(integrated_frame == &composite_body)
548  {
549  propagate_state_from_composite();
550  }
551 
552  else
553  {
554  MessageHandler::fail(__FILE__,
555  __LINE__,
556  DynBodyMessages::invalid_frame,
557  "Body '%s' has an invalid integration frame.",
558  name.c_str());
559  }
560  }

563 // Propagate state to attached bodies starting from the structural frame.
564 void DynBody::propagate_state_from_structure()
565 {
566  // Use the full propagation methods if the state is full.
567  if(structure.initialized_items.is_full())
568  {
569  // Propagate the structure state to the composite and core body states
570  // for this object.
571  compute_derived_state_forward(structure, mass.composite_properties, composite_body);
572 
573  compute_derived_state_forward(structure, mass.core_properties, core_body);
574 
575  // Propagate state to the vehicle points.
576  if(autoupdate_vehicle_points)
577  {
578  compute_vehicle_point_states(RefFrameItems::Pos_Vel_Att_Rate);
579  }
580 
581  // Propagate the structure state to all child bodies.
582  for(auto child : dyn_children)
583  {
584  // Propagate the structure state to the child body's structure state.
585  compute_derived_state_forward(structure, child->mass.structure_point, child->structure);
586 
587  // Mark the child's known states.
