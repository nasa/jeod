83 // Compute the coefficients.
84 void
85 GaussJacksonCoeffs::compute_coeffs (
86  unsigned int order_in)
87 {
88  assert (order_in <= max_order);
89  order = order_in;
90 
91  // Compute the non-summed Adams corrector coefficients.
92  // For now there's an extra coefficient at the end, needed for the
93  // Stormer-Cowell coefficients.
94  GaussJacksonRationalCoefficients ac;
95  ac.configure_adams_corrector (order+3);
96 
97  // Compute the non-summed Stormer-Cowell corrector coefficients.
98  GaussJacksonRationalCoefficients sc =
99  ac.construct_stormer_cowell_corrector ();
100 
101  // Compute the non-summed predictor coefficients.
102  GaussJacksonRationalCoefficients ap = ac.construct_predictor();
103  GaussJacksonRationalCoefficients sp = sc.construct_predictor();
104 
105  // Get rid of the extra coefficients.
106  // The Adams coefficients contain one extra on each side,
107  // the Stormer-Cowell, two extra leading coefficients.
108  ac.discard_extra_terms (1, 1);
109  ap.discard_extra_terms (1, 1);
110  sc.discard_extra_terms (2, 0);
111  sp.discard_extra_terms (2, 0);
112 
113 
114  // Convert the coefficients to ordinate form.
115  er7_utils::NChooseM n_choose_m;
116 
117  // print_array (std::cout, ap.coefficients, "ap");
118  ap.convert_to_ordinate_form (n_choose_m, predictor.sa_coefs);
119 
120  // print_array (std::cout, sp.coefficients, "sp");
121  sp.convert_to_ordinate_form (n_choose_m, predictor.gj_coefs);
122 
123  // print_array (std::cout, ac.coefficients, "ac", order);
124  ac.convert_to_ordinate_form (n_choose_m, corrector[order].sa_coefs);
125 
126  // print_array (std::cout, sc.coefficients, "sc", order);
127  sc.convert_to_ordinate_form (n_choose_m, corrector[order].gj_coefs);
128 
129  for (unsigned int ii = 1; ii <= order; ++ii) {
130  ac.displace_back ();
131  // print_array (std::cout, ac.coefficients, "ac", ii);
132  ac.convert_to_ordinate_form (n_choose_m, corrector[order-ii].sa_coefs);
133 
134  sc.displace_back ();
135  // print_array (std::cout, sc.coefficients, "sc", ii);
136  sc.convert_to_ordinate_form (n_choose_m, corrector[order-ii].gj_coefs);
137  }
138  // std::cout << *this << "\n";
139 }
