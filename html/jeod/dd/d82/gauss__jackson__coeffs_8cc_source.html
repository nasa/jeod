74 // Compute the coefficients.
75 void GaussJacksonCoeffs::compute_coeffs(unsigned int order_in)
76 {
77  assert(order_in <= max_order);
78  order = order_in;
79 
80  // Compute the non-summed Adams corrector coefficients.
81  // For now there's an extra coefficient at the end, needed for the
82  // Stormer-Cowell coefficients.
83  GaussJacksonRationalCoefficients ac;
84  ac.configure_adams_corrector(order + 3);
85 
86  // Compute the non-summed Stormer-Cowell corrector coefficients.
87  GaussJacksonRationalCoefficients sc = ac.construct_stormer_cowell_corrector();
88 
89  // Compute the non-summed predictor coefficients.
90  GaussJacksonRationalCoefficients ap = ac.construct_predictor();
91  GaussJacksonRationalCoefficients sp = sc.construct_predictor();
92 
93  // Get rid of the extra coefficients.
94  // The Adams coefficients contain one extra on each side,
95  // the Stormer-Cowell, two extra leading coefficients.
96  ac.discard_extra_terms(1, 1);
97  ap.discard_extra_terms(1, 1);
98  sc.discard_extra_terms(2, 0);
99  sp.discard_extra_terms(2, 0);
100 
101  // Convert the coefficients to ordinate form.
102  er7_utils::NChooseM n_choose_m;
103 
104  // print_array (std::cout, ap.coefficients, "ap");
105  ap.convert_to_ordinate_form(n_choose_m, predictor.sa_coefs);
106 
107  // print_array (std::cout, sp.coefficients, "sp");
108  sp.convert_to_ordinate_form(n_choose_m, predictor.gj_coefs);
109 
110  // print_array (std::cout, ac.coefficients, "ac", order);
111  ac.convert_to_ordinate_form(n_choose_m, corrector[order].sa_coefs);
112 
113  // print_array (std::cout, sc.coefficients, "sc", order);
114  sc.convert_to_ordinate_form(n_choose_m, corrector[order].gj_coefs);
115 
116  for(unsigned int ii = 1; ii <= order; ++ii)
117  {
118  ac.displace_back();
119  // print_array (std::cout, ac.coefficients, "ac", ii);
120  ac.convert_to_ordinate_form(n_choose_m, corrector[order - ii].sa_coefs);
121 
122  sc.displace_back();
123  // print_array (std::cout, sc.coefficients, "sc", ii);
124  sc.convert_to_ordinate_form(n_choose_m, corrector[order - ii].gj_coefs);
125  }
126  // std::cout << *this << "\n";
127 }
