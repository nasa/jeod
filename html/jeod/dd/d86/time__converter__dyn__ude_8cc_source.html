101  ude_ptr = dynamic_cast<TimeUDE *> (child_ptr);
102  if (ude_ptr == nullptr) {

105  "Object '%s' is not a %s object as expected with int_dir = 1\n",

107  return;

109  }
110 
111  // This converter does not accommodate a reverse conversion.
112  else if (int_dir == -1) {

115  "There is no converter available for converting UDE to Dyn.\n");

120  else {
