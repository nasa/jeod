69  Purpose:
70  (calculates the flux vector from the vehicle's position.)
71 *****************************************************************************/
72 void RadiationSource::calculate_flux(RefFrame & veh_struc_frame, const double center_gravity[3])
73 {
74  if(luminosity < 1.0E-6)
75  {
76  flux_mag = 0.0;
77  return;
78  }
79 
80  // Calculate the position of the vehicle with respect to Primary Source.
81  // Start by getting the position of the origin of the vehicle structural
82  // frame.
83  veh_struc_frame.compute_position_from(*inertial_frame_ptr, source_to_struc_origin);
84 
85  // center_gravity is in structural. Need it in inertial to complete the
86  // position vector (center_gravity is more closely set to the vehicle than the
87  // origin of the structural frame). Since the parent-frame of any root
88  // vehicle is an inertial-frame, the vehicle's T_parent_this will convert
89  // structural to inertial.
90  Vector3::transform_transpose(veh_struc_frame.state.rot.T_parent_this, center_gravity, inertial_cg);
91 
92  // Add the vector to structural origin to the vector from structural origin to
93  // vehicle center.
94  Vector3::sum(source_to_struc_origin, inertial_cg, source_to_cg);
95 
96  // Now normalize this vector. Since we need the magnitude later, do this in
97  // two steps, rather than use the available "normalize" function.
98 
99  d_source_to_cg = Vector3::vmag(source_to_cg);
100 
101  Vector3::scale(source_to_cg, 1 / d_source_to_cg, flux_hat);
102 
103  flux_mag = luminosity / (d_source_to_cg * d_source_to_cg * 4 * M_PI);
104 
105  Vector3::scale(flux_hat, flux_mag, flux_inertial);
106 
107  // Transform this vector back into structural reference and normalize.
108  Vector3::transform(veh_struc_frame.state.rot.T_parent_this, flux_inertial, flux_struc);
109 
110  Vector3::normalize(flux_struc, flux_struc_hat);
111 }
