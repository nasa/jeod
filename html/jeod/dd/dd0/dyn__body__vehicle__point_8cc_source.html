49 // Add a mass point to the dyn body's list of such and
50 // make a vehicle point that corresponds to the added mass point.
51 void DynBody::add_mass_point(const MassPointInit & mass_point_init)
52 {
53  MassPoint * mass_point;
54  BodyRefFrame * point_frame;
55 
56  // Sanity check: The subject must be registered with the dynamics manager.
57  if((dyn_manager == nullptr) || (!dyn_manager->is_dyn_body_registered(this)))
58  {
59  MessageHandler::fail(__FILE__,
60  __LINE__,
61  DynBodyMessages::invalid_body,
62  "DynBody '%s' has not been initialized.\n",
63  name.c_str());
64  }
65 
66  // Add the mass point as a mass point.
67  mass.add_mass_point(mass_point_init);
68 
69  // The newly-constructed will be the last point in the list.
70  mass_point = mass.mass_points.back();
71 
72  // Allocate the vehicle point (a reference frame).
73  point_frame = JEOD_ALLOC_CLASS_OBJECT(BodyRefFrame, ());
74 
75  // Give the vehicle point the same name as the mass point's name.
76  point_frame->set_name(mass_point->get_name());
77 
78  // Associate the vehicle point with the mass point.
79  point_frame->mass_point = static_cast<MassPoint *>(mass_point);
80 
81  // Associate the vehicle point with the vehicle.
82  point_frame->set_owner(this);
83 
84  // Attach the vehicle point frame to the integration frame.
85  if(integ_frame != nullptr)
86  {
87  integ_frame->add_child(*point_frame);
88  }
89 
90  // Add the vehicle point to the body's list of such.
91  vehicle_points.push_back(point_frame);
92 
93  // Register the frame with the dynamics manager.
94  dyn_manager->add_ref_frame(*point_frame);
95 }

97 // Find the vehicle point with the given name.
98 const BodyRefFrame * DynBody::find_vehicle_point(const std::string & pt_name) const
99 {
100  std::size_t search_offset = name.size() + 1;
101  const BodyRefFrame * found_point = nullptr;
102 
103  // Only search if the name is valid. (No else; found point is already NULL)
104  if(!pt_name.empty())
105  {
106  std::string pt_suffix = name.suffix(pt_name);
107 
108  // Search for the point.
109  auto name_compare = [&search_offset, &pt_suffix](const BodyRefFrame * point)
110  {
111  return point->get_name().compare(search_offset, std::string::npos, pt_suffix) == 0;
112  };
113 
114  const auto iter = std::find_if(vehicle_points.begin(), vehicle_points.end(), name_compare);
115 
116  if(iter != vehicle_points.end())
117  {
118  found_point = *iter;
119  }
120  }
121 
122  return found_point;
123 }

125 /**
126  * Compute the state derivatives at a vehicle point.
127  *
128  * @param[in] vehicle_pt Vehicle point reference frame
129  * @param[out] pt_derivs Computed derivatives
130  */
131 void DynBody::compute_vehicle_point_derivatives(const BodyRefFrame & vehicle_pt, FrameDerivs & pt_derivs)
132 {
133  // The root body for this body.
134  // Only the root body's derivs element contains valid data.
135  const DynBody * root_body = get_root_body();
136 
137  // The root body's composite mass properties.
138  const MassProperties & root_ppty = root_body->mass.composite_properties;
139 
140  // The root body's composite body rotational state.
141  const RefFrameRot & root_rot = root_body->composite_body.state.rot;
142 
143  // The derivatives for this root body.
144  const FrameDerivs & root_derivs = root_body->derivs;
145 
146  // Other local variables
147  MassPointState cm_to_point; // -- Vehicle pt state wrt CoM
148  double cm_to_point_inrtl[3]; // M Point wrt CoM, inertial coordinates
149  double wxr[3]; // M/s Omega cross r
150  double wxwxr[3]; // M/s2 Centrifugal accel
151  double axr[3]; // M/s2 Acceleration due to angular accel
152  double accel_rot_body[3]; // M/s2 Acceleration due to rotation
153  double accel_rot_inrtl[3]; // M/s2 Acceleration due to rotation
154  double delta_acc_grav[3]; // M/s2 Acceleration due to gravity gradient
155  double delta_acc_non_g[3]; // M/s2 Delta acceleration, non-gravity
156 
157  // Sanity check: Never update the root body's derivs structure. Ever.
158  if(&pt_derivs == &root_derivs)
159  {
160  // Complain vehemently if the attempt is utterly invalid.
161  // (Requesting the root body to update its composite derivatives
162  // is silly, but no harm, no foul, so no complaint.)
163  if(&vehicle_pt != &root_body->composite_body)
164  {
165  MessageHandler::error(__FILE__,
166  __LINE__,
167  DynBodyMessages::invalid_frame,
168  "Illegal attempt to update the root body '%s' FrameDerivs\n",
169  root_body->name.c_str());
170  }
171  return;
172  }
173 
174  // Compute the vehicle point state wrt the composite center of mass.
175  vehicle_pt.mass_point->compute_relative_state(root_ppty, cm_to_point);
176 
177  // Transform to position to inertial.
178  Vector3::transform_transpose(root_rot.T_parent_this, cm_to_point.position, cm_to_point_inrtl);
179 
180  // Compute accelerations caused by vehicle rotation.
181  Vector3::cross(root_rot.ang_vel_this, cm_to_point.position, wxr);
182  Vector3::cross(root_rot.ang_vel_this, wxr, wxwxr);
183  Vector3::cross(root_derivs.rot_accel, cm_to_point.position, axr);
184  Vector3::sum(wxwxr, axr, accel_rot_body);
185  Vector3::transform_transpose(root_rot.T_parent_this, accel_rot_body, accel_rot_inrtl);
186 
187  // Compute delta gravitational acceleration.
188  Vector3::transform(grav_interaction.grav_grad, cm_to_point_inrtl, delta_acc_grav);
189 
190  // Compute accelerations.
191  Vector3::diff(accel_rot_inrtl, delta_acc_grav, delta_acc_non_g);
192  Vector3::sum(root_derivs.non_grav_accel, delta_acc_non_g, pt_derivs.non_grav_accel);
193  Vector3::sum(root_derivs.trans_accel, accel_rot_inrtl, pt_derivs.trans_accel);
194  Vector3::transform(cm_to_point.T_parent_this, root_derivs.rot_accel, pt_derivs.rot_accel);
195  vehicle_pt.state.rot.Q_parent_this.compute_left_quat_deriv(vehicle_pt.state.rot.ang_vel_this,
196  pt_derivs.Qdot_parent_this);
197 }
