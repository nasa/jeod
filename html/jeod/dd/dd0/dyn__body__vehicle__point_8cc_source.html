50 // Add a mass point to the dyn body's list of such and
51 // make a vehicle point that corresponds to the added mass point.
52 void
53 DynBody::add_mass_point (
54  const MassPointInit & mass_point_init)
55 {
56  MassPoint * mass_point;
57  BodyRefFrame * point_frame;
58 
59  // Sanity check: The subject must be registered with the dynamics manager.
60  if ((dyn_manager == nullptr) ||
61  (! dyn_manager->is_dyn_body_registered (this))) {
62  MessageHandler::fail (
63  __FILE__, __LINE__, DynBodyMessages::invalid_body,
64  "DynBody '%s' has not been initialized.\n",
65  name.c_str());
66  }
67 
68  // Add the mass point as a mass point.
69  mass.add_mass_point (mass_point_init);
70 
71  // The newly-constructed will be the last point in the list.
72  mass_point = mass.mass_points.back ();
73 
74  // Allocate the vehicle point (a reference frame).
75  point_frame = JEOD_ALLOC_CLASS_OBJECT (BodyRefFrame, ());
76 
77  // Give the vehicle point the same name as the mass point's name.
78  point_frame->set_name (mass_point->get_name());
79 
80  // Associate the vehicle point with the mass point.
81  point_frame->mass_point = static_cast <MassPoint *> (mass_point);
82 
83  // Associate the vehicle point with the vehicle.
84  point_frame->set_owner (this);
85 
86  // Attach the vehicle point frame to the integration frame.
87  if (integ_frame != nullptr) {
88  integ_frame->add_child (*point_frame);
89  }
90 
91  // Add the vehicle point to the body's list of such.
92  vehicle_points.push_back (point_frame);
93 
94  // Register the frame with the dynamics manager.
95  dyn_manager->add_ref_frame (*point_frame);
96 }

99 // Find the vehicle point with the given name.
100 const BodyRefFrame *
101 DynBody::find_vehicle_point (
102  const char * pt_name)
103 const
104 {
105  std::size_t search_offset = name.size() + 1;
106  const BodyRefFrame * found_point = nullptr;
107 
108  // Only search if the name is valid. (No else; found point is already NULL)
109  if ((pt_name != nullptr) && (*pt_name != '\0')) {
110  const char * pt_suffix = name.suffix(pt_name);
111  // Search for the point.
112  const auto iter = std::find_if(vehicle_points.begin(), vehicle_points.end(),
113  [&search_offset, &pt_suffix](const BodyRefFrame* point) { return std::strcmp (pt_suffix, point->get_name()+search_offset) == 0; });
114  if (iter != vehicle_points.end()) {
115  found_point = *iter;
116  }
117  }
118 
119  return found_point;
120 }

122 /**
123  * Compute the state derivatives at a vehicle point.
124  *
125  * @param[in] vehicle_pt Vehicle point reference frame
126  * @param[out] pt_derivs Computed derivatives
127  */
128 void
129 DynBody::compute_vehicle_point_derivatives (
130  const BodyRefFrame & vehicle_pt,
131  FrameDerivs & pt_derivs)
132 {
133  // The root body for this body.
134  // Only the root body's derivs element contains valid data.
135  const DynBody * root_body = get_root_body();
136 
137  // The root body's composite mass properties.
138  const MassProperties & root_ppty = root_body->mass.composite_properties;
139 
140  // The root body's composite body rotational state.
141  const RefFrameRot & root_rot = root_body->composite_body.state.rot;
142 
143  // The derivatives for this root body.
144  const FrameDerivs & root_derivs = root_body->derivs;
145 
146  // Other local variables
147  MassPointState cm_to_point; // -- Vehicle pt state wrt CoM
148  double cm_to_point_inrtl[3]; // M Point wrt CoM, inertial coordinates
149  double wxr[3]; // M/s Omega cross r
150  double wxwxr[3]; // M/s2 Centrifugal accel
151  double axr[3]; // M/s2 Acceleration due to angular accel
152  double accel_rot_body[3]; // M/s2 Acceleration due to rotation
153  double accel_rot_inrtl[3]; // M/s2 Acceleration due to rotation
154  double delta_acc_grav[3]; // M/s2 Acceleration due to gravity gradient
155  double delta_acc_non_g[3]; // M/s2 Delta acceleration, non-gravity
156 
157 
158  // Sanity check: Never update the root body's derivs structure. Ever.
159  if (&pt_derivs == &root_derivs) {
160  // Complain vehemently if the attempt is utterly invalid.
161  // (Requesting the root body to update its composite derivatives
162  // is silly, but no harm, no foul, so no complaint.)
163  if (&vehicle_pt != &root_body->composite_body) {
164  MessageHandler::error (
165  __FILE__, __LINE__, DynBodyMessages::invalid_frame,
166  "Illegal attempt to update the root body '%s' FrameDerivs\n",
167  root_body->name.c_str());
168  }
169  return;
170  }
171 
172 
173  // Compute the vehicle point state wrt the composite center of mass.
174  vehicle_pt.mass_point->compute_relative_state (root_ppty, cm_to_point);
175 
176  // Transform to position to inertial.
177  Vector3::transform_transpose (root_rot.T_parent_this,
178  cm_to_point.position,
179  cm_to_point_inrtl);
180 
181  // Compute accelerations caused by vehicle rotation.
182  Vector3::cross (root_rot.ang_vel_this, cm_to_point.position, wxr);
183  Vector3::cross (root_rot.ang_vel_this, wxr, wxwxr);
184  Vector3::cross (root_derivs.rot_accel, cm_to_point.position, axr);
185  Vector3::sum (wxwxr, axr, accel_rot_body);
186  Vector3::transform_transpose (root_rot.T_parent_this,
187  accel_rot_body,
188  accel_rot_inrtl);
189 
190  // Compute delta gravitational acceleration.
191  Vector3::transform (grav_interaction.grav_grad, cm_to_point_inrtl,
192  delta_acc_grav);
193 
194  // Compute accelerations.
195  Vector3::diff (accel_rot_inrtl, delta_acc_grav, delta_acc_non_g);
196  Vector3::sum (root_derivs.non_grav_accel, delta_acc_non_g,
197  pt_derivs.non_grav_accel);
198  Vector3::sum (root_derivs.trans_accel, accel_rot_inrtl,
199  pt_derivs.trans_accel);
200  Vector3::transform (cm_to_point.T_parent_this, root_derivs.rot_accel,
201  pt_derivs.rot_accel);
202  vehicle_pt.state.rot.Q_parent_this.compute_left_quat_deriv (
203  vehicle_pt.state.rot.ang_vel_this, pt_derivs.Qdot_parent_this);
204 
205  return;
206 }
