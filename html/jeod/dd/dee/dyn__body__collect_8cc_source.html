52  * Accumulate forces acting on a vehicle.

54  * \param[in] vec Forces
55  * \param[out] cumulation Accumulated force
56  **/
57 static inline void accumulate_forces(const JeodPointerVector<CollectForce>::type & vec, double * cumulation)
58 {
59  unsigned int nitems = vec.size();
60 
61  Vector3::initialize(cumulation);
62  for(unsigned int ii = 0; ii < nitems; ii++)
63  {
64  if(vec[ii]->is_active())
65  {
66  Vector3::incr(vec[ii]->force, cumulation);
67  }
68  }
69 }

91 // Collect forces and torques acting on the vehicle.
92 void DynBody::collect_forces_and_torques()
93 {
94  // Translational dynamics on: Accumulate the external forces on this DynBody.
95  if(translational_dynamics)
96  {
97  accumulate_forces(collect.collect_effector_forc, collect.effector_forc);
98  accumulate_forces(collect.collect_environ_forc, collect.environ_forc);
99  accumulate_forces(collect.collect_no_xmit_forc, collect.no_xmit_forc);
100  }
101 
102  // Translational dynamics off: Zero-out force accumulators.
103  else
104  {
105  Vector3::initialize(collect.effector_forc);
106  Vector3::initialize(collect.environ_forc);
107  Vector3::initialize(collect.no_xmit_forc);
108  }
109 
110  // Rotational dynamics on: Accumulate the external torques on this DynBody.
111  if(rotational_dynamics)
112  {
113  accumulate_torques(collect.collect_effector_torq, collect.effector_torq);
114  accumulate_torques(collect.collect_environ_torq, collect.environ_torq);
115  accumulate_torques(collect.collect_no_xmit_torq, collect.no_xmit_torq);
116  }
117 
118  // Rotational dynamics off: Zero-out torque accumulators.
119  else
120  {
121  Vector3::initialize(collect.effector_torq);
122  Vector3::initialize(collect.environ_torq);
123  Vector3::initialize(collect.no_xmit_torq);
124  }
125 
126  // Collect forces and torques on attached bodies and add them to the
127  // force and torque collections for this body.
128  for(auto child : dyn_children)
129  {
130  child->collect_forces_and_torques();
131  }
132 
133  // At this point forces and torques have been accumulated with the body
134  // and the method will have been recursively invoked on all child bodies.
135  // The remaining actions depend on whether this body is a child body
136  // or a root body.
137 
138  // Child body: Propagate forces, torques to the parent body.
139  if(dyn_parent != nullptr)
140  {
141  double effector_forc_pstr[3];
142  double environ_forc_pstr[3];
143 
144  // Translational dynamics is on:
145  // Transmit forces to the parent (but in the parent's structural frame).
146  if(translational_dynamics)
147  {
148  // Transform transmittable forces to parent structural.
149  // The inherited (Mass) structure_point object provides the
150  // transformation from parent structural to child structural.
151  // The transpose is needed to transform from child to parent.
152  Vector3::transform_transpose(mass.structure_point.T_parent_this, collect.effector_forc, effector_forc_pstr);
153  Vector3::transform_transpose(mass.structure_point.T_parent_this, collect.environ_forc, environ_forc_pstr);
154 
155  // Transmit these forces to the parent dyn body.
156  Vector3::incr(effector_forc_pstr, dyn_parent->collect.effector_forc);
157  Vector3::incr(environ_forc_pstr, dyn_parent->collect.environ_forc);
158  }
159 
160  // Translational dynamics is off:
161  // Zero out the transmitted forces as these become torques in the parent.
162  else
163  {
164  Vector3::initialize(effector_forc_pstr);
165  Vector3::initialize(environ_forc_pstr);
166  }
167 
168  // Rotational dynamics is on:
169  // Transmit torques to the parent (but in the parent's structural frame).
170  if(rotational_dynamics)
171  {
172  double effector_torq_pstr[3];
173  double environ_torq_pstr[3];
174  double pcm_to_ccm[3];
175 
176  // Transform transmittable torques to parent structural.
177  Vector3::transform_transpose(mass.structure_point.T_parent_this, collect.effector_torq, effector_torq_pstr);
178  Vector3::transform_transpose(mass.structure_point.T_parent_this, collect.environ_torq, environ_torq_pstr);
179 
180  // Compute the parent cm to child cm offset in parent structural.
181  Vector3::diff(mass.composite_wrt_pstr.position, dyn_parent->mass.composite_properties.position, pcm_to_ccm);
182 
183  // Compute the torque contributions of the child forces.
184  Vector3::cross_incr(pcm_to_ccm, effector_forc_pstr, effector_torq_pstr);
185  Vector3::cross_incr(pcm_to_ccm, environ_forc_pstr, environ_torq_pstr);
186 
187  // Transmit the torques to the parent dyn body.
188  Vector3::incr(effector_torq_pstr, dyn_parent->collect.effector_torq);
189  Vector3::incr(environ_torq_pstr, dyn_parent->collect.environ_torq);
190  }
191 
192  // There is nothing to do here if rotational dynamics is off.
193  else
194  {
195  }
196 
197  // If requested, compute child body pt derivatives
198  if(compute_point_derivative)
199  {
200  compute_vehicle_point_derivatives(composite_body, derivs);
201  }
202  }
203 
204  // Root body: Compute total forces and torques and resultant accelerations.
205  else
206  {
207  // Translational dynamics is on:
208  // Compute total force on the body, structural and inertial referenced,
209  // and from this, compute translational accelerations.
210  // Note: The gravitational acceleration must have already been computed.
211  if(translational_dynamics)
212  {
213  // Compute the total force, structural referenced.
214  Vector3::sum(collect.effector_forc, collect.environ_forc, collect.no_xmit_forc, collect.extern_forc_struct);
215 
216  // Transform the total external force to inertial.
217  // The structure member provides the transformation from inertial to
218  // structural. The transpose is needed to transform to inertial.
219  Vector3::transform_transpose(structure.state.rot.T_parent_this,
220  collect.extern_forc_struct,
221  collect.extern_forc_inrtl);
222 
223  // Compute the translational acceleration.
224  Vector3::scale(collect.extern_forc_inrtl, mass.composite_properties.inverse_mass, derivs.non_grav_accel);
225  Vector3::sum(derivs.non_grav_accel, grav_interaction.grav_accel, derivs.trans_accel);
226  }
227 
228  // Translational dynamics is off:
229  // Zero out the collected forces and the translational accelerations.
230  else
231  {
232  Vector3::initialize(collect.extern_forc_struct);
233  Vector3::initialize(collect.extern_forc_inrtl);
234  Vector3::initialize(derivs.non_grav_accel);
235  Vector3::initialize(derivs.trans_accel);
236  }
237 
238  // Rotational dynamics is on:
239  if(rotational_dynamics)
240  {
241  double ang_mom[3]; // kg*M^2/s Angular momentum, body referenced
242  double torque_body[3]; // kg*M^2/s^2 Total torque, body referenced
243 
244  // Compute the total torque, structural referenced.
245  Vector3::sum(collect.effector_torq, collect.environ_torq, collect.no_xmit_torq, collect.extern_torq_struct);
246 
247  // Transform the structural reference torque to body.
248  // The inherited (Mass) composite_properties member provides the
249  // transformation from structure to body.
250  Vector3::transform(mass.composite_properties.T_parent_this,
251  collect.extern_torq_struct,
252  collect.extern_torq_body);
253 
254  // Compute the inertial torque, w x L, where w is the body-referenced
255  // angular velocity and L is the body-referenced angular momentum.
256  Vector3::transform(mass.composite_properties.inertia, composite_body.state.rot.ang_vel_this, ang_mom);
257  Vector3::cross(composite_body.state.rot.ang_vel_this, ang_mom, collect.inertial_torq);
258 
259  // Subtract from the external torque to form the body-frame
260  // apparent torque.
261  Vector3::diff(collect.extern_torq_body, collect.inertial_torq, torque_body);
262 
263  // Solve the rotational EOM for body accelerations.
264  Vector3::transform(mass.composite_properties.inverse_inertia, torque_body, derivs.rot_accel);
265 
266  // Truncate small rotational accelerations to avoid numerical problems.
267  Vector3::zero_small(1e-20, derivs.rot_accel);
268  }
269 
270  // Rotational dynamics is off:
271  // Zero out the collected torques and the rotational acceleration.
272  else
273  {
274  Vector3::initialize(collect.extern_torq_struct);
275  Vector3::initialize(collect.extern_torq_body);
276  Vector3::initialize(collect.inertial_torq);
277  Vector3::initialize(derivs.rot_accel);
278  }
279  }
280 }
