88 /**
89  * This is a structure used only in the surface model to aid in relative
90  * state calculations for articulation. For each mass body that needs a
91  * relative state calculated w.r.t. the mass body named in struct_body_name,
92  * one of these objects will be instantiated. That way, the relative state
93  * information must only be calculated once per mass body.
94  */
95 struct FacetStateInfo {
96 
97  JEOD_MAKE_SIM_INTERFACES(FacetStateInfo)
98  /**
99  * The resulting relative mass point state between the
100  * structural body named in struct_body_name and the
101  * MassBody pointed to in this structure's mass_body
102  */
103  MassPointState mass_state; //!< trick_io(**)
104  /**
105  * The MassBody object whose state, relative and w.r.t.
106  * the MassBody named in struct_body_name, is
107  * being calculated
108  */
109  MassBody* mass_body; //!< trick_io(**)
110 
111  /**
112  * Default constructor to keep the memory manager happy.
113  */
114  FacetStateInfo(){}
115 
116  /**
117  * FacetStateInfo non-default constructor.
118  * @param new_mass_body The mass body to which this object will refer.
119  */
120  explicit FacetStateInfo(MassBody* new_mass_body)
121  :
122  mass_body(new_mass_body)
123  {}
124 
125  /**
126  * FacetStateInfo non-default constructor.
127  * @param new_mass_body The mass body to which this object will refer.
128  */
129  explicit FacetStateInfo(MassBody& new_mass_body)
130  :
131  mass_body(&new_mass_body)
132  {}
133 
134  /**
135  * Compare this FacetStateInfo object to another.
136  * The two are 'equal' if they refer to the same mass body.
137  * @param rhs Object to be compared with this object.
138  */
139  bool operator == (const FacetStateInfo& rhs) const {
140  return mass_body == rhs.mass_body;
141  }
142 };
