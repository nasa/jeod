87 /**
88  * This is a structure used only in the surface model to aid in relative
89  * state calculations for articulation. For each mass body that needs a
90  * relative state calculated w.r.t. the mass body named in struct_body_name,
91  * one of these objects will be instantiated. That way, the relative state
92  * information must only be calculated once per mass body.
93  */
94 struct FacetStateInfo
95 {
96  JEOD_MAKE_SIM_INTERFACES(jeod, FacetStateInfo)
97  /**
98  * The resulting relative mass point state between the
99  * structural body named in struct_body_name and the
100  * MassBody pointed to in this structure's mass_body
101  */
102  MassPointState mass_state; //!< trick_io(**)
103  /**
104  * The MassBody object whose state, relative and w.r.t.
105  * the MassBody named in struct_body_name, is
106  * being calculated
107  */
108  MassBody * mass_body{}; //!< trick_io(**)
109 
110  /**
111  * Default constructor to keep the memory manager happy.
112  */
113  FacetStateInfo() = default;
114 
115  /**
116  * FacetStateInfo non-default constructor.
117  * @param new_mass_body The mass body to which this object will refer.
118  */
119  explicit FacetStateInfo(MassBody * new_mass_body)
120  : mass_body(new_mass_body)
121  {
122  }
123 
124  /**
125  * FacetStateInfo non-default constructor.
126  * @param new_mass_body The mass body to which this object will refer.
127  */
128  explicit FacetStateInfo(MassBody & new_mass_body)
129  : mass_body(&new_mass_body)
130  {
131  }
132 
133  /**
134  * Compare this FacetStateInfo object to another.
135  * The two are 'equal' if they refer to the same mass body.
136  * @param rhs Object to be compared with this object.
137  */
138  bool operator==(const FacetStateInfo & rhs) const
139  {
140  return mass_body == rhs.mass_body;
141  }
142 };
