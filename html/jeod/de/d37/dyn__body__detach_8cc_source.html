46 // Break the logical connectivity between a parent and child DynBody.
47 // Returns false if unable to disconnect.
48 bool DynBody::detach(DynBody & other_body)
49 {
50  DynBody * parent = nullptr;
51  DynBody * detacher = nullptr;
52 
53  bool success = true;
54 
55  // Find the body that is to be detached from its immediate parent.
56  // First search from 'this' up for the input mass_body, and if that fails
57  // search from the input mass_body up for 'this'. Both searches failing is
58  // an error.
59  // 1) search from this up to other_body
60  for(auto * link : TreeLinksAscendRange<MassBodyLinks>(mass.links))
61  {
62  MassBody * iter_body = &(link->container());
63  if(iter_body == &other_body.mass)
64  {
65  parent = &other_body;
66  break;
67  }
68  detacher = iter_body->dyn_owner;
69  }
70 
71  // 2) parent_body is null => First search failed.
72  if(parent == nullptr)
73  {
74  // Search from other_body up to this
75  for(auto * link : TreeLinksAscendRange<MassBodyLinks>(other_body.mass.links))
76  {
77  MassBody * iter_body = &(link->container());
78  if(iter_body == &mass)
79  {
80  parent = this;
81  break;
82  }
83  detacher = iter_body->dyn_owner;
84  }
85  }
86  // Else: bodies or not attached, or detach point is arbitrary
87  // In either case, unable to proceed. Return false.
88  if(parent == nullptr || detacher == nullptr)
89  {
90  success = false;
91  MessageHandler::inform(__FILE__,
92  __LINE__,
93  DynBodyMessages::invalid_attachment,
94  "\nWarning: Unable to find detach point between\n"
95  "bodies '%s' and '%s'. No detachment was performed.",
96  name.c_str(),
97  other_body.name.c_str());
98  return success;
99  }
100 
101  // Delete child from parent
102  std::list<DynBody *>::iterator it;
103  for(it = parent->dyn_children.begin(); it != parent->dyn_children.end(); ++it)
104  {
105  if(*it == detacher)
106  {
107  parent->dyn_children.erase(it);
108  break;
109  }
110  }
111  if(it == parent->dyn_children.end())
112  {
113  success = false;
114  MessageHandler::fail(__FILE__,
115  __LINE__,
116  DynBodyMessages::invalid_attachment,
117  "\nDuring detach process:\n"
118  "%s recognizes %s as its dyn_parent, "
119  "but %s does not recognize it as a child.\n"
120  "An invalid attachment hierarchy has been detected.",
121  detacher->name.c_str(),
122  parent->name.c_str(),
123  parent->name.c_str());
124 
125  // Unreached
126  return success;
127  }
128 
129  // Delete parent from child
130  detacher->dyn_parent = nullptr;
131 
132  // Delete mass links
133  parent->detach_mass_internal(detacher->mass);
134  return success;
135 }

137 // Detach this DynBody from its parent
138 bool DynBody::detach()
139 {
140  bool success = false;
141  if(frame_attach.isAttached())
142  {
143  frame_attach.clear_attachment();
144  }
145  else if(dyn_parent != nullptr && dyn_parent != this)
146  {

153  __LINE__,
154  DynBodyMessages::invalid_technique,

158  name.c_str());
159  }
160 
161  return success;

164 // Break the logical connectivity between parent DynBody and child submass.
165 bool DynBody::remove_mass_body(MassBody & child)
166 {
167  // Look up references
168  MassBody * child_root; // the root MassBody of the detaching subassembly
169  MassBody * bdy; // MassBody iteration pointer
170  DynBody * child_body; // look up if MassBody belongs to a DynBody
171 
172  // Status variables
173  bool success = true;
174 
175  // First, find the root body that is to be detached from this parent.
176  // This is performed because the user is able to call attach for a
177  // subassembly of mass bodies, in which case the root of the subassembly
178  // is attached if the user specified a subbody in the subassembly. In the
179  // same manner, this process conveniently allows the user to detach the
180  // root of the subassembly by requesting a detach of the subbody.
181  // This way, the user essentially doesn't have to worry about the root of
182  // the subassembly.
183  child_root = nullptr;
184  bdy = nullptr;
185  for(auto * link : TreeLinksAscendRange<MassBodyLinks>(child.links))
186  {
187  bdy = &(link->container());
188  if(bdy->get_parent_body() == &mass)
189  {
190  child_root = bdy;
191  break;
192  }
193  }
194  if(child_root == nullptr)
195  {
196  success = false;
197  MessageHandler::inform(__FILE__,
198  __LINE__,
199  DynBodyMessages::invalid_technique,
200  "\nDuring detach process:\n"
201  "%s ordered to detach sub-body MassBody %s, \n"
202  "But no corresponding subassembly was found. "
203  "No detachment performed.\n",
204  name.c_str(),
205  child.name.c_str());
206  return success;
207  }
208 
209  // Check if mass is actually belonging to DynBody
210  // If a DynBody, warn user and reroute detach method.
211  child_body = child_root->dyn_owner;
212  if(child_body != nullptr)
213  {
214  MessageHandler::inform(__FILE__,
215  __LINE__,
216  DynBodyMessages::invalid_technique,
217  "\nDuring detach process:\n"
218  "%s ordered to detach child MassBody %s, which refers\n"
219  "to a DynBody attachment. Attempting to detach child "
220  "DynBody %s instead.\n",
221  name.c_str(),
222  child.name.c_str(),
223  child_body->name.c_str());
224  success = detach(*child_body);
225  }
226  else
227  {
228  detach_mass_body_frames(*child_root);
229  detach_mass_internal(*child_root);
230  mass_children.remove(child_root);
231  }
232 
233  return success;
234 }

236 // Remove BodyRefFrame for each MassBody point from vehicle_points recursively
237 void DynBody::detach_mass_body_frames(MassBody & subbody)
238 {
239  // Remove points for current subbody
240  BodyRefFrame * pt_frame = nullptr;
241  std::list<BodyRefFrame *>::iterator veh_pt;
242  for(auto & mass_point : subbody.mass_points)
243  {
244  pt_frame = find_body_frame(std::string(mass_point->get_name()));
245  if(pt_frame != nullptr)
246  {
247  veh_pt = std::find(vehicle_points.begin(), vehicle_points.end(), pt_frame);
248  vehicle_points.erase(veh_pt);
249  dyn_manager->remove_ref_frame(*pt_frame);
250  }
251  JEOD_DELETE_OBJECT(pt_frame);
252  }
253 
254  // Call for each of subbody's children
255  for(auto * child_link : TreeLinksChildrenRange<MassBodyLinks>(subbody.links))
256  {
257  detach_mass_body_frames(child_link->container());
258  }
259 }
