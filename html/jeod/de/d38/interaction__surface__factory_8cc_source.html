86 void InteractionSurfaceFactory::create_surface(SurfaceModel * surface, InteractionSurface * inter_surface)
87 {
88  // can't hand this un-allocated things. check for that.
89  if(surface == nullptr)
90  {
91  MessageHandler::fail(__FILE__,
92  __LINE__,
93  SurfaceModelMessages::initialization_error,
94  "The SurfaceModel pointer supplied to "
95  "InteractionSurfaceFactory::create_surface was NULL. Please "
96  "provide a SurfaceModel pointer.");
97 
98  return;
99  }
100 
101  if(inter_surface == nullptr)
102  {
103  MessageHandler::fail(__FILE__,
104  __LINE__,
105  SurfaceModelMessages::initialization_error,
106  "The InteractionSurface pointer supplied to "
107  "InteractionSurfaceFactory::create_surface was NULL. Please "
108  "provide an InteractionSurface pointer.");
109  return;
110  }
111 
112  // Allocate the array, through this virtual function, in the
113  // interaction surface
114  inter_surface->allocate_array(surface->facets.size());
115 
116  // For all facets in the surface, match up an interaction facet factory
117  for(unsigned int ii = 0; ii < surface->facets.size(); ++ii)
118  {
119  InteractionFacetFactory * facet_factory = nullptr;
120 
121  for(unsigned int jj = 0; jj < factories.size(); ++jj)
122  {
123  if(factories[jj]->is_correct_factory(surface->facets[ii]))
124  {
125  facet_factory = factories[jj];
126  break;
127  }
128  }
129 
130  if(facet_factory == nullptr)
131  {
132  MessageHandler::fail(__FILE__,
133  __LINE__,
134  SurfaceModelMessages::initialization_error,
135  "No InteractionFacetFactory was found for the facet at index "
136  "%d. Please provide an appropriate InteractionFacetFactory",
137  ii);
138 
139  return;
140  }
141 
142  // For each facet, try to match up a facet_params that matches
143  // them.
144 
145  FacetParams * facet_params = nullptr;
146 
147  for(unsigned int jj = 0; jj < params.size(); ++jj)
148  {
149  if(surface->facets[ii]->param_name.empty())
150  {
151  MessageHandler::fail(__FILE__,
152  __LINE__,
153  SurfaceModelMessages::initialization_error,
154  "A Facet was found with no param_name set.");
155  }
156 
157  if(params[jj]->name == surface->facets[ii]->param_name)
158  {
159  facet_params = params[jj];
160  break;
161  }
162  }
163 
164  if(facet_params == nullptr)
165  {
166  MessageHandler::fail(__FILE__,
167  __LINE__,
168  SurfaceModelMessages::initialization_error,
169  "No facet_params with name %s was found.",
170  surface->facets[ii]->param_name.c_str());
171  }
172 
173  inter_surface->allocate_interaction_facet(surface->facets[ii], facet_factory, facet_params, ii);
174  } // for(unsigned int ii = 0)
175 }
