92 InteractionSurfaceFactory::create_surface (
93  SurfaceModel* surface,
94  InteractionSurface* inter_surface)
95 {
96 
97  // can't hand this un-allocated things. check for that.
98  if (surface == nullptr) {
99 
100  MessageHandler::fail (
101  __FILE__, __LINE__, SurfaceModelMessages::initialization_error,
102  "The SurfaceModel pointer supplied to "
103  "InteractionSurfaceFactory::create_surface was NULL. Please "
104  "provide a SurfaceModel pointer.");
105 
106  return;
107  }
108 
109  if (inter_surface == nullptr) {
110 
111  MessageHandler::fail (
112  __FILE__, __LINE__, SurfaceModelMessages::initialization_error,
113  "The InteractionSurface pointer supplied to "
114  "InteractionSurfaceFactory::create_surface was NULL. Please "
115  "provide an InteractionSurface pointer.");
116  return;
117  }
118 
119 
120 
121  // Allocate the array, through this virtual function, in the
122  // interaction surface
123  inter_surface->allocate_array (surface->facets.size());
124 
125 
126 
127  // For all facets in the surface, match up an interaction facet factory
128  for (unsigned int ii = 0; ii < surface->facets.size(); ++ii) {
129 
130  InteractionFacetFactory* facet_factory = nullptr;
131 
132  for (unsigned int jj = 0; jj < factories.size(); ++jj) {
133 
134  if (factories[jj]->is_correct_factory (surface->facets[ii])) {
135  facet_factory = factories[jj];
136  break;
137  }
138 
139  }
140 
141  if (facet_factory == nullptr) {
142 
143  MessageHandler::fail (
144  __FILE__, __LINE__, SurfaceModelMessages::initialization_error,
145  "No InteractionFacetFactory was found for the facet at index "
146  "%d. Please provide an appropriate InteractionFacetFactory",
147  ii);
148 
149  return;
150  }
151 
152  // For each facet, try to match up a facet_params that matches
153  // them.
154 
155  FacetParams* facet_params = nullptr;
156 
157  for (unsigned int jj = 0; jj < params.size(); ++jj) {
158 
159  if (surface->facets[ii]->param_name == nullptr) {
160  MessageHandler::fail (
161  __FILE__, __LINE__,
162  SurfaceModelMessages::initialization_error,
163  "A Facet was found with no param_name set.");
164  }
165 
166  if (strcmp (params[jj]->name.c_str(), surface->facets[ii]->param_name)
167  == 0) {
168  facet_params = params[jj];
169  break;
170  }
171  }
172 
173  if (facet_params == nullptr) {
174 
175  MessageHandler::fail (
176  __FILE__, __LINE__,
177  SurfaceModelMessages::initialization_error,
178  "No facet_params with name %s was found.",
179  surface->facets[ii]->param_name);
180 
181  }
182 
183 
184  inter_surface->allocate_interaction_facet (surface->facets[ii],
185  facet_factory,
186  facet_params,
187  ii);
188  } // for(unsigned int ii = 0)
189  return;
190 }
