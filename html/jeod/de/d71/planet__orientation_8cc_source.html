56 /**
57  * Goes to the dyn manager given and searches for the planet
58  * indicated by the user inputted name, in the given dyn manager.
59  * Will send a fail message if the named planet is not found.
60  * Additionally, this will register this ephemeris interface
61  * with the given DynManager, and will set up the contained
62  * EphemerisOrientation to control the named planet.
63  * \param[in,out] dyn_manager DynManager where the attitude will be applied
64  */
65 void PlanetOrientation::initialize(DynManager & dyn_manager)
66 {
67  if(name.empty())
68  {
69  MessageHandler::fail(__FILE__,
70  __LINE__,
71  RNPMessages::initialization_error,
72  "The planet name was not set in the PlanetOrientation"
73  " object. This is required for initialization of the"
74  " PlanetOrientation object\n");
75  }
76 
77  planet = dyn_manager.find_planet(name);
78 
79  if(planet == nullptr)
80  {
81  MessageHandler::fail(__FILE__,
82  __LINE__,
83  RNPMessages::initialization_error,
84  "Planet %s specified in PlanetOrientation::initialize"
85  " was not found",
86  name.c_str());
87  return;
88  }
89 
90  // initialize everything relevant in the
91  planet->pfix.set_owner(this);
92  planet_rot_state = &(planet->pfix.state.rot);
93 
94  planet_rot_state->ang_vel_unit[0] = 0;
95  planet_rot_state->ang_vel_unit[1] = 0;
96  planet_rot_state->ang_vel_unit[2] = 1;
97  planet_rot_state->ang_vel_this[0] = 0;
98  planet_rot_state->ang_vel_this[1] = 0;
99  planet_rot_state->ang_vel_this[2] = planet_omega;
100  planet_rot_state->ang_vel_mag = planet_omega;
101 
102  // Initialize the lunar orientation
103  orient_interface.set_name(name, "pfix");
104  orient_interface.set_owner(this);
105  orient_interface.enable();
106 
107  dyn_manager.add_ephemeris(*this);
108  dyn_manager.add_ephem_item(orient_interface);
109 }
