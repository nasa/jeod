386 bool DynBody::attach_child(const std::string & this_point_name, const std::string & child_point_name, DynBody & child)
387 {
388  // References looked up from input args
389  const BodyRefFrame * parent_vehicle_pt;
390  const BodyRefFrame * child_vehicle_pt;
391  const MassPoint * parent_point;
392  const MassPoint * child_point;
393 
394  // status variables
395  std::string reason;
396  bool success = true;
397 
398  // Temporary points used for transformation math
399  MassPointState child_struct_wrt_child_point;
400  MassPointState child_point_wrt_parent_point;
401  MassPointState child_struct_wrt_parent_struct;
402  double T_tmp[3][3];
403  Matrix3x3::initialize(T_tmp);
404 
405  // Validate later when we call the attach by pos/orientatation method. For
406  // now, find the mass points corresponding to the input names and verify
407  // that they exist.
408  parent_vehicle_pt = find_vehicle_point(this_point_name);
409  if(parent_vehicle_pt == nullptr)
410  {
411  parent_point = mass.find_mass_point(this_point_name);
412  }
413  else
414  {
415  parent_point = parent_vehicle_pt->mass_point;
416  }
417  child_vehicle_pt = child.find_vehicle_point(child_point_name);
418  if(child_vehicle_pt == nullptr)
419  {
420  child_point = child.mass.find_mass_point(child_point_name);
421  }
422  else
423  {
424  child_point = child_vehicle_pt->mass_point;
425  }
426 
427  // Handle failed searches as non-fatal errors (warning + failure return).
428  if(parent_point == nullptr)
429  {
430  reason = "Failed to find attach point on parent DynBody";
431  success = false;
432  }
433  if(child_point == nullptr)
434  {
435  if(success == false) // already found issue with this_point
436  {
437  reason = "Failed to find attach points on both parent and child";
438  }
439  else
440  {
441  reason = "Failed to find attach point on child MassBody";
442  }
443  success = false;
444  }
445  if(!success)
446  {
447  MessageHandler::warn(__FILE__,
448  __LINE__,
449  DynBodyMessages::invalid_attachment,
450  "%s\n"
451  "Attachment of '%s' to '%s' was not performed.\n",
452  reason.c_str(),
453  child_point_name.c_str(),
454  this_point_name.c_str());
455  return success;
456  }
457 
458  // Construct the affine transformation from the mass point location and
459  // frame of the attach point on child body to child body's structural
460  // origin and frame (i.e.: the inverse of the mass point specification
461  // within its own body)
462  Vector3::transform(child_point->T_parent_this, child_point->position, child_struct_wrt_child_point.position);
463  Vector3::negate(child_struct_wrt_child_point.position);
464  Matrix3x3::transpose(child_point->T_parent_this, child_struct_wrt_child_point.T_parent_this);
465  child_point->Q_parent_this.conjugate(child_struct_wrt_child_point.Q_parent_this);
466 
467  // Construct the affine transformation from the attach point on the child
468  // body to the attach point on the parent dynamic body.
469  // Per JEOD convention, this is always defined as a zero offset with a
470  // corresponding 180 degree yaw orientation.
471  // Thus, the point is initialized with zero offset and identity transform.
472  child_point_wrt_parent_point.T_parent_this[0][0] = -1.0;
473  child_point_wrt_parent_point.T_parent_this[1][1] = -1.0;
474 
475  // Calculate the compound affine transformation from the attach point on
476  // the parent structure to the child structure
477  // 1) create position vector
478  // 1a) Create transform for childpt_wrt_childstruct -> parent_struct frame
479  Matrix3x3::product_transpose_transpose(parent_point->T_parent_this,
480  child_point_wrt_parent_point.T_parent_this,
481  T_tmp);
482  // 1b) Transform childpt_wrt_childstruct into parent struct frame
483  Vector3::transform(T_tmp, child_struct_wrt_child_point.position, child_struct_wrt_parent_struct.position);
484  // 1c) sum position vectors in parent frame
485  Vector3::incr(parent_point->position, child_struct_wrt_parent_struct.position);
486  // 2) Create transformation
487  // T_pstr_cstr = T_pstr_ppt -> T_ppt_cpt -> T_cpt_cstr
488  // 2a) T_pstr_ppr -> T_ppt_cpt
489  Matrix3x3::product(child_point_wrt_parent_point.T_parent_this, parent_point->T_parent_this, T_tmp);
490  // 2b) -> T_cpt_cstr
491  Matrix3x3::product(child_struct_wrt_child_point.T_parent_this, T_tmp, child_struct_wrt_parent_struct.T_parent_this);
492 
493  // Attach using geometry-based attachment
494  success = attach_child(child_struct_wrt_parent_struct.position,
495  child_struct_wrt_parent_struct.T_parent_this,
496  child);
497 
498  return success;
499 }

514

520  }

538 
539  // Give a low-level warning that the attaching body is not a root body.
540  MessageHandler::inform(__FILE__,

562  Vector3::copy(xyz_cstr_wrt_pstr, xyz_rstr_wrt_pstr);

566  attach_update_properties(xyz_rstr_wrt_pstr, T_pstr_to_rstr, *child_root);
567  success = true;
568  }
569

571 }

639  // origin and frame (i.e.: the inverse of the mass point specification
640  // within its own body)
641  Vector3::transform(child_point->T_parent_this, child_point->position, child_struct_wrt_child_point.position);
642  Vector3::negate(child_struct_wrt_child_point.position);
643  Matrix3x3::transpose(child_point->T_parent_this, child_struct_wrt_child_point.T_parent_this);
644  child_point->Q_parent_this.conjugate(child_struct_wrt_child_point.Q_parent_this);
645 
646  // Construct the affine transformation from the attach point on the child
647  // body to the attach point on the parent dynamic body.
648  // Per JEOD convention, this is always defined as a zero offset with a
649  // corresponding 180 degree yaw orientation.
650  // Thus, the point is initialized with zero offset and identity transform.
651  child_point_wrt_parent_point.T_parent_this[0][0] = -1.0;
652  child_point_wrt_parent_point.T_parent_this[1][1] = -1.0;
653 
654  // Calculate the compound affine transformation from the attach point on
655  // the parent structure to the child structure
656  // 1) create position vector
657  // 1a) Create transform for childpt_wrt_childstruct -> parent_struct frame
658  Matrix3x3::product_transpose_transpose(parent_point->T_parent_this,
659  child_point_wrt_parent_point.T_parent_this,
660  T_tmp);
661  // 1b) Transform childpt_wrt_childstruct into parent struct frame
662  Vector3::transform(T_tmp, child_struct_wrt_child_point.position, child_struct_wrt_parent_struct.position);
663  // 1c) sum position vectors in parent frame
664  Vector3::incr(parent_point->position, child_struct_wrt_parent_struct.position);
665  // 2) Create transformation
666  // T_pstr_cstr = T_pstr_ppt -> T_ppt_cpt -> T_cpt_cstr
667  // 2a) T_pstr_ppr -> T_ppt_cpt
668  Matrix3x3::product(child_point_wrt_parent_point.T_parent_this, parent_point->T_parent_this, T_tmp);
669  // 2b) -> T_cpt_cstr
670  Matrix3x3::product(child_struct_wrt_child_point.T_parent_this, T_tmp, child_struct_wrt_parent_struct.T_parent_this);
671 
672  // Attach using geometry-based attachment
673  success = add_mass_body(child_struct_wrt_parent_struct.position,
674  child_struct_wrt_parent_struct.T_parent_this,
675  child);
676 
677  return success;

681 bool DynBody::add_mass_body(const double xyz_cstr_wrt_pstr[3], const double T_pstr_to_cstr[3][3], MassBody & child)

684  MassBody * child_root;
685 
686  // Status variables
687  bool success;

743  add_mass_body_frames(*child_root);
744

747  success = true;
748  return success;

798 // Set the relation between parent and child and update the mass
799 // properties.
800 void DynBody::attach_update_properties(const double offset_pstr_cstr_pstr[3],
801  const double T_pstr_cstr[3][3],
802  DynBody & child)
803 {
804  DynBody * root_body = get_root_body_internal();
805  bool child_has_state;
806  bool root_has_state;
807 
808  // Determine if the child has state (pos/vel/att/rate) information.
809  if(child.initialized_states.is_empty())
810  {
811  child_has_state = false;
812  }
813  else
814  {
815  child_has_state = true;
816  }
817 
818  // Determine if the root body has state information.
819  if(root_body->initialized_states.is_empty())
820  {
821  root_has_state = false;
822  }
823  else
824  {
825  root_has_state = true;
826  }
827 
828  // There are three cases to handle:
829  // - Simple attachment:
830  // -- A DynBody with an uninitialized state attaches
831  // to a parent DynBody with an uninitialized state.
832  // -- Since the state is uninitialized there is no state to update.
833  // - Attachment with simple update
834  // -- A DynBody with an uninitialized state attaches
835  // to a parent DynBody with a fully initialized state.
836  // -- The parent state is updated per assumption that the parent structure
837  // state is unchanged by the attachment.
838  // - Dynamic attachment
839  // -- A DynBody with a fully initialized state attaches to a parent
840  // DynBody with a fully initialized state.
841  // -- State is updated to conserve momenta.
842 
843  // Simple attachment:
844  // No state yet, so just update mass properties.
845  if((!child_has_state) && (!root_has_state))
846  {
847  mass.attach_update_properties(offset_pstr_cstr_pstr, T_pstr_cstr, child.mass);
848  }
849 
850  // Attachment, simple update:
851  // Update properties and set source from structure.
852  // Even this simple attachment may cause a jolt to state integration,
853  // so reset integrators for this body.
854  else if((!child_has_state) && root_has_state)
855  {
856  mass.attach_update_properties(offset_pstr_cstr_pstr, T_pstr_cstr, child.mass);
857 
858  root_body->set_state_source_internal(RefFrameItems::Pos_Vel_Att_Rate, root_body->structure);
859  root_body->propagate_state();
860 
861  get_dynamics_integration_group()->reset_integrators();
862  }
863 
864  // Dynamic attachment:
865  // Perform attachment, conserving momenta.
866  // This will definitely jolt state integration, so reset integrators for
867  // this body.
868  else
869  {
870  process_dynamic_attachment(offset_pstr_cstr_pstr, T_pstr_cstr, *root_body, child);
871 
872  get_dynamics_integration_group()->reset_integrators();
873  }
874 }

876 // Process the attachment event of one body from another.
877 void DynBody::process_dynamic_attachment(const double offset_pstr_cstr_pstr[3],
878  const double T_pstr_cstr[3][3],
879  DynBody & root_body,
880  DynBody & child_body)
881 {
882  // Definitions:
883  // child - The DynBody that is being attached to this body.
884  // parent - The body to which the child is being attached; this body.
885  // root - The DynBody at the head of this body's attachment tree.
886 
887  // Mass
888  double mass_parent; /* kg
889  mass of the parent-composite. */
890 
891  // Positions
892  double orig_cm[3]; /* M
893  The location of the root body's composite center of mass prior to
894  attachment, expressed in root structural coordinates. */
895  double new_cm[3]; /* M
896  The location of the root body's composite center of mass after
897  attachment, expressed in root structural coordinates. */
898  double cm_delta[3]; /* M
899  The change in the center of mass, new_cm - orig_cm. */
900  double cm_delta_inrtl[3]; /* M
901  cm_delta expressed in inertial coordinates. */
902  double child_pos_rel_newcm_inrtl[3]; /* M
903  The position of the child body's composite center of mass with respect to
904  the post-attachment root body center of mass, expressed in inertial. */
905 
906  // Velocities
907  double parent_vel_inrtl[3]; /* M/s
908  The inertial velocity of the root body's composite CoM prior to attach. */
909  double child_vel_inrtl[3]; /* M/s
910  The inertial velocity of the child body's composite CoM. */
911  double child_vel_relative[3]; /* M/s
912  The child body's velocity relative to the combined CoM. */
913  double parent_vel_relative[3]; /* M/s
914  The root body's velocity relative to the combined CoM. */
915 
916  // Linear momenta
917  double linear_momentum_child_composite[3]; /* N*s
918  The linear momentum of child-composite, expressed in inertial axes,
919  measured w.r.t. the integration frame origin. */
920  double child_lin_mom_relative[3]; /* N*s
921  The linear momentum of child-composite, expressed in inertial axes,
922  measured w.r.t. the center of mass of the combined body. */
923  double linear_momentum_parent_composite[3]; /* N*s
924  The linear momentum of parent-composite, expressed in inertial axes,
925  measured w.r.t. the integration frame origin. */
926  double parent_lin_mom_relative[3]; /* N*s
927  The linear momentum of parent-composite, expressed in inertial axes,
928  measured w.r.t. the center of mass of the combined body. */
929  double linear_momentum[3]; /* N*s
930  The linear momentum of combined-body. Expressed in the
931  inertial axes w.r.t. the integration frame origin. */
932 
933  // Angular momenta
934  double parent_ang_mom_pbdy[3]; /* J*s
935  Angular momentum of parent-composite. Expressed in the body axes of
936  parent-composite, and measured with respect to the momentary (and
937  inertial) location of the center of mass of that same composite body
938  (so includes only rotation terms, with no angular momentum due to
939  linear motion on a moment arm). */
940  double child_ang_mom_cbdy[3]; /* J*s
941  Angular momentum of child-composite. Expressed in the body axes of
942  child-composite, and measured with respect to the momentary (and
943  inertial) location of the center of mass of that same composite body
944  (so includes only rotation terms, with no angular momentum due to
945  linear motion on a moment arm). */
946  double child_ang_mom_inrtl[3]; /* J*s
947  The transform of child_ang_mom_cbdy so that it is expressed in inertial
948  axes. */
949  double child_ang_mom_vel[3]; /* J*s
950  The angular momentum of child-composite due to the linear motion of
951  same body past the center of mass of combined-body. Measured w.r.t the
952  momentary (and inertial) location of the center of mass of combined-body,
953  expresssed in inertial axes.*/
954  double parent_ang_mom_vel[3]; /* J*s
955  The angular momentum of parent-composite due to the linear motion of
956  same body past the center of mass of combined-body. Measured w.r.t the
957  momentary (and inertial) location of the center of mass of combined-body,
958  expresssed in inertial axes.*/
959  double angular_momentum[3]; /* J*s
960  Multiple angular momentum representations in different frames. Ultimately
961  becomes the angular momentum of combined-body, w.r.t. inertial axes and
962  its own center of mass, expressed in the body axes of combined-body.*/
963 
964  // Get pointers to the root bodies corresponding to the parent and
965  // child bodies.
966 
967  // Compute the linear momentum for these bodies (individually & combined).
968  Vector3::scale(child_body.composite_body.state.trans.velocity,
969  child_body.mass.composite_properties.mass,
970  linear_momentum_child_composite);
971  Vector3::scale(root_body.composite_body.state.trans.velocity,
972  root_body.mass.composite_properties.mass,
973  linear_momentum_parent_composite);
974  Vector3::sum(linear_momentum_child_composite, linear_momentum_parent_composite, linear_momentum);
975 
976  // Compute the angular momentum of the parent composite body.
977  // NOTE: This is about the parent's composite body CM prior to attachment,
978  // expressed in parent composite body coordinates.
979  Vector3::transform(root_body.mass.composite_properties.inertia,
980  root_body.composite_body.state.rot.ang_vel_this,
981  parent_ang_mom_pbdy);
982 
983  // Compute the angular momentum of the child composite body.
984  // NOTE: This is about the child's composite body CM prior to attachment,
985  // expressed in child composite body coordinates.
986  Vector3::transform(child_body.mass.composite_properties.inertia,
987  child_body.composite_body.state.rot.ang_vel_this,
988  child_ang_mom_cbdy);
989 
990  // Store the parent's and child's linear velocities and parent mass.
991  Vector3::copy(root_body.composite_body.state.trans.velocity, parent_vel_inrtl);
992  Vector3::copy(child_body.composite_body.state.trans.velocity, child_vel_inrtl);
993  mass_parent = root_body.mass.composite_properties.mass;
994 
995  // Store the original CM of the parent's root body (in parent's root
996  // structural frame).
997  Vector3::copy(root_body.mass.composite_properties.position, orig_cm);
998 
999  // Attach the bodies as MassBodies.
1000  mass.attach_update_properties(offset_pstr_cstr_pstr, T_pstr_cstr, child_body.mass);
1001 
1002  // Store the CM of the new root body
1003  // (i.e. the combined-body body-frame origin w.r.t., and
1004  // expressed in, its structural-frame).
1005  Vector3::copy(root_body.mass.composite_properties.position, new_cm);
1006 
1007  // Compute the change in CM. */
1008  Vector3::diff(new_cm, orig_cm, cm_delta);
1009 
1010  // Update the translational position of the new root body.
1011  // NOTE: The parent reference frame of a root body is inertial, so
1012  // T_parent_this is inertial to structural, we need structural to inertial)
1013  Vector3::transform_transpose(root_body.structure.state.rot.T_parent_this, cm_delta, cm_delta_inrtl);
1014 
1015  Vector3::incr(cm_delta_inrtl, root_body.composite_body.state.trans.position);
1016 
1017  // Compute linear velocity from conservation of linear momentum.
1018  // NOTE: These computations must be in the inertial reference frame.
1019  // Total linear momentum was already calculated before the attachment
1020  Vector3::scale(linear_momentum,
1021  root_body.mass.composite_properties.inverse_mass,
1022  root_body.composite_body.state.trans.velocity);
1023 
1024  // Compute angular velocity from conservation of angular momentum.
1025  // NOTE: subscript c = child
1026  // p = parent
1027  // t = total (combined)
1028  // L = (r_p x (m_p v_p)) + I_p w_p + (r_c x (m_c v_c)) + I_c w_c
1029  // = (r_t x (m_t v_t)) + I_t w_t
1030  //
1031  // Shift to a reference frame in which v_t = 0 and r_t = 0 (due to
1032  // "magical" migration of child, this is a non-physical problem, and
1033  // angular momentum cannot be conserved in all frames simultaneously, this
1034  // is the best choice).
1035  //
1036  // L = (r_p - r_t) x (m_p (v_p - v_t)) + I_p w_p +
1037  // (r_c - r_t) x (m_c (v_c - v_t)) + I_c w_c
1038  // = I_t w_t
1039  //
1040  // hence,
1041  // w_t = inv(I_t) *
1042  // ((r_p-r_t) x (m_p (v_p - v_t)) +
1043  // (r_c-r_t) x (m_c (v_c - v_t)) + I_c w_c + I_p w_p)
1044  // r_t and v_t already calculated
1045 
1046  // Compute the relative position of the child's (pre-attachment) root
1047  // composite body with respect to the new center of mass.
1048  // NOTE: Wanting the position of child prior to the "snap" w.r.t updated
1049  // position of parent c.o.m..
1050  // Child position has not been updated yet, but parent has, and it reflects
1051  // the new composite CoM.
1052  Vector3::diff(child_body.composite_body.state.trans.position,
1053  root_body.composite_body.state.trans.position,
1054  child_pos_rel_newcm_inrtl);
1055 
1056  // Compute the relative momentum of the child and (pre-attachment) parent
1057  // with respect to the new center of mass.
1058  Vector3::diff(child_vel_inrtl, root_body.composite_body.state.trans.velocity, child_vel_relative);
1059 
1060  Vector3::scale(child_vel_relative, child_body.mass.composite_properties.mass, child_lin_mom_relative);
1061 
1062  Vector3::diff(parent_vel_inrtl, root_body.composite_body.state.trans.velocity, parent_vel_relative);
1063 
1064  Vector3::scale(parent_vel_relative, mass_parent, parent_lin_mom_relative);
1065 
1066  // Compute the angular momentum components due to relative velocities.
1067 
1068  // Compute the contribution made by the child's relative velocity to the
1069  // overall angular momentum. L = r x p
1070  Vector3::cross(child_pos_rel_newcm_inrtl, child_lin_mom_relative, child_ang_mom_vel);
1071 
1072  // Compute the contribution made by the (unattached) parent's relative
1073  // velocity to the overall angular momentum.
1074  // Use L = p x (-r) since -r is already known.
1075  Vector3::cross(parent_lin_mom_relative, cm_delta_inrtl, parent_ang_mom_vel);
1076 
1077  // Transform the child's inherent angular momentum (i.e. I*w) to the
1078  // inertial frame so that it can be added.
1079  // NOTE: Parent's angular momentum stays in parent body frm for later
1080 
1081  Vector3::transform_transpose(child_body.composite_body.state.rot.T_parent_this,
1082  child_ang_mom_cbdy,
1083  child_ang_mom_inrtl);
1084 
1085  // Add together the 3 components calculated so far:
1086  // angular momentum due to child's rotation
1087  // angular momentum due to child's relative motion
1088  // angular momentum due to parent's relative motion
1089 
1090  Vector3::sum(child_ang_mom_vel, parent_ang_mom_vel, child_ang_mom_inrtl, angular_momentum);
1091 
1092  // Convert back to the body frame of the parent (now the composite)
1093  // NOTE: the new parent body frame is colinear with the original body frame
1094  // (just offset and moving), so T_inertial_body is unchanged.
1095  // NOTE: angular_momentum is in the body frame after this transformation.
1096  Vector3::transform(root_body.composite_body.state.rot.T_parent_this, angular_momentum);
1097 
1098  // now add in the parent's Iw term for total angular momentum
1099  // NOTE: parent_ang_mom_pbdy is already in the new parent body frame.
1100  Vector3::incr(parent_ang_mom_pbdy, angular_momentum);
1101 
1102  // Compute the body rate from the angular momentum (inverse inertia already
1103  // calculated as part of mass_update)
1104  Vector3::transform(root_body.mass.composite_properties.inverse_inertia,
1105  angular_momentum,
1106  root_body.composite_body.state.rot.ang_vel_this);
1107 
1108  // Propagate state down through the tree:
1109  // Tell the function where to obtain the data from.
1110  // Position and attitude from the core body,
1111  // Velocity and rate from the composite (values computed above).
1112  root_body.set_state_source_internal(RefFrameItems::Pos_Att, root_body.core_body);
1113  root_body.set_state_source_internal(RefFrameItems::Vel_Rate, root_body.composite_body);
1114 
1115  // Finally, propagate the state.
1116  root_body.propagate_state();
1117 }
