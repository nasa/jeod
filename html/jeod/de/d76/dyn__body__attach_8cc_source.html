373 bool DynBody::attach_child (
374  const char * this_point_name,
375  const char * child_point_name,
376  DynBody & child )
377 {
378  // References looked up from input args
379  const BodyRefFrame* parent_vehicle_pt;
380  const BodyRefFrame* child_vehicle_pt;
381  const MassPoint* parent_point;
382  const MassPoint* child_point;
383 
384  // status variables
385  std::string reason;
386  bool success = true;
387 
388  // Temporary points used for transformation math
389  MassPointState child_struct_wrt_child_point;
390  MassPointState child_point_wrt_parent_point;
391  MassPointState child_struct_wrt_parent_struct;
392  double T_tmp[3][3];
393  Matrix3x3::initialize(T_tmp);
394 
395  // Validate later when we call the attach by pos/orientatation method. For
396  // now, find the mass points corresponding to the input names and verify
397  // that they exist.
398  parent_vehicle_pt = find_vehicle_point( this_point_name );
399  if( parent_vehicle_pt == nullptr )
400  {
401  parent_point = mass.find_mass_point( this_point_name );
402  }
403  else
404  {
405  parent_point = parent_vehicle_pt->mass_point;
406  }
407  child_vehicle_pt = child.find_vehicle_point( child_point_name );
408  if( child_vehicle_pt == nullptr )
409  {
410  child_point = child.mass.find_mass_point( child_point_name );
411  }
412  else
413  {
414  child_point = child_vehicle_pt->mass_point;
415  }
416 
417 
418  // Handle failed searches as non-fatal errors (warning + failure return).
419  if( parent_point == nullptr )
420  {
421  reason = "Failed to find attach point on parent DynBody";
422  success = false;
423 
424  }
425  if( child_point == nullptr )
426  {
427  if( success == false ) // already found issue with this_point
428  {
429  reason = "Failed to find attach points on both parent and child";
430  }
431  else
432  {
433  reason = "Failed to find attach point on child MassBody";
434  }
435  success = false;
436  }
437  if(!success) {
438  MessageHandler::warn (
439  __FILE__, __LINE__, DynBodyMessages::invalid_attachment,
440  "%s\n"
441  "Attachment of '%s' to '%s' was not performed.\n",
442  reason.c_str(), child_point_name, this_point_name);
443  return success;
444  }
445 
446  // Construct the affine transformation from the mass point location and
447  // frame of the attach point on child body to child body's structural
448  // origin and frame (i.e.: the inverse of the mass point specification
449  // within its own body)
450  Vector3::transform( child_point->T_parent_this,
451  child_point->position,
452  child_struct_wrt_child_point.position );
453  Vector3::negate( child_struct_wrt_child_point.position );
454  Matrix3x3::transpose( child_point->T_parent_this,
455  child_struct_wrt_child_point.T_parent_this );
456  child_point->Q_parent_this.conjugate(
457  child_struct_wrt_child_point.Q_parent_this);
458 
459  // Construct the affine transformation from the attach point on the child
460  // body to the attach point on the parent dynamic body.
461  // Per JEOD convention, this is always defined as a zero offset with a
462  // corresponding 180 degree yaw orientation.
463  // Thus, the point is initialized with zero offset and identity transform.
464  child_point_wrt_parent_point.T_parent_this[0][0] = -1.0;
465  child_point_wrt_parent_point.T_parent_this[1][1] = -1.0;
466 
467  // Calculate the compound affine transformation from the attach point on
468  // the parent structure to the child structure
469  // 1) create position vector
470  // 1a) Create transform for childpt_wrt_childstruct -> parent_struct frame
471  Matrix3x3::product_transpose_transpose(
472  parent_point->T_parent_this,
473  child_point_wrt_parent_point.T_parent_this,
474  T_tmp);
475  // 1b) Transform childpt_wrt_childstruct into parent struct frame
476  Vector3::transform(
477  T_tmp,
478  child_struct_wrt_child_point.position,
479  child_struct_wrt_parent_struct.position);
480  // 1c) sum position vectors in parent frame
481  Vector3::incr(
482  parent_point->position,
483  child_struct_wrt_parent_struct.position);
484  // 2) Create transformation
485  // T_pstr_cstr = T_pstr_ppt -> T_ppt_cpt -> T_cpt_cstr
486  // 2a) T_pstr_ppr -> T_ppt_cpt
487  Matrix3x3::product(
488  child_point_wrt_parent_point.T_parent_this,
489  parent_point->T_parent_this,
490  T_tmp);
491  // 2b) -> T_cpt_cstr
492  Matrix3x3::product(
493  child_struct_wrt_child_point.T_parent_this,
494  T_tmp,
495  child_struct_wrt_parent_struct.T_parent_this);
496 
497  // Attach using geometry-based attachment
498  success = attach_child( child_struct_wrt_parent_struct.position,
499  child_struct_wrt_parent_struct.T_parent_this,
500  child );
501 
502  return success;
503 }

569  child_struct_wrt_root_struct.T_parent_this,
570  T_pstr_to_cstr,
571  T_pstr_to_rstr);
572

574  xyz_rstr_wrt_pstr );

576  T_pstr_to_rstr,
577  child_struct_wrt_root_struct.position,
578  xyz_rstr_wrt_pstr );

581  attach_update_properties( xyz_rstr_wrt_pstr,
582  T_pstr_to_rstr,
583  *child_root );
584  success = true;
585  }
586

588 }

657  // frame of the attach point on child body to child body's structural
658  // origin and frame (i.e.: the inverse of the mass point specification
659  // within its own body)

661  child_point->position,

664  Matrix3x3::transpose( child_point->T_parent_this,
665  child_struct_wrt_child_point.T_parent_this );

667  child_struct_wrt_child_point.Q_parent_this);

672  // corresponding 180 degree yaw orientation.
673  // Thus, the point is initialized with zero offset and identity transform.
674  child_point_wrt_parent_point.T_parent_this[0][0] = -1.0;
675  child_point_wrt_parent_point.T_parent_this[1][1] = -1.0;

678  // the parent structure to the child structure
679  // 1) create position vector

688  child_struct_wrt_child_point.position,
689  child_struct_wrt_parent_struct.position);
690  // 1c) sum position vectors in parent frame
691  Vector3::incr(
692  parent_point->position,
693  child_struct_wrt_parent_struct.position);
694  // 2) Create transformation
695  // T_pstr_cstr = T_pstr_ppt -> T_ppt_cpt -> T_cpt_cstr

698  child_point_wrt_parent_point.T_parent_this,
699  parent_point->T_parent_this,
700  T_tmp);
701  // 2b) -> T_cpt_cstr
702  Matrix3x3::product(
703  child_struct_wrt_child_point.T_parent_this,
704  T_tmp,
705  child_struct_wrt_parent_struct.T_parent_this);
706

708  success = add_mass_body( child_struct_wrt_parent_struct.position,
709  child_struct_wrt_parent_struct.T_parent_this,
710  child );

868 // Set the relation between parent and child and update the mass
869 // properties.
870 void
871 DynBody::attach_update_properties (
872  const double offset_pstr_cstr_pstr[3],
873  const double T_pstr_cstr[3][3],
874  DynBody & child)
875 {
876  DynBody * root_body = get_root_body_internal();
877  bool child_has_state;
878  bool root_has_state;
879 
880  // Determine if the child has state (pos/vel/att/rate) information.
881  if( child.initialized_states.is_empty() )
882  {
883  child_has_state = false;
884  }
885  else {
886  child_has_state = true;
887 
888  }
889 
890  // Determine if the root body has state information.
891  if (root_body->initialized_states.is_empty() )
892  {
893  root_has_state = false;
894  }
895  else {
896  root_has_state = true;
897  }
898 
899  // There are three cases to handle:
900  // - Simple attachment:
901  // -- A DynBody with an uninitialized state attaches
902  // to a parent DynBody with an uninitialized state.
903  // -- Since the state is uninitialized there is no state to update.
904  // - Attachment with simple update
905  // -- A DynBody with an uninitialized state attaches
906  // to a parent DynBody with a fully initialized state.
907  // -- The parent state is updated per assumption that the parent structure
908  // state is unchanged by the attachment.
909  // - Dynamic attachment
910  // -- A DynBody with a fully initialized state attaches to a parent
911  // DynBody with a fully initialized state.
912  // -- State is updated to conserve momenta.
913 
914  // Simple attachment:
915  // No state yet, so just update mass properties.
916  if( (!child_has_state) && (!root_has_state) )
917  {
918  mass.attach_update_properties (
919  offset_pstr_cstr_pstr, T_pstr_cstr, child.mass);
920  }
921 
922  // Attachment, simple update:
923  // Update properties and set source from structure.
924  // Even this simple attachment may cause a jolt to state integration,
925  // so reset integrators for this body.
926  else if( (!child_has_state) && root_has_state )
927  {
928  mass.attach_update_properties (
929  offset_pstr_cstr_pstr, T_pstr_cstr, child.mass);
930 
931  root_body->set_state_source_internal (RefFrameItems::Pos_Vel_Att_Rate,
932  root_body->structure);
933  root_body->propagate_state ();
934 
935  get_dynamics_integration_group()->reset_integrators();
936  }
937 
938  // Dynamic attachment:
939  // Perform attachment, conserving momenta.
940  // This will definitely jolt state integration, so reset integrators for
941  // this body.
942  else {
943  process_dynamic_attachment (offset_pstr_cstr_pstr, T_pstr_cstr,
944  *root_body, child);
945 
946  get_dynamics_integration_group()->reset_integrators();
947  }
948 }

1074  Vector3::copy (root_body.composite_body.state.trans.velocity,

1126  // is the best choice).

1130  // = I_t w_t
1131  //

1176  // velocity to the overall angular momentum.
1177  // Use L = p x (-r) since -r is already known.
1178  Vector3::cross (parent_lin_mom_relative,
1179  cm_delta_inrtl,
1180  parent_ang_mom_vel);
1181

1185  // inertial frame so that it can be added.
1186  // NOTE: Parent's angular momentum stays in parent body frm for later
1187 
1188  Vector3::transform_transpose (
1189  child_body.composite_body.state.rot.T_parent_this,
1190  child_ang_mom_cbdy,
1191  child_ang_mom_inrtl);
1192 
1193

1195  // angular momentum due to child's rotation
1196  // angular momentum due to child's relative motion
1197  // angular momentum due to parent's relative motion
1198 
1199  Vector3::sum (child_ang_mom_vel,
1200  parent_ang_mom_vel,
1201  child_ang_mom_inrtl,
1202  angular_momentum);
1203

1205  // NOTE: the new parent body frame is colinear with the original body frame
1206  // (just offset and moving), so T_inertial_body is unchanged.
1207  // NOTE: angular_momentum is in the body frame after this transformation.
1208  Vector3::transform (root_body.composite_body.state.rot.T_parent_this,
1209  angular_momentum);
1210

1212  // NOTE: parent_ang_mom_pbdy is already in the new parent body frame.
1213  Vector3::incr (parent_ang_mom_pbdy,
1214  angular_momentum);
1215 
1216  // Compute the body rate from the angular momentum (inverse inertia already
1217  // calculated as part of mass_update)
1218  Vector3::transform (root_body.mass.composite_properties.inverse_inertia,
1219  angular_momentum,
1220  root_body.composite_body.state.rot.ang_vel_this);

1224  // Tell the function where to obtain the data from.
1225  // Position and attitude from the core body,
1226  // Velocity and rate from the composite (values computed above).

1229  root_body.set_state_source_internal (RefFrameItems::Vel_Rate,
1230  root_body.composite_body);
