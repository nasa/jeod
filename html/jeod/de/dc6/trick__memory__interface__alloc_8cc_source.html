84 
85  // Register for checkpoint/restart in the allocation map.
86  allocation_map.insert (
87  AllocationMap::value_type (
88  unique_id,
89  AllocationMapEntry (
90  tdesc.get_typeid(), nelems, item.get_is_array())));
91 
92  // Construct a string that emulates the declaration of the memory.
93  std::stringstream sstream;
94  sstream << tdesc.get_name()
95  << " " << construct_identifier(unique_id)
96  << "[" << nelems << "]";
97 
98  // Construct a context for parsing the above string.
99  // Note: Requires Trick 10.2 or greater.
100  Trick::ADefParseContext context(&sstream);
101 
102  // Parse the declaration context and register the memory.
103  if ((ADEF_parse (&context) != 0) ||
104  (trick_MM->declare_extern_var (
105  const_cast <void *>(addr),
106  context.type,
107  context.user_type_name,
108  context.n_stars,
109  context.var_name,
110  context.n_cdims,
111  context.cdims) == nullptr)) {
112 
113  // Handle failed parse / failed registration.
114  MessageHandler::error (
115  __FILE__, __LINE__, SimInterfaceMessages::interface_error,
116  "Memory registration failed with allocation at %s:%d",
117  file, line);
118  return false;
119  }
