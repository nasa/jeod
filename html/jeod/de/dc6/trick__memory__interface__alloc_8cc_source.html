81 
82  // Register for checkpoint/restart in the allocation map.
83  allocation_map.insert(
84  AllocationMap::value_type(unique_id, AllocationMapEntry(tdesc.get_typeid(), nelems, item.get_is_array())));
85 
86  // Construct a string that emulates the declaration of the memory.

88  sstream << tdesc.get_name() << " " << construct_identifier(unique_id) << "[" << nelems << "]";
89 
90  // Construct a context for parsing the above string.

92  Trick::ADefParseContext context(&sstream);

96  context.type,

98  context.n_stars,
99  context.var_name,
100  context.n_cdims,

102  {
103  // Handle failed parse / failed registration.
104  MessageHandler::error(__FILE__,
105  __LINE__,
106  SimInterfaceMessages::interface_error,
