82 /**
83  * Initialize this GravityControl.
84  * \param[in] grav_manager Ref to Gravity Manager
85  */
86 void SphericalHarmonicsGravityControls::initialize_control(GravityManager & grav_manager)
87 {
88  // Perform basic initializations from base class
89  GravityControls::initialize_control(grav_manager);
90 
91  // Cast the base class GravitySource pointer to derived type
92  harmonics_source = dynamic_cast<SphericalHarmonicsGravitySource *>(body);
93 
94  // Check the validity of the provided spherical harmonics controls settings
95  check_validity();
96 
97  // If degree > 0 in the gravity body then create and fill Gottlieb
98  // LeGendre polynomial array. Otherwise only spherical gravity available.
99  unsigned int alloc_degree = harmonics_source->degree;
100 
101  if(alloc_degree > 0)
102  {
103  Pnm = JEOD_ALLOC_PRIM_ARRAY(alloc_degree + 1, double *);
104  double * int_to_double = JEOD_ALLOC_PRIM_ARRAY(alloc_degree + 2, double);
105 
106  for(unsigned int ii = 0; ii <= alloc_degree; ++ii)
107  {
108  int_to_double[ii] = static_cast<double>(ii);
109  Pnm[ii] = JEOD_ALLOC_PRIM_ARRAY(ii + 3, double);
110  }
111 
112  int_to_double[alloc_degree + 1] = static_cast<double>(alloc_degree + 1);
113 
114  // In the code below, the equation numbers and page numbers refer to
115  // the Gottlieb 1993 paper.
116 
117  // Bottom of page 47 and page 48, and see equation (7-8)
118  Pnm[0][0] = 1.0;
119  Pnm[0][1] = 0.0;
120  Pnm[0][2] = 0.0;
121  Pnm[1][1] = sqrt(3.0);
122  Pnm[1][2] = 0.0;
123  Pnm[1][3] = 0.0;
124 
125  // Pages 46-47
126  for(unsigned int ii = 2; ii <= alloc_degree; ++ii)
127  {
128  // P(n,n) term, equation (7-8)
129  Pnm[ii][ii] = sqrt((2.0 * int_to_double[ii] + 1.0) / (2.0 * int_to_double[ii])) * Pnm[ii - 1][ii - 1];
130 
131  // P(n,n+1) and P(n,n+2) terms, table 1 (p. 14)
132  Pnm[ii][ii + 1] = 0.0;
133  Pnm[ii][ii + 2] = 0.0;
134  }
135 
136  // Clean up local variable int_to_double (needed to init Pnm above)
137  JEOD_DELETE_ARRAY(int_to_double);
138  }
139 }

145 void SphericalHarmonicsGravityControls::add_deltacontrol(SphericalHarmonicsDeltaControls * delta_control)
146 {
147  SphericalHarmonicsDeltaCoeffs * temp_deltacoeff;
148  temp_deltacoeff = delta_control->grav_effect;
149 
150  // Add the gravity control to end of list
151  var_effects.push_back(delta_control);
152 
153  /* If delta_Cnm & delta_Snm need larger dimensions than currently exist,
154  delete and re-allocate them */
155  if((temp_deltacoeff->degree > delta_degree) || (temp_deltacoeff->order > delta_order))
156  {
157  unsigned int new_degree = temp_deltacoeff->degree;
158  unsigned int new_order = temp_deltacoeff->order;
159 
160  JEOD_DELETE_2D(delta_Cnm, delta_degree, true);
161  JEOD_DELETE_2D(delta_Snm, delta_degree, true);
162 
163  delta_Cnm = JEOD_ALLOC_PRIM_ARRAY(new_degree, double *);
164  delta_Snm = JEOD_ALLOC_PRIM_ARRAY(new_degree, double *);
165 
166  for(unsigned int ii = 0; ii < new_degree; ++ii)
167  {
168  delta_Cnm[ii] = JEOD_ALLOC_PRIM_ARRAY(new_order, double);
169  delta_Snm[ii] = JEOD_ALLOC_PRIM_ARRAY(new_order, double);
170  }
171  }
172 
173  // Update top-level max degree & order and array size if needed
174  if(temp_deltacoeff->degree > delta_degree)
175  {
176  delta_degree = temp_deltacoeff->degree;
177  }
178  if(temp_deltacoeff->order > delta_order)
179  {
180  delta_order = temp_deltacoeff->order;
181  }

319 /**
320  * Check the validity of the gravity controls.
321  */
322 void SphericalHarmonicsGravityControls::check_validity()
323 {
324  if(harmonics_source == nullptr)
325  {
326  MessageHandler::fail(__FILE__,
327  __LINE__,
328  GravityMessages::invalid_object,
329  "Gravity field '%s' has no associated source.",
330  source_name.c_str());
331  return;
332  }
333 
334  if(!spherical)
335  {
336  // Check if maximum degree to be used for computations is greater than
337  // degree of gravity field
338  if(degree > harmonics_source->degree)
339  {
340  MessageHandler::fail(__FILE__,
341  __LINE__,
342  GravityMessages::invalid_limit,
343  "Gravity field degree requested(%i) for %s "
344  "is greater than max gravity field degree(%i).",
345  degree,
346  harmonics_source->name.c_str(),
347  harmonics_source->degree);
348  return;
349  }
350 
351  // check if maximum order to be used for computations is greater than
352  // maximum order
353  if(order > harmonics_source->order)
354  {
355  MessageHandler::fail(__FILE__,
356  __LINE__,
357  GravityMessages::invalid_limit,
358  "Gravity field order requested(%i) for %s "
359  "is greater than max gravity field order(%i).",
360  order,
361  harmonics_source->name.c_str(),
362  harmonics_source->order);
363  return;
364  }
365 
366  // check if maximum order to be used for computations is greater than
367  // maximum degree
368  if(order > degree)
369  {
370  MessageHandler::fail(__FILE__,
371  __LINE__,
372  GravityMessages::invalid_limit,
373  "Gravity field order(%i) for %s is "
374  "greater than gravity field degree(%i).",
375  order,
376  harmonics_source->name.c_str(),
377  degree);
378  return;
379  }
380 
381  // GRADIENT ERRORS
382  // These are all recoverable.
383  if(gradient)
384  {
385  // check if gradient degree exceeds accel degree
386  if(gradient_degree > degree)
387  {
388  MessageHandler::fail(__FILE__,
389  __LINE__,
390  GravityMessages::invalid_limit,
391  "Gravity gradient degree (%i) for %s is "
392  "greater than gravity degree (%i).\n"
393  "Setting the gradient degree to the gravity degree.",
394  gradient_degree,
395  harmonics_source->name.c_str(),
396  degree);
397  gradient_degree = degree;
398  }
399 
400  // check if gradient degree equals 1
401  if(gradient_degree == 1)
402  {
403  MessageHandler::error(__FILE__,
404  __LINE__,
405  GravityMessages::invalid_limit,
406  "Gravity gradient degree for %s must not equal 1.\n"
407  "Resetting to zero.\n",
408  harmonics_source->name.c_str());
409  gradient_degree = 0;
410  }
411 
412  // check if gradient order to be used for computations is
413  // greater than the gradient degree
414  if(gradient_order > gradient_degree)
415  {
416  MessageHandler::error(__FILE__,
417  __LINE__,
418  GravityMessages::invalid_limit,
419  "Gravity gradient order (%i) for %s is "
420  "greater than gravity gradient degree (%i).\n"
421  "Setting the gradient order to the gradient degree.",
422  gradient_order,
423  harmonics_source->name.c_str(),
424  gradient_degree);
425  gradient_order = gradient_degree;
426  }
427 
428  // check if gradient order exceeds accel order
429  if(gradient_order > order)
430  {
431  MessageHandler::error(__FILE__,
432  __LINE__,
433  GravityMessages::invalid_limit,
434  "Gravity gradient order (%i) for %s is "
435  "greater than gravity order (%i).\n"
436  "Setting the gradient order to the gravity order.",
437  gradient_order,
438  harmonics_source->name.c_str(),
439  order);
440  gradient_order = order;
441  }
442  }
443  }

464 /**
465  * Loop over all of the active gravitational variation effects models
466  * and aggregate their changes to the gravity coefficients into the
467  * top-level delta-coeffs "bin" for this gravity body.
468  */
469 void SphericalHarmonicsGravityControls::sum_deltacoeffs()
470 {
471  unsigned int n_deltacoeffs = harmonics_source->delta_coeffs.size();
472  SphericalHarmonicsDeltaControls * temp_delta_control;
473  SphericalHarmonicsDeltaCoeffs * temp_deltacoeff;
474 
475  // Zero out Cnm, Snm elements and top-level dC20 term
476  total_dC20 = 0.0;
477  for(unsigned int ii = 0; ii < delta_degree; ++ii)
478  {
479  for(unsigned int jj = 0; jj < delta_order; ++jj)
480  {
481  delta_Cnm[ii][jj] = 0;
482  delta_Snm[ii][jj] = 0;
483  }
484  }
485 
486  // Loop over all active delta-coeffs and sum the coefficient-changing values
487  for(unsigned int ii = 0; ii < n_deltacoeffs; ++ii)
488  {
489  temp_delta_control = var_effects[ii];
490  if(temp_delta_control->active)
491  {
492  temp_deltacoeff = temp_delta_control->grav_effect;
493 
494  if(temp_delta_control->first_order_only)
495  {
496  total_dC20 += temp_deltacoeff->dC20;
497  }
498  else
499  {
500  for(unsigned int jj = 0; jj < temp_delta_control->degree; ++jj)
501  {
502  for(unsigned int kk = 0; kk < temp_delta_control->order; ++kk)
503  {
504  delta_Cnm[jj][kk] += temp_deltacoeff->delta_Cnm[jj][kk];
505  delta_Snm[jj][kk] += temp_deltacoeff->delta_Snm[jj][kk];
506  }
507  }
508  }
509  }
510  }
511 }
