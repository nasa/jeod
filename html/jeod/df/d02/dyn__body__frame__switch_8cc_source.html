115 /**
116  * Switch reference frames.
117  * \param[in,out] dyn_manager Jeod manager
118  */
119 void DynBodyFrameSwitch::apply(DynManager & dyn_manager)
120 {
121  // Debug.
122  MessageHandler::debug(__FILE__,
123  __LINE__,
124  BodyActionMessages::trace,
125  "%s: %s integration frame switched to %s.",
126  action_identifier.c_str(),
127  dyn_subject->name.c_str(),
128  integ_frame_name.c_str());
129 
130  // Switch frames.
131  dyn_subject->switch_integration_frames(*integ_frame);
132 
133  // Compute gravitation and sort if requested.
134  if(sort_grav_controls)
135  {
136  // Compute the gravitational acceleration from each gravity control.
137  const double * integ_pos = dyn_subject->composite_body.state.trans.position;
138  GravityInteraction & grav = dyn_subject->grav_interaction;
139  unsigned int n_controls = grav.grav_controls.size();
140  for(unsigned int ii = 0; ii < n_controls; ++ii)
141  {
142  GravityControls & control_ii = *(grav.grav_controls[ii]);
143  if(control_ii.active)
144  {
145  control_ii.gravitation(integ_pos,
146  grav.integ_frame_index,
147  control_ii.grav_accel,
148  control_ii.grav_grad,
149  &control_ii.grav_pot);
150  }
151  }
152 
153  // Sort the controls based on the newly computed grav accels.
154  grav.sort_controls();
155  }
156 
157  // Forward the action up the chain.
158  BodyAction::apply(dyn_manager);
159 }

161 /**
162  * Determine whether it is time to switch frames.
163  * A frame-switch action is ready if it is activated and if the
164  * vehicle has entered/left the appropriate sphere of influence.
165  * @return Can action be applied?
166  */
167 bool DynBodyFrameSwitch::is_ready()
168 {
169  bool ready;
170 
171  if(BodyAction::is_ready())
172  {
173  if(switch_sense == SwitchOnApproach)
174  {
175  double rel_pos[3];
176  dyn_subject->composite_body.compute_position_from(*integ_frame, rel_pos);
177  ready = Vector3::vmagsq(rel_pos) < switch_distance * switch_distance;
178  }
179  else
180  {
181  ready = Vector3::vmagsq(dyn_subject->composite_body.state.trans.position) >
182  switch_distance * switch_distance;
183  }
184  }
185  else
186  {
187  ready = false;
188  }
189 
190  return ready;
191 }
