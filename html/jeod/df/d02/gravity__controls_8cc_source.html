258 /**
259  * Compute the gravitation at a given position toward a gravity body.
260  * \param[in] point_of_interest Point of interest, as a reference frame.
261  * \param[in] integ_frame_idx Integ frame index.
262  * \param[out] body_grav_accel Accel for given grav body.
263  * \param[out] dgdx Gradient for given grav body.
264  * \param[out] pot Specific gravitational potential for given grev body.
265  */
266 void
267 GravityControls::gravitation (
268  const RefFrame& point_of_interest,
269  unsigned int integ_frame_idx,
270  double body_grav_accel[3],
271  double dgdx[3][3],
272  double& pot)
273 {
274  double rel_pos[3]; // M Vehicle inertial position wrt planet
275  GravityIntegFrame & grav_source_frame = // -- Grav frame for this integ frame
276  body->frames[integ_frame_idx];
277  const RefFrame & integ_frame = // -- Integration frame
278  *(grav_source_frame.ref_frame);
279  RefFrameState grav_source_state;
280 
281  // Compute state of the planet center wrt integ. frame origin.
282  body->inertial->compute_relative_state (integ_frame, grav_source_state);
283 
284  // Compute position of the vehicle CoM wrt the planet center.
285  Vector3::diff (
286  point_of_interest.state.trans.position, grav_source_state.trans.position,
287  rel_pos);
288 
289  // Compute contributions from non-spherical gravity if requested.
290  if (! spherical) {
291  calc_nonspherical (
292  point_of_interest.state.trans.position, rel_pos, grav_source_frame,
293  body_grav_accel, dgdx, pot);
294  }
295 
296  // Spherical only: Initialize gradient, accel, and potential to zero.
297  else {
298  Matrix3x3:: initialize (dgdx);
299  Vector3::initialize (body_grav_accel);
300  pot = 0.0;
301  }
302 
303  // Compute relativistic gravitational acceleration correction if requested.
304  // The relativistic contributions to gravity gradient and potential are
305  // not computed.
306  if (relativistic)
307  {
308  double rel_vel[3];
309  double relativistic_accel[3];
310  Vector3::diff (
311  point_of_interest.state.trans.velocity, grav_source_state.trans.velocity,
312  rel_vel);
313  calc_relativistic (
314  point_of_interest, rel_pos, rel_vel,
315  relativistic_accel);
316  Vector3::incr (relativistic_accel, body_grav_accel);
317  }
318 
319  // Compute spherical gravity contributions if needed.
320  if (! perturbing_only && ! skip_spherical) {
321 
322  // Compute state of integ. frame origin wrt the planet center.
323  Vector3::negate (grav_source_state.trans.position, grav_source_frame.pos);
324 
325  // Calculate spherical gravity, in the integration frame.
326  calc_spherical (
327  point_of_interest.state.trans.position, rel_pos, grav_source_frame,
328  body_grav_accel, dgdx, pot);
329  }

333 void
334 GravityControls::calc_spherical (
335  const double integ_pos[3],
336  const double posn[3],
337  const GravityIntegFrame & grav_source_frame,
338  double body_grav_accel[3],
339  double dgdx[3][3],
340  double& pot)
341 {
342  double mu_local = body->mu; // M3/s2 Locally cache mu for current grav body
343 
344  // Compute terms leading to spherical gravity acceleration.
345  double acc_local[3];
346  double r_sq = Vector3::vmagsq (posn);
347  double r_mag = std::sqrt(r_sq);
348  double r_3rd = r_sq * r_mag;
349 
350  // Compute 3rd body acceleration offset.
351  if (grav_source_frame.is_third_body) {
352  double rho_sq = Vector3::vmagsq (grav_source_frame.pos);
353  double rho_3rd = rho_sq * std::sqrt(rho_sq);
354  double r_dot_rho = Vector3::dot (integ_pos, grav_source_frame.pos);
355 
356  // Issue: Does the 3rd body effect also apply to potential?
357  // pot += mu_local * r_dot_rho / rho_3rd;
358 
359  // Battin's method: See writeup.
360  if (battin_method) {
361  double q = (Vector3::vmagsq(integ_pos) + 2.0 * r_dot_rho) / rho_sq;
362 
363  // Use Battin's method only where it yields better results.
364  // Note: updated for v4.0 based on additional experimental analysis.
365  if (q > -0.38) {
366  q = q * (3.0 + q * (3.0 + q));
367  q = q / (1.0 + std::sqrt(1.0 + q));
368  Vector3::scale (
369  -mu_local / (rho_3rd * (1.0 + q)),
370  Vector3::scale_decr (
371  grav_source_frame.pos, q,
372  Vector3::copy (integ_pos, acc_local)));
373  }
374 
375  // Use the original method where Battin's method yields poor results.
376  else {
377  Vector3::scale (posn, -mu_local/r_3rd, acc_local);
378  Vector3::scale_incr (
379  grav_source_frame.pos, mu_local/rho_3rd, acc_local);
380  }
381  }
382 
383  // Original method: a = -(mu/d^3) * \vec d + (mu/\rho^3) * \vec{\rho}.
384  else {
385  Vector3::scale (posn, -mu_local/r_3rd, acc_local);
386  Vector3::scale_incr (
387  grav_source_frame.pos, mu_local/rho_3rd, acc_local);
388  }
389  }
390 
391  // Non-third body: a = -mu/r^3 * \vec r.
392  else {
393  Vector3::scale (posn, -mu_local / r_3rd, acc_local);
394  }
395 
396  Vector3::incr (acc_local, body_grav_accel);
397 
398  // Compute inertial frame specific potential energy.
399  pot += mu_local / r_mag;
400 
401  // Compute spherical gravity gradient (Gottlieb p.23)
402  if (gradient) {
403  double r_5th = r_3rd * r_sq; // M^5
404  double mu_div_r5th = mu_local / r_5th;
405  dgdx[0][0] += mu_div_r5th * (3.0 * posn[0] * posn[0] - r_sq);
406  dgdx[0][1] += mu_div_r5th * 3.0 * posn[0] * posn[1];
407  dgdx[0][2] += mu_div_r5th * 3.0 * posn[0] * posn[2];
408  dgdx[1][1] += mu_div_r5th * (3.0 * posn[1] * posn[1] - r_sq);
409  dgdx[1][2] += mu_div_r5th * 3.0 * posn[1] * posn[2];
410  dgdx[2][2] += mu_div_r5th * (3.0 * posn[2] * posn[2] - r_sq);
411  dgdx[1][0] = dgdx[0][1];
412  dgdx[2][0] = dgdx[0][2];
413  dgdx[2][1] = dgdx[1][2];
414  }
415 }

418 /**
419  *
420  * \details
421  * Implements equation 27 (Folkner) / equation 4 (Genova) to compute a
422  * parameteric post-Newtonian correction to gravitation. The referenced equation
423  * is of the form
424  * @f{align*}{
425  * \begin{split}
426  * \boldsymbol a_{\text{A,pm-pm}} =&
427  * \phantom{+} \sum_{B\ne A}
428  * \frac{GM_B}{r^3_{AB}}(\boldsymbol r_B - \boldsymbol r_A)
429  * \bigl(1+\frac {s_1} {c^2}\bigr) \\
430  * & + \frac 1 {c^2} \sum_{B\ne A}
431  * \frac{GM_B}{r^3_{AB}}(\boldsymbol v_A - \boldsymbol v_B)
432  * \Bigl(
433  * \bigl( \boldsymbol r_A - \boldsymbol r_B \bigr) \cdot
434  * \bigl( (2+2\gamma)\boldsymbol v_A - (1+2\gamma)\boldsymbol v_B \bigr)
435  * \Bigr) \\
436  * & + \frac {3+4\gamma} {2c^2}
437  * \sum_{B\ne A} \frac{GM_B}{r_{AB}} \boldsymbol a_B
438  * \end{split}
439  * @f}
440  * where @f$ \boldsymbol a_B @f$ is the net acceleration of gravitating body *B*
441  * toward the other gravitating bodies (typically taken to be the Newtonian
442  * gravitational acceleration) and
443  * @f{equation*}{
444  * \begin{split}
445  * s_1 =
446  * & - 2(\beta+\gamma) \sum_{C\ne A}\frac{GM_C}{r_{AC}}
447  * - (2\beta-1) \sum_{C\ne A}\frac{GM_C}{r_{BC}} \\
448  * & + \gamma v_a^2
449  * + (1+\gamma) v_b^2
450  * - 2(1+\gamma)\boldsymbol v_A \cdot \boldsymbol v_B \\
451  * & - \frac32
452  * \left(
453  * \frac{(\boldsymbol r_A - \boldsymbol r_B) \cdot \boldsymbol v_B}{r_B}
454  * \right)^2 \\
455  * & + \frac12(\boldsymbol r_B - \boldsymbol r_A)\cdot \boldsymbol a_B
456  * \end{split}
457  * @f}
458  * In terms of the referenced equation, the intent of this function is to compute
459  * the relativistic portion of the acceleration of body *A* toward body *B*.
460  * With this, the expression this function computes is
461  * @f{equation*}{
462  * \begin{split}
463  * \boldsymbol \Delta a_{\text{A,B}} =
464  * \frac 1 c^2 \frac {GM_B}{r_{AB}} \Biggl\{ &
465  * \phantom{+} \frac{\boldsymbol r_B - \boldsymbol r_A}{r^2_{AB}} s_1 \\
466  * & + \frac{\boldsymbol v_A - \boldsymbol v_B}{r^2_{AB}}
467  * \Bigl(
468  * \bigl( \boldsymbol r_A - \boldsymbol r_B \bigr) \cdot
469  * \bigl( (2+2\gamma)\boldsymbol v_A - (1+2\gamma)\boldsymbol v_B \bigr)
470  * \Bigr) \\
471  * & + \frac {3+4\gamma}{2} \boldsymbol a_B
472  * \Biggr\}
473  * \end{split}
474  * @f}
475  * Note that the common factor @f$ \frac 1 c^2 \frac {GM_B}{r_{AB}} @f$
476  * is unitless and thus each of the three terms in the braced expression
477  * has units of acceleration.
478  *
479  * The referenced equations explicitly involve the parameterized post-Newtonian
480  * (PPN) factors @f$ \beta @f$ and @f$ \gamma @f$. This function hardcodes both
481  * of those factors as one, consistent with general relativity.
482  * For example, the factor @f$ (3+4\gamma)/2 @f$ becomes 3.5 in the code.
483  * Magic numbers that involve @f$ \beta @f$ or @f$ \gamma @f$, including a
484  * factor of one (e.g., @f$ 2 \beta - 1 @f$), are noted in comments.
485  *
486  * @par References:
487  * - William M. Folkner, et al., *Planetary and Lunar Ephemerides DE430 and
488  * DE431*, IPN Progress Report 42-196, 15 February 2014.\n
489  * ftp://naif.jpl.nasa.gov/pub/naif/generic_kernels/spk/planets/de430_and_de431.pdf
490  * - Antonio Genova, et al., *Solar system expansion and strong equivalence
491  * principle as seen by the NASA MESSENGER mission*,
492  * Nature Communications 9:289, 18 January 2018,
493  * DOI: 10.1038/s41467-017-02558-1.\n
494  * https://www.nature.com/articles/s41467-017-02558-1
495  */
496 void
497 GravityControls::calc_relativistic (
498  const RefFrame& point_of_interest,
499  const double rel_pos[3],
500  const double rel_vel[3],
501  double perturbing_accel[3])
502 {
503  typedef std::vector<GravitySource*>::const_iterator BodyIterator;
504  double rsq_AB = Vector3::vmagsq(rel_pos);
505  double r_AB_mag = std::sqrt(rsq_AB);
506 
507  double body_accel[3] = {0.0, 0.0, 0.0};
508  double body_potential = 0.0;
509  double point_potential = 0.0;
510 
511  // Compute
512  // - The gravitational acceleration and the specific gravitational potential
513  // energy at the location of the gravitational body associated with this
514  // control with respect to the the other bodies in the simulation, and
515  // - The specific gravitational potential energy at the point of interest
516  // with respect to the the other bodies in the simulation.
517  const RefFrame& body_inertial = *(body->inertial);
518  const std::vector<GravitySource*>& all_bodies = grav_manager->get_bodies();
519  for (BodyIterator iter = all_bodies.begin(), end = all_bodies.end();
520  iter != end;
521  ++iter)
522  {
523  const GravitySource* other_body = *iter;
524  const RefFrame& other_inertial = *(other_body->inertial);
525 
526  // Add the other body's contribution to the gravitational acceleration
527  // and potential at this body's location.
528  if (other_body != body) {
529  double r_CB[3];
530  other_inertial.compute_position_from (body_inertial, r_CB);
531  double r_CB_sq = Vector3::vmagsq(r_CB);
532  double other_potential = other_body->mu / std::sqrt(r_CB_sq);
533  Vector3::scale_incr (r_CB, other_potential/r_CB_sq, body_accel);
534  body_potential += other_potential;
535  }
536 
537  // Add the other body's contribution to the gravitational potential
538  // at the point of interest. Note: The point of interest might be
539  // that of a propagated planet, which needs to be excluded from the sum.
540  double r_CA[3];
541  other_inertial.compute_position_from (point_of_interest, r_CA);
542  double rsq_CA = Vector3::vmagsq(r_CA);
543  if (rsq_CA > 1.0) {
544  point_potential += other_body->mu / std::sqrt(rsq_CA);
545  }
546  }
547 
548  // The relativistic point mass interaction (equation 27, Folkner) involves
549  // three sums over the gravitating bodies. As noted above, this function
550  // calculates the contributions from gravitating body B, sans the Newtonian
551  // acceleration.
552  // yielding accelerations. The first is parallel to relative position,
553  // the second is parallel to relative velocity, and the third is parallel
554  // to the Newtonian acceleration of the gravitating body toward the other
555  // other gravitating bodies in the simulation.
556  double accel1[3];
557  double accel2[3];
558  double accel3[3];
559 
560  // Compute the contribution parallel to relative position.
561  double accel1_scale;
562 
563  // The first and second terms in the sum leading to accel1 are
564  // @f$ -2(\beta+\gamma) \sum_{C\ne A} \frac{\mu_C}{r_{AC}} @f$ and
565  // @f$ -(2\beta-1) \sum_{C\ne B} \frac{\mu_C}{r_{BC}} @f$.
566  // The scale factors @f$ 2(\beta+\gamma) @f$ and @f$ (2\beta-1) @f$
567  // are 4.0 and 1.0, respectively, when @f$ \beta=\gamma=1 @f$.
568  // Sans the scale factors, these have already been calculated
569  // as point_potential and body_potential, respectively.
570  accel1_scale = -(4.0*point_potential + body_potential);
571 
572  // The next set of terms in the sum leading to accel1 are
573  // @f$ \gamma*v_A^2 + (1+\gamma)*v_B^2 - 2*(1+\gamma)*v_A \cdot v_B @f$.
574  // These are calculated as @f$ (1+\gamma)||v_A-v_B||^2 - v_A^2 @f$.
575  accel1_scale +=
576  2.0*Vector3::vmagsq(rel_vel) // 1+gamma = 2
577  - Vector3::vmagsq(point_of_interest.state.trans.velocity);
578 
579  // The next term in the sum leading to accel1 is
580  // @f$ -3/2 (((r_A-r_B)\cdot v_B)/(||r_A-r_B||))^2
581  // We don't have v_B, but it is calculable from v_A and v_A-v_B.
582  double v_B[3];
583  Vector3::diff (point_of_interest.state.trans.velocity, rel_vel, v_B);
584  double rhat_dot_vb = Vector3::dot(rel_pos, v_B) / r_AB_mag;
585  accel1_scale -= 1.5*rhat_dot_vb*rhat_dot_vb;
586 
587  // The last term in the sum leading to accel1 is
588  // @f$ 1/2 (((r_B-r_A)\cdot a_B) @f$.
589  // We have @f$ r_A-r_B @F$ in rel_pos, hence the change in sign.
590  accel1_scale -= 0.5*Vector3::dot(rel_pos, body_accel);
591 
592  // Finally, calculate the contribution parallel to relative position.
593  // The reference equation uses
594  // @f$ \frac{\boldsymbol r_B - \boldsymbol r_A}{r^2_{AB}} s_1 @f$ .
595  // The @f$ s_1 @f$ in the above is accel1_scale. We once again
596  // have @f$ r_A-r_B @F$ in rel_pos, hence the change in sign.
597  Vector3::scale (rel_pos, -accel1_scale/rsq_AB, accel1);
598 
599  // Compute the contribution parallel to relative velocity.
600  double term2_vel[3];
601  Vector3::scale (rel_vel, 3.0, term2_vel);
602  Vector3::incr (point_of_interest.state.trans.velocity, term2_vel);
603  double term2_scale = Vector3::dot(rel_pos, term2_vel) / rsq_AB;
604  Vector3::scale (rel_vel, term2_scale, accel2);
605 
606  // Compute the contribution parallel to the acceleration of the grav body.
607  Vector3::scale (body_accel, 3.5, accel3);
608 
609  Vector3::sum (accel1, accel2, accel3, perturbing_accel);
610  Vector3::scale (body->mu / r_AB_mag / speed_of_light_sq, perturbing_accel);
611 }
