187 void GravityControls::gravitation(const double integ_pos[3],

190  double dgdx[3][3],
191  double Pot[1])
192 {

224 /**
225  * Compute the gravitation at a given position toward a gravity body.
226  * \param[in] point_of_interest Point of interest, as a reference frame.
227  * \param[in] integ_frame_idx Integ frame index.
228  * \param[out] body_grav_accel Accel for given grav body.
229  * \param[out] dgdx Gradient for given grav body.
230  * \param[out] pot Specific gravitational potential for given grev body.
231  */
232 void GravityControls::gravitation(const RefFrame & point_of_interest,
233  unsigned int integ_frame_idx,
234  double body_grav_accel[3],
235  double dgdx[3][3],
236  double & pot)
237 {
238  double rel_pos[3]; // M Vehicle inertial position wrt planet
239  GravityIntegFrame & grav_source_frame = body->frames[integ_frame_idx]; // Grav frame for this integ frame
240  const RefFrame & integ_frame = *(grav_source_frame.ref_frame); // Integration frame
241  RefFrameState grav_source_state;
242 
243  // Compute state of the planet center wrt integ. frame origin.
244  body->inertial->compute_relative_state(integ_frame, grav_source_state);
245 
246  // Compute position of the vehicle CoM wrt the planet center.
247  Vector3::diff(point_of_interest.state.trans.position, grav_source_state.trans.position, rel_pos);
248 
249  // Compute contributions from non-spherical gravity if requested.
250  if(!spherical)
251  {
252  calc_nonspherical(point_of_interest.state.trans.position,
253  rel_pos,
254  grav_source_frame,
255  body_grav_accel,
256  dgdx,
257  pot);
258  }
259 
260  // Spherical only: Initialize gradient, accel, and potential to zero.
261  else
262  {
263  Matrix3x3::initialize(dgdx);
264  Vector3::initialize(body_grav_accel);
265  pot = 0.0;
266  }
267 
268  // Compute relativistic gravitational acceleration correction if requested.
269  // The relativistic contributions to gravity gradient and potential are
270  // not computed.
271  if(relativistic)
272  {
273  double rel_vel[3];
274  double relativistic_accel[3];
275  Vector3::diff(point_of_interest.state.trans.velocity, grav_source_state.trans.velocity, rel_vel);
276  calc_relativistic(point_of_interest, rel_pos, rel_vel, relativistic_accel);
277  Vector3::incr(relativistic_accel, body_grav_accel);
278  }
279 
280  // Compute spherical gravity contributions if needed.
281  if(!perturbing_only && !skip_spherical)
282  {
283  // Compute state of integ. frame origin wrt the planet center.
284  Vector3::negate(grav_source_state.trans.position, grav_source_frame.pos);
285 
286  // Calculate spherical gravity, in the integration frame.
287  calc_spherical(point_of_interest.state.trans.position, rel_pos, grav_source_frame, body_grav_accel, dgdx, pot);
288  }

291 void GravityControls::calc_spherical(const double integ_pos[3],
292  const double posn[3],
293  const GravityIntegFrame & grav_source_frame,
294  double body_grav_accel[3],
295  double dgdx[3][3],
296  double & pot)
297 {
298  double mu_local = body->mu; // M3/s2 Locally cache mu for current grav body
299 
300  // Compute terms leading to spherical gravity acceleration.
301  double acc_local[3];
302  double r_sq = Vector3::vmagsq(posn);
303  double r_mag = std::sqrt(r_sq);
304  double r_3rd = r_sq * r_mag;
305 
306  // Compute 3rd body acceleration offset.
307  if(grav_source_frame.is_third_body)
308  {
309  double rho_sq = Vector3::vmagsq(grav_source_frame.pos);
310  double rho_3rd = rho_sq * std::sqrt(rho_sq);
311  double r_dot_rho = Vector3::dot(integ_pos, grav_source_frame.pos);
312 
313  // Issue: Does the 3rd body effect also apply to potential?
314  // pot += mu_local * r_dot_rho / rho_3rd;
315 
316  // Battin's method: See writeup.
317  if(battin_method)
318  {
319  // Note that grav_source_frame.pos is used in place of the rho listed in the documentation.
320  // This is noteworthy because grav_source_frame.pos is directionally opposite of rho.
321  // Therefor the sign in this calculation (+) is opposite of the one found in the documentation (-).
322  double q = (Vector3::vmagsq(integ_pos) + 2.0 * r_dot_rho) / rho_sq;
323 
324  // Use Battin's method only where it yields better results.
325  // Note: updated for v4.0 based on additional experimental analysis.
326  if(q > -0.38)
327  {
328  q = q * (3.0 + q * (3.0 + q));
329  q = q / (1.0 + std::sqrt(1.0 + q));
330  Vector3::scale(-mu_local / (rho_3rd * (1.0 + q)),
331  Vector3::scale_decr(grav_source_frame.pos, q, Vector3::copy(integ_pos, acc_local)));
332  }
333 
334  // Use the original method where Battin's method yields poor results.
335  else
336  {
337  Vector3::scale(posn, -mu_local / r_3rd, acc_local);
338  Vector3::scale_incr(grav_source_frame.pos, mu_local / rho_3rd, acc_local);
339  }
340  }
341 
342  // Original method: a = -(mu/d^3) * \vec d + (mu/\rho^3) * \vec{\rho}.
343  else
344  {
345  Vector3::scale(posn, -mu_local / r_3rd, acc_local);
346  Vector3::scale_incr(grav_source_frame.pos, mu_local / rho_3rd, acc_local);
347  }
348  }
349 
350  // Non-third body: a = -mu/r^3 * \vec r.
351  else
352  {
353  Vector3::scale(posn, -mu_local / r_3rd, acc_local);
354  }
355 
356  Vector3::incr(acc_local, body_grav_accel);
357 
358  // Compute inertial frame specific potential energy.
359  pot += mu_local / r_mag;
360 
361  // Compute spherical gravity gradient (Gottlieb p.23)
362  if(gradient)
363  {
364  double r_5th = r_3rd * r_sq; // M^5
365  double mu_div_r5th = mu_local / r_5th;
366  dgdx[0][0] += mu_div_r5th * (3.0 * posn[0] * posn[0] - r_sq);
367  dgdx[0][1] += mu_div_r5th * 3.0 * posn[0] * posn[1];
368  dgdx[0][2] += mu_div_r5th * 3.0 * posn[0] * posn[2];
369  dgdx[1][1] += mu_div_r5th * (3.0 * posn[1] * posn[1] - r_sq);
370  dgdx[1][2] += mu_div_r5th * 3.0 * posn[1] * posn[2];
371  dgdx[2][2] += mu_div_r5th * (3.0 * posn[2] * posn[2] - r_sq);
372  dgdx[1][0] = dgdx[0][1];
373  dgdx[2][0] = dgdx[0][2];
374  dgdx[2][1] = dgdx[1][2];
375  }

378 /**
379  *
380  * \details
381  * Implements equation 27 (Folkner) / equation 4 (Genova) to compute a
382  * parameteric post-Newtonian correction to gravitation. The referenced equation
383  * is of the form
384  * @f{align*}{
385  * \begin{split}
386  * \boldsymbol a_{\text{A,pm-pm}} =&
387  * \phantom{+} \sum_{B\ne A}
388  * \frac{GM_B}{r^3_{AB}}(\boldsymbol r_B - \boldsymbol r_A)
389  * \bigl(1+\frac {s_1} {c^2}\bigr) \\
390  * & + \frac 1 {c^2} \sum_{B\ne A}
391  * \frac{GM_B}{r^3_{AB}}(\boldsymbol v_A - \boldsymbol v_B)
392  * \Bigl(
393  * \bigl( \boldsymbol r_A - \boldsymbol r_B \bigr) \cdot
394  * \bigl( (2+2\gamma)\boldsymbol v_A - (1+2\gamma)\boldsymbol v_B \bigr)
395  * \Bigr) \\
396  * & + \frac {3+4\gamma} {2c^2}
397  * \sum_{B\ne A} \frac{GM_B}{r_{AB}} \boldsymbol a_B
398  * \end{split}
399  * @f}
400  * where @f$ \boldsymbol a_B @f$ is the net acceleration of gravitating body *B*
401  * toward the other gravitating bodies (typically taken to be the Newtonian
402  * gravitational acceleration) and
403  * @f{equation*}{
404  * \begin{split}
405  * s_1 =
406  * & - 2(\beta+\gamma) \sum_{C\ne A}\frac{GM_C}{r_{AC}}
407  * - (2\beta-1) \sum_{C\ne A}\frac{GM_C}{r_{BC}} \\
408  * & + \gamma v_a^2
409  * + (1+\gamma) v_b^2
410  * - 2(1+\gamma)\boldsymbol v_A \cdot \boldsymbol v_B \\
411  * & - \frac32
412  * \left(
413  * \frac{(\boldsymbol r_A - \boldsymbol r_B) \cdot \boldsymbol v_B}{r_B}
414  * \right)^2 \\
415  * & + \frac12(\boldsymbol r_B - \boldsymbol r_A)\cdot \boldsymbol a_B
416  * \end{split}
417  * @f}
418  * In terms of the referenced equation, the intent of this function is to compute
419  * the relativistic portion of the acceleration of body *A* toward body *B*.
420  * With this, the expression this function computes is
421  * @f{equation*}{
422  * \begin{split}
423  * \boldsymbol \Delta a_{\text{A,B}} =
424  * \frac 1 c^2 \frac {GM_B}{r_{AB}} \Biggl\{ &
425  * \phantom{+} \frac{\boldsymbol r_B - \boldsymbol r_A}{r^2_{AB}} s_1 \\
426  * & + \frac{\boldsymbol v_A - \boldsymbol v_B}{r^2_{AB}}
427  * \Bigl(
428  * \bigl( \boldsymbol r_A - \boldsymbol r_B \bigr) \cdot
429  * \bigl( (2+2\gamma)\boldsymbol v_A - (1+2\gamma)\boldsymbol v_B \bigr)
430  * \Bigr) \\
431  * & + \frac {3+4\gamma}{2} \boldsymbol a_B
432  * \Biggr\}
433  * \end{split}
434  * @f}
435  * Note that the common factor @f$ \frac 1 c^2 \frac {GM_B}{r_{AB}} @f$
436  * is unitless and thus each of the three terms in the braced expression
437  * has units of acceleration.
438  *
439  * The referenced equations explicitly involve the parameterized post-Newtonian
440  * (PPN) factors @f$ \beta @f$ and @f$ \gamma @f$. This function hardcodes both
441  * of those factors as one, consistent with general relativity.
442  * For example, the factor @f$ (3+4\gamma)/2 @f$ becomes 3.5 in the code.
443  * Magic numbers that involve @f$ \beta @f$ or @f$ \gamma @f$, including a
444  * factor of one (e.g., @f$ 2 \beta - 1 @f$), are noted in comments.
445  *
446  * @par References:
447  * - William M. Folkner, et al., *Planetary and Lunar Ephemerides DE430 and
448  * DE431*, IPN Progress Report 42-196, 15 February 2014.\n
449  * ftp://naif.jpl.nasa.gov/pub/naif/generic_kernels/spk/planets/de430_and_de431.pdf
450  * - Antonio Genova, et al., *Solar system expansion and strong equivalence
451  * principle as seen by the NASA MESSENGER mission*,
452  * Nature Communications 9:289, 18 January 2018,
453  * DOI: 10.1038/s41467-017-02558-1.\n
454  * https://www.nature.com/articles/s41467-017-02558-1
455  */
456 void GravityControls::calc_relativistic(const RefFrame & point_of_interest,
457  const double rel_pos[3],
458  const double rel_vel[3],
459  double perturbing_accel[3])
460 {
461  double rsq_AB = Vector3::vmagsq(rel_pos);
462  double r_AB_mag = std::sqrt(rsq_AB);
463 
464  double body_accel[3] = {0.0, 0.0, 0.0};
465  double body_potential = 0.0;
466  double point_potential = 0.0;
467 
468  // Compute
469  // - The gravitational acceleration and the specific gravitational potential
470  // energy at the location of the gravitational body associated with this
471  // control with respect to the the other bodies in the simulation, and
472  // - The specific gravitational potential energy at the point of interest
473  // with respect to the the other bodies in the simulation.
474  const RefFrame & body_inertial = *(body->inertial);
475  const std::vector<GravitySource *> & all_bodies = grav_manager->get_bodies();
476  for(auto other_body : all_bodies)
477  {
478  const RefFrame & other_inertial = *(other_body->inertial);
479 
480  // Add the other body's contribution to the gravitational acceleration
481  // and potential at this body's location.
482  if(other_body != body)
483  {
484  double r_CB[3];
485  other_inertial.compute_position_from(body_inertial, r_CB);
486  double r_CB_sq = Vector3::vmagsq(r_CB);
487  double other_potential = other_body->mu / std::sqrt(r_CB_sq);
488  Vector3::scale_incr(r_CB, other_potential / r_CB_sq, body_accel);
489  body_potential += other_potential;
490  }
491 
492  // Add the other body's contribution to the gravitational potential
493  // at the point of interest. Note: The point of interest might be
494  // that of a propagated planet, which needs to be excluded from the sum.
495  double r_CA[3];
496  other_inertial.compute_position_from(point_of_interest, r_CA);
497  double rsq_CA = Vector3::vmagsq(r_CA);
498  if(rsq_CA > 1.0)
499  {
500  point_potential += other_body->mu / std::sqrt(rsq_CA);
501  }
502  }
503 
504  // The relativistic point mass interaction (equation 27, Folkner) involves
505  // three sums over the gravitating bodies. As noted above, this function
506  // calculates the contributions from gravitating body B, sans the Newtonian
507  // acceleration.
508  // yielding accelerations. The first is parallel to relative position,
509  // the second is parallel to relative velocity, and the third is parallel
510  // to the Newtonian acceleration of the gravitating body toward the other
511  // other gravitating bodies in the simulation.
512  double accel1[3];
513  double accel2[3];
514  double accel3[3];
515 
516  // Compute the contribution parallel to relative position.
517  double accel1_scale;
518 
519  // The first and second terms in the sum leading to accel1 are
520  // @f$ -2(\beta+\gamma) \sum_{C\ne A} \frac{\mu_C}{r_{AC}} @f$ and
521  // @f$ -(2\beta-1) \sum_{C\ne B} \frac{\mu_C}{r_{BC}} @f$.
522  // The scale factors @f$ 2(\beta+\gamma) @f$ and @f$ (2\beta-1) @f$
523  // are 4.0 and 1.0, respectively, when @f$ \beta=\gamma=1 @f$.
524  // Sans the scale factors, these have already been calculated
525  // as point_potential and body_potential, respectively.
526  accel1_scale = -(4.0 * point_potential + body_potential);
527 
528  // The next set of terms in the sum leading to accel1 are
529  // @f$ \gamma*v_A^2 + (1+\gamma)*v_B^2 - 2*(1+\gamma)*v_A \cdot v_B @f$.
530  // These are calculated as @f$ (1+\gamma)||v_A-v_B||^2 - v_A^2 @f$.
531  accel1_scale += 2.0 * Vector3::vmagsq(rel_vel) // 1+gamma = 2
532  - Vector3::vmagsq(point_of_interest.state.trans.velocity);
533 
534  // The next term in the sum leading to accel1 is
535  // @f$ -3/2 (((r_A-r_B)\cdot v_B)/(||r_A-r_B||))^2
536  // We don't have v_B, but it is calculable from v_A and v_A-v_B.
537  double v_B[3];
538  Vector3::diff(point_of_interest.state.trans.velocity, rel_vel, v_B);
539  double rhat_dot_vb = Vector3::dot(rel_pos, v_B) / r_AB_mag;
540  accel1_scale -= 1.5 * rhat_dot_vb * rhat_dot_vb;
541 
542  // The last term in the sum leading to accel1 is
543  // @f$ 1/2 (((r_B-r_A)\cdot a_B) @f$.
544  // We have @f$ r_A-r_B @F$ in rel_pos, hence the change in sign.
545  accel1_scale -= 0.5 * Vector3::dot(rel_pos, body_accel);
546 
547  // Finally, calculate the contribution parallel to relative position.
548  // The reference equation uses
549  // @f$ \frac{\boldsymbol r_B - \boldsymbol r_A}{r^2_{AB}} s_1 @f$ .
550  // The @f$ s_1 @f$ in the above is accel1_scale. We once again
551  // have @f$ r_A-r_B @F$ in rel_pos, hence the change in sign.
552  Vector3::scale(rel_pos, -accel1_scale / rsq_AB, accel1);
553 
554  // Compute the contribution parallel to relative velocity.
555  double term2_vel[3];
556  Vector3::scale(rel_vel, 3.0, term2_vel);
557  Vector3::incr(point_of_interest.state.trans.velocity, term2_vel);
558  double term2_scale = Vector3::dot(rel_pos, term2_vel) / rsq_AB;
559  Vector3::scale(rel_vel, term2_scale, accel2);
560 
561  // Compute the contribution parallel to the acceleration of the grav body.
562  Vector3::scale(body_accel, 3.5, accel3);
563
