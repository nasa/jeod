120 int OrbitalElements::from_cartesian(double mu, const double pos[3], const double vel[3])
121 {
122  // Declare local variables
123  int i; // Loop counter
124  double ang_momntm[3]; // Angular momentum vector
125  double line_of_nodes[3]; // Line-of-nodes vector
126  double pos_dot_vel; // Dot product of the position and velocity
127  double eccentricity[3]; // Orbit eccentricity vector
128  double K[3]; // Unit vector in K direction of inertial frame
129  double I[3]; // Unit vector in I direction of inertial frame
130  double v2; // Magnitude of the velocity squared
131  double tolerance; // Tolerance for e and i
132  double orbit_switch_tol; // Tolerance for orbit type switch
133  double cross_vec[3]; // cross product used for angle calculations
134  double sin_in; // sine value for input to atan2
135  double cos_in; // cosine value for input to atan2
136 
137  // Set tolerance values
138  tolerance = 1.0e-13;
139  orbit_switch_tol = 1.0e-2;
140 
141  // Define unit vector in K direction of inertial frame
142  K[0] = 0.0;
143  K[1] = 0.0;
144  K[2] = 1.0;
145 
146  /* NOTE: In December of 2005, the code was changed to calculate
147  angles using atan2 instead of acos for robustness. Some new
148  variables were added and some unecessary ones were removed.
149  The changes to the code are denoted by SWP 2005 */
150 
151  // Define unit vector in I direction of inertial frame (SWP 2005)
152  I[0] = 1.0;
153  I[1] = 0.0;
154  I[2] = 0.0;
155 
156  // Compute the magnitude of position and velocity
157  r_mag = Vector3::vmag(pos);
158  vel_mag = Vector3::vmag(vel);
159 
160  // Compute the angular momentum vector and magnitude
161  Vector3::cross(pos, vel, ang_momntm);
162  orb_ang_momentum = Vector3::vmag(ang_momntm);
163 
164  // Compute the line of nodes vector
165  Vector3::cross(K, ang_momntm, line_of_nodes);
166 
167  // Compute the eccentricity vector and magnitude
168  pos_dot_vel = Vector3::dot(pos, vel);
169  v2 = vel_mag * vel_mag;
170  for(i = 0; i < 3; i++)
171  {
172  eccentricity[i] = (v2 - mu / r_mag) * pos[i] / mu - (pos_dot_vel * vel[i] / mu);
173  }
174  e_mag = Vector3::vmag(eccentricity);
175 
176  // Compute the specific mechanical energy
177  orb_energy = (v2 / 2.0) - (mu / r_mag);
178 
179  // Compute semi-major-axis, semiparameter, and mean motion
180  if(e_mag < tolerance)
181  { // Circular orbit
182 
183  semi_major_axis = r_mag;
184  semiparam = r_mag;
185  mean_motion = sqrt(mu / semi_major_axis) / semi_major_axis;
186  }
187  else if(e_mag < (1.0 - orbit_switch_tol))
188  { // Eccentric orbit
189 
190  semi_major_axis = -mu / (2.0 * orb_energy);
191  semiparam = semi_major_axis * (1.0 - e_mag * e_mag);
192  mean_motion = sqrt(mu / semi_major_axis) / semi_major_axis;
193  }
194  else if(e_mag > (1.0 + orbit_switch_tol))
195  { // Hyperbolic orbit
196 
197  semi_major_axis = -mu / (2.0 * orb_energy);
198  semiparam = semi_major_axis * (1.0 - e_mag * e_mag);
199  mean_motion = sqrt(mu / -semi_major_axis) / -semi_major_axis;
200  }
201  else
202  { // Parabolic orbit
203 
204  // Null out semi-major axis since it is actually infinity
205  semi_major_axis = 0.0;
206 
207  semiparam = orb_ang_momentum * orb_ang_momentum / mu;
208  mean_motion = 2.0 * sqrt(mu / semiparam) / semiparam;
209  }
210 
211  // Compute the orbital inclination (SWP 2005)
212  Vector3::cross(K, ang_momntm, cross_vec);
213  sin_in = Vector3::vmag(cross_vec);
214  cos_in = Vector3::dot(K, ang_momntm);
215  inclination = atan2(sin_in, cos_in);
216 
217  // Compute longitude of ascending node, argument of periapsis, & true anomaly
218  if((inclination < tolerance) || ((M_PI - tolerance) < inclination))
219  { // Equatorial orbit
220 
221  if(e_mag < tolerance)
222  { // Equatorial and circular orbit
223  long_asc_node = 0.0;
224  arg_periapsis = 0.0;
225 
226  // SWP 2005
227  Vector3::cross(I, pos, cross_vec);
228  sin_in = Vector3::vmag(cross_vec);
229  cos_in = Vector3::dot(I, pos);
230  true_anom = atan2(sin_in, cos_in);
231 
232  if(inclination < tolerance)
233  {
234  if(pos[1] < 0.0)
235  {
236  true_anom = 2.0 * M_PI - true_anom;
237  }
238  }
239  else
240  {
241  if(pos[1] > 0.0)
242  {
243  true_anom = 2.0 * M_PI - true_anom;
244  }
245  }
246  }
247 
248  else
249  { // Equatorial and non-circular orbit
250  long_asc_node = 0.0;
251 
252  // SWP 2005
253  Vector3::cross(I, eccentricity, cross_vec);
254  sin_in = Vector3::vmag(cross_vec);
255  cos_in = Vector3::dot(I, eccentricity);
256  arg_periapsis = atan2(sin_in, cos_in);
257 
258  if(inclination < tolerance)
259  {
260  if(eccentricity[1] < 0.0)
261  {
262  arg_periapsis = 2.0 * M_PI - arg_periapsis;
263  }
264  }
265  else
266  {
267  if(eccentricity[1] > 0.0)
268  {
269  arg_periapsis = 2.0 * M_PI - arg_periapsis;
270  }
271  }
272 
273  // SWP 2005
274  Vector3::cross(eccentricity, pos, cross_vec);
275  sin_in = Vector3::vmag(cross_vec);
276  cos_in = Vector3::dot(eccentricity, pos);
277  true_anom = atan2(sin_in, cos_in);
278 
279  if(pos_dot_vel < 0.0)
280  {
281  true_anom = 2.0 * M_PI - true_anom;
282  }
283  }
284  }
285 
286  else if(e_mag < tolerance)
287  { // Circular and non-equatorial orbit
288 
289  // SWP 2005
290  Vector3::cross(I, line_of_nodes, cross_vec);
291  sin_in = Vector3::vmag(cross_vec);
292  cos_in = Vector3::dot(I, line_of_nodes);
293  long_asc_node = atan2(sin_in, cos_in);
294 
295  if(line_of_nodes[1] < 0.0)
296  {
297  long_asc_node = 2.0 * M_PI - long_asc_node;
298  }
299 
300  arg_periapsis = 0.0;
301 
302  // SWP 2005
303  Vector3::cross(line_of_nodes, pos, cross_vec);
304  sin_in = Vector3::vmag(cross_vec);
305  cos_in = Vector3::dot(line_of_nodes, pos);
306  true_anom = atan2(sin_in, cos_in);
307 
308  if(pos[2] < 0.0)
309  {
310  true_anom = 2.0 * M_PI - true_anom;
311  }
312  }
313 
314  else
315  { // Non-circular and non-equatorial orbit
316 
317  // SWP 2005
318  Vector3::cross(I, line_of_nodes, cross_vec);
319  sin_in = Vector3::vmag(cross_vec);
320  cos_in = Vector3::dot(I, line_of_nodes);
321  long_asc_node = atan2(sin_in, cos_in);
322 
323  if(line_of_nodes[1] < 0.0)
324  {
325  long_asc_node = 2.0 * M_PI - long_asc_node;
326  }
327 
328  // SWP 2005
329  Vector3::cross(line_of_nodes, eccentricity, cross_vec);
330  sin_in = Vector3::vmag(cross_vec);
331  cos_in = Vector3::dot(line_of_nodes, eccentricity);
332  arg_periapsis = atan2(sin_in, cos_in);
333 
334  if(eccentricity[2] < 0.0)
335  {
336  arg_periapsis = 2.0 * M_PI - arg_periapsis;
337  }
338 
339  // SWP 2005
340  Vector3::cross(eccentricity, pos, cross_vec);
341  sin_in = Vector3::vmag(cross_vec);
342  cos_in = Vector3::dot(eccentricity, pos);
343  true_anom = atan2(sin_in, cos_in);
344 
345  if(pos_dot_vel < 0.0)
346  {
347  true_anom = 2.0 * M_PI - true_anom;
348  }
349  }
350 
351  // Compute the mean anomaly & the eccentric, hyperbolic, or parabolic anomaly
352  nu_to_anomalies();
353 
354  return (0);
355 }

357 /******************************************************************************
358 PURPOSE:
359  (Calculation of Inertial Cartesian Coordinates from Orbital Elements
360  semiparameter -p-, eccentricity -e-, inclination -i-, longitude of the
361  ascending node -Omega-, argument of periapsis -w-, mean anomaly -M-,
362  and gravitational parameter -mu-.)
363 
364 REFERENCE:
365  (((Vallado, D.) (Fundamentals of Astrodynamics and Applications, 2nd Ed.)
366  (Kluwer Acedemic Publishers) (2001) (Pages 135-230) (ISBN:0-07-066834-5))
367  ((Montenbruck, O. and Gill, E.) (Satellite Orbits)
368  (Springer) (2000) (Pages 157-190) (ISBN:3 540 67280))
369  ((Pedro Ramon Escobal) (Methods of Orbital Determination)
370  (Krieger Publishing) (1976) (pages 393-422) (ISBN:0-88275-319-3)))
371 
372 ASSUMPTIONS AND LIMITATIONS:
373  ((This routine is intended to be used at initialization only.)
374  (Output cartesian coordinates are in an inertial frame.)
375  (This routine works for all types of orbits.))
376 
377 CLASS:
378  (initialization)
379 
380 
381 ******************************************************************************/
382 int OrbitalElements::to_cartesian(double mu, double pos[3], double vel[3])
383 {
384  // Declare local variables
385  double tolerance = 1.0e-13; // Tolerance for e and i
386  double p; // Semiparameter
387  double e; // Magnitude of eccentricity
388  double r_pqw[3]; // Position vector in the PQW coordinate frame
389  double v_pqw[3]; // Velocity vector in the PQW coordinate frame
390  double T_PQW_Inrtl[3][3]; // Transformation matrix from PQW to Inertial
391  double sO; // Sine of the longitude of ascending node
392  double cO; // Cosine of the longitude of ascending node
393  double sw; // Sine of the argument of periapsis
394  double cw; // Cosine of the argument of periapsis
395  double si; // Sine of the inclination
396  double ci; // Cosine of the inclination
397 
398  // Assign input values for clarity
399  p = semiparam;
400  e = e_mag;
401 
402  // Check for "meaningful" value of semi-parameter
403  if(p <= 0.0)
404  {
405  MessageHandler::fail(__FILE__,
406  __LINE__,
407  OrbitalElementsMessages::domain_error,
408  "Semi-parameter %g is zero or negative.",
409  p);
410  return (-1);
411  }
412 
413  // Check for "meaningful" values of sin_v and cos_v
414  double rss_sincos_v = sqrt(sin_v * sin_v + cos_v * cos_v);
415  if(fabs(rss_sincos_v - 1.0) > tolerance)
416  {
417  MessageHandler::fail(__FILE__,
418  __LINE__,
419  OrbitalElementsMessages::domain_error,
420  "Sine (%g) and cosine (%g) of the true anomaly are inconsistent.",
421  sin_v,
422  cos_v);
423  return (-1);
424  }
425 
426  // Check for meaningful value of central mass
427  if(mu <= 0.0)
428  {
429  MessageHandler::fail(__FILE__,
430  __LINE__,
431  OrbitalElementsMessages::domain_error,
432  "Grav mass constant (mu) is zero or negative (%f).\n"
433  "This is likely a user-induced initialization error.\n",
434  mu);
435  return (-1);
436  }
437 
438  // Test for equatorial and circular orbits
439  if((inclination < tolerance) || ((M_PI - tolerance) < inclination))
440  { // Equatorial orbit
441 
442  if(e < tolerance)
443  { // Equatorial and Circular orbit
444 
445  sO = 0.0;
446  cO = 1.0;
447  sw = 0.0;
448  cw = 1.0;
449  }
450  else
451  { // Equatorial and non-circular orbit
452 
453  sO = 0.0;
454  cO = 1.0;
455  sw = sin(arg_periapsis);
456  cw = cos(arg_periapsis);
457  }
458 
459  if(inclination < tolerance)
460  { // Inclination is approx. zero
461 
462  si = 0.0;
463  ci = 1.0;
464  }
465  else
466  { // Inclination is approx. pi
467 
468  si = 0.0;
469  ci = -1.0;
470  }
471  }
472 
473  else if(e < tolerance)
474  { // Circular and non-equatorial orbit
475 
476  sO = sin(long_asc_node);
477  cO = cos(long_asc_node);
478  sw = 0.0;
479  cw = 1.0;
480  si = sin(inclination);
481  ci = cos(inclination);
482  }
483 
484  else
485  { // Non-circular and non-equatorial orbit
486 
487  sO = sin(long_asc_node);
488  cO = cos(long_asc_node);
489  sw = sin(arg_periapsis);
490  cw = cos(arg_periapsis);
491  si = sin(inclination);
492  ci = cos(inclination);
493  }
494 
495  // Compute position and velocity in perifocal frame (PQW)
496  r_pqw[0] = (p * cos_v) / (1.0 + e * cos_v);
497  r_pqw[1] = (p * sin_v) / (1.0 + e * cos_v);
498  r_pqw[2] = 0.0;
499 
500  v_pqw[0] = -sqrt(mu / p) * sin_v;
501  v_pqw[1] = sqrt(mu / p) * (e + cos_v);
502  v_pqw[2] = 0.0;
503 
504  // Compute transformation matrix from PQW to inertial
505  T_PQW_Inrtl[0][0] = cO * cw - sO * sw * ci;
506  T_PQW_Inrtl[0][1] = -cO * sw - sO * cw * ci;
507  T_PQW_Inrtl[0][2] = sO * si;
508  T_PQW_Inrtl[1][0] = sO * cw + cO * sw * ci;
509  T_PQW_Inrtl[1][1] = -sO * sw + cO * cw * ci;
510  T_PQW_Inrtl[1][2] = -cO * si;
511  T_PQW_Inrtl[2][0] = sw * si;
512  T_PQW_Inrtl[2][1] = cw * si;
513  T_PQW_Inrtl[2][2] = ci;
514 
515  // Compute position and velocity in inertial frame
516  Vector3::transform(T_PQW_Inrtl, r_pqw, pos);
517  Vector3::transform(T_PQW_Inrtl, v_pqw, vel);
518 
519  return (0);
520 }

522 /******************************************************************************
523 PURPOSE:
524  (Calculation of the Mean Anomaly and the Eccentric, Hyperbolic, or
525  Parabolic Anomalies, given the True Anomaly and eccentricity)
526 
527 REFERENCE:
528  (((Vallado, D.) (Fundamentals of Astrodynamics and Applications, 2nd Ed.)
529  (Kluwer Acedemic Publishers) (2001) (Pages 135-230) (ISBN:0-07-066834-5))
530  ((Montenbruck, O. and Gill, E.) (Satellite Orbits)
531  (Springer) (2000) (Pages 157-190) (ISBN:3 540 67280))
532  ((Pedro Ramon Escobal) (Methods of Orbital Determination)
533  (Krieger Publishing) (1976) (pages 393-422) (ISBN:0-88275-319-3)))
534 
535 ASSUMPTIONS AND LIMITATIONS:
536  ((This routine works for all types of orbits.))
537 
538 CLASS:
539  (N/A)
540 
541 
542 ******************************************************************************/
543 int OrbitalElements::nu_to_anomalies()
544 {
545  // Declare local variables
546  double cir_tol = 1.0e-13; // Tolerance for circular orbit
547  double switch_tol = 1.0e-2; // Tolerance for use of ecc, par, hyp anom
548  double v; // True Anomaly
549  double e; // Magnitude of eccentricity
550  double sin_E; // Sine of the Eccentric Anomaly
551  double cos_E; // Cosine of the Eccentric Anomaly
552  double sinh_H; // Hyperbolic sine of the Hyperbolic Anomaly
553 
554  // Assign input parameters for clarity
555  v = true_anom;
556  e = e_mag;
557  sin_v = sin(true_anom);
558  cos_v = cos(true_anom);
559 
560  // Circular orbit
561  if(e < cir_tol)
562  {
563  orbital_anom = v;
564  mean_anom = v;
565  }
566 
567  // Eccentric orbit
568  else if(e < (1.0 - switch_tol))
569  {
570  sin_E = sqrt(1.0 - e * e) * sin_v / (1.0 + e * cos_v);
571  cos_E = (e + cos_v) / (1.0 + e * cos_v);
572  orbital_anom = atan2(sin_E, cos_E);
573 
574  if(orbital_anom < 0.0)
575  {
576  orbital_anom += 2.0 * M_PI;
577  }
578 
579  mean_anom = orbital_anom - e * sin(orbital_anom);
580  }
581 
582  // Hyperbolic orbit
583  else if(e > (1.0 + switch_tol))
584  {
585  sinh_H = sqrt(e * e - 1.0) * sin_v / (1.0 + e * cos_v);
586  orbital_anom = asinh(sinh_H);
587  mean_anom = e * sinh(orbital_anom) - orbital_anom;
588  }
589 
590  // Parabolic orbit
591  else
592  {
593  orbital_anom = tan(v / 2.0);
594  mean_anom = orbital_anom + pow(orbital_anom, 3.0) / 3.0;
595  }
596 
597  return (0);
598 }

600 /******************************************************************************
601 PURPOSE:
602  (Calculation of the True Anomaly from the Mean Anomaly and Eccentricity.)
603 
604 REFERENCE:
605  (((Vallado, David A.) (Fundamentals of Astrodynamics and Applications)
606  (McGraw-Hill) (New York) (1997)))
607 
608 ASSUMPTIONS AND LIMITATIONS:
609  ((This routine works for all types of orbits.))
610 
611 CLASS:
612  (N/A)
613 
614 
615 ******************************************************************************/
616 int OrbitalElements::mean_anom_to_nu()
617 {
618  // Declare local variables
619  int Converge; // Convergence flag for iteration
620  double cir_tol = 1.0e-13; // Tolerance for circular orbit
621  double switch_tol = 1.0e-2; // Tolerance for use of ecc, par, hyp anom
622  double M; // Mean Anomaly
623  double e; // Eccentricity magnitude
624  double B; // Parabolic Anomaly
625  double sin_E; // Sine of the Eccentric Anomaly
626  double cos_E; // Cosine of the Eccentric Anomaly
627  double sinh_H; // Hyperbolic Sine of the Hyperbolic Anomaly
628  double cosh_H; // Hyperbolic Cosine of Hyperbolic Anomaly
629 
630  // Assign input parameters for clarity
631  M = mean_anom;
632  e = e_mag;
633 
634  // Circular orbit
635  if(e < cir_tol)
636  {
637  // Compute the eccentric anomaly
638  orbital_anom = M;
639 
640  // Compute the true anomaly
641  sin_v = sin(M);
642  cos_v = cos(M);
643  true_anom = M;
644  }
645 
646  // Eccentric orbit
647  else if(e < (1.0 - switch_tol))
648  {
649  // Use Kepler's Equation to compute the eccentric anomaly
650  Converge = KepEqtnE(M, e, &(orbital_anom));
651 
652  // Check iteration convergence
653  if(Converge == -1)
654  {
655  MessageHandler::fail(__FILE__,
656  __LINE__,
657  OrbitalElementsMessages::convergence_error,
658  "Attempted solution for eccentric Kepler's equation "
659  "failed to converge.");
660  return (-1);
661  }
662 
663  // Compute the true anomaly
664  sin_E = sin(orbital_anom);
665  cos_E = cos(orbital_anom);
666  sin_v = sqrt(1.0 - e * e) * sin_E / (1.0 - e * cos_E);
667  cos_v = (cos_E - e) / (1.0 - e * cos_E);
668  true_anom = atan2(sin_v, cos_v);
669 
670  if(true_anom < 0.0)
671  {
672  true_anom += 2.0 * M_PI;
673  }
674  }
675 
676  // Hyperbolic orbit
677  else if(e > (1.0 + switch_tol))
678  {
679  // Use Kepler's Equation to compute the hyperbolic anomaly

695  cosh_H = cosh(orbital_anom);
