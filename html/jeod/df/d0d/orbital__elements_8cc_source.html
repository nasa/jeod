164 PURPOSE:
165  (Calculation of Orbital Elements from Inertial Cartesian Coordinates and the
166  planetary gravitational parameter. Elements include semi-major-axis -a-,
167  semiparameter -p-, eccentricity -e-, inclination -i-, longitude of the
168  ascending node -Omega-, argument of periapsis -w-, true anomaly -v-,
169  mean anomaly -M, and eccentric -E-, hyperbolic -H-, or parabolic anomaly
170  -B-.)
171 
172 REFERENCE:
173  (((Vallado, David A.) (Fundamentals of Astrodynamics and Applications)
174  (McGraw-Hill) (New York) (1997)))
175 
176 ASSUMPTIONS AND LIMITATIONS:
177  ((Input cartesian coordinates are assumed to be in an inertial frame.))
178 
179 CLASS:
180  (scheduled)
181 
182 
183 ******************************************************************************/
184 int OrbitalElements::from_cartesian(
185  double mu,
186  const double pos[3],
187  const double vel[3])
188 {
189 
190  // Declare local variables
191  int i; // Loop counter
192  double ang_momntm[3]; // Angular momentum vector
193  double line_of_nodes[3]; // Line-of-nodes vector
194  double pos_dot_vel; // Dot product of the position and velocity
195  double eccentricity[3]; // Orbit eccentricity vector
196  double K[3]; // Unit vector in K direction of inertial frame
197  double I[3]; // Unit vector in I direction of inertial frame
198  double v2; // Magnitude of the velocity squared
199  double tolerance; // Tolerance for e and i
200  double orbit_switch_tol; // Tolerance for orbit type switch
201  double cross_vec[3]; // cross product used for angle calculations
202  double sin_in; // sine value for input to atan2
203  double cos_in; // cosine value for input to atan2
204 
205  // Set tolerance values
206  tolerance = 1.0e-13;
207  orbit_switch_tol = 1.0e-2;
208 
209  // Define unit vector in K direction of inertial frame
210  K[0] = 0.0;
211  K[1] = 0.0;
212  K[2] = 1.0;
213 
214  /* NOTE: In December of 2005, the code was changed to calculate
215  angles using atan2 instead of acos for robustness. Some new
216  variables were added and some unecessary ones were removed.
217  The changes to the code are denoted by SWP 2005 */
218 
219  // Define unit vector in I direction of inertial frame (SWP 2005)
220  I[0] = 1.0;
221  I[1] = 0.0;
222  I[2] = 0.0;
223 
224  // Compute the magnitude of position and velocity
225  r_mag = Vector3::vmag (pos);
226  vel_mag = Vector3::vmag (vel);
227 
228  // Compute the angular momentum vector and magnitude
229  Vector3::cross (pos, vel, ang_momntm);
230  orb_ang_momentum = Vector3::vmag (ang_momntm);
231 
232  // Compute the line of nodes vector
233  Vector3::cross (K, ang_momntm, line_of_nodes);
234 
235  // Compute the eccentricity vector and magnitude
236  pos_dot_vel = Vector3::dot (pos, vel);
237  v2 = vel_mag * vel_mag;
238  for (i = 0; i < 3; i++) {
239  eccentricity[i] = (v2 - mu/r_mag) * pos[i]/mu -
240  (pos_dot_vel * vel[i]/mu);
241  }
242  e_mag = Vector3::vmag (eccentricity);
243 
244  // Compute the specific mechanical energy
245  orb_energy = (v2 / 2.0) - (mu / r_mag);
246 
247 
248  // Compute semi-major-axis, semiparameter, and mean motion
249  if (e_mag < tolerance) { // Circular orbit
250 
251  semi_major_axis = r_mag;
252  semiparam = r_mag;
253  mean_motion = sqrt(mu/semi_major_axis) / semi_major_axis;
254  }
255  else if (e_mag < (1.0 - orbit_switch_tol)) { // Eccentric orbit
256 
257  semi_major_axis = - mu / (2.0 * orb_energy);
258  semiparam = semi_major_axis * (1.0 - e_mag*e_mag);
259  mean_motion = sqrt(mu/semi_major_axis) / semi_major_axis;
260  }
261  else if (e_mag > (1.0 + orbit_switch_tol)) { // Hyperbolic orbit
262 
263  semi_major_axis = - mu / (2.0 * orb_energy);
264  semiparam = semi_major_axis * (1.0 - e_mag*e_mag);
265  mean_motion = sqrt(mu / -semi_major_axis) / -semi_major_axis;
266  }
267  else { // Parabolic orbit
268 
269  // Null out semi-major axis since it is actually infinity
270  semi_major_axis = 0.0;
271 
272  semiparam = orb_ang_momentum*orb_ang_momentum / mu;
273  mean_motion = 2.0 * sqrt(mu/semiparam) / semiparam;
274  }
275 
276 
277  // Compute the orbital inclination (SWP 2005)
278  Vector3::cross (K, ang_momntm, cross_vec);
279  sin_in = Vector3::vmag (cross_vec);
280  cos_in = Vector3::dot(K, ang_momntm);
281  inclination = atan2(sin_in, cos_in);
282 
283  // Compute longitude of ascending node, argument of periapsis, & true anomaly
284  if ((inclination < tolerance) ||
285  ((M_PI-tolerance) < inclination)) { // Equatorial orbit
286 
287  if (e_mag < tolerance) { // Equatorial and circular orbit
288  long_asc_node = 0.0;
289  arg_periapsis = 0.0;
290 
291  // SWP 2005
292  Vector3::cross (I, pos, cross_vec);
293  sin_in = Vector3::vmag (cross_vec);
294  cos_in = Vector3::dot (I, pos);
295  true_anom = atan2(sin_in, cos_in);
296 
297  if (inclination < tolerance) {
298  if (pos[1] < 0.0) {
299  true_anom = 2.0*M_PI - true_anom;
300  }
301  }
302  else {
303  if (pos[1] > 0.0) {
304  true_anom = 2.0*M_PI - true_anom;
305  }
306  }
307  }
308 
309  else { // Equatorial and non-circular orbit
310  long_asc_node = 0.0;
311 
312  // SWP 2005
313  Vector3::cross (I, eccentricity, cross_vec);
314  sin_in = Vector3::vmag (cross_vec);
315  cos_in = Vector3::dot (I, eccentricity);
316  arg_periapsis = atan2(sin_in, cos_in);
317 
318  if (inclination < tolerance) {
319  if (eccentricity[1] < 0.0) {
320  arg_periapsis = 2.0*M_PI - arg_periapsis;
321  }
322  }
323  else {
324  if (eccentricity[1] > 0.0) {
325  arg_periapsis = 2.0*M_PI - arg_periapsis;
326  }
327  }
328 
329  // SWP 2005
330  Vector3::cross (eccentricity, pos, cross_vec);
331  sin_in = Vector3::vmag (cross_vec);
332  cos_in = Vector3::dot (eccentricity, pos);
333  true_anom = atan2(sin_in, cos_in);
334 
335  if (pos_dot_vel < 0.0) {
336  true_anom = 2.0*M_PI - true_anom;
337  }
338  }
339  }
340 
341  else if (e_mag < tolerance) { // Circular and non-equatorial orbit
342 
343  // SWP 2005
344  Vector3::cross (I, line_of_nodes, cross_vec);
345  sin_in = Vector3::vmag (cross_vec);
346  cos_in = Vector3::dot (I, line_of_nodes);
347  long_asc_node = atan2(sin_in, cos_in);
348 
349  if (line_of_nodes[1] < 0.0) {
350  long_asc_node = 2.0*M_PI - long_asc_node;
351  }
352 
353  arg_periapsis = 0.0;
354 
355  // SWP 2005
356  Vector3::cross (line_of_nodes, pos, cross_vec);
357  sin_in = Vector3::vmag (cross_vec);
358  cos_in = Vector3::dot (line_of_nodes, pos);
359  true_anom = atan2(sin_in, cos_in);
360 
361  if (pos[2] < 0.0) {
362  true_anom = 2.0*M_PI - true_anom;
363  }
364  }
365 
366  else { // Non-circular and non-equatorial orbit
367 
368  // SWP 2005
369  Vector3::cross (I, line_of_nodes, cross_vec);
370  sin_in = Vector3::vmag (cross_vec);
371  cos_in = Vector3::dot (I, line_of_nodes);
372  long_asc_node = atan2(sin_in, cos_in);
373 
374  if (line_of_nodes[1] < 0.0) {
375  long_asc_node = 2.0*M_PI - long_asc_node;
376  }
377 
378  // SWP 2005
379  Vector3::cross (line_of_nodes, eccentricity, cross_vec);
380  sin_in = Vector3::vmag (cross_vec);
381  cos_in = Vector3::dot (line_of_nodes, eccentricity);
382  arg_periapsis = atan2(sin_in, cos_in);
383 
384  if (eccentricity[2] < 0.0) {
385  arg_periapsis = 2.0*M_PI - arg_periapsis;
386  }
387 
388  // SWP 2005
389  Vector3::cross (eccentricity, pos, cross_vec);
390  sin_in = Vector3::vmag (cross_vec);
391  cos_in = Vector3::dot (eccentricity, pos);
392  true_anom = atan2(sin_in, cos_in);
393 
394  if (pos_dot_vel < 0.0) {
395  true_anom = 2.0*M_PI - true_anom;
396  }
397  }
398 
399  // Compute the mean anomaly & the eccentric, hyperbolic, or parabolic anomaly
400  nu_to_anomalies();
401 
402  return (0);
403 }

406 /******************************************************************************
407 PURPOSE:
408  (Calculation of Inertial Cartesian Coordinates from Orbital Elements
409  semiparameter -p-, eccentricity -e-, inclination -i-, longitude of the
410  ascending node -Omega-, argument of periapsis -w-, mean anomaly -M-,
411  and gravitational parameter -mu-.)
412 
413 REFERENCE:
414  (((Vallado, D.) (Fundamentals of Astrodynamics and Applications, 2nd Ed.)
415  (Kluwer Acedemic Publishers) (2001) (Pages 135-230) (ISBN:0-07-066834-5))
416  ((Montenbruck, O. and Gill, E.) (Satellite Orbits)
417  (Springer) (2000) (Pages 157-190) (ISBN:3 540 67280))
418  ((Pedro Ramon Escobal) (Methods of Orbital Determination)
419  (Krieger Publishing) (1976) (pages 393-422) (ISBN:0-88275-319-3)))
420 
421 ASSUMPTIONS AND LIMITATIONS:
422  ((This routine is intended to be used at initialization only.)
423  (Output cartesian coordinates are in an inertial frame.)
424  (This routine works for all types of orbits.))
425 
426 CLASS:
427  (initialization)
428 
429 
430 ******************************************************************************/
431 int OrbitalElements::to_cartesian(
432  double mu,
433  double pos[3],
434  double vel[3])
435 {
436 
437  // Declare local variables
438  double tolerance = 1.0e-13; // Tolerance for e and i
439  double p; // Semiparameter
440  double e; // Magnitude of eccentricity
441  double r_pqw[3]; // Position vector in the PQW coordinate frame
442  double v_pqw[3]; // Velocity vector in the PQW coordinate frame
443  double T_PQW_Inrtl[3][3]; // Transformation matrix from PQW to Inertial
444  double sO; // Sine of the longitude of ascending node
445  double cO; // Cosine of the longitude of ascending node
446  double sw; // Sine of the argument of periapsis
447  double cw; // Cosine of the argument of periapsis
448  double si; // Sine of the inclination
449  double ci; // Cosine of the inclination
450 
451  // Assign input values for clarity
452  p = semiparam;
453  e = e_mag;
454 
455  // Check for "meaningful" value of semi-parameter
456  if (p <= 0.0) {
457  MessageHandler::fail (
458  __FILE__, __LINE__, OrbitalElementsMessages::domain_error,
459  "Semi-parameter %g is zero or negative.", p);
460  return (-1);
461  }
462 
463  // Check for "meaningful" values of sin_v and cos_v
464  double rss_sincos_v = sqrt(sin_v*sin_v + cos_v*cos_v);
465  if (fabs (rss_sincos_v - 1.0) > tolerance) {
466  MessageHandler::fail (
467  __FILE__, __LINE__, OrbitalElementsMessages::domain_error,
468  "Sine (%g) and cosine (%g) of the true anomaly are inconsistent.",
469  sin_v, cos_v);
470  return (-1);
471  }
472 
473  // Check for meaningful value of central mass
474  if (mu <= 0.0) {
475  MessageHandler::fail (
476  __FILE__, __LINE__, OrbitalElementsMessages::domain_error,
477  "Grav mass constant (mu) is zero or negative (%f).\n"
478  "This is likely a user-induced initialization error.\n",mu);
479  return (-1);
480  }
481 
482  // Test for equatorial and circular orbits
483  if ((inclination < tolerance) ||
484  ((M_PI-tolerance) < inclination)) { // Equatorial orbit
485 
486  if (e < tolerance) { // Equatorial and Circular orbit
487 
488  sO = 0.0;
489  cO = 1.0;
490  sw = 0.0;
491  cw = 1.0;
492  }
493  else { // Equatorial and non-circular orbit
494 
495  sO = 0.0;
496  cO = 1.0;
497  sw = sin(arg_periapsis);
498  cw = cos(arg_periapsis);
499  }
500 
501  if (inclination < tolerance) { // Inclination is approx. zero
502 
503  si = 0.0;
504  ci = 1.0;
505  }
506  else { // Inclination is approx. pi
507 
508  si = 0.0;
509  ci = -1.0;
510  }
511  }
512 
513  else if (e < tolerance) { // Circular and non-equatorial orbit
514 
515  sO = sin(long_asc_node);
516  cO = cos(long_asc_node);
517  sw = 0.0;
518  cw = 1.0;
519  si = sin(inclination);
520  ci = cos(inclination);
521  }
522 
523  else { // Non-circular and non-equatorial orbit
524 
525  sO = sin(long_asc_node);
526  cO = cos(long_asc_node);
527  sw = sin(arg_periapsis);
528  cw = cos(arg_periapsis);
529  si = sin(inclination);
530  ci = cos(inclination);
531  }
532 
533  // Compute position and velocity in perifocal frame (PQW)
534  r_pqw[0] = (p * cos_v) / (1.0 + e*cos_v);
535  r_pqw[1] = (p * sin_v) / (1.0 + e*cos_v);
536  r_pqw[2] = 0.0;
537 
538  v_pqw[0] = -sqrt(mu/p) * sin_v;
539  v_pqw[1] = sqrt(mu/p) * (e+cos_v);
540  v_pqw[2] = 0.0;
541 
542  // Compute transformation matrix from PQW to inertial
543  T_PQW_Inrtl[0][0] = cO * cw - sO * sw * ci;
544  T_PQW_Inrtl[0][1] = -cO * sw - sO * cw * ci;
545  T_PQW_Inrtl[0][2] = sO * si;
546  T_PQW_Inrtl[1][0] = sO * cw + cO * sw * ci;
547  T_PQW_Inrtl[1][1] = -sO * sw + cO * cw * ci;
548  T_PQW_Inrtl[1][2] = -cO * si;
549  T_PQW_Inrtl[2][0] = sw * si;
550  T_PQW_Inrtl[2][1] = cw * si;
551  T_PQW_Inrtl[2][2] = ci;
552 
553  // Compute position and velocity in inertial frame
554  Vector3::transform (T_PQW_Inrtl, r_pqw, pos);
555  Vector3::transform (T_PQW_Inrtl, v_pqw, vel);
556 
557  return (0);
558 }

561 /******************************************************************************
562 PURPOSE:
563  (Calculation of the Mean Anomaly and the Eccentric, Hyperbolic, or
564  Parabolic Anomalies, given the True Anomaly and eccentricity)
565 
566 REFERENCE:
567  (((Vallado, D.) (Fundamentals of Astrodynamics and Applications, 2nd Ed.)
568  (Kluwer Acedemic Publishers) (2001) (Pages 135-230) (ISBN:0-07-066834-5))
569  ((Montenbruck, O. and Gill, E.) (Satellite Orbits)
570  (Springer) (2000) (Pages 157-190) (ISBN:3 540 67280))
571  ((Pedro Ramon Escobal) (Methods of Orbital Determination)
572  (Krieger Publishing) (1976) (pages 393-422) (ISBN:0-88275-319-3)))
573 
574 ASSUMPTIONS AND LIMITATIONS:
575  ((This routine works for all types of orbits.))
576 
577 CLASS:
578  (N/A)
579 
580 
581 ******************************************************************************/
582 int OrbitalElements::nu_to_anomalies()
583 {
584 
585  // Declare local variables
586  double cir_tol = 1.0e-13 ; // Tolerance for circular orbit
587  double switch_tol = 1.0e-2; // Tolerance for use of ecc, par, hyp anom
588  double v; // True Anomaly
589  double e; // Magnitude of eccentricity
590  double sin_E; // Sine of the Eccentric Anomaly
591  double cos_E; // Cosine of the Eccentric Anomaly
592  double sinh_H; // Hyperbolic sine of the Hyperbolic Anomaly
593 
594  // Assign input parameters for clarity
595  v = true_anom;
596  e = e_mag;
597  sin_v = sin(true_anom);
598  cos_v = cos(true_anom);
599 
600  // Circular orbit
601  if (e < cir_tol) {
602 
603  orbital_anom = v;
604  mean_anom = v;
605  }
606 
607  // Eccentric orbit
608  else if (e < (1.0 - switch_tol)) {
609 
610  sin_E = sqrt(1.0 - e*e)*sin_v / (1.0 + e*cos_v);
611  cos_E = (e + cos_v) / (1.0 + e*cos_v);
612  orbital_anom = atan2(sin_E, cos_E);
613 
614  if (orbital_anom < 0.0) {
615  orbital_anom += 2.0*M_PI;
616  }
617 
618  mean_anom = orbital_anom - e*sin(orbital_anom);
619  }
620 
621  // Hyperbolic orbit
622  else if (e > (1.0 + switch_tol)) {
623 
624  sinh_H = sqrt(e*e - 1.0) * sin_v / (1.0 + e*cos_v);
625  orbital_anom = asinh(sinh_H);
626  mean_anom = e*sinh(orbital_anom) - orbital_anom;
627  }
628 
629  // Parabolic orbit
630  else {
631  orbital_anom = tan( v / 2.0);
632  mean_anom = orbital_anom + pow(orbital_anom, 3.0)/ 3.0;
633  }
634 
635  return (0);
636 }

736  true_anom += 2.0 * M_PI;
737  }
738  }
739 
740  // Parabolic orbit
741  else {

753 
754  if (true_anom < 0.0) {
755  true_anom += 2.0 * M_PI;
756  }
757  }

759  return (0);
