97 void RadiationPressure::update(RefFrame & veh_struc_frame, double center_grav[3], double scale_factor, double real_time)
98 {
99  if(!active)
100  {
101  return;
102  }
103 
104  // ThermalFacetRider cycle-time is the time needed for integration purposes,
105  // and includes "catchup" whenever the rate at which time advances changes.
106  ThermalFacetRider::cycle_time = JeodSimulationInterface::get_job_cycle() * scale_factor;
107 
108  source.calculate_flux(veh_struc_frame, center_grav);
109 
110  if(third_bodies_active)
111  {
112  third_body_adjustments(real_time, veh_struc_frame);
113  }
114 
115  // It appears that this would be better handled by polymorphism - that a call
116  // to, say, update_surface, would branch to the appropriate method, which
117  // should be in RadiationSurface, not RadiationPressure. However, the
118  // methods called here are more managerial that operational. The operational
119  // methods are in the appropriate Surface class, and are called by these
120  // managerial methods. They do belong in RadiationPressure.
121  if(surface_ptr == nullptr)
122  {
123  update_default_surface();
124  }
125  else
126  {
127  update_facet_surface();
128  }

131 /**
132  * Used to modify the flux for shadowing.
133  * \param[in] real_time The current time, e.g. TAI or UT1.\n Units: s
134  * \param[in] veh_struc_frame Structure frame of vehicle
135  */
136 void RadiationPressure::third_body_adjustments(double real_time, RefFrame & veh_struc_frame)
137 {
138  illum_factor = 1.0;
139  // multi_shadows = false;
140  for(unsigned int ii_body = 0; ii_body < num_third_bodies; ++ii_body)
141  {
142  if(third_bodies[ii_body]->active)
143  {
144  illum_factor *= third_bodies[ii_body]->process_third_body(real_time, veh_struc_frame);
145  }
146  }
147  source.flux_mag = source.flux_mag * illum_factor;
148  Vector3::scale(illum_factor, source.flux_inertial);
149  Vector3::scale(illum_factor, source.flux_struc);
150 }

152 /**
153  * Adds a third body to the vector storage of third bodies
154  * \param[in] third_body_ptr pointer to ThirdBody
155  */
156 void RadiationPressure::add_third_body(RadiationThirdBody * third_body_ptr)
157 {
158  if(third_body_ptr == nullptr)
159  {
160  MessageHandler::fail(__FILE__,
161  __LINE__,
162  RadiationMessages::invalid_setup_error,
163  "\n"
164  "Attempted to push a NULL object onto the ThirdBody vector in "
165  "the Radiation Pressure model.\n");
166  return;
167  }
168  // third_body_ptr is not NULL
169  // If third body is not named, then fail at initialization, and flag a
170  // warning that the adding has failed (but continue with the simulation)
171  // otherwise.
172  if(third_body_ptr->name.empty())
173  {
174  if(initialized)
175  {
176  MessageHandler::error(__FILE__,
177  __LINE__,
178  RadiationMessages::incomplete_setup_error,
179  "\n"
180  "Attempting to add a ThirdBody with no defined name.\n"
181  "This cannot be done, since the name is needed to identify the "
182  "reference frame.\n"
183  "Addition of ThirdBody has failed. Simulation proceeding without it."
184  "\n");
185  return;
186  }
187  else
188  {
189  MessageHandler::fail(__FILE__,
190  __LINE__,
191  RadiationMessages::incomplete_setup_error,
192  "\n"
193  "Attempting to add a ThirdBody with no defined name.\n"
194  "This cannot be done, since the name is needed to identify the "
195  "reference frame.\n");
196  }
197  }
198  // Check that the Third Body has not already been added. The
199  // RadiaitonThirdBody points to the RadiationSource, which provides
200  // information about the local radiation environment. Therefore, the
201  // RadiationSource is model specific, and therefore the RadiationThirdBody
202  // must also be model-specific. They cannot be shared across models.
203  if(third_body_ptr->get_added_to_model())
204  {
205  MessageHandler::fail(__FILE__,
206  __LINE__,
207  RadiationMessages::invalid_setup_error,
208  "\n"
209  "Attempting to add a ThirdBody to more than one Radiation Pressure "
210  "model.\n"
211  "Offending object is called(%s).\n"
212  "ThirdBody objects can only be added to one Radiation Pressure model.\n"
213  "The same entity (vehicle, planet) can be represented as a ThirdBody\n"
214  "in multiple RadiationPressure models, but each representation must\n"
215  "be a separate instance of a ThirdBody.\n"
216  "E.g., there can be two ThirdBody objects called(%s), each linked to the\n"
217  "appropriate object (Planet or DynBody) of the same name, so that two \n"
218  "vehicles can each have their own RadiationPressure model with a \n"
219  "representation of the effects that(%s) has on their local radiation "
220  "field.\n",
221  third_body_ptr->name.c_str(),
222  third_body_ptr->name.c_str(),
223  third_body_ptr->name.c_str());
224  }
225 
226  // Don't try adding two ThirdBody objects with the same name to one model.
227  // This is usually an error of duplicity in setting up the model, so
228  // duplicates are just ignored.
229  for(unsigned int ii_t_b = 0; ii_t_b < num_third_bodies; ii_t_b++)
230  {
231  if(third_body_ptr->name == third_bodies[ii_t_b]->name.c_str())
232  {
233  MessageHandler::error(__FILE__,
234  __LINE__,
235  RadiationMessages::invalid_setup_error,
236  "\n"
237  "In the Radiation Pressure Model, two RadiationThirdBody "
238  "objects \n were found with identical names (%s).\n "
239  "Omitting one.\n",
240  third_body_ptr->name.c_str());
241  return;
242  }
243  }
244 
245  third_bodies.push_back(third_body_ptr);
246  third_body_ptr->set_added_to_model(true);
247 
248  // If the model has already been initialized, then initialize the third
249  // body (else, the third bodies will all be initialized with the
250  // model, so no action needed).
251 
252  if(initialized)
253  {
254  third_body_ptr->initialize(dyn_manager_ptr);
255  }
256 
257  // If any ThirdBody is active, the model should be looking for it/them.
258  if(third_body_ptr->active)
259  {
260  third_bodies_active = true;
261  }
262 
263  num_third_bodies = third_bodies.size();
264 }

270 void RadiationPressure::set_third_body_active(const std::string & third_body_name)
271 {
272  int body_index = find_third_body(third_body_name);
273  if(body_index >= 0)
274  {
275  if(third_bodies[body_index]->active)
276  {
277  MessageHandler::warn(__FILE__,
278  __LINE__,
279  RadiationMessages::invalid_function_call,
280  "\n"
281  "Call made to set the RadiationThirdBody(%s) active, but it is "
282  "already active.\n",
283  third_body_name.c_str());
284  }
285  else
286  {
287  third_bodies_active = true;
288  third_bodies[body_index]->active = true;
289  if(third_bodies[body_index]->local_frame_ptr != nullptr)
290  {
291  dyn_manager_ptr->subscribe_to_frame(*third_bodies[body_index]->local_frame_ptr);
292  }
293  }
294  }
295  else
296  {
297  // Continuation from find_third_body error message
298  MessageHandler::error(__FILE__,
299  __LINE__,
300  RadiationMessages::invalid_function_call,
301  "\n"
302  "Failure to find third body by name means it cannot be activated.");
303  }
