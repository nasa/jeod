35 // Configure the coefficients as an Adams corrector in difference form.
36 void GaussJacksonRationalCoefficients::configure_adams_corrector(unsigned int nelem)
37 {
38  // Resize the coefficients array to accommodate nelem elements.
39  coefficients.resize(nelem);
40 
41  // Calculate the coefficients using
42  // c_0 = 1
43  // c_n = -\sum_{i=0}^{n-1} \frac {c_i} {n+1-i}
44  // See Berry eqn 2.37 and 2.38 for details.
45  coefficients[0] = 1;
46  for(unsigned int nn = 1; nn < nelem; ++nn)
47  {
48  er7_utils::Ratio128 sum = 0;
49  for(unsigned int ii = 0; ii < nn; ++ii)
50  {
51  sum -= coefficients[ii] / er7_utils::Ratio128(nn + 1 - ii);
52  }
53  coefficients[nn] = sum;
54  }
55 }

57 // Compute the Stormer-Cowell corrector coefficients.
58 GaussJacksonRationalCoefficients GaussJacksonRationalCoefficients::construct_stormer_cowell_corrector() const
59 {
60  // Create a set of unpopulated rational coefficients sized the same as this.
61  GaussJacksonRationalCoefficients result;
62  unsigned int nelem = coefficients.size();
63  result.coefficients.resize(nelem);
64 
65  // Populate the result using
66  // q_i = \sum_{k=0}^i c_k c_{i-k}
67  // See Berry eqn 2.55 for details.
68  for(unsigned int ii = 0; ii < nelem; ++ii)
69  {
70  er7_utils::Ratio128 sum = 0;
71  for(unsigned int kk = 0; kk <= ii; ++kk)
72  {
73  sum += coefficients[kk] * coefficients[ii - kk];
74  }
75  result.coefficients[ii] = sum;
76  }
77  return result;
78 }

102 // Convert the coefficients to ordinate form.
103 // Note: This assumes that the correct number of leading terms have been
104 // discarded from the front of the coefficients vector.
105 void GaussJacksonRationalCoefficients::convert_to_ordinate_form(er7_utils::NChooseM & n_choose_m, double * result) const
106 {
107  // Create a set of unpopulated rational coefficients sized the same as this.
108  std::vector<er7_utils::Ratio128> rational_coefs;
109  unsigned int nelem = coefficients.size();
110  rational_coefs.resize(nelem);
111 
112  // Populate the result using
113  // z_{Nm} = (-1)^m \sum_{i=m}^N z'_i \binom i m
114  // See Berry eqn 2.79 for details.
115  for(unsigned int mm = 0; mm < nelem; ++mm)
116  {
117  er7_utils::Ratio128 sum = 0;
118  for(unsigned int ii = mm; ii < nelem; ++ii)
119  {
120  sum += n_choose_m(ii, mm) * coefficients[ii];
121  }
122  if((mm % 2) != 0)
123  {
124  sum = -sum;
125  }
126 
127  // Store in reverse order to simplify calculations.
128  rational_coefs[nelem - 1 - mm] = sum;
129  }
130 
131  // Copy the computed coefficient to the result.
132  // Note that copy converts the rational coefficients to doubles.
133  std::copy(rational_coefs.begin(), rational_coefs.end(), result);
134 }

147 // Apply the operator (1-\nabla) to the coefficients.
148 // Note: This assumes that the correct number of leading terms have been
149 // discarded from the front of the coefficients vector.
150 void GaussJacksonRationalCoefficients::displace_back()
151 {
152  // Create a set of unpopulated rational coefficients sized the same as this.
153  std::vector<er7_utils::Ratio128> new_coefficients;
154  unsigned int nelem = coefficients.size();
155  new_coefficients.resize(nelem);
156 
157  // Apply (1-\nabla): c'_0 = c_0, c'_i = c_i - c_{i-1}
158  new_coefficients[0] = coefficients[0];
159  for(unsigned int ii = 1; ii < nelem; ++ii)
160  {
161  new_coefficients[ii] = coefficients[ii] - coefficients[ii - 1];
162  }
163 
164  // Replace the current coefficients with the new ones.
165  coefficients = new_coefficients;
166 }
