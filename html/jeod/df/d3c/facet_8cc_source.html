130 void Facet::update_articulation_internal()
131 {
132  // mass_rel_struct should now contain the position and orientation
133  // of mass_body's structural, with respect to struct_body's
134  // structural, in struct body's structural. If it doesn't, then
135  // this function shouldn't have been called yet (since it's
136  // protected we'll make that assumption)
137 
138  // convert the local position from the local mass body struct
139  // frame to the integrating mass body struct frame (the frame
140  // associated with the mass body named in struct_body_name
141  // first rotate it into that frame
142 
143  Vector3::transform_transpose(mass_rel_struct->T_parent_this, local_position, int_pos);
144 
145  // now add the position of the mass frame w.r.t. the struct
146  // frame, in the struct frame, to the local_position in the
147  // struct frame (contained in int_pos since we just calculated
148  // it)
149 
150  Vector3::sum(mass_rel_struct->position, int_pos, position);
151 }
