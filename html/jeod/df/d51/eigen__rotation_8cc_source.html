58 /**
59  * Compute the transformation matrix from the eigen rotation.
60  * Given a rotation by an angle \f$\phi\f$ about an axis
61  * \f$\hat{\mathbf u}\f$,
62  * the [i][j] element of the transformation matrix is given by
63  * \f[
64  * T_{ij} =
65  * \cos\phi\,\delta_{ij} +
66  * (1-\cos\phi)\,\hat u_i \hat u_j +
67  * \epsilon_{ijk}\sin\phi\,\hat u_k
68  * \f]
69  * where
70  * - \f$\delta_{ij}\f$ is the Kronecker delta,
71  * - @e k is \f$(i+j) \bmod 3\f$, and
72  * - \f$\epsilon_{ijk}\f$ is the Levi-Civita symbol taken with respect
73  * to (0,1,2).
74  *
75  * \par Assumptions and Limitations
76  * - The eigen axis is a unit vector.
77  * \param[in] eigen_angle Rotation angle\n Units: r
78  * \param[in] eigen_axis Rotation axis, unit vector
79  * \param[out] trans Resultant transformation matrix
80  */
81 void
82 Orientation::compute_matrix_from_eigen_rotation (
83  double eigen_angle,
84  const double eigen_axis[3],
85  double trans[3][3])
86 {
87  double cphi = std::cos (eigen_angle);
88  double sphi = std::sin (eigen_angle);
89  double omcp = 1.0 - cphi;
90  double del;
91 
92  // Compute the diagonal elements of the matrix, which are of the form
93  // T[i][i] = cos(phi) + (1-cos(phi)) * uhat[i]*uhat[i]
94  trans[0][0] = cphi + omcp * eigen_axis[0] * eigen_axis[0];
95  trans[1][1] = cphi + omcp * eigen_axis[1] * eigen_axis[1];
96  trans[2][2] = cphi + omcp * eigen_axis[2] * eigen_axis[2];
97 
98  // The off-diagonal elements are of the form
99  // T[i][j] = (1-cos(phi))*uhat[i]*uhat[j] + epsilon_ijk*sin(phi)*uhat[k]
100  // where k is the omitted axis and epsilon_ijk is the Levi-Civita symbol.
101  // epsilon_ijk is +1/-1 if (i,j,k) is an even/odd permutation of (0,1,2).
102  // First compute the terms that don't involve epsilon_ijk.
103  trans[0][1] =
104  trans[1][0] = omcp * eigen_axis[0] * eigen_axis[1];
105  trans[0][2] =
106  trans[2][0] = omcp * eigen_axis[0] * eigen_axis[2];
107  trans[1][2] =
108  trans[2][1] = omcp * eigen_axis[1] * eigen_axis[2];
109 
110  // Now add/subtract the sin(phi)*uhat[k] term.
111 
112  // (i,j) = (0,1) and (1,0) => k=2. (0,1,2) is even, (1,0,2) is odd.
113  del = sphi * eigen_axis[2];
114  trans[0][1] += del;
115  trans[1][0] -= del;
116 
117  // (i,j) = (2,0) and (0,2) => k=1. (2,0,1) is even, (0,2,1) is odd.
118  del = sphi * eigen_axis[1];
119  trans[2][0] += del;
120  trans[0][2] -= del;
121 
122  // (i,j) = (1,2) and (2,1) => k=0. (1,2,0) is even, (2,1,0) is odd.
123  del = sphi * eigen_axis[0];
124  trans[1][2] += del;
125  trans[2][1] -= del;
126 }

129 /**
130  * Compute the eigen rotation from the transformation matrix.
131  * There are several alternate expressions for computing the eigen rotation
132  * from a matrix, all of which are equivalent in infinite precision arithmetic.
133  * The use of finite precision arithmetic means that care must be taken
134  * in choosing the algorithm to be used. The starting point is the
135  * generic expression
136  * \f[
137  * T_{ij} =
138  * \cos\phi\,\delta_{ij} +
139  * (1-\cos\phi)\,\hat u_i \hat u_j +
140  * \epsilon_{ijk}\sin\phi\,\hat u_k
141  * \f]
142  * From this, the trace of the matrix and the difference between and sum of
143  * pairs of off-diagonal elements are
144  * \f{eqnarray*}{
145  * \mathrm{tr}(T) &=& 2\cos\phi + 1 \\
146  * T_{ij}-T_{ji} &=& 2\epsilon_{ijk}\sin\phi\,\hat u_k \\
147  * T_{ij}+T_{ji} &=& 2(1-\cos\phi)\,\hat u_i \hat u_j
148  * \f}
149  *
150  * <b>Method 1</b>\n
151  * One approach to determining the eigen rotation involves the construction
152  * of a vector of differences between pairs of off-diagonal elements of the
153  * transformation matrix,
154  * \f[
155  * d_k = T_{ij}-T_{ji} = 2 \sin\phi\,\hat u_k
156  * \f]
157  * where (i,j,k) is an even permutation of (0,1,2). With this,
158  * \f{eqnarray*}{
159  * \phi &=& \arcsin\left(\frac{||\mathbf d||} 2\right) \\
160  * \hat{\mathbf u} &=& \frac{\mathbf d}{||\mathbf d||}
161  * \f}
162  * Note that the above of the inverse sine will restrict the rotation angle to
163  * be between 0 and 90 degrees. Special processing is needed when the rotation
164  * angle is between 90 and 180 degrees.
165  * Note also that the symmetric difference vector will
166  * be identically zero if the rotation angle is 0 or 180 degrees and will be
167  * very small for rotation angles close to 0 or 180 degrees.
168  * The precision loss for rotation angles near 0 and 180 degrees means the
169  * individual components of the eigen axis will not be as precise with this
170  * approach compared to alternatives.
171  *
172  * <b>Method 2</b>\n
173  * The diagonal elements of the matrix yields another method for determining the
174  * single axis rotation angle and the rotation axis:
175  * \f{eqnarray*}{
176  * \phi &=& \arccos\left(\frac{\mathrm{tr}(T)-1} 2\right) \\
177  * |\hat u_i| &=& \sqrt{\frac{T_{ii} - \cos \phi}{1-\cos\phi}}
178  * \f}
179  * Note that this approach determines the magnitudes but not the signs of the
180  * components of the eigen axis vector. Because this method is based on the
181  * inverse cosine, the calculated phi angle will be less precise than that
182  * obtained by method 1 for angles near 0 or 180 degrees. The unit vector
183  * however will be more accurate than that obtained from method 1 for small
184  * rotation angles.
185  *
186  * <b>Method 3</b>\n
187  * Yet another alternative for computing components of the eigen axis is to
188  * use the sum of pairs of off-diagonal elements of the transformation metrix,
189  * \f{eqnarray*}{
190  * T_{ij}+T_{ji} &=& 2(1-\cos\phi)\,\hat u_i \hat u_j \\
191  * T_{ik}+T_{ki} &=& 2(1-\cos\phi)\,\hat u_i \hat u_k
192  * \f}
193  * This enables the calculation of two components of the unit vector. One
194  * component needs to be computed by one of the two previous methods.
195  *
196  * \par Assumptions and Limitations
197  * - The matrix is a proper transformation matrix.
198  * \param[in] trans Transformation matrix
199  * \param[out] eigen_angle Resultant rotation angle\n Units: r
200  * \param[out] eigen_axis Resultant rotation axis
201  */
202 void
203 Orientation::compute_eigen_rotation_from_matrix (
204  const double trans[3][3],
205  double * eigen_angle,
206  double eigen_axis[3])
207 {
208  double cosphi; // Cosing of the eigen angle
209  double sinphi; // Sine of the eigen angle
210  double diff[3]; // Symmetric difference vector
211  double dmag; // Magnitude of the symmetric difference vector
212 
213  // Calculate the symmetric difference vector and its magnitude.
214  diff[0] = trans[1][2] - trans[2][1];
215  diff[1] = trans[2][0] - trans[0][2];
216  diff[2] = trans[0][1] - trans[1][0];
217  dmag = Vector3::vmag(diff);
218 
219  // Determine the sine and cosine of the rotation angle.
220  sinphi = 0.5 * dmag;
221  cosphi = 0.5 * (trans[0][0] + trans[1][1] + trans[2][2] - 1.0);
222 
223 
224  // Now see which solution to use.
225  // There are four cases.
226  // Use the appropriate method and solve for the angle and the axis.
227 
228 
229  // Case 1: The trivial case, rotation angle = 0.
230  // This case is evidenced by sinphi = 0, cosphi = 1.
231  // The rotation is ill-defined in this case.
232  // Arbitrarily set the rotation vector to xhat.
233  if ((std::fpclassify(dmag) == FP_ZERO) && (cosphi > 0.0)) {
234  *eigen_angle = 0.0;
235  eigen_axis[0] = 1.0;
236  eigen_axis[1] = 0.0;
237  eigen_axis[2] = 0.0;
238  }
239 
240  // Case 2: Rotation angle between 0 and 135 degrees (exclusive).
241  // This case is evidenced by sinphi greater than -cosphi.
242  // Here the symmetric difference (method 1) does the best job of determining
243  // the eigen axis. Method 1 yields for best results for the rotation angle
244  // for small rotation angles (0 to 45 degrees); method 2 is better for
245  // larger values.
246  else if (sinphi >= -cosphi) {
247  // The rotation angle is given by
248  // phi = acos(cosphi)
249  // or
250  // phi = asin(sinphi)
251  // The inverse sine yields better results for angles less than 45 degrees
252  // while the inverse cosine is more accurate for larger angles.
253  if (sinphi < cosphi) {
254  *eigen_angle = std::asin(sinphi);
255  }
256  else {
257  *eigen_angle = std::acos(cosphi);
258  }
259 
260  // Use method 1 to construct the eigen axis.
261  Vector3::scale (diff, 1.0/dmag, eigen_axis);
262  }
263 
264  // Case 3: Rotation angle between 135 and 180 degrees (inclusive).
265  // Here the symmetric difference does the best job of determining the
266  // rotation angle and the symmetric sum does the best job of determining
267  // the eigen axis.
268  else {
269  unsigned int ii; // Index of largest diagonal element
270  double omcp = 1.0 - cosphi; // 1 - cos(phi)
271 
272  // Use the trace to determine the eigen rotation angle,
273  // taking care to place the angle in the correct quadrant.
274  *eigen_angle = M_PI - std::asin(sinphi);
275 
276  // The symmetric sum can only determine two of the three components
277  // of the eigen axis. The symmetric difference needs to be used for
278  // one of the components. The best accuracy results from choosing the
279  // the component correponding to the largest diagonal element.
280  // Find the index of the largest diagonal element.
281  ii = 0;
282  if (trans[1][1] > trans[ii][ii]) {
283  ii = 1;
284  }
285  if (trans[2][2] > trans[ii][ii]) {
286  ii = 2;
287  }
288 
289  // Construct the iith element of the eigen axis vector using
290  // method 2 to determine the magnitude, method 1 to determine the sign.
291  eigen_axis[ii] = std::sqrt ((trans[ii][ii] - cosphi) / omcp);
292  if (diff[ii] < 0.0) {
293  eigen_axis[ii] = -eigen_axis[ii];
294  }
295 
296  // Determine the remaining elements of the eigen axis via method 3.
297  double vscale = 1. / (2 * omcp * eigen_axis[ii]);
298  for (unsigned int jj = 0; jj < 3; jj++) {
299  if (jj != ii) {
300  eigen_axis[jj] = (trans[ii][jj] + trans[jj][ii]) * vscale;
301  }
302  }
303  }
304 
305  return;
306 }
