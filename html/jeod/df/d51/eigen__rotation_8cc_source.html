58 /**
59  * Compute the transformation matrix from the eigen rotation.
60  * Given a rotation by an angle \f$\phi\f$ about an axis
61  * \f$\hat{\mathbf u}\f$,
62  * the [i][j] element of the transformation matrix is given by
63  * \f[
64  * T_{ij} =
65  * \cos\phi\,\delta_{ij} +
66  * (1-\cos\phi)\,\hat u_i \hat u_j +
67  * \epsilon_{ijk}\sin\phi\,\hat u_k
68  * \f]
69  * where
70  * - \f$\delta_{ij}\f$ is the Kronecker delta,
71  * - @e k is \f$(i+j) \bmod 3\f$, and
72  * - \f$\epsilon_{ijk}\f$ is the Levi-Civita symbol taken with respect
73  * to (0,1,2).
74  *
75  * \par Assumptions and Limitations
76  * - The eigen axis is a unit vector.
77  * \param[in] eigen_angle Rotation angle\n Units: r
78  * \param[in] eigen_axis Rotation axis, unit vector
79  * \param[out] trans Resultant transformation matrix
80  */
81 void Orientation::compute_matrix_from_eigen_rotation(double eigen_angle, const double eigen_axis[3], double trans[3][3])
82 {
83  double cphi = std::cos(eigen_angle);
84  double sphi = std::sin(eigen_angle);
85  double omcp = 1.0 - cphi;
86  double del;
87 
88  // Compute the diagonal elements of the matrix, which are of the form
89  // T[i][i] = cos(phi) + (1-cos(phi)) * uhat[i]*uhat[i]
90  trans[0][0] = cphi + omcp * eigen_axis[0] * eigen_axis[0];
91  trans[1][1] = cphi + omcp * eigen_axis[1] * eigen_axis[1];
92  trans[2][2] = cphi + omcp * eigen_axis[2] * eigen_axis[2];
93 
94  // The off-diagonal elements are of the form
95  // T[i][j] = (1-cos(phi))*uhat[i]*uhat[j] + epsilon_ijk*sin(phi)*uhat[k]
96  // where k is the omitted axis and epsilon_ijk is the Levi-Civita symbol.
97  // epsilon_ijk is +1/-1 if (i,j,k) is an even/odd permutation of (0,1,2).
98  // First compute the terms that don't involve epsilon_ijk.
99  trans[0][1] = trans[1][0] = omcp * eigen_axis[0] * eigen_axis[1];
100  trans[0][2] = trans[2][0] = omcp * eigen_axis[0] * eigen_axis[2];
101  trans[1][2] = trans[2][1] = omcp * eigen_axis[1] * eigen_axis[2];
102 
103  // Now add/subtract the sin(phi)*uhat[k] term.
104 
105  // (i,j) = (0,1) and (1,0) => k=2. (0,1,2) is even, (1,0,2) is odd.
106  del = sphi * eigen_axis[2];
107  trans[0][1] += del;
108  trans[1][0] -= del;
109 
110  // (i,j) = (2,0) and (0,2) => k=1. (2,0,1) is even, (0,2,1) is odd.
111  del = sphi * eigen_axis[1];
112  trans[2][0] += del;
113  trans[0][2] -= del;
114 
115  // (i,j) = (1,2) and (2,1) => k=0. (1,2,0) is even, (2,1,0) is odd.
116  del = sphi * eigen_axis[0];
117  trans[1][2] += del;
118  trans[2][1] -= del;
119 }

121 /**
122  * Compute the eigen rotation from the transformation matrix.
123  * There are several alternate expressions for computing the eigen rotation
124  * from a matrix, all of which are equivalent in infinite precision arithmetic.
125  * The use of finite precision arithmetic means that care must be taken
126  * in choosing the algorithm to be used. The starting point is the
127  * generic expression
128  * \f[
129  * T_{ij} =
130  * \cos\phi\,\delta_{ij} +
131  * (1-\cos\phi)\,\hat u_i \hat u_j +
132  * \epsilon_{ijk}\sin\phi\,\hat u_k
133  * \f]
134  * From this, the trace of the matrix and the difference between and sum of
135  * pairs of off-diagonal elements are
136  * \f{eqnarray*}{
137  * \mathrm{tr}(T) &=& 2\cos\phi + 1 \\
138  * T_{ij}-T_{ji} &=& 2\epsilon_{ijk}\sin\phi\,\hat u_k \\
139  * T_{ij}+T_{ji} &=& 2(1-\cos\phi)\,\hat u_i \hat u_j
140  * \f}
141  *
142  * <b>Method 1</b>\n
143  * One approach to determining the eigen rotation involves the construction
144  * of a vector of differences between pairs of off-diagonal elements of the
145  * transformation matrix,
146  * \f[
147  * d_k = T_{ij}-T_{ji} = 2 \sin\phi\,\hat u_k
148  * \f]
149  * where (i,j,k) is an even permutation of (0,1,2). With this,
150  * \f{eqnarray*}{
151  * \phi &=& \arcsin\left(\frac{||\mathbf d||} 2\right) \\
152  * \hat{\mathbf u} &=& \frac{\mathbf d}{||\mathbf d||}
153  * \f}
154  * Note that the above of the inverse sine will restrict the rotation angle to
155  * be between 0 and 90 degrees. Special processing is needed when the rotation
156  * angle is between 90 and 180 degrees.
157  * Note also that the symmetric difference vector will
158  * be identically zero if the rotation angle is 0 or 180 degrees and will be
159  * very small for rotation angles close to 0 or 180 degrees.
160  * The precision loss for rotation angles near 0 and 180 degrees means the
161  * individual components of the eigen axis will not be as precise with this
162  * approach compared to alternatives.
163  *
164  * <b>Method 2</b>\n
165  * The diagonal elements of the matrix yields another method for determining the
166  * single axis rotation angle and the rotation axis:
167  * \f{eqnarray*}{
168  * \phi &=& \arccos\left(\frac{\mathrm{tr}(T)-1} 2\right) \\
169  * |\hat u_i| &=& \sqrt{\frac{T_{ii} - \cos \phi}{1-\cos\phi}}
170  * \f}
171  * Note that this approach determines the magnitudes but not the signs of the
172  * components of the eigen axis vector. Because this method is based on the
173  * inverse cosine, the calculated phi angle will be less precise than that
174  * obtained by method 1 for angles near 0 or 180 degrees. The unit vector
175  * however will be more accurate than that obtained from method 1 for small
176  * rotation angles.
177  *
178  * <b>Method 3</b>\n
179  * Yet another alternative for computing components of the eigen axis is to
180  * use the sum of pairs of off-diagonal elements of the transformation metrix,
181  * \f{eqnarray*}{
182  * T_{ij}+T_{ji} &=& 2(1-\cos\phi)\,\hat u_i \hat u_j \\
183  * T_{ik}+T_{ki} &=& 2(1-\cos\phi)\,\hat u_i \hat u_k
184  * \f}
185  * This enables the calculation of two components of the unit vector. One
186  * component needs to be computed by one of the two previous methods.
187  *
188  * \par Assumptions and Limitations
189  * - The matrix is a proper transformation matrix.
190  * \param[in] trans Transformation matrix
191  * \param[out] eigen_angle Resultant rotation angle\n Units: r
192  * \param[out] eigen_axis Resultant rotation axis
193  */
194 void Orientation::compute_eigen_rotation_from_matrix(const double trans[3][3],
195  double * eigen_angle,
196  double eigen_axis[3])
197 {
198  double cosphi; // Cosing of the eigen angle
199  double sinphi; // Sine of the eigen angle
200  double diff[3]; // Symmetric difference vector
201  double dmag; // Magnitude of the symmetric difference vector
202 
203  // Calculate the symmetric difference vector and its magnitude.
204  diff[0] = trans[1][2] - trans[2][1];
205  diff[1] = trans[2][0] - trans[0][2];
206  diff[2] = trans[0][1] - trans[1][0];
207  dmag = Vector3::vmag(diff);
208 
209  // Determine the sine and cosine of the rotation angle.
210  sinphi = 0.5 * dmag;
211  cosphi = 0.5 * (trans[0][0] + trans[1][1] + trans[2][2] - 1.0);
212 
213  // Now see which solution to use.
214  // There are four cases.
215  // Use the appropriate method and solve for the angle and the axis.
216 
217  // Case 1: The trivial case, rotation angle = 0.
218  // This case is evidenced by sinphi = 0, cosphi = 1.
219  // The rotation is ill-defined in this case.
220  // Arbitrarily set the rotation vector to xhat.
221  if((std::fpclassify(dmag) == FP_ZERO) && (cosphi > 0.0))
222  {
223  *eigen_angle = 0.0;
224  eigen_axis[0] = 1.0;
225  eigen_axis[1] = 0.0;
226  eigen_axis[2] = 0.0;
227  }
228 
229  // Case 2: Rotation angle between 0 and 135 degrees (exclusive).
230  // This case is evidenced by sinphi greater than -cosphi.
231  // Here the symmetric difference (method 1) does the best job of determining
232  // the eigen axis. Method 1 yields for best results for the rotation angle
233  // for small rotation angles (0 to 45 degrees); method 2 is better for
234  // larger values.
235  else if(sinphi >= -cosphi)
236  {
237  // The rotation angle is given by
238  // phi = acos(cosphi)
239  // or
240  // phi = asin(sinphi)
241  // The inverse sine yields better results for angles less than 45 degrees
242  // while the inverse cosine is more accurate for larger angles.
243  if(sinphi < cosphi)
244  {
245  *eigen_angle = std::asin(sinphi);
246  }
247  else
248  {
249  *eigen_angle = std::acos(cosphi);
250  }
251 
252  // Use method 1 to construct the eigen axis.
253  Vector3::scale(diff, 1.0 / dmag, eigen_axis);
254  }
255 
256  // Case 3: Rotation angle between 135 and 180 degrees (inclusive).
257  // Here the symmetric difference does the best job of determining the
258  // rotation angle and the symmetric sum does the best job of determining
259  // the eigen axis.
260  else
261  {
262  unsigned int ii; // Index of largest diagonal element
263  double omcp = 1.0 - cosphi; // 1 - cos(phi)
264 
265  // Use the trace to determine the eigen rotation angle,
266  // taking care to place the angle in the correct quadrant.
267  *eigen_angle = M_PI - std::asin(sinphi);
268 
269  // The symmetric sum can only determine two of the three components
270  // of the eigen axis. The symmetric difference needs to be used for
271  // one of the components. The best accuracy results from choosing the
272  // the component correponding to the largest diagonal element.
273  // Find the index of the largest diagonal element.
274  ii = 0;
275  if(trans[1][1] > trans[ii][ii])
276  {
277  ii = 1;
278  }
279  if(trans[2][2] > trans[ii][ii])
280  {
281  ii = 2;
282  }
283 
284  // Construct the iith element of the eigen axis vector using
285  // method 2 to determine the magnitude, method 1 to determine the sign.
286  eigen_axis[ii] = std::sqrt((trans[ii][ii] - cosphi) / omcp);
287  if(diff[ii] < 0.0)
288  {
289  eigen_axis[ii] = -eigen_axis[ii];
290  }
291 
292  // Determine the remaining elements of the eigen axis via method 3.
293  double vscale = 1. / (2 * omcp * eigen_axis[ii]);
294  for(unsigned int jj = 0; jj < 3; jj++)
295  {
296  if(jj != ii)
297  {
298  eigen_axis[jj] = (trans[ii][jj] + trans[jj][ii]) * vscale;
299  }
300  }
301  }
302 }
