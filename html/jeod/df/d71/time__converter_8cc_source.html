104 /**
105  * Check whether this converter is able to handle the requested conversion(s).
106  * If query is compound (e.g. CONV_ALL, CONV_A_TO_B_UPDATE|CONV_B_TO_A_UPDATE)
107  * then return true only if capable of all conversions
108  * @return whether this converter can do all the conversions
109  * \param[in] query converter directions to check
110  */
111 bool TimeConverter::can_convert(Direction query)
112 {
113  if(query == NO_DIRECTION)
114  {
115  if(valid_directions == NO_DIRECTION)
116  {
117  return true;
118  }
119  else
120  {
121  return false;
122  }
123  }
124  else
125  {
126  return (query & valid_directions) == query;
127  }
