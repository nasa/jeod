74  * Initialize a Planet object.

76  * \par Assumptions and Limitations
77  * - Planet::register_model has already been invoked.
78  */
79 void Planet::initialize()
80 {
81  // Sanity check: The name must be valid.
82  // Note that this is a redundant check (the redundancy is necessary).
83  if(name.empty())
84  {
85  MessageHandler::fail(__FILE__, __LINE__, PlanetMessages::name_error, "Planet object has a null or empty name.");
86  return;
87  }
88 
89  // Sanity check: The model must have been registered.
90  if(grav_source == nullptr)
91  {
92  MessageHandler::fail(__FILE__,
93  __LINE__,
94  PlanetMessages::registration_error,
95  "Gravity Source has not been registered.");
96  return;
97  }
98 
99  // Require equatorial radius value to be provided
100  if(std::fpclassify(r_eq) == FP_ZERO)
101  {
102  MessageHandler::fail(__FILE__,
103  __LINE__,
104  PlanetMessages::registration_error,
105  "Equatorial radius for planet %s not provided.\n"
106  "Please specify value greater than 0.",
107  name.c_str());
108  return;
109  }
110 
111  // Set other data fields based on what was given by the user.
112 
113  // Flattening coefficient must be between 0 and 1 (exclusive).
114  // Flattening is related to the eccentricity via e^2 = f*(2-f).
115  if(std::fpclassify(flat_coeff) != FP_ZERO)
116  {
117  if((flat_coeff < 0.0) || (flat_coeff >= 1.0))
118  {
119  MessageHandler::fail(__FILE__,
120  __LINE__,
121  PlanetMessages::domain_error,
122  "Flattening coefficient %g for planet %s is out of range.\n"
123  "The value must be between 0 (inclusive) and 1 (exclusive).",
124  flat_coeff,
125  name.c_str());
126  return;
127  }
128 
129  flat_inv = 1.0 / flat_coeff;
130  e_ellip_sq = (2.0 * flat_coeff) - (flat_coeff * flat_coeff);
131  e_ellipsoid = std::sqrt(e_ellip_sq);
132  r_pol = r_eq * (1.0 - flat_coeff);
133  }
134 
135  // Inverse flattening must be between 1 and infinity.
136  else if(std::fpclassify(flat_inv) != FP_ZERO)
137  {
138  if(flat_inv <= 1.0)
139  {
140  MessageHandler::fail(__FILE__,
141  __LINE__,
142  PlanetMessages::domain_error,
143  "Inverse flattening coefficient %g for planet %s is out of range.\n"
144  "The value must more than 1.",
145  flat_inv,
146  name.c_str());
147  return;
148  }
149  flat_coeff = 1.0 / flat_inv;
150  e_ellip_sq = (2.0 * flat_coeff) - (flat_coeff * flat_coeff);
151  e_ellipsoid = std::sqrt(e_ellip_sq);
152  r_pol = r_eq * (1.0 - flat_coeff);
153  }
154 
155  // Eccentricity must be between 0 and 1 (exclusive).
156  else if(std::fpclassify(e_ellipsoid) != FP_ZERO)
157  {
158  if((e_ellipsoid < 0.0) || (e_ellipsoid >= 1.0))
159  {
160  MessageHandler::fail(__FILE__,
161  __LINE__,
162  PlanetMessages::domain_error,
163  "Ellipsoid eccentricity %g for planet %s is out of range.\n"
164  "The value must be between 0 (inclusive) and 1 (exclusive).",
165  e_ellipsoid,
166  name.c_str());
167  return;
168  }
169 
170  e_ellip_sq = e_ellipsoid * e_ellipsoid;
171  flat_coeff = 1.0 - std::sqrt(1.0 - e_ellip_sq);
172  flat_inv = 1.0 / flat_coeff;
173  r_pol = r_eq * (1.0 - flat_coeff);
174  }
175 
176  // Eccentricity squared must be between 0 and 1 (exclusive).
177  else if(std::fpclassify(e_ellip_sq) != FP_ZERO)
178  {
179  if((e_ellip_sq < 0.0) || (e_ellip_sq >= 1.0))
180  {
181  MessageHandler::fail(__FILE__,
182  __LINE__,
183  PlanetMessages::domain_error,
184  "Eccentricity squared %g for planet %s is out of range.\n"
185  "The value must be between 0 (inclusive) and 1 (exclusive).",
186  e_ellip_sq,
187  name.c_str());
188  return;
189  }
190 
191  e_ellipsoid = std::sqrt(e_ellip_sq);
192  flat_coeff = 1.0 - std::sqrt(1.0 - e_ellip_sq);
193  flat_inv = 1.0 / flat_coeff;
194  r_pol = r_eq * (1.0 - flat_coeff);
195  }
196 
197  // Polar radius must be between 0 (exclusive) and the equatorial radius.
198  else if(std::fpclassify(r_pol) != FP_ZERO)
199  {
200  if((r_pol < 0.0) || (r_pol > r_eq))
201  {
202  MessageHandler::fail(__FILE__,
203  __LINE__,
204  PlanetMessages::domain_error,
205  "Polar radius %g for planet %s is out of range.\n"
206  "The value must not exceed the equatorial radius.",
207  name.c_str(),
208  r_pol);
209  return;
210  }
211 
212  flat_coeff = 1.0 - r_pol / r_eq;
213  flat_inv = 1.0 / flat_coeff;
214  e_ellip_sq = (2.0 * flat_coeff) - (flat_coeff * flat_coeff);
215  e_ellipsoid = std::sqrt(e_ellip_sq);
216  }
217 
218  // Everything empty -- defaults to perfectly spherical planet
219  else
220  {
221  flat_coeff = 0.0;
222  flat_inv = 0.0;
223  e_ellipsoid = 0.0;
224  e_ellip_sq = 0.0;
225  r_pol = r_eq;
226  }
227 }
