106  * Initialize a Planet object.
107  *
108  * \par Assumptions and Limitations
109  * - Planet::register_model has already been invoked.
110  */
111 void
112 Planet::initialize (
113  void)
114 {
115 
116 
117  // Sanity check: The name must be valid.
118  // Note that this is a redundant check (the redundancy is necessary).
119  if (name.empty()) {
120  MessageHandler::fail (
121  __FILE__, __LINE__, PlanetMessages::name_error,
122  "Planet object has a null or empty name.");
123  return;
124  }
125 
126  // Sanity check: The model must have been registered.
127  if (grav_source == nullptr) {
128  MessageHandler::fail (
129  __FILE__, __LINE__, PlanetMessages::registration_error,
130  "Gravity Source has not been registered.");
131  return;
132  }
133 
134  // Require equatorial radius value to be provided
135  if (std::fpclassify(r_eq) == FP_ZERO) {
136  MessageHandler::fail (
137  __FILE__, __LINE__, PlanetMessages::registration_error,
138  "Equatorial radius for planet %s not provided.\n"
139  "Please specify value greater than 0.",
140  name.c_str());
141  return;
142  }
143 
144  // Set other data fields based on what was given by the user.
145 
146  // Flattening coefficient must be between 0 and 1 (exclusive).
147  // Flattening is related to the eccentricity via e^2 = f*(2-f).
148  if (std::fpclassify(flat_coeff) != FP_ZERO) {
149  if ((flat_coeff < 0.0) || (flat_coeff >= 1.0)) {
150  MessageHandler::fail (
151  __FILE__, __LINE__, PlanetMessages::domain_error,
152  "Flattening coefficient %g for planet %s is out of range.\n"
153  "The value must be between 0 (inclusive) and 1 (exclusive).",
154  flat_coeff, name.c_str());
155  return;
156  }
157 
158  flat_inv = 1.0 / flat_coeff;
159  e_ellip_sq = (2.0 * flat_coeff) - (flat_coeff * flat_coeff);
160  e_ellipsoid = std::sqrt (e_ellip_sq);
161  r_pol = r_eq * (1.0 - flat_coeff);
162  }
163 
164  // Inverse flattening must be between 1 and infinity.
165  else if (std::fpclassify(flat_inv) != FP_ZERO) {
166  if (flat_inv <= 1.0) {
167  MessageHandler::fail (
168  __FILE__, __LINE__, PlanetMessages::domain_error,
169  "Inverse flattening coefficient %g for planet %s is out of range.\n"
170  "The value must more than 1.",
171  flat_inv, name.c_str());
172  return;
173  }
174  flat_coeff = 1.0 / flat_inv;
175  e_ellip_sq = (2.0 * flat_coeff) - (flat_coeff * flat_coeff);
176  e_ellipsoid = std::sqrt (e_ellip_sq);
177  r_pol = r_eq * (1.0 - flat_coeff);
178  }
179 
180  // Eccentricity must be between 0 and 1 (exclusive).
181  else if (std::fpclassify(e_ellipsoid) != FP_ZERO) {
182  if ((e_ellipsoid < 0.0) || (e_ellipsoid >= 1.0)) {
183  MessageHandler::fail (
184  __FILE__, __LINE__, PlanetMessages::domain_error,
185  "Ellipsoid eccentricity %g for planet %s is out of range.\n"
186  "The value must be between 0 (inclusive) and 1 (exclusive).",
187  e_ellipsoid, name.c_str());
188  return;
189  }
190 
191  e_ellip_sq = e_ellipsoid * e_ellipsoid;
192  flat_coeff = 1.0 - std::sqrt (1.0 - e_ellip_sq);
193  flat_inv = 1.0 / flat_coeff;
194  r_pol = r_eq * (1.0 - flat_coeff);
195  }
196 
197  // Eccentricity squared must be between 0 and 1 (exclusive).
198  else if (std::fpclassify(e_ellip_sq) != FP_ZERO) {
199  if ((e_ellip_sq < 0.0) || (e_ellip_sq >= 1.0)) {
200  MessageHandler::fail (
201  __FILE__, __LINE__, PlanetMessages::domain_error,
202  "Eccentricity squared %g for planet %s is out of range.\n"
203  "The value must be between 0 (inclusive) and 1 (exclusive).",
204  e_ellip_sq, name.c_str());
205  return;
206  }
207 
208  e_ellipsoid = std::sqrt (e_ellip_sq);
209  flat_coeff = 1.0 - std::sqrt (1.0 - e_ellip_sq);
210  flat_inv = 1.0 / flat_coeff;
211  r_pol = r_eq * (1.0 - flat_coeff);
212  }
213 
214  // Polar radius must be between 0 (exclusive) and the equatorial radius.
215  else if (std::fpclassify(r_pol) != FP_ZERO) {
216  if ((r_pol < 0.0) || (r_pol > r_eq)) {
217  MessageHandler::fail (
218  __FILE__, __LINE__, PlanetMessages::domain_error,
219  "Polar radius %g for planet %s is out of range.\n"
220  "The value must not exceed the equatorial radius.",
221  name.c_str(), r_pol);
222  return;
223  }
224 
225  flat_coeff = 1.0 - r_pol / r_eq;
226  flat_inv = 1.0 / flat_coeff;
227  e_ellip_sq = (2.0 * flat_coeff) - (flat_coeff * flat_coeff);
228  e_ellipsoid = std::sqrt (e_ellip_sq);
229  }
230 
231  // Everything empty -- defaults to perfectly spherical planet
232  else {
233  flat_coeff = 0.0;
234  flat_inv = 0.0;
235  e_ellipsoid = 0.0;
236  e_ellip_sq = 0.0;
237  r_pol = r_eq;
238  }
239 }
