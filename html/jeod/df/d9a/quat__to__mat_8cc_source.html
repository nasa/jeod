26 /*------------------------------------------------------------------------------
27  * Overview
28  *
29  * Nomenclature
30  * q Left quaternion
31  * a Transformation matrix based on q
32  * qs The scalar part of the quaternion
33  * qv The vector part of the quaternion
34  * qv_i Element i of qv
35  * a_ij Element i,j of a
36  * t Rotation angle
37  * cost Cosine of t
38  * sint Sine of t
39  * u Rotation axis unit vector
40  * eps_ijk Permutation tensor
41  *
42  * Given a single axis rotation t from reference frame 1 to reference frame 2,
43  * the transformation matrix from frame 1 to frame 2 is
44  * a_ii = 1 - (1-cost)*(1-u_i^2)
45  * = cost + (1-cost)*u_i^2
46  * a_ij = (1-cost) u_i u_j + sint eps_ijk u_k i!=j
47  * The left transformation quaternion from frame 1 to frame 2 is
48  * qs = cos(t/2)
49  * qv_i = -sin(t/2)*u_i
50  * The transformation matrix can be expressed in terms of the left
51  * quaternion by using the half-angle formulae
52  * cost = 2*cos^2(t/2) - 1
53  * sint = 2*cos(t/2)*sin(t/2)
54  * 1-cost = 2*sin^2(t/2)
55  * With a little manipulation,
56  * cost = 2*qs^2 - 1
57  * a_ii = cost + 2*qv_i^2
58  * a_ij = 2*(qv_i*qv_j - eps_ijk*qs*qv_k) i!=j
59  *
60  *----------------------------------------------------------------------------*/

74 /**
75  * Compute the parent-to-child transformation matrix from the
76  * parent-to-child left quaternion.
77  *
78  * \par Assumptions and Limitations
79  * - Quaternion is normalized.
80  * \param[out] T Transformation matrix
81  */
82 void Quaternion::left_quat_to_transformation(double T[3][3]) const
83 {
84  double cost; // Cosine of rotation angle
85  double qvx2[3]; // qvx2_i = 2 qv_i (See discussion above)
86  double qsqv2[3]; // qsqv2_i = 2 qs qv_i
87  double qvqv2[3]; // qvqv2_i = 2 qv_j qv_k
88 
89  // Compute the cosine of the rotation angle.
90  cost = 2.0 * scalar * scalar - 1.0;
91 
92  // Form intermediate vectors qvx2, qsqv2, qvqv2.
93  Vector3::sum(vector, vector, qvx2);
94  Vector3::scale(qvx2, scalar, qsqv2);
95  qvqv2[0] = vector[1] * qvx2[2];
96  qvqv2[1] = vector[2] * qvx2[0];
97  qvqv2[2] = vector[0] * qvx2[1];
98 
99  // Construct the transformation matrix diagonal:
100  // T_ii = cost + 2 qv_i^2 = cost + qv_i*qvx2_i
101  T[0][0] = cost + vector[0] * qvx2[0];
102  T[1][1] = cost + vector[1] * qvx2[1];
103  T[2][2] = cost + vector[2] * qvx2[2];
104 
105  // Contruct off-diagonal transformation matrix elements:
106  // T_ij = 2 (qv_i qv_j - eps_ijk qs qv_k)
107  // = qvqv2_k - eps_ijk qsqv2_k
108  T[0][1] = qvqv2[2] - qsqv2[2];
109  T[1][0] = qvqv2[2] + qsqv2[2];
110  T[1][2] = qvqv2[0] - qsqv2[0];
111  T[2][1] = qvqv2[0] + qsqv2[0];
112  T[2][0] = qvqv2[1] - qsqv2[1];
113  T[0][2] = qvqv2[1] + qsqv2[1];
114 }
