64 InteractionFacet * PointContactFacetFactory::create_facet(Facet * facet, FacetParams * params)
65 {
66  ContactParams * contact_params = nullptr;
67  FlatPlateCircular * plate = nullptr;
68 
69  contact_params = dynamic_cast<ContactParams *>(params);
70  plate = dynamic_cast<FlatPlateCircular *>(facet);
71 
72  // We have tried casting the facet and params to the correct types.
73  // if they were not the correct type, send out an error message
74  if(contact_params == nullptr)
75  {
76  MessageHandler::fail(__FILE__,
77  __LINE__,
78  ContactMessages::initialization_error,
79  "the params supplied to "
80  "PointContactFacetFactory::create_facet was not of type "
81  "ContactParams, as is required");
82  return nullptr;
83  }
84  if(plate == nullptr)
85  {
86  MessageHandler::fail(__FILE__,
87  __LINE__,
88  ContactMessages::initialization_error,
89  "the Facet supplied to "
90  "PointContactFacetFactory::create_facet was not of type "
91  "CircularFlatPlate, as is required");
92  return nullptr;
93  }
94 
95  // Create the interaction facet
96  PointContactFacet * inter_facet = JEOD_ALLOC_CLASS_OBJECT(PointContactFacet, ());
97 
98  // Fill it out from the parameters and from the facet itself
99  inter_facet->base_facet = facet;
100 
101  inter_facet->radius = plate->radius;
102  inter_facet->set_max_dimension();
103 
104  inter_facet->surface_type = contact_params;
105 
106  Vector3::copy(plate->position, inter_facet->position);
107  Vector3::copy(plate->normal, inter_facet->normal);
108 
109  inter_facet->vehicle_body = facet->get_mass_body_ptr()->dyn_owner;
110  if(inter_facet->vehicle_body == nullptr)
111  {
112  MessageHandler::fail(__FILE__,
113  __LINE__,
114  ContactMessages::initialization_error,
115  "mass_body_ptr contained in the facet supplied to"
116  "PointContactFacetFactory::create_facet was not of type "
117  "DynBody, as is required");
118  return nullptr;
119  }
120 
121  inter_facet->create_vehicle_point();
122 
123  return inter_facet;
124 }
