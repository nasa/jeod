89 /**
90  * Class template that provides static functions @a construct and @a destruct
91  * that construct an array of objects.
92  * @tparam T Type
93  * @tparam is_class True if the type T is a class.
94  * @tparam is_abstract True if the type T is an abstract class.
95  *
96  * This class template is intended for used by jeod_alloc_construct_array and
97  * jeod_alloc_destruct_array. Use in any other context is caveat emptor.
98  *
99  * This template provides do-nothing implementations, which is about all one
100  * can do for arrays of abstract objects (which can't exist).
101  */
102 template<typename T, bool is_class, bool is_abstract> class JeodAllocHelperConstructDestruct
103 {
104 public:
105  /**
106  * Construct an array of objects.
107  * @return Constructed array.
108  * @param nelem Number of elements in the array
109  * @param addr Address to be constructed
110  */
111  static void * construct(std::size_t nelem JEOD_UNUSED, void * addr)
112  {
113  return addr;
114  }
115 
116  /**
117  * Destruct an array of objects.
118  * @param nelem Number of elements in the array
119  * @param addr Address to be destructed
120  */
121  static void destruct(std::size_t nelem JEOD_UNUSED, void * addr JEOD_UNUSED)
122  {
123  ; // Nothing to do.
124  }
125 };

156 /**
157  * Partial instantiation for non-abstract classes.
158  */
159 template<typename T> class JeodAllocHelperConstructDestruct<T, true, false>
160 {
161 public:
162  /**
163  * Construct an array of objects.
164  * @return Constructed array.
165  * @param nelem Number of elements in the array
166  * @param addr Address to be constructed
167  */
168  static void * construct(std::size_t nelem, void * addr)
169  {
170  return new(addr) T[nelem];
171  }
172 
173  /**
174  * Destruct an array of objects.
175  * @param nelem Number of elements in the array
176  * @param addr Address to be destructed
177  */
178  static void destruct(std::size_t nelem, void * addr)
179  {
180  auto * array = reinterpret_cast<T *>(addr);
181  for(unsigned int ii = 0; ii < nelem; ++ii)
182  {
183  array[ii].~T();
184  }
185  }
186 };
