111  // Call parent-class initializer

115 /**
116  * Update the complete RNP and supply it to the Dynamics Manager.
117  * \param[in] time_tt Current Terrestrial Time
118  */
119 void RNPMars::update_rnp(TimeTT & time_tt)
120 {
121  // Do nothing if model inactive
122  if(!active)
123  {
124  return;
125  }
126 
127  // If the DynTime pointer is empty, connect it
128  if(time_dyn_ptr == nullptr)
129  {
130  get_dyn_time_ptr(time_tt);
131  }
132 
133  // If the TimeTT pointer hasn't been cached off yet, do so
134  if(tt_ptr == nullptr)
135  {
136  tt_ptr = &time_tt;
137  }
138 
139  // Check if RNP needs update; if it has been previously updated and the
140  // timestamp still matches, then it doesn't
141  if(!never_updated_full && Numerical::compare_exact(last_updated_time_full, time_dyn_ptr->seconds))
142  {
143  return;
144  }
145 
146  // Proceed with full RNP update, including rotational portion
147  never_updated_full = false;
148  never_updated_rotational = false;
149  last_updated_time_full = last_updated_time_rotational = time_dyn_ptr->seconds;
150 
151  double time = 0.0;
152  time = time_tt.seconds;
153 
154  // Pass in TT to update all if user requested full RNP, or call rotational
155  // update only and pass in simulation time to it if only rotation requested
156  if(rotation != nullptr)
157  {
158  if(rnp_type == FullRNP)
159  {
160  rotation->update_time(time);
161  nutation->update_time(time);
162  precession->update_time(time);
163  }
164  else
165  {
166  time = time_tt.time_manager->dyn_time.seconds;
167  rotation->update_time(time);
168  }
169  }
170 
171  // Update the timestamp of the controlled reference frame
172  planet->pfix.set_timestamp(time_dyn_ptr->seconds);
173 
174  // Pass the function call to the parent object
175  PlanetRNP::update_rnp();
176 }

275 void RNPMars::get_dyn_time_ptr(TimeTT & time_tt)
276 {
277  // Check if pointer needs to be set; exit if not
278  if(time_dyn_ptr != nullptr)

288  MessageHandler::fail(__FILE__,
289  __LINE__,
