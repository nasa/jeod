137 /**
138  * Update the complete RNP and supply it to the Dynamics Manager.
139  * \param[in] time_tt Current Terrestrial Time
140  */
141 void
142 RNPMars::update_rnp (
143  TimeTT& time_tt)
144 {
145 
146  // Do nothing if model inactive
147  if(!active) {
148  return;
149  }
150 
151  // If the DynTime pointer is empty, connect it
152  if(time_dyn_ptr == nullptr) {
153  get_dyn_time_ptr (time_tt);
154  }
155 
156  // If the TimeTT pointer hasn't been cached off yet, do so
157  if(tt_ptr == nullptr) {
158  tt_ptr = &time_tt;
159  }
160 
161  // Check if RNP needs update; if it has been previously updated and the
162  // timestamp still matches, then it doesn't
163  if(!never_updated_full &&
164  Numerical::compare_exact(last_updated_time_full, time_dyn_ptr->seconds)) {
165  return;
166  }
167 
168  // Proceed with full RNP update, including rotational portion
169  never_updated_full = false;
170  never_updated_rotational = false;
171  last_updated_time_full = last_updated_time_rotational = time_dyn_ptr->seconds;
172 
173  double time = 0.0;
174  time = time_tt.seconds;
175 
176  // Pass in TT to update all if user requested full RNP, or call rotational
177  // update only and pass in simulation time to it if only rotation requested
178  if (rotation != nullptr) {
179  if (rnp_type == FullRNP) {
180  rotation->update_time (time);
181  nutation->update_time (time);
182  precession->update_time (time);
183  }
184  else {
185  time = time_tt.time_manager->dyn_time.seconds;
186  rotation->update_time (time);
187  }
188  }
189 
190  // Update the timestamp of the controlled reference frame
191  planet->pfix.set_timestamp(time_dyn_ptr->seconds);
192 
193  // Pass the function call to the parent object
194  PlanetRNP::update_rnp();
195 
196  return;
197 }

302 RNPMars::get_dyn_time_ptr(

304 {
305 
306  // Check if pointer needs to be set; exit if not

318  "The TimeManager associated with the given TimeTT "
