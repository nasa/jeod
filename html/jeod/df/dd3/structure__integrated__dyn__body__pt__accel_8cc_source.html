30 // Compute the state derivatives at a vehicle point.
31 void StructureIntegratedDynBody::compute_vehicle_point_derivatives(const BodyRefFrame & vehicle_pt,
32  FrameDerivs & pt_derivs)
33 {
34  // The root body for this body.
35  // Only the root body's derivs element contains valid data.
36  const auto * root_body = dynamic_cast<const StructureIntegratedDynBody *>(get_root_body());
37 
38  // The root body's structure.
39  const MassPoint & root_ppty = root_body->mass.structure_point;
40 
41  // The root body's structural rotational state.
42  const RefFrameRot & root_rot = root_body->structure.state.rot;
43 
44  // The derivatives for this root body.
45  const FrameDerivs & root_derivs = root_body->struct_derivs;
46 
47  // Other local variables
48  MassPointState str_to_point; // -- Vehicle pt state wrt structure
49  double str_to_point_inrtl[3]; // M Point wrt CoM, inertial coordinates
50  double wxr[3]; // M/s Omega cross r
51  double wxwxr[3]; // M/s2 Centrifugal accel
52  double axr[3]; // M/s2 Acceleration due to angular accel
53  double accel_rot_body[3]; // M/s2 Acceleration due to rotation
54  double accel_rot_inrtl[3]; // M/s2 Acceleration due to rotation
55  double delta_acc_grav[3]; // M/s2 Acceleration due to gravity gradient
56  double delta_acc_non_grav[3]; // M/s2 Delta acceleration, non-gravity
57 
58  // Sanity check: Never update the root body's derivs structure. Ever.
59  if(&pt_derivs == &root_derivs)
60  {
61  // Complain vehemently if the attempt is utterly invalid.
62  // (Requesting the root body to update its composite derivatives
63  // is silly, but no harm, no foul, so no complaint.)
64  if(&vehicle_pt != &root_body->composite_body)
65  {
66  MessageHandler::error(__FILE__,
67  __LINE__,
68  DynBodyMessages::invalid_frame,
69  "Illegal attempt to update the root body '%s' FrameDerivs\n",
70  root_body->name.c_str());
71  }
72  return;
73  }
74 
75  // Compute the vehicle point state wrt the composite center of mass.
76  vehicle_pt.mass_point->compute_relative_state(root_ppty, str_to_point);
77 
78  // Transform position to inertial.
79  Vector3::transform_transpose(root_rot.T_parent_this, str_to_point.position, str_to_point_inrtl);
80 
81  // Compute accelerations caused by vehicle rotation.
82  Vector3::cross(root_rot.ang_vel_this, str_to_point.position, wxr);
83  Vector3::cross(root_rot.ang_vel_this, wxr, wxwxr);
84  Vector3::cross(root_derivs.rot_accel, str_to_point.position, axr);
85  Vector3::sum(wxwxr, axr, accel_rot_body);
86  Vector3::transform_transpose(root_rot.T_parent_this, accel_rot_body, accel_rot_inrtl);
87 
88  // Compute delta gravitational acceleration.
89  Vector3::transform(grav_interaction.grav_grad, str_to_point_inrtl, delta_acc_grav);
90 
91  // Compute accelerations.
92  Vector3::diff(accel_rot_inrtl, delta_acc_grav, delta_acc_non_grav);
93  Vector3::sum(root_derivs.non_grav_accel, delta_acc_non_grav, pt_derivs.non_grav_accel);
94  Vector3::sum(root_derivs.trans_accel, accel_rot_inrtl, pt_derivs.trans_accel);
95  Vector3::transform(str_to_point.T_parent_this, root_derivs.rot_accel, pt_derivs.rot_accel);
96  vehicle_pt.state.rot.Q_parent_this.compute_left_quat_deriv(vehicle_pt.state.rot.ang_vel_this,
97  pt_derivs.Qdot_parent_this);
98 }
