57 /**
58  * Attach this mass body's root body as a child of the specified mass
59  * body such that the specified mass points on the two bodies are
60  * coincident and the frames associated with those mass points are
61  * related by a 180 degree yaw.
62  * @return Success indicator: true=success, false=attachment not performed.
63  * \param[in] this_point_name The name of a mass point contained in this mass body's list of mass points.
64  * \param[in] parent_point_name The name of a mass point contained in the new parent body's list of mass points.
65  * \param[in,out] parent The new parent body; the body to which this body's root body is to be attached.
66  */
67 bool MassBody::attach_to(const std::string & this_point_name, const std::string & parent_point_name, MassBody & parent)
68 {
69  const MassPoint * this_point;
70  const MassPoint * parent_point;
71  MassPoint this_struct_wrt_this_point;
72  MassPoint this_point_wrt_parent_point;
73  MassPointState this_struct_wrt_parent_struct;
74  bool success;
75 
76  if(!attach_validate(parent, true))
77  {
78  success = false;
79  return success;
80  }
81 
82  // Find the mass points corresponding to the input names.
83  this_point = find_mass_point(this_point_name);
84  parent_point = parent.find_mass_point(parent_point_name);
85 
86  // Handle failed searches as non-fatal errors (warning + failure return).
87  if((this_point == nullptr) || (parent_point == nullptr))
88  {
89  generate_bad_point_message(__FILE__,
90  __LINE__,
91  name.c_str(),
92  this_point_name,
93  this_point,
94  parent.name.c_str(),
95  parent_point_name,
96  parent_point);
97 
98  return false;
99  }
100 
101  // Construct the affine transformation from the mass point location and
102  // frame of the attach point on this body to this body's structural origin
103  // and frame; the inverse of the mass point specification.
104  Vector3::transform(this_point->T_parent_this, this_point->position, this_struct_wrt_this_point.position);
105  Vector3::negate(this_struct_wrt_this_point.position);
106  Matrix3x3::transpose(this_point->T_parent_this, this_struct_wrt_this_point.T_parent_this);
107  this_point->Q_parent_this.conjugate(this_struct_wrt_this_point.Q_parent_this);
108 
109  // Construct the affine transformation from the attach point on this body
110  // to the attach point on the new parent body.
111  // This is always a zero offset and a 180 degree yaw.
112  // The point is initialized with a zero offset and an identity transform.
113  this_point_wrt_parent_point.T_parent_this[0][0] = -1.0;
114  this_point_wrt_parent_point.T_parent_this[1][1] = -1.0;
115  this_point_wrt_parent_point.Q_parent_this.scalar = 0.0;
116  this_point_wrt_parent_point.Q_parent_this.vector[2] = 1.0;
117 
118  // Link the constructed points via the parent elements.
119  // The result will be a chain from this body's structural origin/frame
120  // to the new parent body's structural origin/frame.
121  this_point_wrt_parent_point.attach(const_cast<MassPoint &>(*parent_point));
122  this_struct_wrt_this_point.attach(this_point_wrt_parent_point);
123 
124  // Compute the affine transformation from the parent body's structural
125  // origin/frame to this body's structural origin/frame.
126  this_struct_wrt_this_point.compute_state_wrt_pred(parent.structure_point, this_struct_wrt_parent_struct);
127 
128  // Complete attachment using the geometry-based attach method,
129  // returning the success indicator from that method as this method's status.
130  success = attach_to(this_struct_wrt_parent_struct.position, this_struct_wrt_parent_struct.T_parent_this, parent);
131 
132  // Detach the temporary points.
133  this_struct_wrt_this_point.detach();
134  this_point_wrt_parent_point.detach();
135 
136  return success;
137 }

139 /**
140  * Attach this mass body's root body as a child of the specified mass
141  * body such that this body's structural origin is offset from the
142  * parent body's structural origin and this body's structural axes
143  * are oriented with respect to the parent body's structural axes as
144  * specified.
145  * @return Success indicator: true=success, false=attachment not performed.
146  * \param[in] offset_pstr_cstr_pstr Location of this body's structural origin with respect to the new parent body's
147  * structural origin, specified in structural coordinates of the new parent body.\n Units: M \param[in] T_pstr_cstr
148  * Transformation matrix from the new parent body's structural frame to this body's structural frame. \param[in,out]
149  * parent The new parent body; the body to which this body's root body is to be attached.
150  */
151 bool MassBody::attach_to(double offset_pstr_cstr_pstr[3], double T_pstr_cstr[3][3], MassBody & parent)
152 {
153  MassBody * root_body; // This body's root body (pre-attachment).
154  bool success;
155 
156  if(!attach_validate(parent, true))
157  {
158  success = false;
159  return success;
160  }
161 
162  // Find this body's root body. It is the root body that is attached.
163  root_body = MassBody::get_root_body_internal();
164 
165  // Sanity check: Don't make an invalid tree.
166  if(root_body == parent.get_root_body())
167  {
168  MessageHandler::error(__FILE__,
169  __LINE__,
170  MassBodyMessages::invalid_attach,
171  "Bodies '%s' and '%s' are part of the same mass tree.\n"
172  "Attachment of body '%s' to '%s' was not performed.",
173  name.c_str(),
174  parent.name.c_str(),
175  name.c_str(),
176  parent.name.c_str());
177 
178  return false;
179  }
180 
181  // This is a root body: Use the attach_root_body primitive directly.
182  if(root_body == this)
183  {
184  success = attach_root_body(offset_pstr_cstr_pstr, T_pstr_cstr, parent);
185  }
186 
187  // This is not a root body: Offset the input transformation by the
188  // transformation to the root, then attach.
189  else
190  {
191  double T_pstr_rstr[3][3]; // Parent struct to root struct xform
192  double offset_pstr_rstr_pstr[3]; // Parent to child-root offset in parent
193  // structure frame
194  MassPointState this_struct_wrt_root_struct;
195 
196  // Give a low-level warning that the attaching body is not a root body.
197  MessageHandler::inform(__FILE__,
198  __LINE__,
199  MassBodyMessages::attach_info,
200  "Body '%s' is already attached to '%s'. Reattaching to '%s' would\n"
201  "either discard this existing attachment or create an invalid tree.\n"
202  "Attaching the root body for '%s', '%s', to '%s'.",
203  name.c_str(),
204  links.parent()->name.c_str(),
205  parent.name.c_str(),
206  name.c_str(),
207  root_body->name.c_str(),
208  parent.name.c_str());
209 
210  // Compute the affine transformation from the root structural frame
211  // to this body's structural frame.
212  structure_point.compute_state_wrt_pred(root_body->structure_point, this_struct_wrt_root_struct);
213 
214  // Recast the attachment transformation / offset in terms of the root.
215  Matrix3x3::product_left_transpose(this_struct_wrt_root_struct.T_parent_this, T_pstr_cstr, T_pstr_rstr);
216 
217  Vector3::copy(offset_pstr_cstr_pstr, offset_pstr_rstr_pstr);
218  Vector3::transform_transpose_decr(T_pstr_rstr, this_struct_wrt_root_struct.position, offset_pstr_rstr_pstr);
219 
220  // Attach.
221  success = root_body->attach_root_body(offset_pstr_rstr_pstr, T_pstr_rstr, parent);
222  }
223 
224  return success;
225 }

267 bool MassBody::attach_root_body(double offset_pstr_cstr_pstr[3], double T_pstr_cstr[3][3], MassBody & parent)
268 {
269  bool success; // -- Was attachment performed?
270 
271  // Enforce that this must be a root body.
272  // Note that this is a fatal error. This condition will only occur if
273  // some non-JEOD method erroneous invokes
274  if(!links.is_root())
275  {
276  MessageHandler::error(__FILE__,
277  __LINE__,
278  MassBodyMessages::invalid_attach,
279  "MassBody::attach_root_body erroneously invoked on non-root body.\n"
280  "Body name: '%s'",
281  name.c_str());
282  return false;
283  }
284 
285  // Perform the attachment, but only if it is valid to do so.
286  if(attach_validate(parent, true))
287  {
288  // Tell the child (this) to establish the connectivity.
289  attach_establish_links(parent);
290 
291  // Tell the parent to update properties/state/...
292  parent.attach_update_properties(offset_pstr_cstr_pstr, T_pstr_cstr, *this);
293 
294  // Note that the attachment was performed.
295  success = true;
296  }
297 
298  // Validation failed. Note that the attachment was not performed.
299  else
300  {
301  success = false;
302  }
303 
304  return success;

472 /**
473  * Set the relation between parent and child and update the mass
474  * properties.
475  *
476  * Extensibility comments --
477  * - This method is sent to the parent body of the attachment after
478  * the child body has established the logical connectivity between
479  * the parent body and child body.
480  * - The generic purpose of this method is to establish the physical
481  * relation between parent and child and to update any physical
482  * properties that change as a result of the attachment.
483  * - Any class that overrides this method must either invoke this method
484  * or perform the actions performed herein.
485  *
486  * \par Assumptions and Limitations
487  * - The attachment is valid and logical connectivity has been established.
488  * Neither assumption is checked.
489  * \param[in] offset_pstr_cstr_pstr Location of this body's structural origin with respect to the new parent body's
490  * structural origin, specified in structural coordinates of the new parent body.\n Units: M \param[in] T_pstr_cstr
491  * Transformation matrix from the new parent body's structural frame to this body's structural frame. \param[in,out]
492  * child The child body; the body newly attached to this body.
493  */
494 void MassBody::attach_update_properties(const double offset_pstr_cstr_pstr[3],
495  const double T_pstr_cstr[3][3],
496  MassBody & child)
497 {
498  // Set the child's attachment transformation and offset w.r.t. this body.
499  child.structure_point.update_orientation(T_pstr_cstr);
500  child.structure_point.update_point(offset_pstr_cstr_pstr);
501 
502  // Construct the transformation from the parent body's structural frame
503  // to child body's composite body frame:
504  // T_pstr_cbdy = T_cstr_cbdy * T_pstr_cstr
505  child.composite_properties.Q_parent_this.multiply(child.structure_point.Q_parent_this,
506  child.composite_wrt_pstr.Q_parent_this);
507  child.composite_wrt_pstr.compute_transformation();
508 
509  // Compute the displace from the parent body's structural origin to the
510  // child body's composite CoM in the parent body's structural frame:
511  // x_pstr_cbdy = x_pstr_cstr + (T_cstr_pstr^T)*x_cstr_cbdy
512  Vector3::transform_transpose(child.structure_point.T_parent_this,
513  child.composite_properties.position,
514  child.composite_wrt_pstr.position);
515  Vector3::incr(child.structure_point.position, child.composite_wrt_pstr.position);
516 
517  // Construct the transformation from the parent body's body frame
518  // to child body's composite body frame.
519  child.composite_wrt_pstr.Q_parent_this.multiply_conjugate(composite_properties.Q_parent_this,
520  child.composite_wrt_pbdy.Q_parent_this);
521  child.composite_wrt_pbdy.compute_transformation();
522 
523  // Update this body's mass properties.
524  // Note that this sets child.composite_wrt_pbdy.position.
525  set_update_flag();
526  get_root_body_internal()->update_mass_properties();
527 }
